173 void MRGenerator::collectModRefForLoadStore()
174 {
175 
176  SVFModule* svfModule = pta->getModule();
177  for (SVFModule::const_iterator fi = svfModule->begin(), efi = svfModule->end(); fi != efi;
178  ++fi)
179  {
180  const SVFFunction& fun = **fi;
181 
183  if (Options::IgnoreDeadFun() && fun.isUncalledFunction())
184  continue;
185 
186  for (SVFFunction::const_iterator iter = fun.begin(), eiter = fun.end();
187  iter != eiter; ++iter)
188  {
189  const SVFBasicBlock* bb = *iter;
190  for (const auto& inst: bb->getICFGNodeList())
191  {
192  SVFStmtList& pagEdgeList = getPAGEdgesFromInst(inst);
193  for (SVFStmtList::iterator bit = pagEdgeList.begin(), ebit =
194  pagEdgeList.end(); bit != ebit; ++bit)
195  {
196  const PAGEdge* inst = *bit;
197  pagEdgeToFunMap[inst] = &fun;
198  if (const StoreStmt *st = SVFUtil::dyn_cast<StoreStmt>(inst))
199  {
200  NodeBS cpts(pta->getPts(st->getLHSVarID()).toNodeBS());
201  // TODO: change this assertion check later when we have conditional points-to set
202  if (cpts.empty())
203  continue;
204  assert(!cpts.empty() && "null pointer!!");
205  addCPtsToStore(cpts, st, &fun);
206  }
207 
208  else if (const LoadStmt *ld = SVFUtil::dyn_cast<LoadStmt>(inst))
209  {
210  NodeBS cpts(pta->getPts(ld->getRHSVarID()).toNodeBS());
211  // TODO: change this assertion check later when we have conditional points-to set
212  if (cpts.empty())
213  continue;
214  assert(!cpts.empty() && "null pointer!!");
215  addCPtsToLoad(cpts, ld, &fun);
216  }
217  }
218  }
219  }
220  }

283 {
284 
285  if(cptsToRepCPtsMap.find(cpts)!=cptsToRepCPtsMap.end())
286  return;
287 
288  PointsToList subSetList;

477 void MRGenerator::collectCallSitePts(const CallICFGNode* cs)
478 {
480  NodeBS& argsPts = csToCallSiteArgsPtsMap[cs];
481  SVFIR* pag = pta->getPAG();
482  CallICFGNode* callBlockNode = pag->getICFG()->getCallICFGNode(cs->getCallSite());
483  RetICFGNode* retBlockNode = pag->getICFG()->getRetICFGNode(cs->getCallSite());
484 
485  WorkList worklist;
486  if (pag->hasCallSiteArgsMap(callBlockNode))
487  {
488  const SVFIR::SVFVarList& args = pta->getPAG()->getCallSiteArgsList(callBlockNode);
489  for(SVFIR::SVFVarList::const_iterator itA = args.begin(), ieA = args.end(); itA!=ieA; ++itA)
490  {
491  const PAGNode* node = *itA;
492  if(node->isPointer())
493  worklist.push(node->getId());
494  }
495  }
496 
497  while(!worklist.empty())
498  {
499  NodeID nodeId = worklist.pop();
500  const NodeBS& tmp = pta->getPts(nodeId).toNodeBS();
501  for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
502  argsPts |= CollectPtsChain(*it);
503  }
504 
506  NodeBS& retPts = csToCallSiteRetPtsMap[cs];
507 
508  if (pta->getPAG()->callsiteHasRet(retBlockNode))
509  {
510  const PAGNode* node = pta->getPAG()->getCallSiteRet(retBlockNode);
511  if(node->isPointer())
512  {
513  const NodeBS& tmp = pta->getPts(node->getId()).toNodeBS();
514  for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
515  retPts |= CollectPtsChain(*it);
516  }
517  }

601 bool MRGenerator::handleCallsiteModRef(NodeBS& mod, NodeBS& ref, const CallICFGNode* cs, const SVFFunction* callee)
602 {
604  if(isHeapAllocExtCall(cs))
605  {
606  SVFStmtList& pagEdgeList = getPAGEdgesFromInst(cs);
607  for (SVFStmtList::const_iterator bit = pagEdgeList.begin(),
608  ebit = pagEdgeList.end(); bit != ebit; ++bit)
609  {
610  const PAGEdge* edge = *bit;
611  if (const AddrStmt* addr = SVFUtil::dyn_cast<AddrStmt>(edge))
612  mod.set(addr->getRHSVarID());
613  }
614  }
616  else
617  {
618  mod = getModSideEffectOfFunction(callee);
619  ref = getRefSideEffectOfFunction(callee);
620  }
621  // add ref set
622  bool refchanged = addRefSideEffectOfCallSite(cs, ref);
623  // add mod set
624  bool modchanged = addModSideEffectOfCallSite(cs, mod);
625 
626  return refchanged || modchanged;

633 void MRGenerator::modRefAnalysis(CallGraphNode* callGraphNode, WorkList& worklist)
634 {
635 
637  for(CallGraphNode::iterator it = callGraphNode->InEdgeBegin(), eit = callGraphNode->InEdgeEnd();
638  it!=eit; ++it)
639  {
640  CallGraphEdge* edge = *it;
641 
643  for(CallGraphEdge::CallInstSet::iterator cit = edge->getDirectCalls().begin(),
644  ecit = edge->getDirectCalls().end(); cit!=ecit; ++cit)
645  {
646  NodeBS mod, ref;
647  const CallICFGNode* cs = (*cit);
648  bool modrefchanged = handleCallsiteModRef(mod, ref, cs, callGraphNode->getFunction());
649  if(modrefchanged)
650  worklist.push(edge->getSrcID());
651  }
653  for(CallGraphEdge::CallInstSet::iterator cit = edge->getIndirectCalls().begin(),
654  ecit = edge->getIndirectCalls().end(); cit!=ecit; ++cit)
655  {
656  NodeBS mod, ref;
657  const CallICFGNode* cs = (*cit);
658  bool modrefchanged = handleCallsiteModRef(mod, ref, cs, callGraphNode->getFunction());
659  if(modrefchanged)

661  }
662  }
