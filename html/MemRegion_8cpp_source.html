70void MRGenerator::createMR(const SVFFunction* fun, const NodeBS& cpts)
71{
72 const NodeBS& repCPts = getRepPointsTo(cpts);
73 MemRegion mr(repCPts);
74 MRSet::const_iterator mit = memRegSet.find(&mr);
75 if(mit!=memRegSet.end())
76 {
77 const MemRegion* mr = *mit;
78 MRSet& mrs = funToMRsMap[fun];
79 if(mrs.find(mr)==mrs.end())
80 mrs.insert(mr);
81 }
82 else
83 {
84 MemRegion* m = new MemRegion(repCPts);
85 memRegSet.insert(m);
86 funToMRsMap[fun].insert(m);
87 }

174void MRGenerator::collectModRefForLoadStore()
175{
176
177 CallGraph* svfirCallGraph = PAG::getPAG()->getCallGraph();
178 for (const auto& item: *svfirCallGraph)
179 {
180 const SVFFunction& fun = *item.second->getFunction();
181
183 if (Options::IgnoreDeadFun() && fun.isUncalledFunction())
184 continue;
185
186 for (SVFFunction::const_iterator iter = fun.begin(), eiter = fun.end();
187 iter != eiter; ++iter)
188 {
189 const SVFBasicBlock* bb = *iter;
190 for (const auto& inst: bb->getICFGNodeList())
191 {
192 SVFStmtList& pagEdgeList = getPAGEdgesFromInst(inst);
193 for (SVFStmtList::iterator bit = pagEdgeList.begin(), ebit =
194 pagEdgeList.end(); bit != ebit; ++bit)
195 {
196 const PAGEdge* inst = *bit;
197 pagEdgeToFunMap[inst] = &fun;
198 if (const StoreStmt *st = SVFUtil::dyn_cast<StoreStmt>(inst))
199 {
200 NodeBS cpts(pta->getPts(st->getLHSVarID()).toNodeBS());
201 // TODO: change this assertion check later when we have conditional points-to set
202 if (cpts.empty())
203 continue;
204 assert(!cpts.empty() && "null pointer!!");
205 addCPtsToStore(cpts, st, &fun);
206 }
207
208 else if (const LoadStmt *ld = SVFUtil::dyn_cast<LoadStmt>(inst))
209 {
210 NodeBS cpts(pta->getPts(ld->getRHSVarID()).toNodeBS());
211 // TODO: change this assertion check later when we have conditional points-to set
212 if (cpts.empty())
213 continue;
214 assert(!cpts.empty() && "null pointer!!");
215 addCPtsToLoad(cpts, ld, &fun);
216 }
217 }
218 }
219 }
220 }

227void MRGenerator::collectModRefForCall()
228{
229
230 DBOUT(DGENERAL, outs() << pasMsg("\t\tCollect Callsite PointsTo \n"));
231
233 for(SVFIR::CallSiteSet::const_iterator it = pta->getPAG()->getCallSiteSet().begin(),
234 eit = pta->getPAG()->getCallSiteSet().end(); it!=eit; ++it)
235 {
236 collectCallSitePts((*it));
237 }
238
239 DBOUT(DGENERAL, outs() << pasMsg("\t\tPerform Callsite Mod-Ref \n"));
240
241 WorkList worklist;
242 getCallGraphSCCRevTopoOrder(worklist);
243
244 while(!worklist.empty())
245 {
246 NodeID callGraphNodeID = worklist.pop();
248 const NodeBS& subNodes = callGraphSCC->subNodes(callGraphNodeID);
249 for(NodeBS::iterator it = subNodes.begin(), eit = subNodes.end(); it!=eit; ++it)
250 {
251 PTACallGraphNode* subCallGraphNode = callGraph->getCallGraphNode(*it);
253 modRefAnalysis(subCallGraphNode,worklist);
254 }
255 }
256
257 DBOUT(DGENERAL, outs() << pasMsg("\t\tAdd PointsTo to Callsites \n"));
258
259 for (const CallICFGNode* callBlockNode : pta->getPAG()->getCallSiteSet())
260 {
261 if(hasRefSideEffectOfCallSite(callBlockNode))
262 {
263 NodeBS refs = getRefSideEffectOfCallSite(callBlockNode);
264 addCPtsToCallSiteRefs(refs,callBlockNode);
265 }
266 if(hasModSideEffectOfCallSite(callBlockNode))
267 {
268 NodeBS mods = getModSideEffectOfCallSite(callBlockNode);
270 addCPtsToCallSiteMods(mods,callBlockNode);
271 addCPtsToCallSiteRefs(mods,callBlockNode);
272 }
273 }

282void MRGenerator::sortPointsTo(const NodeBS& cpts)
283{
284
285 if(cptsToRepCPtsMap.find(cpts)!=cptsToRepCPtsMap.end())
286 return;
287
288 PointsToList subSetList;
289 NodeBS repCPts = cpts;
290 for(PtsToRepPtsSetMap::iterator it = cptsToRepCPtsMap.begin(),
291 eit = cptsToRepCPtsMap.end(); it!=eit; ++it)
292 {
293 NodeBS& existCPts = it->second;
294 if(cpts.contains(existCPts))
295 {
296 subSetList.insert(it->first);
297 }
298 else if(existCPts.contains(cpts))
299 {
300 repCPts = existCPts;
301 }
302 }
303
304 for(PointsToList::iterator it = subSetList.begin(), eit = subSetList.end(); it!=eit; ++it)
305 {
306 cptsToRepCPtsMap[*it] = cpts;
307 }
308
309 cptsToRepCPtsMap[cpts] = repCPts;
310}

346void MRGenerator::updateAliasMRs()
347{
348
350 for(StoresToPointsToMap::const_iterator it = storesToPointsToMap.begin(), eit = storesToPointsToMap.end(); it!=eit; ++it)
351 {
352 MRSet aliasMRs;
353 const SVFFunction* fun = getFunction(it->first);
354 const NodeBS& storeCPts = it->second;
355 getAliasMemRegions(aliasMRs,storeCPts,fun);
356 for(MRSet::iterator ait = aliasMRs.begin(), eait = aliasMRs.end(); ait!=eait; ++ait)
357 {
358 storesToMRsMap[it->first].insert(*ait);
359 }
360 }
361
362 for(LoadsToPointsToMap::const_iterator it = loadsToPointsToMap.begin(), eit = loadsToPointsToMap.end(); it!=eit; ++it)
363 {
364 MRSet aliasMRs;
365 const SVFFunction* fun = getFunction(it->first);
366 const NodeBS& loadCPts = it->second;
367 getMRsForLoad(aliasMRs, loadCPts, fun);
368 for(MRSet::iterator ait = aliasMRs.begin(), eait = aliasMRs.end(); ait!=eait; ++ait)
369 {
370 loadsToMRsMap[it->first].insert(*ait);
371 }
372 }
373
375 for(CallSiteToPointsToMap::const_iterator it = callsiteToModPointsToMap.begin(),
376 eit = callsiteToModPointsToMap.end(); it!=eit; ++it)
377 {
378 const SVFFunction* fun = it->first->getCaller();
379 MRSet aliasMRs;
380 const NodeBS& callsiteModCPts = it->second;
381 getAliasMemRegions(aliasMRs,callsiteModCPts,fun);
382 for(MRSet::iterator ait = aliasMRs.begin(), eait = aliasMRs.end(); ait!=eait; ++ait)
383 {
384 callsiteToModMRsMap[it->first].insert(*ait);
385 }
386 }
387 for(CallSiteToPointsToMap::const_iterator it = callsiteToRefPointsToMap.begin(),
388 eit = callsiteToRefPointsToMap.end(); it!=eit; ++it)
389 {
390 const SVFFunction* fun = it->first->getCaller();
391 MRSet aliasMRs;
392 const NodeBS& callsiteRefCPts = it->second;
393 getMRsForCallSiteRef(aliasMRs, callsiteRefCPts, fun);
394 for(MRSet::iterator ait = aliasMRs.begin(), eait = aliasMRs.end(); ait!=eait; ++ait)
395 {
396 callsiteToRefMRsMap[it->first].insert(*ait);
397 }
398 }

429bool MRGenerator::addRefSideEffectOfCallSite(const CallICFGNode* cs, const NodeBS& refs)
430{
431 if(!refs.empty())
432 {
433 NodeBS refset = refs;
434 refset &= getCallSiteArgsPts(cs);
435 getEscapObjviaGlobals(refset,refs);
436 addRefSideEffectOfFunction(cs->getCaller(),refset);
437 return csToRefsMap[cs] |= refset;
438 }
439 return false;

477void MRGenerator::collectCallSitePts(const CallICFGNode* cs)
478{
480 NodeBS& argsPts = csToCallSiteArgsPtsMap[cs];
481 SVFIR* pag = pta->getPAG();
482 CallICFGNode* callBlockNode = const_cast<CallICFGNode*>(cs);
483 const RetICFGNode* retBlockNode = cs->getRetICFGNode();
484
485 WorkList worklist;
486 if (pag->hasCallSiteArgsMap(callBlockNode))
487 {
488 const SVFIR::SVFVarList& args = pta->getPAG()->getCallSiteArgsList(callBlockNode);
489 for(SVFIR::SVFVarList::const_iterator itA = args.begin(), ieA = args.end(); itA!=ieA; ++itA)
490 {
491 const PAGNode* node = *itA;
492 if(node->isPointer())
493 worklist.push(node->getId());
494 }
495 }
496
497 while(!worklist.empty())
498 {
499 NodeID nodeId = worklist.pop();
500 const NodeBS& tmp = pta->getPts(nodeId).toNodeBS();
501 for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
502 argsPts |= CollectPtsChain(*it);
503 }
504
506 NodeBS& retPts = csToCallSiteRetPtsMap[cs];
507
508 if (pta->getPAG()->callsiteHasRet(retBlockNode))
509 {
510 const PAGNode* node = pta->getPAG()->getCallSiteRet(retBlockNode);
511 if(node->isPointer())
512 {
513 const NodeBS& tmp = pta->getPts(node->getId()).toNodeBS();
514 for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
515 retPts |= CollectPtsChain(*it);
516 }
517 }
518
519}

525NodeBS& MRGenerator::CollectPtsChain(NodeID id)
526{
527 NodeID baseId = pta->getPAG()->getBaseObjVar(id);
528 NodeToPTSSMap::iterator it = cachedPtsChainMap.find(baseId);
529 if(it!=cachedPtsChainMap.end())
530 return it->second;
531 else
532 {
533 NodeBS& pts = cachedPtsChainMap[baseId];
534 pts |= pta->getPAG()->getFieldsAfterCollapse(baseId);
535
536 WorkList worklist;
537 for(NodeBS::iterator it = pts.begin(), eit = pts.end(); it!=eit; ++it)
538 worklist.push(*it);
539
540 while(!worklist.empty())
541 {
542 NodeID nodeId = worklist.pop();
543 const NodeBS& tmp = pta->getPts(nodeId).toNodeBS();
544 for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
545 {
546 pts |= CollectPtsChain(*it);
547 }
548 }
549 return pts;
550 }

576bool MRGenerator::isNonLocalObject(NodeID id, const SVFFunction* curFun) const
577{
578 //ABTest
579 const BaseObjVar* obj = pta->getPAG()->getBaseObject(id);
580 assert(obj && "object not found!!");
582 const BaseObjVar* pVar = pta->getPAG()->getBaseObject(id);
583 assert(pVar && "object not found!");
584 if(obj->isGlobalObj() || SVFUtil::isa<HeapObjVar, DummyObjVar>(pVar))
585 return true;
588 else if(SVFUtil::isa<StackObjVar>(pVar))
589 {
590 if(const SVFFunction* svffun = pVar->getFunction())
591 {
592 if(svffun!=curFun)
593 return true;
594 else
595 return callGraphSCC->isInCycle(callGraph->getCallGraphNode(svffun)->getId());
596 }
597 }
598
599 return false;
600}

605bool MRGenerator::handleCallsiteModRef(NodeBS& mod, NodeBS& ref, const CallICFGNode* cs, const SVFFunction* callee)
606{
608 if(isHeapAllocExtCall(cs))
609 {
610 SVFStmtList& pagEdgeList = getPAGEdgesFromInst(cs);
611 for (SVFStmtList::const_iterator bit = pagEdgeList.begin(),
612 ebit = pagEdgeList.end(); bit != ebit; ++bit)
613 {
614 const PAGEdge* edge = *bit;
615 if (const AddrStmt* addr = SVFUtil::dyn_cast<AddrStmt>(edge))
616 mod.set(addr->getRHSVarID());
617 }
618 }
620 else
621 {
622 mod = getModSideEffectOfFunction(callee);
623 ref = getRefSideEffectOfFunction(callee);
624 }
625 // add ref set
626 bool refchanged = addRefSideEffectOfCallSite(cs, ref);
627 // add mod set
628 bool modchanged = addModSideEffectOfCallSite(cs, mod);
629
630 return refchanged || modchanged;
631}

637void MRGenerator::modRefAnalysis(PTACallGraphNode* callGraphNode, WorkList& worklist)
638{
639
641 for(PTACallGraphNode::iterator it = callGraphNode->InEdgeBegin(), eit = callGraphNode->InEdgeEnd();
642 it!=eit; ++it)
643 {
644 PTACallGraphEdge* edge = *it;
645
647 for(PTACallGraphEdge::CallInstSet::iterator cit = edge->getDirectCalls().begin(),
648 ecit = edge->getDirectCalls().end(); cit!=ecit; ++cit)
649 {
650 NodeBS mod, ref;
651 const CallICFGNode* cs = (*cit);
652 bool modrefchanged = handleCallsiteModRef(mod, ref, cs, callGraphNode->getFunction());
653 if(modrefchanged)
654 worklist.push(edge->getSrcID());
655 }
657 for(PTACallGraphEdge::CallInstSet::iterator cit = edge->getIndirectCalls().begin(),
658 ecit = edge->getIndirectCalls().end(); cit!=ecit; ++cit)
659 {
660 NodeBS mod, ref;
661 const CallICFGNode* cs = (*cit);
662 bool modrefchanged = handleCallsiteModRef(mod, ref, cs, callGraphNode->getFunction());
663 if(modrefchanged)
664 worklist.push(edge->getSrcID());
665 }
666 }
667}

740ModRefInfo MRGenerator::getModRefInfo(const CallICFGNode* cs, const SVFValue* V)
741{
742 bool ref = false;
743 bool mod = false;
744
745 if (pta->getPAG()->hasValueNode(V))
746 {
747 const NodeBS pts(pta->getPts(pta->getPAG()->getValueNode(V)).toNodeBS());
748 const NodeBS csRef = getRefInfoForCall(cs);
749 const NodeBS csMod = getModInfoForCall(cs);
750 NodeBS ptsExpanded, csRefExpanded, csModExpanded;
751 pta->expandFIObjs(pts, ptsExpanded);
752 pta->expandFIObjs(csRef, csRefExpanded);
753 pta->expandFIObjs(csMod, csModExpanded);
754
755 if (csRefExpanded.intersects(ptsExpanded))
756 ref = true;
757 if (csModExpanded.intersects(ptsExpanded))
758 mod = true;
759 }
760
761 if (mod && ref)
762 return ModRefInfo::ModRef;
763 else if (ref)
764 return ModRefInfo::Ref;
765 else if (mod)
766 return ModRefInfo::Mod;
767 else
768 return ModRefInfo::NoModRef;
769}

774ModRefInfo MRGenerator::getModRefInfo(const CallICFGNode* cs1, const CallICFGNode* cs2)
775{
776 bool ref = false;
777 bool mod = false;
778
780 if (getModRefInfo(cs1) == ModRefInfo::NoModRef || getModRefInfo(cs2) == ModRefInfo::NoModRef)
781 return ModRefInfo::NoModRef;
782
783 const NodeBS cs1Ref = getRefInfoForCall(cs1);
784 const NodeBS cs1Mod = getModInfoForCall(cs1);
785 const NodeBS cs2Ref = getRefInfoForCall(cs2);
786 const NodeBS cs2Mod = getModInfoForCall(cs2);
787 NodeBS cs1RefExpanded, cs1ModExpanded, cs2RefExpanded, cs2ModExpanded;
788 pta->expandFIObjs(cs1Ref, cs1RefExpanded);
789 pta->expandFIObjs(cs1Mod, cs1ModExpanded);
790 pta->expandFIObjs(cs2Ref, cs2RefExpanded);
791 pta->expandFIObjs(cs2Mod, cs2ModExpanded);
792
794 if (cs1RefExpanded.intersects(cs2ModExpanded))
795 ref = true;
797 if (cs1ModExpanded.intersects(cs2RefExpanded) || cs1ModExpanded.intersects(cs2ModExpanded))
798 mod = true;
800 if (cs1RefExpanded.intersects(cs2ModExpanded) && cs1ModExpanded.intersects(cs2ModExpanded))
801 ref = mod = true;
802
803 if (ref && mod)
804 return ModRefInfo::ModRef;
805 else if (ref)
806 return ModRefInfo::Ref;
807 else if (mod)
808 return ModRefInfo::Mod;
809 else
810 return ModRefInfo::NoModRef;
811}
