300  /*
301  * function: pass bug type (i.e., GenericBug::NEVERFREE) and eventStack as parameter,
302  * it will add the bug into bugQueue.
303  * usage: addSaberBug(GenericBug::NEVERFREE, eventStack)
304  */
305  void addSaberBug(GenericBug::BugType bugType, const GenericBug::EventStack &eventStack)
306  {
308  GenericBug *newBug = nullptr;
309  switch(bugType)
310  {
311  case GenericBug::NEVERFREE:
312  {
313  newBug = new NeverFreeBug(eventStack);
314  bugSet.insert(newBug);
315  break;
316  }
317  case GenericBug::PARTIALLEAK:
318  {
319  newBug = new PartialLeakBug(eventStack);
320  bugSet.insert(newBug);
321  break;
322  }
323  case GenericBug::DOUBLEFREE:
324  {
325  newBug = new DoubleFreeBug(eventStack);
326  bugSet.insert(newBug);
327  break;
328  }
329  case GenericBug::FILENEVERCLOSE:
330  {
331  newBug = new FileNeverCloseBug(eventStack);
332  bugSet.insert(newBug);
333  break;
334  }
335  case GenericBug::FILEPARTIALCLOSE:
336  {
337  newBug = new FilePartialCloseBug(eventStack);
338  bugSet.insert(newBug);
339  break;
340  }
341  default:
342  {
343  assert(false && "saber does NOT have this bug type!");
344  break;
345  }
346  }
347 
348  // when add a bug, also print it to terminal
349  newBug->printBugToTerminal();
350  }
351 
352  /*
353  * function: pass bug type (i.e., GenericBug::FULLBUFOVERFLOW) and eventStack as parameter,
354  * it will add the bug into bugQueue.
355  * usage: addAbsExecBug(GenericBug::FULLBUFOVERFLOW, eventStack, 0, 10, 11, 11)
356  */
357  void addAbsExecBug(GenericBug::BugType bugType, const GenericBug::EventStack &eventStack,
358  s64_t allocLowerBound, s64_t allocUpperBound, s64_t accessLowerBound, s64_t accessUpperBound)
359  {
361  GenericBug *newBug = nullptr;
362  switch(bugType)
363  {
364  case GenericBug::FULLBUFOVERFLOW:
365  {
366  newBug = new FullBufferOverflowBug(eventStack, allocLowerBound, allocUpperBound, accessLowerBound, accessUpperBound);
367  bugSet.insert(newBug);
368  break;
369  }
370  case GenericBug::PARTIALBUFOVERFLOW:
371  {
372  newBug = new PartialBufferOverflowBug(eventStack, allocLowerBound, allocUpperBound, accessLowerBound, accessUpperBound);
373  bugSet.insert(newBug);
374  break;
375  }
376  case GenericBug::FULLNULLPTRDEREFERENCE:
377  {
378  newBug = new FullNullPtrDereferenceBug(eventStack);
379  bugSet.insert(newBug);
380  break;
381  }
382  case GenericBug::PARTIALNULLPTRDEREFERENCE:
383  {
384  newBug = new PartialNullPtrDereferenceBug(eventStack);
385  bugSet.insert(newBug);
386  break;
387  }
388  default:
389  {
390  assert(false && "Abstract Execution does NOT have this bug type!");
391  break;
392  }
393  }
394 
395  // when add a bug, also print it to terminal
396  newBug->printBugToTerminal();
397  }
