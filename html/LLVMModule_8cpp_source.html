50/*
51 svf.main() is used to model the real entry point of a C++ program, which
52 initializes all global C++ objects and then call main().
53 LLVM may generate two global arrays @llvm.global_ctors and @llvm.global_dtors
54 that contain constructor and destructor functions for global variables. They
55 are not called explicitly, so we have to add them in the svf.main function.
56 The order to call these constructor and destructor functions are also
57 specified in the global arrays.
58 Related part in LLVM language reference:
59 https://llvm.org/docs/LangRef.html#the-llvm-global-ctors-global-variable
60 For example, given a "int main(int argc, char * argv[])", the corresponding
61 svf.main will be generated as follows:
62 define void @svf.main(i32, i8**, i8**) {
63 entry:
64 call void @ctor1()
65 call void @ctor2()
66 %3 = call i32 @main(i32 %0, i8** %1)
67 call void @dtor1()
68 call void @dtor2()
69 ret void
70 }
71*/
72
73#define SVF_MAIN_FUNC_NAME "svf.main"
74#define SVF_GLOBAL_CTORS "llvm.global_ctors"
75#define SVF_GLOBAL_DTORS "llvm.global_dtors"

202void LLVMModuleSet::createSVFDataStructure()
203{
204 SVFType::svfI8Ty = getSVFType(getTypeInference()->int8Type());
205 SVFType::svfPtrTy = getSVFType(getTypeInference()->ptrType());
206 // Functions need to be retrieved in the order of insertion
207 // candidateDefs is the vector for all used defined functions
208 // candidateDecls is the vector for all used declared functions
209 std::vector<const Function*> candidateDefs, candidateDecls;
210
211 for (Module& mod : modules)
212 {
214 for (Function& func : mod.functions())
215 {
216 if (func.isDeclaration())
217 {
218 candidateDecls.push_back(&func);
219 }
220 else
221 {
222 candidateDefs.push_back(&func);
223 }
224 }
225 }
226
227 for (const Function* func: candidateDefs)
228 {
229 createSVFFunction(func);
230 }
231 for (const Function* func: candidateDecls)
232 {
233 createSVFFunction(func);
234 }
235
236 // Store annotations of functions in extapi.bc
237 for (const auto& pair : ExtFun2Annotations)
238 {
239 const SVFFunction* svffun = getSVFFunction(pair.first);
240 ExtAPI::getExtAPI()->setExtFuncAnnotations(svffun, pair.second);
241 }
242
244 for (const Module& mod : modules)
245 {
247 for (const GlobalVariable& global : mod.globals())
248 {
249 SVFGlobalValue* svfglobal = new SVFGlobalValue(
250 global.getName().str(), getSVFType(global.getType()));
251 svfModule->addGlobalSet(svfglobal);
252 addGlobalValueMap(&global, svfglobal);
253 }
254
256 for (const GlobalAlias& alias : mod.aliases())
257 {
258 SVFGlobalValue* svfalias = new SVFGlobalValue(
259 alias.getName().str(), getSVFType(alias.getType()));
260 svfModule->addAliasSet(svfalias);
261 addGlobalValueMap(&alias, svfalias);
262 }
263
265 for (const GlobalIFunc& ifunc : mod.ifuncs())
266 {
267 SVFGlobalValue* svfifunc = new SVFGlobalValue(
268 ifunc.getName().str(), getSVFType(ifunc.getType()));
269 svfModule->addAliasSet(svfifunc);
270 addGlobalValueMap(&ifunc, svfifunc);
271 }
272 }

275void LLVMModuleSet::createSVFFunction(const Function* func)
276{
277 SVFFunction* svfFunc = new SVFFunction(
278 getSVFType(func->getType()),
279 SVFUtil::cast<SVFFunctionType>(
280 getSVFType(func->getFunctionType())),
281 func->isDeclaration(), LLVMUtil::isIntrinsicFun(func),
282 func->hasAddressTaken(), func->isVarArg(), new SVFLoopAndDomInfo);
283 BasicBlockGraph* bbGraph = new BasicBlockGraph(svfFunc);
284 svfFunc->setBasicBlockGraph(bbGraph);
285 svfModule->addFunctionSet(svfFunc);
286 addFunctionMap(func, svfFunc);
287
288 for (const Argument& arg : func->args())
289 {
290 SVFArgument* svfarg = new SVFArgument(
291 getSVFType(arg.getType()), svfFunc, arg.getArgNo(),
292 LLVMUtil::isArgOfUncalledFunction(&arg));
293 // Setting up arg name
294 if (!arg.hasName())
295 svfarg->setName(std::to_string(arg.getArgNo()));
296
297 svfFunc->addArgument(svfarg);
298 addArgumentMap(&arg, svfarg);
299 }
300
301 for (const BasicBlock& bb : *func)
302 {
303 addBasicBlock(svfFunc, &bb);
304 for (const Instruction& inst : bb)
305 {
306 SVFInstruction* svfInst = nullptr;
307 if (const CallBase* call = SVFUtil::dyn_cast<CallBase>(&inst))
308 {
309 svfInst = new SVFCallInst(
310 getSVFType(call->getType()), getSVFBasicBlock(&bb),
311 call->getFunctionType()->isVarArg(),
312 inst.isTerminator());
313 }
314 else
315 {
316 svfInst =
317 new SVFInstruction(getSVFType(inst.getType()),
318 getSVFBasicBlock(&bb), inst.isTerminator(),
319 SVFUtil::isa<ReturnInst>(inst));
320 }
321
322 addInstructionMap(&inst, svfInst);
323 }
324 }

345void LLVMModuleSet::initSVFBasicBlock(const Function* func)
346{
347 SVFFunction *svfFun = getSVFFunction(func);
348 for (Function::const_iterator bit = func->begin(), ebit = func->end(); bit != ebit; ++bit)
349 {
350 const BasicBlock* bb = &*bit;
351 SVFBasicBlock* svfbb = getSVFBasicBlock(bb);
352 for (succ_const_iterator succ_it = succ_begin(bb); succ_it != succ_end(bb); succ_it++)
353 {
354 const SVFBasicBlock* svf_scc_bb = getSVFBasicBlock(*succ_it);
355 svfbb->addSuccBasicBlock(svf_scc_bb);
356 }
357 for (const_pred_iterator pred_it = pred_begin(bb); pred_it != pred_end(bb); pred_it++)
358 {
359 const SVFBasicBlock* svf_pred_bb = getSVFBasicBlock(*pred_it);
360 svfbb->addPredBasicBlock(svf_pred_bb);
361 }
362
364 if (svfbb->getSuccessors().empty())
365 {
366 if (LLVMUtil::basicBlockHasRetInst(bb))
367 {
368 assert((LLVMUtil::functionDoesNotRet(func) ||
369 SVFUtil::isa<ReturnInst>(bb->back())) &&
370 "last inst must be return inst");
371 svfFun->setExitBlock(svfbb);
372 }
373 }
374
375 for (BasicBlock::const_iterator iit = bb->begin(), eiit = bb->end(); iit != eiit; ++iit)
376 {
377 const Instruction* inst = &*iit;
378 if(const CallBase* call = SVFUtil::dyn_cast<CallBase>(inst))
379 {
380 SVFInstruction* svfinst = getSVFInstruction(call);
381 SVFCallInst* svfcall = SVFUtil::cast<SVFCallInst>(svfinst);
382 auto called_llvmval = call->getCalledOperand()->stripPointerCasts();
383 if (const Function* called_llvmfunc = SVFUtil::dyn_cast<Function>(called_llvmval))
384 {
385 SVFFunction* callee = getSVFFunction(called_llvmfunc);
386 svfcall->setCalledOperand(callee);
387 }
388 else
389 {
390 svfcall->setCalledOperand(getSVFValue(called_llvmval));
391 }
392 for(u32_t i = 0; i < call->arg_size(); i++)
393 {
394 SVFValue* svfval = getSVFValue(call->getArgOperand(i));
395 svfcall->addArgument(svfval);
396 }
397 }
398 }
399 }
400 // For no return functions, we set the last block as exit BB
401 // This ensures that each function that has definition must have an exit BB
402 if (svfFun->exitBlock == nullptr && svfFun->hasBasicBlock())
403 {
404 SVFBasicBlock* retBB = const_cast<SVFBasicBlock*>(svfFun->back());
405 assert((LLVMUtil::functionDoesNotRet(func) ||
406 SVFUtil::isa<ReturnInst>(&func->back().back())) &&
407 "last inst must be return inst");
408 svfFun->setExitBlock(retBB);
409 }
410}

413void LLVMModuleSet::initDomTree(SVFFunction* svffun, const Function* fun)
414{
415 if (fun->isDeclaration())
416 return;
417 //process and stored dt & df
418 DominanceFrontier df;
419 DominatorTree& dt = getDomTree(fun);
420 df.analyze(dt);
421 LoopInfo loopInfo = LoopInfo(dt);
422 PostDominatorTree pdt = PostDominatorTree(const_cast<Function&>(*fun));
423 SVFLoopAndDomInfo* ld = svffun->getLoopAndDomInfo();
424
425 Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>> & dfBBsMap = ld->getDomFrontierMap();
426 for (DominanceFrontierBase::const_iterator dfIter = df.begin(), eDfIter = df.end(); dfIter != eDfIter; dfIter++)
427 {
428 const BasicBlock* keyBB = dfIter->first;
429 const std::set<BasicBlock* >& domSet = dfIter->second;
430 Set<const SVFBasicBlock*>& valueBasicBlocks = dfBBsMap[getSVFBasicBlock(keyBB)];
431 for (const BasicBlock* bbValue:domSet)
432 {
433 valueBasicBlocks.insert(getSVFBasicBlock(bbValue));
434 }
435 }
436 std::vector<const SVFBasicBlock*> reachableBBs;
437 LLVMUtil::getFunReachableBBs(fun, reachableBBs);
438 ld->setReachableBBs(reachableBBs);
439
440 for (Function::const_iterator bit = fun->begin(), beit = fun->end(); bit!=beit; ++bit)
441 {
442 const BasicBlock &bb = *bit;
443 SVFBasicBlock* svfBB = getSVFBasicBlock(&bb);
444 if (DomTreeNode* dtNode = dt.getNode(&bb))
445 {
446 SVFLoopAndDomInfo::BBSet& bbSet = ld->getDomTreeMap()[svfBB];
447 for (const auto domBB : *dtNode)
448 {
449 const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
450 bbSet.insert(domSVFBB);
451 }
452 }
453
454 if (DomTreeNode* pdtNode = pdt.getNode(&bb))
455 {
456 u32_t level = pdtNode->getLevel();
457 ld->getBBPDomLevel()[svfBB] = level;
458 BasicBlock* idomBB = pdtNode->getIDom()->getBlock();
459 const SVFBasicBlock* idom = idomBB == NULL ? NULL: getSVFBasicBlock(idomBB);
460 ld->getBB2PIdom()[svfBB] = idom;
461
462 SVFLoopAndDomInfo::BBSet& bbSet = ld->getPostDomTreeMap()[svfBB];
463 for (const auto domBB : *pdtNode)
464 {
465 const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
466 bbSet.insert(domSVFBB);
467 }
468 }
469
470 if (const Loop* loop = loopInfo.getLoopFor(&bb))
471 {
472 for (const BasicBlock* loopBlock : loop->getBlocks())
473 {
474 const SVFBasicBlock* loopbb = getSVFBasicBlock(loopBlock);
475 ld->addToBB2LoopMap(svfBB, loopbb);
476 }
477 }
478 }

486{
488 std::unique_ptr<BreakConstantGEPs> p1 = std::make_unique<BreakConstantGEPs>();
489 for (Module &M : getLLVMModules())
490 {

492 }
493

502 if (fun.isDeclaration())
503 continue;
504 p2->runOnFunction(fun);
505 }
506 }

530void LLVMModuleSet::loadModules(const std::vector<std::string> &moduleNameVec)
531{
532
533 // We read SVFIR from LLVM IR
534 if(Options::Graphtxt().empty())
535 {
536 if(moduleNameVec.empty())
537 {
538 SVFUtil::outs() << "no LLVM bc file is found!\n";
539 exit(0);
540 }
541 //assert(!moduleNameVec.empty() && "no LLVM bc file is found!");
542 }
543 // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
544 else
545 SVFModule::setPagFromTXT(Options::Graphtxt());
546
547 //
548 // LLVMContext objects separate global LLVM settings (from which e.g. types are
549 // derived); multiple LLVMContext objects can coexist and each context can "own"
550 // multiple modules (modules can only have one context). Mixing contexts can lead
551 // to unintended inequalities, such as the following:
552 //
553 // ------------------------------------------------------------------
554 // LLVMContext ctxa,ctxb;
555 // IntegerType * t1 = IntegerType::get(ctxa,32);
556 // IntegerType * t2 = IntegerType::get(ctxa,32);
557 // assert(t1 == t2);
558 // IntegerType * t3 = IntegerType::get(ctxb,32);
559 // IntegerType * t4 = IntegerType::get(ctxb,32);
560 // assert(t3 == t4);
561 // assert(t1 != t3);
562 // ------------------------------------------------------------------
563 //
564 // When loading bytecode files, SVF will use the same LLVMContext object for all
565 // modules (i.e. the context owns all loaded modules). This applies to ExtAPI as
566 // well, which *must* be loaded using the same LLVMContext object. Hence, when
567 // loading modules from bitcode files, a new LLVMContext is created (using a
568 // `std::unique_ptr<LLVMContext>` type to ensure automatic garbage collection).
569 //
570 // This garbage collection should be avoided when building an SVF module from an LLVM
571 // module instance; see the comment(s) in `buildSVFModule` and `loadExtAPIModules()`
572
573 owned_ctx = std::make_unique<LLVMContext>();
574 for (const std::string& moduleName : moduleNameVec)
575 {
576 if (!LLVMUtil::isIRFile(moduleName))
577 {
578 SVFUtil::errs() << "not an IR file: " << moduleName << std::endl;
579 abort();
580 }
581
582 SMDiagnostic Err;
583 std::unique_ptr<Module> mod = parseIRFile(moduleName, Err, *owned_ctx);
584 if (mod == nullptr)
585 {
586 SVFUtil::errs() << "load module: " << moduleName << "failed!!\n\n";
587 Err.print("SVFModuleLoader", llvm::errs());
588 abort();
589 }
590 modules.emplace_back(*mod);
591 owned_modules.emplace_back(std::move(mod));
592 }

595void LLVMModuleSet::loadExtAPIModules()
596{
597 // This function loads the ExtAPI bitcode file as an LLVM module. Note that it is important that
598 // the same LLVMContext object is used to load this bitcode file as is used by the other modules
599 // being analysed.
600 // When the modules are loaded from bitcode files (i.e. passing filenames to files containing
601 // LLVM IR to `buildSVFModule({file1.bc, file2.bc, ...})) the context is created while loading
602 // the modules in `loadModules()`, which populates this->modules and this->owned_modules.
603 // If, however, an LLVM Module object is passed to `buildSVFModule` (e.g. from an LLVM pass),
604 // the context should be retrieved from the module itself (note that the garbage collection from
605 // `std::unique_ptr<LLVMContext> LLVMModuleSet::owned_ctx` should be avoided in this case). This
606 // function populates only this->modules.
607 // In both cases, fetching the context from the main LLVM module (through `getContext`) works
608 assert(!empty() && "LLVMModuleSet contains no modules; cannot load ExtAPI module without LLVMContext!");
609
610 // Load external API module (extapi.bc)
611 if (!ExtAPI::getExtAPI()->getExtBcPath().empty())
612 {
613 std::string extModuleName = ExtAPI::getExtAPI()->getExtBcPath();
614 if (!LLVMUtil::isIRFile(extModuleName))
615 {
616 SVFUtil::errs() << "not an external IR file: " << extModuleName << std::endl;
617 abort();
618 }
619 SMDiagnostic Err;
620 std::unique_ptr<Module> mod = parseIRFile(extModuleName, Err, getContext());
621 if (mod == nullptr)
622 {
623 SVFUtil::errs() << "load external module: " << extModuleName << "failed!!\n\n";
624 Err.print("SVFModuleLoader", llvm::errs());
625 abort();
626 }
627 // The module of extapi.bc needs to be inserted before applications modules, like std::vector<std::reference_wrapper<Module>> modules{extapi_module, app_module}.
628 // Otherwise, when overwriting the app function with SVF extern function, the corresponding SVFFunction of the extern function will not be found.
629 modules.insert(modules.begin(), *mod);
630 owned_modules.insert(owned_modules.begin(),std::move(mod));
631 }
632}

634std::vector<const Function* > LLVMModuleSet::getLLVMGlobalFunctions(const GlobalVariable *global)
635{
636 // This function is used to extract constructor and destructor functions
637 // sorted by their priority from @llvm.global_ctors or @llvm.global_dtors.
638 // For example, given following @llvm.global_ctors, the returning sorted
639 // function list should be [ctor3, ctor1, ctor2].
640 // ------------------------------------------------------------------
641 // ; Each struct in the array is {priority, function, associated data}
642 //
643 // @llvm.global_ctors = appending global [2 x { i32, void ()*, i8* }]
644 // [{ i32, void ()*, i8* } { i32 1234, void ()* @ctor1.cpp, i8* null },
645 // { i32, void ()*, i8* } { i32 2345, void ()* @ctor2.cpp, i8* null },
646 // { i32, void ()*, i8* } { i32 345, void ()* @ctor3.cpp, i8* null }]
647 // ------------------------------------------------------------------
648 // TODO: According to LLVM language reference, if the third field is
649 // non-null, and points to a global variable or function, the initializer
650 // function will only run if the associated data from the current module is
651 // not discarded. However the associated data is currently ignored.
652
653
654 // This class is used for the priority queue that sorts the functions by
655 // their priority. Each object of this class stands for an item in the
656 // function array.
657 class LLVMGlobalFunction
658 {
659 public:
660 u32_t priority;
661 const Function* func;
662 LLVMGlobalFunction() {};
663 LLVMGlobalFunction(u32_t _priority, const Function* _func)
664 : priority(_priority), func(_func) {};
665 bool operator>(const LLVMGlobalFunction &other) const
666 {
667 if (priority != other.priority)
668 {
669 return priority > other.priority;
670 }
671 else
672 {
673 return func > other.func;
674 }
675 }
676 };
677
678 std::priority_queue<LLVMGlobalFunction, std::vector<LLVMGlobalFunction>,
679 greater<LLVMGlobalFunction>>
680 queue;
681 std::vector<const Function* > result;
682
683 // The @llvm.global_ctors/dtors global variable is an array of struct. Each
684 // struct has three fields: {i32 priority, void ()* @ctor/dtor, i8* @data}.
685 // First get the array here.
686 if(const ConstantArray *globalFuncArray =
687 SVFUtil::dyn_cast<ConstantArray>(global->getInitializer()))
688 {
689 // Get each struct in the array.
690 for (unsigned int i = 0; i < globalFuncArray->getNumOperands(); ++i)
691 {
692 if (
693 const ConstantStruct *globalFuncItem =
694 SVFUtil::dyn_cast<ConstantStruct>(
695 globalFuncArray->getOperand(i)))
696 {
697
698 // Extract priority and function from the struct
699 const ConstantInt* priority = SVFUtil::dyn_cast<ConstantInt>(
700 globalFuncItem->getOperand(0));
701 const Function* func = SVFUtil::dyn_cast<Function>(
702 globalFuncItem->getOperand(1));
703
704 if (priority && func)
705 {
706 queue.push(LLVMGlobalFunction(LLVMUtil::getIntegerValue(priority).second,
707 func));
708 }
709 }
710 }
711 }
712
713 // Generate a sorted vector of functions from the priority queue.
714 while (!queue.empty())
715 {
716 result.push_back(queue.top().func);
717 queue.pop();
718 }
719 return result;
720}

722void LLVMModuleSet::addSVFMain()
723{
724 std::vector<const Function*> ctor_funcs;
725 std::vector<const Function*> dtor_funcs;
726 Function* orgMain = 0;
727 Module* mainMod = nullptr;
728
729 for (Module &mod : modules)
730 {
731 // Collect ctor and dtor functions
732 for (const GlobalVariable& global : mod.globals())
733 {
734 if (global.getName().equals(SVF_GLOBAL_CTORS) && global.hasInitializer())
735 {
736 ctor_funcs = getLLVMGlobalFunctions(&global);
737 }
738 else if (global.getName().equals(SVF_GLOBAL_DTORS) && global.hasInitializer())
739 {
740 dtor_funcs = getLLVMGlobalFunctions(&global);
741 }
742 }
743
744 // Find main function
745 for (auto &func : mod)
746 {
747 auto funName = func.getName();
748
749 assert(!funName.equals(SVF_MAIN_FUNC_NAME) && SVF_MAIN_FUNC_NAME " already defined");
750
751 if (funName.equals("main"))
752 {
753 orgMain = &func;
754 mainMod = &mod;
755 }
756 }
757 }
758
759 // Only create svf.main when the original main function is found, and also
760 // there are global constructor or destructor functions.
761 if (orgMain && getModuleNum() > 0 &&
762 (ctor_funcs.size() > 0 || dtor_funcs.size() > 0))
763 {
764 assert(mainMod && "Module with main function not found.");
765 Module& M = *mainMod;
766 // char **
767 Type* ptr = PointerType::getUnqual(M.getContext());
768 Type* i32 = IntegerType::getInt32Ty(M.getContext());
769 // define void @svf.main(i32, i8**, i8**)
770#if (LLVM_VERSION_MAJOR >= 9)
771 FunctionCallee svfmainFn = M.getOrInsertFunction(
772 SVF_MAIN_FUNC_NAME,
773 Type::getVoidTy(M.getContext()),
774 i32,ptr,ptr
775 );
776 Function* svfmain = SVFUtil::dyn_cast<Function>(svfmainFn.getCallee());
777#else
778 Function* svfmain = SVFUtil::dyn_cast<Function>(M.getOrInsertFunction(
779 SVF_MAIN_FUNC_NAME,
780 Type::getVoidTy(M.getContext()),
781 i32,i8ptr2,i8ptr2
782 ));
783#endif
784 svfmain->setCallingConv(llvm::CallingConv::C);
785 BasicBlock* block = BasicBlock::Create(M.getContext(), "entry", svfmain);
786 IRBuilder Builder(block);
787 // emit "call void @ctor()". ctor_funcs is sorted so the functions are
788 // emitted in the order of priority
789 for (auto& ctor : ctor_funcs)
790 {
791 auto target = M.getOrInsertFunction(
792 ctor->getName(),
793 Type::getVoidTy(M.getContext())
794 );
795 Builder.CreateCall(target);
796 }
797 // main() should be called after all ctor functions and before dtor
798 // functions.
799 Function::arg_iterator arg_it = svfmain->arg_begin();
800 Value* args[] = {arg_it, arg_it + 1, arg_it + 2};
801 size_t cnt = orgMain->arg_size();
802 assert(cnt <= 3 && "Too many arguments for main()");
803 Builder.CreateCall(orgMain, llvm::ArrayRef<Value*>(args, args + cnt));
804 // emit "call void @dtor()". dtor_funcs is sorted so the functions are
805 // emitted in the order of priority
806 for (auto& dtor : dtor_funcs)
807 {
808 auto target = M.getOrInsertFunction(dtor->getName(), Type::getVoidTy(M.getContext()));
809 Builder.CreateCall(target);
810 }
811 // return;
812 Builder.CreateRetVoid();
813 }
814}

816void LLVMModuleSet::collectExtFunAnnotations(const Module* mod)
817{
818 GlobalVariable *glob = mod->getGlobalVariable("llvm.global.annotations");
819 if (glob == nullptr || !glob->hasInitializer())
820 return;
821
822 ConstantArray *ca = SVFUtil::dyn_cast<ConstantArray>(glob->getInitializer());
823 if (ca == nullptr)
824 return;
825
826 for (unsigned i = 0; i < ca->getNumOperands(); ++i)
827 {
828 ConstantStruct *structAn = SVFUtil::dyn_cast<ConstantStruct>(ca->getOperand(i));
829 if (structAn == nullptr || structAn->getNumOperands() == 0)
830 continue;
831
832 // Check if the annotation is for a function
833 Function* fun = nullptr;
834 GlobalVariable *annotateStr = nullptr;
836 if (ConstantExpr *expr = SVFUtil::dyn_cast<ConstantExpr>(structAn->getOperand(0)))
837 {
838 if (expr->getOpcode() == Instruction::BitCast && SVFUtil::isa<Function>(expr->getOperand(0)))
839 fun = SVFUtil::cast<Function>(expr->getOperand(0));
840
841 ConstantExpr *note = SVFUtil::cast<ConstantExpr>(structAn->getOperand(1));
842 if (note->getOpcode() != Instruction::GetElementPtr)
843 continue;
844
845 annotateStr = SVFUtil::dyn_cast<GlobalVariable>(note->getOperand(0));
846 }
848 else
849 {
850 fun = SVFUtil::dyn_cast<Function>(structAn->getOperand(0));
851 annotateStr = SVFUtil::dyn_cast<GlobalVariable>(structAn->getOperand(1));
852 }
853
854 if (!fun || annotateStr == nullptr || !annotateStr->hasInitializer())
855 continue;;
856
857 ConstantDataSequential *data = SVFUtil::dyn_cast<ConstantDataSequential>(annotateStr->getInitializer());
858 if (data && data->isString())
859 {
860 std::string annotation = data->getAsString().str();
861 if (!annotation.empty())
862 ExtFun2Annotations[fun->getName().str()].push_back(annotation);
863 }
864 }
865}
866
867/*
868 There are three types of functions(definitions) in extapi.c:
869 1. (Fun_Overwrite): Functions with "OVERWRITE" annotion:
870 These functions are used to replace the corresponding function definitions in the application.
871 2. (Fun_Annotation): Functions with annotation(s) but without "OVERWRITE" annotation:
872 These functions are used to tell SVF to do special processing, like malloc().
873 3. (Fun_Noraml): Functions without any annotation:
874 These functions are used to replace the corresponding function declarations in the application.
875
876
877 We will iterate over declarations (appFunDecl) and definitons (appFunDef) of functions in the application and extapi.c to do the following clone or replace operations:
878 1. appFuncDecl --> Fun_Normal: Clone the Fun_Overwrite and replace the appFuncDecl in application.
879 2. appFuncDecl --> Fun_Annotation: Move the annotions on Fun_Annotation to appFuncDecl in application.
880
881 3. appFunDef --> Fun_Overwrite: Clone the Fun_Overwrite and overwrite the appFunDef in application.
882 4. appFunDef --> Fun_Annotation: Replace the appFunDef with appFunDecl and move the annotions to appFunDecl in application
883*/
884void LLVMModuleSet::buildFunToFunMap()
885{
886 Set<const Function*> appFunDecls, appFunDefs, extFuncs, clonedFuncs;
887 OrderedSet<string> appFuncDeclNames, appFuncDefNames, extFunDefNames, intersectNames;
888 Map<const Function*, const Function*> extFuncs2ClonedFuncs;
889 Module* appModule = nullptr;
890 Module* extModule = nullptr;
891
892 for (Module& mod : modules)
893 {
894 // extapi.bc functions
895 if (mod.getName().str() == ExtAPI::getExtAPI()->getExtBcPath())
896 {
897 collectExtFunAnnotations(&mod);
898 extModule = &mod;
899 for (const Function& fun : mod.functions())
900 {
901 // there is main declaration in ext bc, it should be mapped to
902 // main definition in app bc.
903 if (fun.getName().str() == "main")
904 {
905 appFunDecls.insert(&fun);
906 appFuncDeclNames.insert(fun.getName().str());
907 }
909 else if (fun.getName().str() == "svf__main")
910 {
911 ExtFuncsVec.push_back(&fun);
912 }
913 else
914 {
915 extFuncs.insert(&fun);
916 extFunDefNames.insert(fun.getName().str());
917 }
918 }
919 }
920 else
921 {
922 appModule = &mod;
924 for (const Function& fun : mod.functions())
925 {
926 if (fun.isDeclaration())
927 {
928 appFunDecls.insert(&fun);
929 appFuncDeclNames.insert(fun.getName().str());
930 }
931 else
932 {
933 appFunDefs.insert(&fun);
934 appFuncDefNames.insert(fun.getName().str());
935 }
936 }
937 }
938 }
939
940 // Find the intersectNames between appFuncDefNames and externalFunDefNames
941 std::set_intersection(
942 appFuncDefNames.begin(), appFuncDefNames.end(), extFunDefNames.begin(), extFunDefNames.end(),
943 std::inserter(intersectNames, intersectNames.end()));
944
945 auto cloneAndReplaceFunction = [&](const Function* extFunToClone, Function* appFunToReplace, Module* appModule, bool cloneBody) -> Function*
946 {
947 assert(!(appFunToReplace == NULL && appModule == NULL) && "appFunToReplace and appModule cannot both be NULL");
948
949 if (appFunToReplace)
950 {
951 appModule = appFunToReplace->getParent();
952 }
953 // Create a new function with the same signature as extFunToClone
954 Function *clonedFunction = Function::Create(extFunToClone->getFunctionType(), Function::ExternalLinkage, extFunToClone->getName(), appModule);
955 // Map the arguments of the new function to the arguments of extFunToClone
956 llvm::ValueToValueMapTy valueMap;
957 Function::arg_iterator destArg = clonedFunction->arg_begin();
958 for (Function::const_arg_iterator srcArg = extFunToClone->arg_begin(); srcArg != extFunToClone->arg_end(); ++srcArg)
959 {
960 destArg->setName(srcArg->getName()); // Copy the name of the original argument
961 valueMap[&*srcArg] = &*destArg++; // Add a mapping from the old arg to the new arg
962 }
963 if (cloneBody)
964 {
965 // Clone the body of extFunToClone into clonedFunction
966 llvm::SmallVector<ReturnInst*, 8> ignoredReturns;
967 CloneFunctionInto(clonedFunction, extFunToClone, valueMap, llvm::CloneFunctionChangeType::LocalChangesOnly, ignoredReturns, "", nullptr);
968 }
969 if (appFunToReplace)
970 {
971 // Replace all uses of appFunToReplace with clonedFunction
972 appFunToReplace->replaceAllUsesWith(clonedFunction);
973 std::string oldFunctionName = appFunToReplace->getName().str();
974 // Delete the old function
975 appFunToReplace->eraseFromParent();
976 clonedFunction->setName(oldFunctionName);
977 }
978 return clonedFunction;
979 };
980
982 for (const Function* appFunDecl : appFunDecls)
983 {
984 std::string appFunDeclName = LLVMUtil::restoreFuncName(appFunDecl->getName().str());
985 for (const Function* extFun : extFuncs)
986 {
987 if (extFun->getName().str().compare(appFunDeclName) == 0)
988 {
989 auto it = ExtFun2Annotations.find(extFun->getName().str());
990 // Without annotations, this function is normal function with useful function body
991 if (it == ExtFun2Annotations.end())
992 {
993 Function* clonedFunction = cloneAndReplaceFunction(const_cast<Function*>(extFun), const_cast<Function*>(appFunDecl), nullptr, true);
994 extFuncs2ClonedFuncs[extFun] = clonedFunction;
995 clonedFuncs.insert(clonedFunction);
996 }
997 else
998 {
999 ExtFuncsVec.push_back(appFunDecl);
1000 }
1001 break;
1002 }
1003 }
1004 }
1005
1008 for (string sameFuncDef: intersectNames)
1009 {
1010 Function* appFuncDef = appModule->getFunction(sameFuncDef);
1011 Function* extFuncDef = extModule->getFunction(sameFuncDef);
1012 if (appFuncDef == nullptr || extFuncDef == nullptr)
1013 continue;
1014
1015 FunctionType *appFuncDefType = appFuncDef->getFunctionType();
1016 FunctionType *extFuncDefType = extFuncDef->getFunctionType();
1017 if (appFuncDefType != extFuncDefType)
1018 continue;
1019
1020 auto it = ExtFun2Annotations.find(sameFuncDef);
1021 if (it != ExtFun2Annotations.end())
1022 {
1023 std::vector<std::string> annotations = it->second;
1024 if (annotations.size() == 1 && annotations[0].find("OVERWRITE") != std::string::npos)
1025 {
1026 Function* clonedFunction = cloneAndReplaceFunction(const_cast<Function*>(extFuncDef), const_cast<Function*>(appFuncDef), nullptr, true);
1027 extFuncs2ClonedFuncs[extFuncDef] = clonedFunction;
1028 clonedFuncs.insert(clonedFunction);
1029 }
1030 else
1031 {
1032 if (annotations.size() >= 2)
1033 {
1034 for (const auto& annotation : annotations)
1035 {
1036 if(annotation.find("OVERWRITE") != std::string::npos)
1037 {
1038 assert(false && "overwrite and other annotations cannot co-exist");
1039 }
1040 }
1041 }
1042 }
1043 }
1044 }
1045
1046 auto linkFunctions = [&](Function* caller, Function* callee)
1047 {
1048 for (inst_iterator I = inst_begin(caller), E = inst_end(caller); I != E; ++I)
1049 {
1050 Instruction *inst = &*I;
1051
1052 if (CallInst *callInst = SVFUtil::dyn_cast<CallInst>(inst))
1053 {
1054 Function *calledFunc = callInst->getCalledFunction();
1055
1056 if (calledFunc && calledFunc->getName() == callee->getName())
1057 {
1058 callInst->setCalledFunction(callee);
1059 }
1060 }
1061 }
1062 };
1063
1064 std::function<void(const Function*, Function*)> cloneAndLinkFunction;
1065 cloneAndLinkFunction = [&](const Function* extFunToClone, Function* appClonedFun)
1066 {
1067 if (clonedFuncs.find(extFunToClone) != clonedFuncs.end())
1068 return;
1069
1070 Module* appModule = appClonedFun->getParent();
1071 // Check if the function already exists in the parent module
1072 if (appModule->getFunction(extFunToClone->getName()))
1073 {
1074 // The function already exists, no need to clone, but need to link it with the caller
1075 Function* func = appModule->getFunction(extFunToClone->getName());
1076 linkFunctions(appClonedFun, func);
1077 return;
1078 }
1079 // Decide whether to clone the function body based on ExtFun2Annotations
1080 bool cloneBody = true;
1081 auto it = ExtFun2Annotations.find(extFunToClone->getName().str());
1082 if (it != ExtFun2Annotations.end())
1083 {
1084 std::vector<std::string> annotations = it->second;
1085 if (!(annotations.size() == 1 && annotations[0].find("OVERWRITE") != std::string::npos))
1086 {
1087 cloneBody = false;
1088 }
1089 }
1090
1091 Function* clonedFunction = cloneAndReplaceFunction(extFunToClone, nullptr, appModule, cloneBody);
1092
1093 clonedFuncs.insert(clonedFunction);
1094 // Add the cloned function to ExtFuncsVec for further processing
1095 ExtFuncsVec.push_back(clonedFunction);
1096
1097 linkFunctions(appClonedFun, clonedFunction);
1098
1099 std::vector<const Function*> calledFunctions = LLVMUtil::getCalledFunctions(extFunToClone);
1100
1101 for (const auto& calledFunction : calledFunctions)
1102 {
1103 cloneAndLinkFunction(calledFunction, clonedFunction);
1104 }
1105 };
1106
1107 // Recursive clone called functions
1108 for (const auto& pair : extFuncs2ClonedFuncs)
1109 {
1110 Function* extFun = const_cast<Function*>(pair.first);
1111 Function* clonedExtFun = const_cast<Function*>(pair.second);
1112 std::vector<const Function*> extCalledFuns = LLVMUtil::getCalledFunctions(extFun);
1113
1114 for (const auto& extCalledFun : extCalledFuns)
1115 {
1116 cloneAndLinkFunction(extCalledFun, clonedExtFun);
1117 }
1118 }
1119
1120 // Remove unused annotations in ExtFun2Annotations according to the functions in ExtFuncsVec
1121 Fun2AnnoMap newFun2AnnoMap;
1122 for (const Function* extFun : ExtFuncsVec)
1123 {
1124 std::string name = LLVMUtil::restoreFuncName(extFun->getName().str());
1125 auto it = ExtFun2Annotations.find(name);
1126 if (it != ExtFun2Annotations.end())
1127 {
1128 std::string newKey = name;
1129 if (name != extFun->getName().str())
1130 {
1131 newKey = extFun->getName().str();
1132 }
1133 newFun2AnnoMap.insert({newKey, it->second});
1134 }
1135 }
1136 ExtFun2Annotations.swap(newFun2AnnoMap);
1137
1138 // Remove ExtAPI module from modules
1139 auto it = std::find_if(modules.begin(), modules.end(),
1140 [&extModule](const std::reference_wrapper<llvm::Module>& moduleRef)
1141 {
1142 return &moduleRef.get() == extModule;
1143 });
1144
1145 if (it != modules.end())
1146 {
1147 size_t index = std::distance(modules.begin(), it);
1148 modules.erase(it);
1149 owned_modules.erase(owned_modules.begin() + index);
1150 }
1151}

1264{

1266 if(it!=LLVMConst2SVFConst.end())
1267 {
1268 assert(SVFUtil::isa<SVFConstantData>(it->second) && "not a SVFConstantData type!");
1269 return SVFUtil::cast<SVFConstantData>(it->second);
1270 }
1271 else
1272 {

1274 if(const ConstantInt* cint = SVFUtil::dyn_cast<ConstantInt>(cd))
1275 {
1279 if(cint->getBitWidth() == 1)
1280 svfcd = new SVFConstantInt(getSVFType(cint->getType()), cint->getZExtValue(), cint->getZExtValue());

1283 else
1284 svfcd = new SVFConstantInt(getSVFType(cint->getType()), 0, 0);
1285 }
1286 else if(const ConstantFP* cfp = SVFUtil::dyn_cast<ConstantFP>(cd))
1287 {
1288 double dval = 0;

1297 &semantics == &llvm::APFloat::x87DoubleExtended())

1300 }

1302 {

1306 }

1309 // other cfp type, like isZero(), isInfinity(), isNegative(), etc.

1315 svfcd = new SVFConstantNullPtr(getSVFType(cd->getType()));
1316 else if (SVFUtil::isa<UndefValue>(cd))
1317 svfcd = new SVFBlackHoleValue(getSVFType(cd->getType()));
1318 else

1325 }

1498{
1499 assert(LLVMType2SVFType.find(T) == LLVMType2SVFType.end() &&
1500 "SVFType has been added before");
1501

1503 u32_t byteSize = 1;

1514 {

1522 svfIntT->setSignAndWidth(intT->getSignBit() ? -signWidth : signWidth);
1523 svftype = svfIntT;
1524 }

1526 svftype = new SVFFunctionType(getSVFType(ft->getReturnType()));
1527 else if (const StructType* st = SVFUtil::dyn_cast<StructType>(T))
1528 {
1529 auto svfst = new SVFStructType(byteSize);
1530 if (st->hasName())
1531 svfst->setName(st->getName().str());
1532 svftype = svfst;
1533 }
1534 else if (const auto at = SVFUtil::dyn_cast<ArrayType>(T))
1535 {
1536 auto svfat = new SVFArrayType(byteSize);
1537 svfat->setNumOfElement(at->getNumElements());
1538 svfat->setTypeOfElement(getSVFType(at->getElementType()));
1539 svftype = svfat;
1540 }
1541 else
1542 {
1543 std::string buffer;
1544 auto ot = new SVFOtherType(T->isSingleValueType(), byteSize);
1545 llvm::raw_string_ostream(buffer) << *T;
1546 ot->setRepr(std::move(buffer));
1547 svftype = ot;
1548 }

1550 svfir->addTypeInfo(svftype);

1559StInfo* LLVMModuleSet::collectArrayInfo(const ArrayType* ty)
1560{
1561 u64_t totalElemNum = ty->getNumElements();
1562 const Type* elemTy = ty->getElementType();
1563 while (const ArrayType* aty = SVFUtil::dyn_cast<ArrayType>(elemTy))
1564 {
1565 totalElemNum *= aty->getNumElements();
1566 elemTy = aty->getElementType();
1567 }
1568
1569 StInfo* stInfo = new StInfo(totalElemNum);
1570 const SVFType* elemSvfType = getSVFType(elemTy);
1571
1573 if (totalElemNum == 0)
1574 {
1575 stInfo->addFldWithType(0, elemSvfType, 0);
1576 stInfo->setNumOfFieldsAndElems(1, 1);
1577 stInfo->getFlattenFieldTypes().push_back(elemSvfType);
1578 stInfo->getFlattenElementTypes().push_back(elemSvfType);
1579 return stInfo;
1580 }
1581
1584 StInfo* elemStInfo = collectTypeInfo(elemTy);
1585 u32_t nfF = elemStInfo->getNumOfFlattenFields();
1586 u32_t nfE = elemStInfo->getNumOfFlattenElements();
1587 for (u32_t j = 0; j < nfF; j++)
1588 {
1589 const SVFType* fieldTy = elemStInfo->getFlattenFieldTypes()[j];
1590 stInfo->getFlattenFieldTypes().push_back(fieldTy);
1591 }
1592
1595 u32_t outArrayElemNum = ty->getNumElements();
1596 for (u32_t i = 0; i < outArrayElemNum; ++i)
1597 {
1598 auto idx = (i * nfE * totalElemNum) / outArrayElemNum;
1599 stInfo->addFldWithType(0, elemSvfType, idx);
1600 }
1601
1602 for (u32_t i = 0; i < totalElemNum; ++i)
1603 {
1604 for (u32_t j = 0; j < nfE; ++j)
1605 {
1606 const SVFType* et = elemStInfo->getFlattenElementTypes()[j];
1607 stInfo->getFlattenElementTypes().push_back(et);
1608 }
1609 }
1610
1611 assert(stInfo->getFlattenElementTypes().size() == nfE * totalElemNum &&
1612 "typeForArray size incorrect!!!");
1613 stInfo->setNumOfFieldsAndElems(nfF, nfE * totalElemNum);
1614
1615 return stInfo;
1616}

1622StInfo* LLVMModuleSet::collectStructInfo(const StructType* structTy,
1623 u32_t& numFields)
1624{
1626 StInfo* stInfo = new StInfo(1);
1627
1628 // Number of fields after flattening the struct
1629 numFields = 0;
1630 // The offset when considering array stride info
1631 u32_t strideOffset = 0;
1632 for (const Type* elemTy : structTy->elements())
1633 {
1634 const SVFType* elemSvfTy = getSVFType(elemTy);

1642 u32_t nfE = subStInfo->getNumOfFlattenElements();

1650 strideOffset += nfE;
1651 for (u32_t tpj = 0; tpj < nfE; ++tpj)
1652 {
1653 const SVFType* ty = subStInfo->getFlattenElementTypes()[tpj];
1654 stInfo->getFlattenElementTypes().push_back(ty);
1655 }
1656
1657 }
1658 else
1659 {
1660 // Simple type
1661 numFields += 1;
1662 strideOffset += 1;
1663 stInfo->getFlattenFieldTypes().push_back(elemSvfTy);
1664 stInfo->getFlattenElementTypes().push_back(elemSvfTy);
1665 }
1666 }
1667
1668 assert(stInfo->getFlattenElementTypes().size() == strideOffset &&
1669 "typeForStruct size incorrect!");
1670 stInfo->setNumOfFieldsAndElems(numFields,strideOffset);
