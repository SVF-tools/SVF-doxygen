50/*
51 svf.main() is used to model the real entry point of a C++ program, which
52 initializes all global C++ objects and then call main().
53 LLVM may generate two global arrays @llvm.global_ctors and @llvm.global_dtors
54 that contain constructor and destructor functions for global variables. They
55 are not called explicitly, so we have to add them in the svf.main function.
56 The order to call these constructor and destructor functions are also
57 specified in the global arrays.
58 Related part in LLVM language reference:
59 https://llvm.org/docs/LangRef.html#the-llvm-global-ctors-global-variable
60 For example, given a "int main(int argc, char * argv[])", the corresponding
61 svf.main will be generated as follows:
62 define void @svf.main(i32, i8**, i8**) {
63 entry:
64 call void @ctor1()
65 call void @ctor2()
66 %3 = call i32 @main(i32 %0, i8** %1)
67 call void @dtor1()
68 call void @dtor2()
69 ret void
70 }
71*/
72
73#define SVF_MAIN_FUNC_NAME "svf.main"
74#define SVF_GLOBAL_CTORS "llvm.global_ctors"
75#define SVF_GLOBAL_DTORS "llvm.global_dtors"

193void LLVMModuleSet::createSVFDataStructure()
194{
195 SVFType::svfI8Ty = getSVFType(getTypeInference()->int8Type());
196 SVFType::svfPtrTy = getSVFType(getTypeInference()->ptrType());
197 // Functions need to be retrieved in the order of insertion
198 // candidateDefs is the vector for all used defined functions
199 // candidateDecls is the vector for all used declared functions
200 std::vector<const Function*> candidateDefs, candidateDecls;
201
202 for (Module& mod : modules)
203 {
205 for (Function& func : mod.functions())
206 {
207 if (func.isDeclaration())
208 {
209 candidateDecls.push_back(&func);
210 }
211 else
212 {
213 candidateDefs.push_back(&func);
214 }
215 }
216 }
217
218 for (const Function* func: candidateDefs)
219 {
220 createSVFFunction(func);
221 }
222 for (const Function* func: candidateDecls)
223 {
224 createSVFFunction(func);
225 }
226
227 // Store annotations of functions in extapi.bc
228 for (const auto& pair : ExtFun2Annotations)
229 {
230 const SVFFunction* svffun = getSVFFunction(pair.first);
231 setExtFuncAnnotations(svffun, pair.second);
232 }
233
235 for (const Module& mod : modules)
236 {
238 for (const GlobalVariable& global : mod.globals())
239 {
240 SVFGlobalValue* svfglobal = new SVFGlobalValue(
241 global.getName().str(), getSVFType(global.getType()));
242 svfModule->addGlobalSet(svfglobal);
243 addGlobalValueMap(&global, svfglobal);
244 }
245
247 for (const GlobalAlias& alias : mod.aliases())
248 {
249 SVFGlobalValue* svfalias = new SVFGlobalValue(
250 alias.getName().str(), getSVFType(alias.getType()));
251 svfModule->addAliasSet(svfalias);
252 addGlobalValueMap(&alias, svfalias);
253 }
254
256 for (const GlobalIFunc& ifunc : mod.ifuncs())
257 {
258 SVFGlobalValue* svfifunc = new SVFGlobalValue(
259 ifunc.getName().str(), getSVFType(ifunc.getType()));
260 svfModule->addAliasSet(svfifunc);
261 addGlobalValueMap(&ifunc, svfifunc);
262 }
263 }

266void LLVMModuleSet::createSVFFunction(const Function* func)
267{
268 SVFFunction* svfFunc = new SVFFunction(
269 getSVFType(func->getType()),
270 SVFUtil::cast<SVFFunctionType>(
271 getSVFType(func->getFunctionType())),
272 func->isDeclaration(), LLVMUtil::isIntrinsicFun(func),
273 func->hasAddressTaken(), func->isVarArg(), new SVFLoopAndDomInfo);
274 BasicBlockGraph* bbGraph = new BasicBlockGraph();
275 svfFunc->setBasicBlockGraph(bbGraph);
276 svfModule->addFunctionSet(svfFunc);
277 addFunctionMap(func, svfFunc);
278
279 for (const Argument& arg : func->args())
280 {
281 SVFArgument* svfarg = new SVFArgument(
282 getSVFType(arg.getType()), svfFunc, arg.getArgNo(),
283 LLVMUtil::isArgOfUncalledFunction(&arg));
284 // Setting up arg name
285 if (!arg.hasName())
286 svfarg->setName(std::to_string(arg.getArgNo()));
287
288 addArgumentMap(&arg, svfarg);
289 }
290
291 for (const BasicBlock& bb : *func)
292 {
293 addBasicBlock(svfFunc, &bb);
294 for (const Instruction& inst : bb)
295 {
296 SVFInstruction* svfInst = nullptr;
297 if (const CallBase* call = SVFUtil::dyn_cast<CallBase>(&inst))
298 {
299 svfInst = new SVFCallInst(
300 getSVFType(call->getType()), getSVFBasicBlock(&bb),
301 call->getFunctionType()->isVarArg(),
302 inst.isTerminator());
303 }
304 else
305 {
306 svfInst =
307 new SVFInstruction(getSVFType(inst.getType()),
308 getSVFBasicBlock(&bb), inst.isTerminator(),
309 SVFUtil::isa<ReturnInst>(inst));
310 }
311
312 addInstructionMap(&inst, svfInst);
313 }
314 }

335void LLVMModuleSet::initSVFBasicBlock(const Function* func)
336{
337 SVFFunction *svfFun = getSVFFunction(func);
338 for (Function::const_iterator bit = func->begin(), ebit = func->end(); bit != ebit; ++bit)
339 {
340 const BasicBlock* bb = &*bit;
341 SVFBasicBlock* svfbb = getSVFBasicBlock(bb);
342 for (succ_const_iterator succ_it = succ_begin(bb); succ_it != succ_end(bb); succ_it++)
343 {
344 const SVFBasicBlock* svf_scc_bb = getSVFBasicBlock(*succ_it);
345 svfbb->addSuccBasicBlock(svf_scc_bb);
346 }
347 for (const_pred_iterator pred_it = pred_begin(bb); pred_it != pred_end(bb); pred_it++)
348 {
349 const SVFBasicBlock* svf_pred_bb = getSVFBasicBlock(*pred_it);
350 svfbb->addPredBasicBlock(svf_pred_bb);
351 }
352
354 if (svfbb->getSuccessors().empty())
355 {
356 if (LLVMUtil::basicBlockHasRetInst(bb))
357 {
358 assert((LLVMUtil::functionDoesNotRet(func) ||
359 SVFUtil::isa<ReturnInst>(bb->back())) &&
360 "last inst must be return inst");
361 svfFun->setExitBlock(svfbb);
362 }
363 }
364
365 for (BasicBlock::const_iterator iit = bb->begin(), eiit = bb->end(); iit != eiit; ++iit)
366 {
367 const Instruction* inst = &*iit;
368 if(const CallBase* call = SVFUtil::dyn_cast<CallBase>(inst))
369 {
370 SVFInstruction* svfinst = getSVFInstruction(call);
371 SVFCallInst* svfcall = SVFUtil::cast<SVFCallInst>(svfinst);
372 auto called_llvmval = call->getCalledOperand()->stripPointerCasts();
373 if (const Function* called_llvmfunc = SVFUtil::dyn_cast<Function>(called_llvmval))
374 {
375 SVFFunction* callee = getSVFFunction(called_llvmfunc);
376 svfcall->setCalledOperand(callee);
377 }
378 else
379 {
380 svfcall->setCalledOperand(getSVFValue(called_llvmval));
381 }
382 for(u32_t i = 0; i < call->arg_size(); i++)
383 {
384 SVFLLVMValue* svfval = getSVFValue(call->getArgOperand(i));
385 svfcall->addArgument(svfval);
386 }
387 }
388 }
389 }
390 // For no return functions, we set the last block as exit BB
391 // This ensures that each function that has definition must have an exit BB
392 if (svfFun->exitBlock == nullptr && svfFun->hasBasicBlock())
393 {
394 SVFBasicBlock* retBB = const_cast<SVFBasicBlock*>(svfFun->back());
395 assert((LLVMUtil::functionDoesNotRet(func) ||
396 SVFUtil::isa<ReturnInst>(&func->back().back())) &&
397 "last inst must be return inst");
398 svfFun->setExitBlock(retBB);
399 }
400}

403void LLVMModuleSet::initDomTree(SVFFunction* svffun, const Function* fun)
404{
405 if (fun->isDeclaration())
406 return;
407 //process and stored dt & df
408 DominanceFrontier df;
409 DominatorTree& dt = getDomTree(fun);
410 df.analyze(dt);
411 LoopInfo loopInfo = LoopInfo(dt);
412 PostDominatorTree pdt = PostDominatorTree(const_cast<Function&>(*fun));
413 SVFLoopAndDomInfo* ld = svffun->getLoopAndDomInfo();
414
415 Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>> & dfBBsMap = ld->getDomFrontierMap();
416 for (DominanceFrontierBase::const_iterator dfIter = df.begin(), eDfIter = df.end(); dfIter != eDfIter; dfIter++)
417 {
418 const BasicBlock* keyBB = dfIter->first;
419 const std::set<BasicBlock* >& domSet = dfIter->second;
420 Set<const SVFBasicBlock*>& valueBasicBlocks = dfBBsMap[getSVFBasicBlock(keyBB)];
421 for (const BasicBlock* bbValue:domSet)
422 {
423 valueBasicBlocks.insert(getSVFBasicBlock(bbValue));
424 }
425 }
426 std::vector<const SVFBasicBlock*> reachableBBs;
427 LLVMUtil::getFunReachableBBs(fun, reachableBBs);
428 ld->setReachableBBs(reachableBBs);
429
430 for (Function::const_iterator bit = fun->begin(), beit = fun->end(); bit!=beit; ++bit)
431 {
432 const BasicBlock &bb = *bit;
433 SVFBasicBlock* svfBB = getSVFBasicBlock(&bb);
434 if (DomTreeNode* dtNode = dt.getNode(&bb))
435 {
436 SVFLoopAndDomInfo::BBSet& bbSet = ld->getDomTreeMap()[svfBB];
437 for (const auto domBB : *dtNode)
438 {
439 const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
440 bbSet.insert(domSVFBB);
441 }
442 }
443
444 if (DomTreeNode* pdtNode = pdt.getNode(&bb))
445 {
446 u32_t level = pdtNode->getLevel();
447 ld->getBBPDomLevel()[svfBB] = level;
448 BasicBlock* idomBB = pdtNode->getIDom()->getBlock();
449 const SVFBasicBlock* idom = idomBB == NULL ? NULL: getSVFBasicBlock(idomBB);
450 ld->getBB2PIdom()[svfBB] = idom;
451
452 SVFLoopAndDomInfo::BBSet& bbSet = ld->getPostDomTreeMap()[svfBB];
453 for (const auto domBB : *pdtNode)
454 {
455 const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
456 bbSet.insert(domSVFBB);
457 }
458 }
459
460 if (const Loop* loop = loopInfo.getLoopFor(&bb))
461 {
462 for (const BasicBlock* loopBlock : loop->getBlocks())
463 {
464 const SVFBasicBlock* loopbb = getSVFBasicBlock(loopBlock);
465 ld->addToBB2LoopMap(svfBB, loopbb);
466 }
467 }
468 }

492 if (fun.isDeclaration())

494 p2->runOnFunction(fun);
495 }

520void LLVMModuleSet::loadModules(const std::vector<std::string> &moduleNameVec)
521{
522
523 // We read SVFIR from LLVM IR
524 if(Options::Graphtxt().empty())
525 {
526 if(moduleNameVec.empty())
527 {
528 SVFUtil::outs() << "no LLVM bc file is found!\n";
529 exit(0);
530 }
531 //assert(!moduleNameVec.empty() && "no LLVM bc file is found!");
532 }
533 // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
534 else
535 {
536 SVFIR::setPagFromTXT(Options::Graphtxt());
537 }
538 //
539 // LLVMContext objects separate global LLVM settings (from which e.g. types are
540 // derived); multiple LLVMContext objects can coexist and each context can "own"
541 // multiple modules (modules can only have one context). Mixing contexts can lead
542 // to unintended inequalities, such as the following:
543 //
544 // ------------------------------------------------------------------
545 // LLVMContext ctxa,ctxb;
546 // IntegerType * t1 = IntegerType::get(ctxa,32);
547 // IntegerType * t2 = IntegerType::get(ctxa,32);
548 // assert(t1 == t2);
549 // IntegerType * t3 = IntegerType::get(ctxb,32);
550 // IntegerType * t4 = IntegerType::get(ctxb,32);
551 // assert(t3 == t4);
552 // assert(t1 != t3);
553 // ------------------------------------------------------------------
554 //
555 // When loading bytecode files, SVF will use the same LLVMContext object for all
556 // modules (i.e. the context owns all loaded modules). This applies to ExtAPI as
557 // well, which *must* be loaded using the same LLVMContext object. Hence, when
558 // loading modules from bitcode files, a new LLVMContext is created (using a
559 // `std::unique_ptr<LLVMContext>` type to ensure automatic garbage collection).
560 //
561 // This garbage collection should be avoided when building an SVF module from an LLVM
562 // module instance; see the comment(s) in `buildSVFModule` and `loadExtAPIModules()`
563
564 owned_ctx = std::make_unique<LLVMContext>();
565 for (const std::string& moduleName : moduleNameVec)
566 {
567 if (!LLVMUtil::isIRFile(moduleName))
568 {
569 SVFUtil::errs() << "not an IR file: " << moduleName << std::endl;
570 abort();
571 }
572
573 SMDiagnostic Err;
574 std::unique_ptr<Module> mod = parseIRFile(moduleName, Err, *owned_ctx);
575 if (mod == nullptr)
576 {
577 SVFUtil::errs() << "load module: " << moduleName << "failed!!\n\n";
578 Err.print("SVFModuleLoader", llvm::errs());
579 abort();
580 }
581 modules.emplace_back(*mod);
582 owned_modules.emplace_back(std::move(mod));
583 }
584}

586void LLVMModuleSet::loadExtAPIModules()
587{
588 // This function loads the ExtAPI bitcode file as an LLVM module. Note that it is important that
589 // the same LLVMContext object is used to load this bitcode file as is used by the other modules
590 // being analysed.
591 // When the modules are loaded from bitcode files (i.e. passing filenames to files containing
592 // LLVM IR to `buildSVFModule({file1.bc, file2.bc, ...})) the context is created while loading
593 // the modules in `loadModules()`, which populates this->modules and this->owned_modules.
594 // If, however, an LLVM Module object is passed to `buildSVFModule` (e.g. from an LLVM pass),
595 // the context should be retrieved from the module itself (note that the garbage collection from
596 // `std::unique_ptr<LLVMContext> LLVMModuleSet::owned_ctx` should be avoided in this case). This
597 // function populates only this->modules.
598 // In both cases, fetching the context from the main LLVM module (through `getContext`) works
599 assert(!empty() && "LLVMModuleSet contains no modules; cannot load ExtAPI module without LLVMContext!");
600
601 // Load external API module (extapi.bc)
602 if (!ExtAPI::getExtAPI()->getExtBcPath().empty())
603 {
604 std::string extModuleName = ExtAPI::getExtAPI()->getExtBcPath();
605 if (!LLVMUtil::isIRFile(extModuleName))
606 {
607 SVFUtil::errs() << "not an external IR file: " << extModuleName << std::endl;
608 abort();
609 }
610 SMDiagnostic Err;
611 std::unique_ptr<Module> mod = parseIRFile(extModuleName, Err, getContext());
612 if (mod == nullptr)
613 {
614 SVFUtil::errs() << "load external module: " << extModuleName << "failed!!\n\n";
615 Err.print("SVFModuleLoader", llvm::errs());
616 abort();
617 }
618 // The module of extapi.bc needs to be inserted before applications modules, like std::vector<std::reference_wrapper<Module>> modules{extapi_module, app_module}.
619 // Otherwise, when overwriting the app function with SVF extern function, the corresponding SVFFunction of the extern function will not be found.
620 modules.insert(modules.begin(), *mod);
621 owned_modules.insert(owned_modules.begin(),std::move(mod));
622 }

625std::vector<const Function* > LLVMModuleSet::getLLVMGlobalFunctions(const GlobalVariable *global)
626{
627 // This function is used to extract constructor and destructor functions
628 // sorted by their priority from @llvm.global_ctors or @llvm.global_dtors.
629 // For example, given following @llvm.global_ctors, the returning sorted
630 // function list should be [ctor3, ctor1, ctor2].
631 // ------------------------------------------------------------------
632 // ; Each struct in the array is {priority, function, associated data}
633 //
634 // @llvm.global_ctors = appending global [2 x { i32, void ()*, i8* }]
635 // [{ i32, void ()*, i8* } { i32 1234, void ()* @ctor1.cpp, i8* null },
636 // { i32, void ()*, i8* } { i32 2345, void ()* @ctor2.cpp, i8* null },
637 // { i32, void ()*, i8* } { i32 345, void ()* @ctor3.cpp, i8* null }]
638 // ------------------------------------------------------------------
639 // TODO: According to LLVM language reference, if the third field is
640 // non-null, and points to a global variable or function, the initializer
641 // function will only run if the associated data from the current module is
642 // not discarded. However the associated data is currently ignored.
643
644
645 // This class is used for the priority queue that sorts the functions by
646 // their priority. Each object of this class stands for an item in the
647 // function array.
648 class LLVMGlobalFunction
649 {
650 public:
651 u32_t priority;
652 const Function* func;
653 LLVMGlobalFunction() {};
654 LLVMGlobalFunction(u32_t _priority, const Function* _func)
655 : priority(_priority), func(_func) {};
656 bool operator>(const LLVMGlobalFunction &other) const
657 {
658 if (priority != other.priority)
659 {
660 return priority > other.priority;
661 }
662 else
663 {
664 return func > other.func;
665 }
666 }
667 };
668
669 std::priority_queue<LLVMGlobalFunction, std::vector<LLVMGlobalFunction>,
670 greater<LLVMGlobalFunction>>
671 queue;
672 std::vector<const Function* > result;
673
674 // The @llvm.global_ctors/dtors global variable is an array of struct. Each
675 // struct has three fields: {i32 priority, void ()* @ctor/dtor, i8* @data}.
676 // First get the array here.
677 if(const ConstantArray *globalFuncArray =
678 SVFUtil::dyn_cast<ConstantArray>(global->getInitializer()))
679 {
680 // Get each struct in the array.
681 for (unsigned int i = 0; i < globalFuncArray->getNumOperands(); ++i)
682 {
683 if (
684 const ConstantStruct *globalFuncItem =
685 SVFUtil::dyn_cast<ConstantStruct>(
686 globalFuncArray->getOperand(i)))
687 {
688
689 // Extract priority and function from the struct
690 const ConstantInt* priority = SVFUtil::dyn_cast<ConstantInt>(
691 globalFuncItem->getOperand(0));
692 const Function* func = SVFUtil::dyn_cast<Function>(
693 globalFuncItem->getOperand(1));
694
695 if (priority && func)
696 {
697 queue.push(LLVMGlobalFunction(LLVMUtil::getIntegerValue(priority).second,
698 func));
699 }
700 }
701 }
702 }
703
704 // Generate a sorted vector of functions from the priority queue.
705 while (!queue.empty())
706 {
707 result.push_back(queue.top().func);
708 queue.pop();
709 }
710 return result;
711}

713void LLVMModuleSet::addSVFMain()
714{
715 std::vector<const Function*> ctor_funcs;
716 std::vector<const Function*> dtor_funcs;
717 Function* orgMain = 0;
718 Module* mainMod = nullptr;
719
720 for (Module &mod : modules)
721 {
722 // Collect ctor and dtor functions
723 for (const GlobalVariable& global : mod.globals())
724 {
725 if (global.getName().equals(SVF_GLOBAL_CTORS) && global.hasInitializer())
726 {
727 ctor_funcs = getLLVMGlobalFunctions(&global);
728 }
729 else if (global.getName().equals(SVF_GLOBAL_DTORS) && global.hasInitializer())
730 {
731 dtor_funcs = getLLVMGlobalFunctions(&global);
732 }
733 }
734
735 // Find main function
736 for (auto &func : mod)
737 {
738 auto funName = func.getName();
739
740 assert(!funName.equals(SVF_MAIN_FUNC_NAME) && SVF_MAIN_FUNC_NAME " already defined");
741
742 if (funName.equals("main"))
743 {
744 orgMain = &func;
745 mainMod = &mod;
746 }
747 }
748 }
749
750 // Only create svf.main when the original main function is found, and also
751 // there are global constructor or destructor functions.
752 if (orgMain && getModuleNum() > 0 &&
753 (ctor_funcs.size() > 0 || dtor_funcs.size() > 0))
754 {
755 assert(mainMod && "Module with main function not found.");
756 Module& M = *mainMod;
757 // char **
758 Type* ptr = PointerType::getUnqual(M.getContext());
759 Type* i32 = IntegerType::getInt32Ty(M.getContext());
760 // define void @svf.main(i32, i8**, i8**)
761#if (LLVM_VERSION_MAJOR >= 9)
762 FunctionCallee svfmainFn = M.getOrInsertFunction(
763 SVF_MAIN_FUNC_NAME,
764 Type::getVoidTy(M.getContext()),
765 i32,ptr,ptr
766 );
767 Function* svfmain = SVFUtil::dyn_cast<Function>(svfmainFn.getCallee());
768#else
769 Function* svfmain = SVFUtil::dyn_cast<Function>(M.getOrInsertFunction(
770 SVF_MAIN_FUNC_NAME,
771 Type::getVoidTy(M.getContext()),
772 i32,i8ptr2,i8ptr2
773 ));
774#endif
775 svfmain->setCallingConv(llvm::CallingConv::C);
776 BasicBlock* block = BasicBlock::Create(M.getContext(), "entry", svfmain);
777 IRBuilder Builder(block);
778 // emit "call void @ctor()". ctor_funcs is sorted so the functions are
779 // emitted in the order of priority
780 for (auto& ctor : ctor_funcs)
781 {
782 auto target = M.getOrInsertFunction(
783 ctor->getName(),
784 Type::getVoidTy(M.getContext())
785 );
786 Builder.CreateCall(target);
787 }
788 // main() should be called after all ctor functions and before dtor
789 // functions.
790 Function::arg_iterator arg_it = svfmain->arg_begin();
791 Value* args[] = {arg_it, arg_it + 1, arg_it + 2};
792 size_t cnt = orgMain->arg_size();
793 assert(cnt <= 3 && "Too many arguments for main()");
794 Builder.CreateCall(orgMain, llvm::ArrayRef<Value*>(args, args + cnt));
795 // emit "call void @dtor()". dtor_funcs is sorted so the functions are
796 // emitted in the order of priority
797 for (auto& dtor : dtor_funcs)
798 {
799 auto target = M.getOrInsertFunction(dtor->getName(), Type::getVoidTy(M.getContext()));
800 Builder.CreateCall(target);
801 }
802 // return;
803 Builder.CreateRetVoid();
804 }
805}

807void LLVMModuleSet::collectExtFunAnnotations(const Module* mod)
808{
809 GlobalVariable *glob = mod->getGlobalVariable("llvm.global.annotations");
810 if (glob == nullptr || !glob->hasInitializer())
811 return;
812
813 ConstantArray *ca = SVFUtil::dyn_cast<ConstantArray>(glob->getInitializer());
814 if (ca == nullptr)
815 return;
816
817 for (unsigned i = 0; i < ca->getNumOperands(); ++i)
818 {
819 ConstantStruct *structAn = SVFUtil::dyn_cast<ConstantStruct>(ca->getOperand(i));
820 if (structAn == nullptr || structAn->getNumOperands() == 0)
821 continue;
822
823 // Check if the annotation is for a function
824 Function* fun = nullptr;
825 GlobalVariable *annotateStr = nullptr;
827 if (ConstantExpr *expr = SVFUtil::dyn_cast<ConstantExpr>(structAn->getOperand(0)))
828 {
829 if (expr->getOpcode() == Instruction::BitCast && SVFUtil::isa<Function>(expr->getOperand(0)))
830 fun = SVFUtil::cast<Function>(expr->getOperand(0));
831
832 ConstantExpr *note = SVFUtil::cast<ConstantExpr>(structAn->getOperand(1));
833 if (note->getOpcode() != Instruction::GetElementPtr)
834 continue;
835
836 annotateStr = SVFUtil::dyn_cast<GlobalVariable>(note->getOperand(0));
837 }
839 else
840 {
841 fun = SVFUtil::dyn_cast<Function>(structAn->getOperand(0));
842 annotateStr = SVFUtil::dyn_cast<GlobalVariable>(structAn->getOperand(1));
843 }
844
845 if (!fun || annotateStr == nullptr || !annotateStr->hasInitializer())
846 continue;;
847
848 ConstantDataSequential *data = SVFUtil::dyn_cast<ConstantDataSequential>(annotateStr->getInitializer());
849 if (data && data->isString())
850 {
851 std::string annotation = data->getAsString().str();
852 if (!annotation.empty())
853 ExtFun2Annotations[fun->getName().str()].push_back(annotation);
854 }
855 }
856}
857
858/*
859 There are three types of functions(definitions) in extapi.c:
860 1. (Fun_Overwrite): Functions with "OVERWRITE" annotion:
861 These functions are used to replace the corresponding function definitions in the application.
862 2. (Fun_Annotation): Functions with annotation(s) but without "OVERWRITE" annotation:
863 These functions are used to tell SVF to do special processing, like malloc().
864 3. (Fun_Noraml): Functions without any annotation:
865 These functions are used to replace the corresponding function declarations in the application.
866
867
868 We will iterate over declarations (appFunDecl) and definitons (appFunDef) of functions in the application and extapi.c to do the following clone or replace operations:
869 1. appFuncDecl --> Fun_Normal: Clone the Fun_Overwrite and replace the appFuncDecl in application.
870 2. appFuncDecl --> Fun_Annotation: Move the annotions on Fun_Annotation to appFuncDecl in application.
871
872 3. appFunDef --> Fun_Overwrite: Clone the Fun_Overwrite and overwrite the appFunDef in application.
873 4. appFunDef --> Fun_Annotation: Replace the appFunDef with appFunDecl and move the annotions to appFunDecl in application
874*/
875void LLVMModuleSet::buildFunToFunMap()
876{
877 Set<const Function*> appFunDecls, appFunDefs, extFuncs, clonedFuncs;
878 OrderedSet<string> appFuncDeclNames, appFuncDefNames, extFunDefNames, intersectNames;
879 Map<const Function*, const Function*> extFuncs2ClonedFuncs;
880 Module* appModule = nullptr;
881 Module* extModule = nullptr;
882
883 for (Module& mod : modules)
884 {
885 // extapi.bc functions
886 if (mod.getName().str() == ExtAPI::getExtAPI()->getExtBcPath())
887 {
888 collectExtFunAnnotations(&mod);
889 extModule = &mod;
890 for (const Function& fun : mod.functions())
891 {
892 // there is main declaration in ext bc, it should be mapped to
893 // main definition in app bc.
894 if (fun.getName().str() == "main")
895 {
896 appFunDecls.insert(&fun);
897 appFuncDeclNames.insert(fun.getName().str());
898 }
900 else if (fun.getName().str() == "svf__main")
901 {
902 ExtFuncsVec.push_back(&fun);
903 }
904 else
905 {
906 extFuncs.insert(&fun);
907 extFunDefNames.insert(fun.getName().str());
908 }
909 }
910 }
911 else
912 {
913 appModule = &mod;
915 for (const Function& fun : mod.functions())
916 {
917 if (fun.isDeclaration())
918 {
919 appFunDecls.insert(&fun);
920 appFuncDeclNames.insert(fun.getName().str());
921 }
922 else
923 {
924 appFunDefs.insert(&fun);
925 appFuncDefNames.insert(fun.getName().str());
926 }
927 }
928 }
929 }
930
931 // Find the intersectNames between appFuncDefNames and externalFunDefNames
932 std::set_intersection(
933 appFuncDefNames.begin(), appFuncDefNames.end(), extFunDefNames.begin(), extFunDefNames.end(),
934 std::inserter(intersectNames, intersectNames.end()));
935
936 auto cloneAndReplaceFunction = [&](const Function* extFunToClone, Function* appFunToReplace, Module* appModule, bool cloneBody) -> Function*
937 {
938 assert(!(appFunToReplace == NULL && appModule == NULL) && "appFunToReplace and appModule cannot both be NULL");
939
940 if (appFunToReplace)
941 {
942 appModule = appFunToReplace->getParent();
943 }
944 // Create a new function with the same signature as extFunToClone
945 Function *clonedFunction = Function::Create(extFunToClone->getFunctionType(), Function::ExternalLinkage, extFunToClone->getName(), appModule);
946 // Map the arguments of the new function to the arguments of extFunToClone
947 llvm::ValueToValueMapTy valueMap;
948 Function::arg_iterator destArg = clonedFunction->arg_begin();
949 for (Function::const_arg_iterator srcArg = extFunToClone->arg_begin(); srcArg != extFunToClone->arg_end(); ++srcArg)
950 {
951 destArg->setName(srcArg->getName()); // Copy the name of the original argument
952 valueMap[&*srcArg] = &*destArg++; // Add a mapping from the old arg to the new arg
953 }
954 if (cloneBody)
955 {
956 // Clone the body of extFunToClone into clonedFunction
957 llvm::SmallVector<ReturnInst*, 8> ignoredReturns;
958 CloneFunctionInto(clonedFunction, extFunToClone, valueMap, llvm::CloneFunctionChangeType::LocalChangesOnly, ignoredReturns, "", nullptr);
959 }
960 if (appFunToReplace)
961 {
962 // Replace all uses of appFunToReplace with clonedFunction
963 appFunToReplace->replaceAllUsesWith(clonedFunction);
964 std::string oldFunctionName = appFunToReplace->getName().str();
965 // Delete the old function
966 appFunToReplace->eraseFromParent();
967 clonedFunction->setName(oldFunctionName);
968 }
969 return clonedFunction;
970 };
971
973 for (const Function* appFunDecl : appFunDecls)
974 {
975 std::string appFunDeclName = LLVMUtil::restoreFuncName(appFunDecl->getName().str());
976 for (const Function* extFun : extFuncs)
977 {
978 if (extFun->getName().str().compare(appFunDeclName) == 0)
979 {
980 auto it = ExtFun2Annotations.find(extFun->getName().str());
981 // Without annotations, this function is normal function with useful function body
982 if (it == ExtFun2Annotations.end())
983 {
984 Function* clonedFunction = cloneAndReplaceFunction(const_cast<Function*>(extFun), const_cast<Function*>(appFunDecl), nullptr, true);
985 extFuncs2ClonedFuncs[extFun] = clonedFunction;
986 clonedFuncs.insert(clonedFunction);
987 }
988 else
989 {
990 ExtFuncsVec.push_back(appFunDecl);
991 }
992 break;
993 }
994 }
995 }
996
999 for (string sameFuncDef: intersectNames)
1000 {
1001 Function* appFuncDef = appModule->getFunction(sameFuncDef);
1002 Function* extFuncDef = extModule->getFunction(sameFuncDef);
1003 if (appFuncDef == nullptr || extFuncDef == nullptr)
1004 continue;
1005
1006 FunctionType *appFuncDefType = appFuncDef->getFunctionType();
1007 FunctionType *extFuncDefType = extFuncDef->getFunctionType();
1008 if (appFuncDefType != extFuncDefType)
1009 continue;
1010
1011 auto it = ExtFun2Annotations.find(sameFuncDef);
1012 if (it != ExtFun2Annotations.end())
1013 {
1014 std::vector<std::string> annotations = it->second;
1015 if (annotations.size() == 1 && annotations[0].find("OVERWRITE") != std::string::npos)
1016 {
1017 Function* clonedFunction = cloneAndReplaceFunction(const_cast<Function*>(extFuncDef), const_cast<Function*>(appFuncDef), nullptr, true);
1018 extFuncs2ClonedFuncs[extFuncDef] = clonedFunction;
1019 clonedFuncs.insert(clonedFunction);
1020 }
1021 else
1022 {
1023 if (annotations.size() >= 2)
1024 {
1025 for (const auto& annotation : annotations)
1026 {
1027 if(annotation.find("OVERWRITE") != std::string::npos)
1028 {
1029 assert(false && "overwrite and other annotations cannot co-exist");
1030 }
1031 }
1032 }
1033 }
1034 }
1035 }
1036
1037 auto linkFunctions = [&](Function* caller, Function* callee)
1038 {
1039 for (inst_iterator I = inst_begin(caller), E = inst_end(caller); I != E; ++I)
1040 {
1041 Instruction *inst = &*I;
1042
1043 if (CallInst *callInst = SVFUtil::dyn_cast<CallInst>(inst))
1044 {
1045 Function *calledFunc = callInst->getCalledFunction();
1046
1047 if (calledFunc && calledFunc->getName() == callee->getName())
1048 {
1049 callInst->setCalledFunction(callee);
1050 }
1051 }
1052 }
1053 };
1054
1055 std::function<void(const Function*, Function*)> cloneAndLinkFunction;
1056 cloneAndLinkFunction = [&](const Function* extFunToClone, Function* appClonedFun)
1057 {
1058 if (clonedFuncs.find(extFunToClone) != clonedFuncs.end())
1059 return;
1060
1061 Module* appModule = appClonedFun->getParent();
1062 // Check if the function already exists in the parent module
1063 if (appModule->getFunction(extFunToClone->getName()))
1064 {
1065 // The function already exists, no need to clone, but need to link it with the caller
1066 Function* func = appModule->getFunction(extFunToClone->getName());
1067 linkFunctions(appClonedFun, func);
1068 return;
1069 }
1070 // Decide whether to clone the function body based on ExtFun2Annotations
1071 bool cloneBody = true;
1072 auto it = ExtFun2Annotations.find(extFunToClone->getName().str());
1073 if (it != ExtFun2Annotations.end())
1074 {
1075 std::vector<std::string> annotations = it->second;
1076 if (!(annotations.size() == 1 && annotations[0].find("OVERWRITE") != std::string::npos))
1077 {
1078 cloneBody = false;
1079 }
1080 }
1081
1082 Function* clonedFunction = cloneAndReplaceFunction(extFunToClone, nullptr, appModule, cloneBody);
1083
1084 clonedFuncs.insert(clonedFunction);
1085 // Add the cloned function to ExtFuncsVec for further processing
1086 ExtFuncsVec.push_back(clonedFunction);
1087
1088 linkFunctions(appClonedFun, clonedFunction);
1089
1090 std::vector<const Function*> calledFunctions = LLVMUtil::getCalledFunctions(extFunToClone);
1091
1092 for (const auto& calledFunction : calledFunctions)
1093 {
1094 cloneAndLinkFunction(calledFunction, clonedFunction);
1095 }
1096 };
1097
1098 // Recursive clone called functions
1099 for (const auto& pair : extFuncs2ClonedFuncs)
1100 {
1101 Function* extFun = const_cast<Function*>(pair.first);
1102 Function* clonedExtFun = const_cast<Function*>(pair.second);
1103 std::vector<const Function*> extCalledFuns = LLVMUtil::getCalledFunctions(extFun);
1104
1105 for (const auto& extCalledFun : extCalledFuns)
1106 {
1107 cloneAndLinkFunction(extCalledFun, clonedExtFun);
1108 }
1109 }
1110
1111 // Remove unused annotations in ExtFun2Annotations according to the functions in ExtFuncsVec
1112 Fun2AnnoMap newFun2AnnoMap;
1113 for (const Function* extFun : ExtFuncsVec)
1114 {
1115 std::string name = LLVMUtil::restoreFuncName(extFun->getName().str());
1116 auto it = ExtFun2Annotations.find(name);
1117 if (it != ExtFun2Annotations.end())
1118 {
1119 std::string newKey = name;
1120 if (name != extFun->getName().str())
1121 {
1122 newKey = extFun->getName().str();
1123 }
1124 newFun2AnnoMap.insert({newKey, it->second});
1125 }
1126 }
1127 ExtFun2Annotations.swap(newFun2AnnoMap);
1128
1129 // Remove ExtAPI module from modules
1130 auto it = std::find_if(modules.begin(), modules.end(),
1131 [&extModule](const std::reference_wrapper<llvm::Module>& moduleRef)
1132 {
1133 return &moduleRef.get() == extModule;
1134 });
1135
1136 if (it != modules.end())
1137 {
1138 size_t index = std::distance(modules.begin(), it);
1139 modules.erase(it);
1140 owned_modules.erase(owned_modules.begin() + index);
1141 }
1142}

1319SVFConstantData* LLVMModuleSet::getSVFConstantData(const ConstantData* cd)
1320{
1321 LLVMConst2SVFConstMap::const_iterator it = LLVMConst2SVFConst.find(cd);
1322 if(it!=LLVMConst2SVFConst.end())
1323 {
1324 assert(SVFUtil::isa<SVFConstantData>(it->second) && "not a SVFConstantData type!");
1325 return SVFUtil::cast<SVFConstantData>(it->second);
1326 }
1327 else
1328 {
1329 SVFConstantData* svfcd = nullptr;
1330 if(const ConstantInt* cint = SVFUtil::dyn_cast<ConstantInt>(cd))
1331 {
1335 if(cint->getBitWidth() == 1)
1336 svfcd = new SVFConstantInt(getSVFType(cint->getType()), cint->getZExtValue(), cint->getZExtValue());
1337 else if(cint->getBitWidth() <= 64 && cint->getBitWidth() > 1)
1338 svfcd = new SVFConstantInt(getSVFType(cint->getType()), cint->getZExtValue(), cint->getSExtValue());
1339 else
1340 svfcd = new SVFConstantInt(getSVFType(cint->getType()), 0, 0);
1341 }
1342 else if(const ConstantFP* cfp = SVFUtil::dyn_cast<ConstantFP>(cd))
1343 {
1344 double dval = 0;
1345 // TODO: Why only double is considered? What about float?

1631StInfo* LLVMModuleSet::collectArrayInfo(const ArrayType* ty)
1632{
1633 u64_t totalElemNum = ty->getNumElements();
1634 const Type* elemTy = ty->getElementType();
1635 while (const ArrayType* aty = SVFUtil::dyn_cast<ArrayType>(elemTy))
1636 {
1637 totalElemNum *= aty->getNumElements();
1638 elemTy = aty->getElementType();
1639 }
1640
1641 StInfo* stInfo = new StInfo(totalElemNum);
1642 const SVFType* elemSvfType = getSVFType(elemTy);
1643
1645 if (totalElemNum == 0)
1646 {
1647 stInfo->addFldWithType(0, elemSvfType, 0);
1648 stInfo->setNumOfFieldsAndElems(1, 1);
1649 stInfo->getFlattenFieldTypes().push_back(elemSvfType);
1650 stInfo->getFlattenElementTypes().push_back(elemSvfType);
1651 return stInfo;
1652 }
1653
1656 StInfo* elemStInfo = collectTypeInfo(elemTy);
1657 u32_t nfF = elemStInfo->getNumOfFlattenFields();
1658 u32_t nfE = elemStInfo->getNumOfFlattenElements();
1659 for (u32_t j = 0; j < nfF; j++)
1660 {
1661 const SVFType* fieldTy = elemStInfo->getFlattenFieldTypes()[j];
1662 stInfo->getFlattenFieldTypes().push_back(fieldTy);
1663 }
1664
1667 u32_t outArrayElemNum = ty->getNumElements();
1668 for (u32_t i = 0; i < outArrayElemNum; ++i)
1669 {
1670 auto idx = (i * nfE * totalElemNum) / outArrayElemNum;
1671 stInfo->addFldWithType(0, elemSvfType, idx);
1672 }
1673
1674 for (u32_t i = 0; i < totalElemNum; ++i)
1675 {
1676 for (u32_t j = 0; j < nfE; ++j)
1677 {
1678 const SVFType* et = elemStInfo->getFlattenElementTypes()[j];
1679 stInfo->getFlattenElementTypes().push_back(et);
1680 }
1681 }
1682
1683 assert(stInfo->getFlattenElementTypes().size() == nfE * totalElemNum &&
1684 "typeForArray size incorrect!!!");
1685 stInfo->setNumOfFieldsAndElems(nfF, nfE * totalElemNum);
1686
1687 return stInfo;
1688}

1694StInfo* LLVMModuleSet::collectStructInfo(const StructType* structTy,
1695 u32_t& numFields)
1696{
1698 StInfo* stInfo = new StInfo(1);
1699
1700 // Number of fields after flattening the struct
1701 numFields = 0;
1702 // The offset when considering array stride info
1703 u32_t strideOffset = 0;
1704 for (const Type* elemTy : structTy->elements())
1705 {
1706 const SVFType* elemSvfTy = getSVFType(elemTy);
1707 // offset with int_32 (s32_t) is large enough and won't overflow
1708 stInfo->addFldWithType(numFields, elemSvfTy, strideOffset);
1709
1710 if (SVFUtil::isa<StructType, ArrayType>(elemTy))
1711 {
1712 StInfo* subStInfo = collectTypeInfo(elemTy);
1713 u32_t nfF = subStInfo->getNumOfFlattenFields();
1714 u32_t nfE = subStInfo->getNumOfFlattenElements();
1715 // Copy ST's info, whose element 0 is the size of ST itself.
1716 for (u32_t j = 0; j < nfF; ++j)
1717 {
1718 const SVFType* elemTy = subStInfo->getFlattenFieldTypes()[j];
1719 stInfo->getFlattenFieldTypes().push_back(elemTy);
1720 }
1721 numFields += nfF;
1722 strideOffset += nfE;
1723 for (u32_t tpj = 0; tpj < nfE; ++tpj)
1724 {
1725 const SVFType* ty = subStInfo->getFlattenElementTypes()[tpj];
1726 stInfo->getFlattenElementTypes().push_back(ty);
1727 }
1728
1729 }
1730 else
1731 {
1732 // Simple type
1733 numFields += 1;
1734 strideOffset += 1;
1735 stInfo->getFlattenFieldTypes().push_back(elemSvfTy);
1736 stInfo->getFlattenElementTypes().push_back(elemSvfTy);
1737 }
1738 }
1739
1740 assert(stInfo->getFlattenElementTypes().size() == strideOffset &&
1741 "typeForStruct size incorrect!");
1742 stInfo->setNumOfFieldsAndElems(numFields,strideOffset);
1743
1744 return stInfo;
1745}
