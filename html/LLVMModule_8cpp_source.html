46 /*
47  svf.main() is used to model the real entry point of a C++ program, which
48  initializes all global C++ objects and then call main().
49  LLVM may generate two global arrays @llvm.global_ctors and @llvm.global_dtors
50  that contain constructor and destructor functions for global variables. They
51  are not called explicitly, so we have to add them in the svf.main function.
52  The order to call these constructor and destructor functions are also
53  specified in the global arrays.
54  Related part in LLVM language reference:
55  https://llvm.org/docs/LangRef.html#the-llvm-global-ctors-global-variable
56  For example, given a "int main(int argc, char * argv[])", the corresponding
57  svf.main will be generated as follows:
58  define void @svf.main(i32, i8**, i8**) {
59  entry:
60  call void @ctor1()
61  call void @ctor2()
62  %3 = call i32 @main(i32 %0, i8** %1)
63  call void @dtor1()
64  call void @dtor2()
65  ret void
66  }
67 */

175  // Functions need to be retrieved in the order of insertion

197  {

199  }
200  for (const Function* func: candidateDecls)
201  {
202  createSVFFunction(func);

207  {

215  }

237 void LLVMModuleSet::createSVFFunction(const Function* func)
238 {
239  SVFFunction* svfFunc = new SVFFunction(
240  getSVFType(func->getType()),
241  SVFUtil::cast<SVFFunctionType>(
242  getSVFType(func->getFunctionType())),
243  func->isDeclaration(), LLVMUtil::isIntrinsicFun(func),
244  func->hasAddressTaken(), func->isVarArg(), new SVFLoopAndDomInfo);
245  svfModule->addFunctionSet(svfFunc);
246  if (ExtFun2Annotations.find(func) != ExtFun2Annotations.end())
247  svfFunc->setAnnotations(ExtFun2Annotations[func]);
248  addFunctionMap(func, svfFunc);
249 
250  for (const Argument& arg : func->args())
251  {
252  SVFArgument* svfarg = new SVFArgument(
253  getSVFType(arg.getType()), svfFunc, arg.getArgNo(),
254  LLVMUtil::isArgOfUncalledFunction(&arg));
255  // Setting up arg name
256  if (!arg.hasName())
257  svfarg->setName(std::to_string(arg.getArgNo()));
258 
259  svfFunc->addArgument(svfarg);
260  addArgumentMap(&arg, svfarg);
261  }
262 
263  for (const BasicBlock& bb : *func)
264  {
265  SVFBasicBlock* svfBB =
266  new SVFBasicBlock(getSVFType(bb.getType()), svfFunc);
267  svfFunc->addBasicBlock(svfBB);
268  addBasicBlockMap(&bb, svfBB);
269  for (const Instruction& inst : bb)
270  {
271  SVFInstruction* svfInst = nullptr;
272  if (const CallBase* call = SVFUtil::dyn_cast<CallBase>(&inst))
273  {
274  if (cppUtil::isVirtualCallSite(call))
275  svfInst = new SVFVirtualCallInst(
276  getSVFType(call->getType()), svfBB,
277  call->getFunctionType()->isVarArg(),
278  inst.isTerminator());
279  else
280  svfInst = new SVFCallInst(
281  getSVFType(call->getType()), svfBB,
282  call->getFunctionType()->isVarArg(),
283  inst.isTerminator());
284  }
285  else
286  {
287  svfInst =
288  new SVFInstruction(getSVFType(inst.getType()),
289  svfBB, inst.isTerminator(),
290  SVFUtil::isa<ReturnInst>(inst));
291  }
292 
293  addInstructionMap(&inst, svfInst);
294  }
295  }

316 void LLVMModuleSet::initSVFBasicBlock(const Function* func)
317 {
318  SVFFunction *svfFun = getSVFFunction(func);
319  for (Function::const_iterator bit = func->begin(), ebit = func->end(); bit != ebit; ++bit)
320  {
321  const BasicBlock* bb = &*bit;
322  SVFBasicBlock* svfbb = getSVFBasicBlock(bb);
323  for (succ_const_iterator succ_it = succ_begin(bb); succ_it != succ_end(bb); succ_it++)
324  {
325  const SVFBasicBlock* svf_scc_bb = getSVFBasicBlock(*succ_it);
326  svfbb->addSuccBasicBlock(svf_scc_bb);
327  }
328  for (const_pred_iterator pred_it = pred_begin(bb); pred_it != pred_end(bb); pred_it++)
329  {
330  const SVFBasicBlock* svf_pred_bb = getSVFBasicBlock(*pred_it);
331  svfbb->addPredBasicBlock(svf_pred_bb);
332  }
333 
335  if (svfbb->getSuccessors().empty())
336  {
337  if (LLVMUtil::basicBlockHasRetInst(bb))
338  {
339  assert((LLVMUtil::functionDoesNotRet(func) ||
340  SVFUtil::isa<ReturnInst>(bb->back())) &&
341  "last inst must be return inst");
342  svfFun->setExitBlock(svfbb);
343  }
344  }
345 
346  for (BasicBlock::const_iterator iit = bb->begin(), eiit = bb->end(); iit != eiit; ++iit)
347  {
348  const Instruction* inst = &*iit;
349  if(const CallBase* call = SVFUtil::dyn_cast<CallBase>(inst))
350  {
351  SVFInstruction* svfinst = getSVFInstruction(call);
352  SVFCallInst* svfcall = SVFUtil::cast<SVFCallInst>(svfinst);
353  auto called_llvmval = call->getCalledOperand()->stripPointerCasts();
354  if (const Function* called_llvmfunc = SVFUtil::dyn_cast<Function>(called_llvmval))
355  {
356  const Function* llvmfunc_def = LLVMUtil::getDefFunForMultipleModule(called_llvmfunc);
357  SVFFunction* callee = getSVFFunction(llvmfunc_def);
358  svfcall->setCalledOperand(callee);
359  }
360  else
361  {
362  svfcall->setCalledOperand(getSVFValue(called_llvmval));
363  }
364  if(SVFVirtualCallInst* virtualCall = SVFUtil::dyn_cast<SVFVirtualCallInst>(svfcall))
365  {
366  virtualCall->setVtablePtr(getSVFValue(cppUtil::getVCallVtblPtr(call)));
367  virtualCall->setFunIdxInVtable(cppUtil::getVCallIdx(call));
368  virtualCall->setFunNameOfVirtualCall(cppUtil::getFunNameOfVCallSite(call));
369  }
370  for(u32_t i = 0; i < call->arg_size(); i++)
371  {
372  SVFValue* svfval = getSVFValue(call->getArgOperand(i));
373  svfcall->addArgument(svfval);
374  }
375  }
376  LLVMUtil::getNextInsts(inst, getSVFInstruction(inst)->getSuccInstructions());
377  LLVMUtil::getPrevInsts(inst, getSVFInstruction(inst)->getPredInstructions());
378  }
379  }
380  // For no return functions, we set the last block as exit BB
381  // This ensures that each function that has definition must have an exit BB
382  if (svfFun->exitBlock == nullptr && svfFun->hasBasicBlock())
383  {
384  SVFBasicBlock* retBB = const_cast<SVFBasicBlock*>(svfFun->back());
385  assert((LLVMUtil::functionDoesNotRet(func) ||
386  SVFUtil::isa<ReturnInst>(&func->back().back())) &&
387  "last inst must be return inst");
388  svfFun->setExitBlock(retBB);
389  }
390 }

393 void LLVMModuleSet::initDomTree(SVFFunction* svffun, const Function* fun)
394 {
395  if (fun->isDeclaration())
396  return;
397  //process and stored dt & df
398  DominatorTree dt;
399  DominanceFrontier df;
400  dt.recalculate(const_cast<Function&>(*fun));
401  df.analyze(dt);
402  LoopInfo loopInfo = LoopInfo(dt);
403  PostDominatorTree pdt = PostDominatorTree(const_cast<Function&>(*fun));
404  SVFLoopAndDomInfo* ld = svffun->getLoopAndDomInfo();
405 
406  Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>> & dfBBsMap = ld->getDomFrontierMap();
407  for (DominanceFrontierBase::const_iterator dfIter = df.begin(), eDfIter = df.end(); dfIter != eDfIter; dfIter++)
408  {
409  const BasicBlock* keyBB = dfIter->first;
410  const std::set<BasicBlock* >& domSet = dfIter->second;
411  Set<const SVFBasicBlock*>& valueBasicBlocks = dfBBsMap[getSVFBasicBlock(keyBB)];
412  for (const BasicBlock* bbValue:domSet)
413  {
414  valueBasicBlocks.insert(getSVFBasicBlock(bbValue));
415  }
416  }
417  std::vector<const SVFBasicBlock*> reachableBBs;
418  LLVMUtil::getFunReachableBBs(fun, reachableBBs);
419  ld->setReachableBBs(reachableBBs);
420 
421  for (Function::const_iterator bit = fun->begin(), beit = fun->end(); bit!=beit; ++bit)
422  {
423  const BasicBlock &bb = *bit;
424  SVFBasicBlock* svfBB = getSVFBasicBlock(&bb);
425  if (DomTreeNode* dtNode = dt.getNode(&bb))
426  {
427  SVFLoopAndDomInfo::BBSet& bbSet = ld->getDomTreeMap()[svfBB];
428  for (const auto domBB : *dtNode)
429  {
430  const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
431  bbSet.insert(domSVFBB);
432  }
433  }
434 
435  if (DomTreeNode* pdtNode = pdt.getNode(&bb))
436  {
437  u32_t level = pdtNode->getLevel();
438  ld->getBBPDomLevel()[svfBB] = level;
439  BasicBlock* idomBB = pdtNode->getIDom()->getBlock();
440  const SVFBasicBlock* idom = idomBB == NULL ? NULL: getSVFBasicBlock(idomBB);
441  ld->getBB2PIdom()[svfBB] = idom;
442 
443  SVFLoopAndDomInfo::BBSet& bbSet = ld->getPostDomTreeMap()[svfBB];
444  for (const auto domBB : *pdtNode)
445  {
446  const auto* domSVFBB = getSVFBasicBlock(domBB->getBlock());
447  bbSet.insert(domSVFBB);
448  }
449  }
450 
451  if (const Loop* loop = loopInfo.getLoopFor(&bb))
452  {
453  for (const BasicBlock* loopBlock : loop->getBlocks())
454  {
455  const SVFBasicBlock* loopbb = getSVFBasicBlock(loopBlock);
456  ld->addToBB2LoopMap(svfBB, loopbb);
457  }
458  }
459  }
460 }

511 void LLVMModuleSet::loadModules(const std::vector<std::string> &moduleNameVec)
512 {
513 
514  // We read SVFIR from LLVM IR
515  if(Options::Graphtxt().empty())
516  {
517  if(moduleNameVec.empty())
518  {
519  SVFUtil::outs() << "no LLVM bc file is found!\n";
520  exit(0);
521  }
522  //assert(!moduleNameVec.empty() && "no LLVM bc file is found!");
523  }
524  // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
525  else
526  SVFModule::setPagFromTXT(Options::Graphtxt());
527 
528  //
529  // LLVMContext objects separate global LLVM settings (from which e.g. types are
530  // derived); multiple LLVMContext objects can coexist and each context can "own"
531  // multiple modules (modules can only have one context). Mixing contexts can lead
532  // to unintended inequalities, such as the following:
533  //
534  // ------------------------------------------------------------------
535  // LLVMContext ctxa,ctxb;
536  // IntegerType * t1 = IntegerType::get(ctxa,32);
537  // IntegerType * t2 = IntegerType::get(ctxa,32);
538  // assert(t1 == t2);
539  // IntegerType * t3 = IntegerType::get(ctxb,32);
540  // IntegerType * t4 = IntegerType::get(ctxb,32);
541  // assert(t3 == t4);
542  // assert(t1 != t3);
543  // ------------------------------------------------------------------
544  //
545  // When loading bytecode files, SVF will use the same LLVMContext object for all
546  // modules (i.e. the context owns all loaded modules). This applies to ExtAPI as
547  // well, which *must* be loaded using the same LLVMContext object. Hence, when
548  // loading modules from bitcode files, a new LLVMContext is created (using a
549  // `std::unique_ptr<LLVMContext>` type to ensure automatic garbage collection).
550  //
551  // This garbage collection should be avoided when building an SVF module from an LLVM
552  // module instance; see the comment(s) in `buildSVFModule` and `loadExtAPIModules()`
553 
554  owned_ctx = std::make_unique<LLVMContext>();
555  for (const std::string& moduleName : moduleNameVec)
556  {
557  if (!LLVMUtil::isIRFile(moduleName))
558  {
559  SVFUtil::errs() << "not an IR file: " << moduleName << std::endl;
560  abort();
561  }
562 
563  SMDiagnostic Err;
564  std::unique_ptr<Module> mod = parseIRFile(moduleName, Err, *owned_ctx);
565  if (mod == nullptr)
566  {
567  SVFUtil::errs() << "load module: " << moduleName << "failed!!\n\n";
568  Err.print("SVFModuleLoader", llvm::errs());
569  abort();
570  }
571  modules.emplace_back(*mod);
572  owned_modules.emplace_back(std::move(mod));
573  }
574 }

576 void LLVMModuleSet::loadExtAPIModules()
577 {
578  // This function loads the ExtAPI bitcode file as an LLVM module. Note that it is important that
579  // the same LLVMContext object is used to load this bitcode file as is used by the other modules
580  // being analysed.
581  // When the modules are loaded from bitcode files (i.e. passing filenames to files containing
582  // LLVM IR to `buildSVFModule({file1.bc, file2.bc, ...})) the context is created while loading
583  // the modules in `loadModules()`, which populates this->modules and this->owned_modules.
584  // If, however, an LLVM Module object is passed to `buildSVFModule` (e.g. from an LLVM pass),
585  // the context should be retrieved from the module itself (note that the garbage collection from
586  // `std::unique_ptr<LLVMContext> LLVMModuleSet::owned_ctx` should be avoided in this case). This
587  // function populates only this->modules.
588  // In both cases, fetching the context from the main LLVM module (through `getContext`) works
589  assert(!empty() && "LLVMModuleSet contains no modules; cannot load ExtAPI module without LLVMContext!");
590 
591  // Load external API module (extapi.bc)
592  if (!ExtAPI::getExtAPI()->getExtBcPath().empty())
593  {
594  std::string extModuleName = ExtAPI::getExtAPI()->getExtBcPath();
595  if (!LLVMUtil::isIRFile(extModuleName))
596  {
597  SVFUtil::errs() << "not an external IR file: " << extModuleName << std::endl;
598  abort();
599  }
600  SMDiagnostic Err;
601  std::unique_ptr<Module> mod = parseIRFile(extModuleName, Err, getContext());
602  if (mod == nullptr)
603  {
604  SVFUtil::errs() << "load external module: " << extModuleName << "failed!!\n\n";
605  Err.print("SVFModuleLoader", llvm::errs());
606  abort();
607  }
608  // The module of extapi.bc needs to be inserted before applications modules, like std::vector<std::reference_wrapper<Module>> modules{extapi_module, app_module}.
609  // Otherwise, when overwriting the app function with SVF extern function, the corresponding SVFFunction of the extern function will not be found.
610  modules.insert(modules.begin(), *mod);
611  owned_modules.insert(owned_modules.begin(),std::move(mod));
612  }
613 }

615 std::vector<const Function* > LLVMModuleSet::getLLVMGlobalFunctions(const GlobalVariable *global)
616 {
617  // This function is used to extract constructor and destructor functions
618  // sorted by their priority from @llvm.global_ctors or @llvm.global_dtors.
619  // For example, given following @llvm.global_ctors, the returning sorted
620  // function list should be [ctor3, ctor1, ctor2].
621  // ------------------------------------------------------------------
622  // ; Each struct in the array is {priority, function, associated data}
623  //
624  // @llvm.global_ctors = appending global [2 x { i32, void ()*, i8* }]
625  // [{ i32, void ()*, i8* } { i32 1234, void ()* @ctor1.cpp, i8* null },
626  // { i32, void ()*, i8* } { i32 2345, void ()* @ctor2.cpp, i8* null },
627  // { i32, void ()*, i8* } { i32 345, void ()* @ctor3.cpp, i8* null }]
628  // ------------------------------------------------------------------
629  // TODO: According to LLVM language reference, if the third field is
630  // non-null, and points to a global variable or function, the initializer
631  // function will only run if the associated data from the current module is
632  // not discarded. However the associated data is currently ignored.
633 
634 
635  // This class is used for the priority queue that sorts the functions by
636  // their priority. Each object of this class stands for an item in the
637  // function array.
638  class LLVMGlobalFunction
639  {
640  public:
641  u32_t priority;
642  const Function* func;
643  LLVMGlobalFunction() {};
644  LLVMGlobalFunction(u32_t _priority, const Function* _func)
645  : priority(_priority), func(_func) {};
646  bool operator>(const LLVMGlobalFunction &other) const
647  {
648  if (priority != other.priority)
649  {
650  return priority > other.priority;
651  }
652  else
653  {
654  return func > other.func;
655  }
656  }
657  };
658 
659  std::priority_queue<LLVMGlobalFunction, std::vector<LLVMGlobalFunction>,
660  greater<LLVMGlobalFunction>>
661  queue;
662  std::vector<const Function* > result;
663 
664  // The @llvm.global_ctors/dtors global variable is an array of struct. Each
665  // struct has three fields: {i32 priority, void ()* @ctor/dtor, i8* @data}.
666  // First get the array here.
667  if(const ConstantArray *globalFuncArray =
668  SVFUtil::dyn_cast<ConstantArray>(global->getInitializer()))
669  {
670  // Get each struct in the array.
671  for (unsigned int i = 0; i < globalFuncArray->getNumOperands(); ++i)
672  {
673  if (
674  const ConstantStruct *globalFuncItem =
675  SVFUtil::dyn_cast<ConstantStruct>(
676  globalFuncArray->getOperand(i)))
677  {
678 
679  // Extract priority and function from the struct
680  const ConstantInt* priority = SVFUtil::dyn_cast<ConstantInt>(
681  globalFuncItem->getOperand(0));
682  const Function* func = SVFUtil::dyn_cast<Function>(
683  globalFuncItem->getOperand(1));
684 
685  if (priority && func)
686  {
687  queue.push(LLVMGlobalFunction(priority
688  ->getZExtValue(),
689  func));
690  }
691  }
692  }
693  }
694 
695  // Generate a sorted vector of functions from the priority queue.
696  while (!queue.empty())
697  {
698  result.push_back(queue.top().func);
699  queue.pop();
700  }
701  return result;
702 }

849 /*
850  For a more detailed explanation of the Function declaration and definition mapping relationships and how External APIs are handled,
851  please refer to the SVF Wiki: https://github.com/SVF-tools/SVF/wiki/Handling-External-APIs-with-extapi.c
852 
853  Table 1
854  | ------- | ----------------- | --------------- | ----------------- | ----------- |
855  | | AppDef | AppDecl | ExtDef | ExtDecl |
856  | ------- | ----------------- | --------------- | ----------------- | ----------- |
857  | AppDef | X | FunDefToDeclsMap| FunDeclToDefMap | X |
858  | ------- | ----------------- | --------------- | ----------------- | ----------- |
859  | AppDecl | FunDeclToDefMap | X | FunDeclToDefMap | X |
860  | ------- | ----------------- | --------------- | ----------------- | ----------- |
861  | ExtDef | FunDefToDeclsMap | FunDefToDeclsMap| X | X |
862  | ------- | ----------------- | --------------- | ----------------- | ----------- |
863  | ExtDecl | FunDeclToDefMap | X | X | ExtFuncsVec |
864  | ------- | ----------------- | --------------- | ----------------- | ----------- |
865 
866  When a user wants to use functions in extapi.c to overwrite the functions defined in the app code, two relationships, "AppDef -> ExtDef" and "ExtDef -> AppDef," are used.
867  Use Ext function definition to override the App function definition (Ext function with "__attribute__((annotate("OVERWRITE")))" in extapi.c).
868  The app function definition will be changed to an app function declaration.
869  Then, put the app function declaration and its corresponding Ext function definition into FunDeclToDefMap/FunDefToDeclsMap.
870  ------------------------------------------------------
871  AppDef -> ExtDef (overwrite):
872  For example,
873  App function:
874  char* foo(char *a, char *b){return a;}
875  Ext function:
876  __attribute__((annotate("OVERWRITE")))
877  char* foo(char *a, char *b){return b;}
878 
879  When SVF handles the foo function in the App module,
880  the definition of
881  foo: char* foo(char *a, char *b){return a;}
882  will be changed to a declaration
883  foo: char* foo(char *a, char *b);
884  Then,
885  foo: char* foo(char *a, char *b);
886  and
887  __attribute__((annotate("OVERWRITE")))
888  char* foo(char *a, char *b){return b;}
889  will be put into FunDeclToDefMap
890  ------------------------------------------------------
891  ExtDef -> AppDef (overwrite):
892  __attribute__((annotate("OVERWRITE")))
893  char* foo(char *a, char *b){return b;}
894  and
895  foo: char* foo(char *a, char *b);
896  are put into FunDefToDeclsMap;
897  ------------------------------------------------------
898  In principle, all functions in extapi.c have bodies (definitions), but some functions (those starting with "sse_")
899  have only function declarations without definitions. ExtFuncsVec is used to record function declarations starting with "sse_" that are used.
900 
901  ExtDecl -> ExtDecl:
902  For example,
903  App function:
904  foo(){call memcpy();}
905  Ext function:
906  declare sse_check_overflow();
907  memcpy(){sse_check_overflow();}
908 
909  sse_check_overflow() used in the Ext function but not in the App function.
910  sse_check_overflow should be kept in ExtFuncsVec.
911 */

950  [&](const std::string& annotation)

953  std::string::npos;

958  }
959  }
960  }
961  }

988  for (const Function* fdef : funDefs)

999  for (const Function* fdecl : funDecls)

1078  // (1) The types are exactly the same,
1079  // (2) Both are pointer types, and at least one of them is a void*.

1096  Type* argType2 = argIter2->getType();
1097

1102  break;
1103  }
1104  argIter1++;
1105  argIter2++;
1106  }

1112  FunctionType* funType = fun->getFunctionType();
1113  std::string funName = fun->getName().str();

1117  fun->eraseFromParent();

1119  // AppDef -> ExtDef in Table 1, AppDef has been changed to AppDecl
1120  FunDeclToDefMap[declaration] = owfunc;
1121  // ExtDef -> AppDef in Table 1
1122  std::vector<const Function*>& decls = FunDefToDeclsMap[owfunc];
1123  decls.push_back(declaration);
1124  // Keep all called functions in owfunc
1125  // ExtDecl -> ExtDecl in Table 1
1126  std::vector<const Function*> calledFunctions = LLVMUtil::getCalledFunctions(owfunc);
1127  ExtFuncsVec.insert(ExtFuncsVec.end(), calledFunctions.begin(), calledFunctions.end());
1128  break;
1129  }
1130  }
1131  }
