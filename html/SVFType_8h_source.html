69 template <class S, class T> struct Hash<std::pair<S, T>>
70 {
71  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
72  static size_t szudzik(size_t a, size_t b)
73  {
74  return a > b ? b * b + a : a * a + a + b;
75  }
76 
77  size_t operator()(const std::pair<S, T>& t) const
78  {
79  Hash<decltype(t.first)> first;
80  Hash<decltype(t.second)> second;
81  return szudzik(first(t.first), second(t.second));
82  }
83 };
