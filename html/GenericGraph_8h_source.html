108 // and duplicated elements in the set are not inserted (binary tree comparison)
110 typedef struct equalGEdge
111 {
112 bool operator()(const GenericEdge<NodeType>* lhs, const GenericEdge<NodeType>* rhs) const
113 {
114 if (lhs->edgeFlag != rhs->edgeFlag)
115 return lhs->edgeFlag < rhs->edgeFlag;
116 else if (lhs->getSrcID() != rhs->getSrcID())
117 return lhs->getSrcID() < rhs->getSrcID();
118 else
119 return lhs->getDstID() < rhs->getDstID();
120 }
121 } equalGEdge;
122
123 virtual inline bool operator==(const GenericEdge<NodeType>* rhs) const
124 {
125 return (rhs->edgeFlag == this->edgeFlag &&
126 rhs->getSrcID() == this->getSrcID() &&
127 rhs->getDstID() == this->getDstID());
128 }

131protected:
132 static constexpr unsigned char EdgeKindMaskBits = 8;
133 static constexpr u64_t EdgeKindMask = (~0ULL) >> (64 - EdgeKindMaskBits);

750// mapped_iter - This is a simple iterator adapter that causes a function to
751// be applied whenever operator* is invoked on the iterator.
752
753template <typename ItTy, typename FuncTy,
754 typename FuncReturnTy =
755 decltype(std::declval<FuncTy>()(*std::declval<ItTy>()))>
756class mapped_iter
757 : public iter_adaptor_base<
758 mapped_iter<ItTy, FuncTy>, ItTy,
759 typename std::iterator_traits<ItTy>::iterator_category,
760 typename std::remove_reference<FuncReturnTy>::type>
761{
762public:
763 mapped_iter(ItTy U, FuncTy F)
764 : mapped_iter::iter_adaptor_base(std::move(U)), F(std::move(F)) {}
765
766 ItTy getCurrent()
767 {
768 return this->I;
769 }
770
771 FuncReturnTy operator*() const
772 {
773 return F(*this->I);
774 }
775
776private:
777 FuncTy F;
778};

784{
