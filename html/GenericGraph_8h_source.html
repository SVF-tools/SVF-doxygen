108  // and duplicated elements in the set are not inserted (binary tree comparison)
110  typedef struct equalGEdge
111  {
112  bool operator()(const GenericEdge<NodeType>* lhs, const GenericEdge<NodeType>* rhs) const
113  {
114  if (lhs->edgeFlag != rhs->edgeFlag)
115  return lhs->edgeFlag < rhs->edgeFlag;
116  else if (lhs->getSrcID() != rhs->getSrcID())
117  return lhs->getSrcID() < rhs->getSrcID();
118  else
119  return lhs->getDstID() < rhs->getDstID();
120  }
121  } equalGEdge;
122 
123  virtual inline bool operator==(const GenericEdge<NodeType>* rhs) const
124  {
125  return (rhs->edgeFlag == this->edgeFlag &&
126  rhs->getSrcID() == this->getSrcID() &&
127  rhs->getDstID() == this->getDstID());
128  }

133  static constexpr u64_t EdgeKindMask = (~0ULL) >> (64 - EdgeKindMaskBits);

468 // mapped_iter - This is a simple iterator adapter that causes a function to
469 // be applied whenever operator* is invoked on the iterator.
470 
471 template <typename ItTy, typename FuncTy,
472  typename FuncReturnTy =
473  decltype(std::declval<FuncTy>()(*std::declval<ItTy>()))>
474 class mapped_iter
475  : public iter_adaptor_base<
476  mapped_iter<ItTy, FuncTy>, ItTy,
477  typename std::iterator_traits<ItTy>::iterator_category,
478  typename std::remove_reference<FuncReturnTy>::type>
479 {
480 public:
481  mapped_iter(ItTy U, FuncTy F)
482  : mapped_iter::iter_adaptor_base(std::move(U)), F(std::move(F)) {}
483 
484  ItTy getCurrent()
485  {
486  return this->I;
487  }
488 
489  FuncReturnTy operator*() const
490  {
491  return F(*this->I);
492  }
493 
494 private:
495  FuncTy F;
496 };
