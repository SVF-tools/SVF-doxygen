153 void VersionedFlowSensitiveStat::versionStat(void)
154 {
155  // TODO! Need to merge yield/consume.
156  _NumSingleVersion = 0;
157  _MaxVersions = 0;
158 
159  u32_t totalVersionPtsSize = 0;
160  for (const VersionedFlowSensitive::LocVersionMap *lvm :
161  {
162  &vfspta->consume, &vfspta->yield
163  })
164  {
165  for (const VersionedFlowSensitive::ObjToVersionMap &lov : *lvm)
166  {
167  for (const VersionedFlowSensitive::ObjToVersionMap::value_type &ov : lov)
168  {
169  const NodeID o = ov.first;
170  const Version v = ov.second;
171 
172  ++_NumVersions;
173 
174  // If the version was just over-approximate and never accessed, ignore.
175  // TODO: with vPtD changed there is no interface to check if the PTS
176  // exists; an emptiness check is *not* an existence check.
177  if (vfspta->vPtD->getPts(vfspta->atKey(o, v)).empty()) continue;
178 
179  const PointsTo &ovPts = vfspta->vPtD->getPts(vfspta->atKey(o, v));
180  if (!ovPts.empty()) ++_NumNonEmptyVersions;
181  else ++_NumEmptyVersions;
182 
183  _TotalPtsSize += ovPts.count();
184  totalVersionPtsSize += ovPts.count();
185  if (ovPts.count() > _MaxVersionPtsSize) _MaxVersionPtsSize = ovPts.count();
186  }
187  }
188  }
189 
190  _NumUsedVersions = _NumNonEmptyVersions + _NumEmptyVersions;
191 
192  if (_NumNonEmptyVersions != 0) _AvgVersionPtsSize = (double)totalVersionPtsSize / (double)_NumUsedVersions;
193 
194  _TotalPtsSize += totalVersionPtsSize;
195 }

197 void VersionedFlowSensitiveStat::ptsSizeStat()
198 {
199  u32_t totalValidTopLvlPointers = 0;
200  u32_t totalTopLvlPtsSize = 0;
201  for (SVFIR::iterator it = vfspta->getPAG()->begin(); it != vfspta->getPAG()->end(); ++it)
202  {
203  if (!vfspta->getPAG()->isValidTopLevelPtr(it->second)) continue;
204 
205  NodeID p = it->first;
206 
207  totalValidTopLvlPointers++;
208 
209  u32_t size = vfspta->getPts(p).count();
210  totalTopLvlPtsSize += size;
211  if (size > _MaxTopLvlPtsSize) _MaxTopLvlPtsSize = size;
212  }
213 
214  if (totalValidTopLvlPointers != 0) _AvgTopLvlPtsSize = (double)totalTopLvlPtsSize / (double)totalValidTopLvlPointers;
215 
216  _TotalPtsSize += totalTopLvlPtsSize;
217 
218  if (_NumNonEmptyVersions + totalValidTopLvlPointers != 0)
219  {
220  _AvgPtsSize = (double)_TotalPtsSize / (double)(_NumNonEmptyVersions + totalValidTopLvlPointers);
221  }
222 
223  _MaxPtsSize = _MaxVersionPtsSize > _MaxTopLvlPtsSize ? _MaxVersionPtsSize : _MaxTopLvlPtsSize;
224 }
