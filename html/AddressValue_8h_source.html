141 bool join_with(const AddressValue &other)
142 {
143 bool changed = false;
144 for (const auto &addr: other)
145 {
146 if (!_addrs.count(addr))
147 {
148 if (insert(addr).second)
149 changed = true;
150 }
151 }
152 return changed;
153 }
154
156 bool meet_with(const AddressValue &other)
157 {
158 AddrSet s;
159 for (const auto &id: other._addrs)
160 {
161 if (_addrs.find(id) != _addrs.end())
162 {
163 s.insert(id);
164 }
165 }
166 bool changed = (_addrs != s);
167 _addrs = std::move(s);
168 return changed;
169 }
