491  STATIC_OBJ = 0x4, // object is a static variable allocated before main
492  STACK_OBJ = 0x8, // object is a stack variable

497  CONST_ARRAY_OBJ = 0x100, // constant array
498  CONST_GLOBAL_OBJ = 0x200, // global constant object
499  CONST_DATA = 0x400, // constant object str e.g. 5, 10, 1.0
500  HASPTR_OBJ = 0x800 // the object stores a pointer address
