44void MTAStat::performThreadCallGraphStat(ThreadCallGraph* tcg)
45{
46 u32_t numOfForkEdge = 0;
47 u32_t numOfJoinEdge = 0;
48 u32_t numOfIndForksite = 0;
49 u32_t numOfIndForkEdge = 0;
50 for (ThreadCallGraph::CallSiteSet::const_iterator it = tcg->forksitesBegin(), eit = tcg->forksitesEnd(); it != eit; ++it)
51 {
52 bool indirectfork = false;
53 const ValVar* pValVar = tcg->getThreadAPI()->getForkedFun(*it);
54 if(!SVFUtil::isa<FunValVar>(pValVar))
55 {
56 numOfIndForksite++;
57 indirectfork = true;
58 }
59 for (ThreadCallGraph::ForkEdgeSet::const_iterator cgIt = tcg->getForkEdgeBegin(*it), ecgIt =
60 tcg->getForkEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
61 {
62 numOfForkEdge++;
63 if(indirectfork)
64 numOfIndForkEdge++;
65 }
66 }
67
68 for (ThreadCallGraph::CallSiteSet::const_iterator it = tcg->joinsitesBegin(), eit = tcg->joinsitesEnd(); it != eit; ++it)
69 {
70 for (ThreadCallGraph::JoinEdgeSet::const_iterator cgIt = tcg->getJoinEdgeBegin(*it), ecgIt =
71 tcg->getJoinEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
72 {
73 numOfJoinEdge++;
74 }
75 }
76
77
78 PTNumStatMap.clear();
79 PTNumStatMap["NumOfForkSite"] = tcg->getNumOfForksite();
80 PTNumStatMap["NumOfForkEdge"] = numOfForkEdge;
81 PTNumStatMap["NumOfJoinEdge"] = numOfJoinEdge;
82 PTNumStatMap["NumOfJoinSite"] = tcg->getNumOfJoinsite();
83 PTNumStatMap["NumOfIndForkSite"] = numOfIndForksite;
84 PTNumStatMap["NumOfIndForkEdge"] = numOfIndForkEdge;
85 PTNumStatMap["NumOfIndCallEdge"] = tcg->getNumOfResolvedIndCallEdge();
86
87 SVFUtil::outs() << "\n****Thread Call Graph Statistics****\n";
88 PTAStat::printStat();
89}

112void MTAStat::performMHPPairStat(MHP* mhp, LockAnalysis* lsa)
113{
114
115 SVFIR* pag = SVFIR::getPAG();
116 if(Options::AllPairMHP())
117 {
118 Set<const ICFGNode*> instSet1;
119 Set<const ICFGNode*> instSet2;
120
121 for (const auto& item : *PAG::getPAG()->getCallGraph())
122 {
123 const FunObjVar* fun = item.second->getFunction();
124 if(SVFUtil::isExtCall(fun))
125 continue;
126 if(!mhp->isConnectedfromMain(fun))
127 continue;
128 for (FunObjVar::const_bb_iterator bit = fun->begin(), ebit = fun->end(); bit != ebit; ++bit)
129 {
130 const SVFBasicBlock* bb = bit->second;
131 for (const auto& icfgNode : bb->getICFGNodeList())
132 {
133 for(const SVFStmt* stmt : pag->getSVFStmtList(icfgNode))
134 {
135 if(SVFUtil::isa<LoadStmt>(stmt))
136 {
137 instSet1.insert(stmt->getICFGNode());
138 }
139 else if(SVFUtil::isa<StoreStmt>(stmt))
140 {
141 instSet1.insert(stmt->getICFGNode());
142 instSet2.insert(stmt->getICFGNode());
143 }
144 }
145 }
146 }
147 }
148
149
150 for(Set<const ICFGNode*>::const_iterator it1 = instSet1.begin(), eit1 = instSet1.end(); it1!=eit1; ++it1)
151 {
152 for(Set<const ICFGNode*>::const_iterator it2 = instSet2.begin(), eit2 = instSet2.end(); it2!=eit2; ++it2)
153 {
154 mhp->mayHappenInParallel(*it1,*it2);
155 }
156 }
157 }
158
159
160 generalNumMap.clear();
161 PTNumStatMap.clear();
162 timeStatMap.clear();
163 PTNumStatMap["TotalMHPQueries"] = mhp->numOfTotalQueries;
164 PTNumStatMap["NumOfMHPPairs"] = mhp->numOfMHPQueries;
165 PTNumStatMap["TotalLockQueries"] = lsa->numOfTotalQueries;
166 PTNumStatMap["NumOfLockedPairs"] = lsa->numOfLockedQueries;
167 PTNumStatMap["NumOfCxtLocks"] = lsa->getNumOfCxtLocks();
168 timeStatMap["InterlevAnaTime"] = mhp->interleavingTime;
169 timeStatMap["LockAnaTime"] = lsa->lockTime;
170 timeStatMap["InterlevQueryTime"] = mhp->interleavingQueriesTime;
171 timeStatMap["LockQueryTime"] = lsa->lockQueriesTime;
172 timeStatMap["MHPAnalysisTime"] = MHPTime;
173
174 SVFUtil::outs() << "\n****MHP Stmt Pairs Statistics****\n";
175 PTAStat::printStat();
176}
