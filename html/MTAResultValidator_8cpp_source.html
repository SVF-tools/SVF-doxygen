211 bool MTAResultValidator::collectCallsiteTargets()
212 {
213  for (Module& M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
214  {
215  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
216  {
217  for (Function::const_iterator bi = (*F).begin(), ebi = (*F).end(); bi != ebi; ++bi)
218  {
219  const BasicBlock* bb = &*bi;
220  if (!bb->getName().str().compare(0, 2, "cs"))
221  {
222  NodeID csnum = atoi(bb->getName().str().substr(2).c_str());
223  const Instruction* inst = &bb->front();
224  while (1)
225  {
226  if (SVFUtil::isa<CallInst>(inst))
227  {
228  break;
229  }
230  inst = inst->getNextNode();
231  assert(inst && "Wrong cs label, cannot find callsite");
232  }
233  const SVFInstruction* svfInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(inst);
234  csnumToInstMap[csnum] = svfInst;

237  }
238  }

368 bool MTAResultValidator::validateCxtThread()
369 {
370 
371  bool res = true;
372  TCT* tct = mhp->getTCT();
373  if (tct->getTCTNodeNum() != vthdToCxt.size())
374  {
375  res = false;
376  if (Options::PrintValidRes())
377  {
378  outs() << errMsg("\nValidate CxtThread: The number of CxtThread is different from given result!!!\n");
379  outs() << "Given threads:\t" << vthdToCxt.size() << "\nAnalysis result:\t" << tct->getTCTNodeNum() << "\n";
380  assert(false && "test case failed!");
381  }
382  }
383 
384  Set<int> visitedvthd;
385 
386  for (NodeID i = 0; i < tct->getTCTNodeNum(); i++)
387  {
388  const CxtThread rthd = tct->getTCTNode(i)->getCxtThread();
389  bool matched = false;
390  for (Map<NodeID, CallStrCxt>::iterator j = vthdToCxt.begin(), ej = vthdToCxt.end(); j != ej; j++)
391  {
392  NodeID vthdid = (*j).first;
393  if (matchCxt(rthd.getContext(), vthdToCxt[vthdid]))
394  {
395  if (visitedvthd.find(vthdid) != visitedvthd.end())
396  {
397  res = false;
398  if (Options::PrintValidRes())
399  {
400  outs() << "\nValidate CxtThread: Repeat real CxtThread !!!\n";
401  rthd.dump();
402  tct->getTCTNode(vthdTorthd[vthdid])->getCxtThread().dump();
403  }
404  }
405  vthdTorthd[vthdid] = i;
406  rthdTovthd[i] = vthdid;
407  visitedvthd.insert(vthdid);
408  matched = true;
409  break;
410  }
411  }
412  if (!matched)
413  {
414  res = false;
415  if (Options::PrintValidRes())
416  {
417  SVFUtil::errs() << errMsg("\nValidate CxtThread: Cannot match real CxtThread !!!\n");
418  rthd.dump();
419  assert(false && "test case failed!");
420  }
421  }
422  }
423  if (visitedvthd.size() != vthdToCxt.size())
424  {
425  res = false;
426  if (Options::PrintValidRes())
427  {
428  SVFUtil::errs() << errMsg("\nValidate CxtThread: Some given CxtThreads cannot be found !!!\n");
429  assert(false && "test case failed!");
430  for (Map<NodeID, CallStrCxt>::iterator j = vthdToCxt.begin(), ej = vthdToCxt.end(); j != ej; j++)
431  {
432  NodeID vthdid = (*j).first;
433  if (visitedvthd.find(vthdid) == visitedvthd.end())
434  {
435  dumpCxt(vthdToCxt[vthdid]);
436  }
437  }
438  }
439  }
440  return res;
441 }

511  for (MHP::CxtThreadStmtSet::iterator thdlevi = (*seti).second.begin(), ethdlevi = (*seti).second.end(); thdlevi != ethdlevi;
512  ++thdlevi)
513  {
514  outs() << "TID " << rthdTovthd[(*thdlevi).getTid()] << ": ";
515  dumpCxt((*thdlevi).getContext());
516  }
517  outs() << "\n----Analysis threads:\n";
518  for (MHP::CxtThreadStmtSet::const_iterator it = tsSet.begin(), eit = tsSet.end(); it != eit; ++it)
519  {
520  outs() << "TID " << rthdTovthd[(*it).getTid()] << ": ";
521  dumpCxt((*it).getContext());
522  }
523  outs() << "\n";
524  }
525  res = MTAResultValidator::INTERLEV_UNSOUND;
526  }
527 
528  for (MHP::CxtThreadStmtSet::const_iterator it = tsSet.begin(), eit = tsSet.end(); it != eit; ++it)
529  {
530  const CxtThreadStmt& ts = *it;
531  bool matched = false;
532  for (MHP::CxtThreadStmtSet::iterator it2 = (*seti).second.begin(), eit2 = (*seti).second.end(); it2 != eit2; ++it2)
533  {
534  const CxtThreadStmt& ts2 = *it2;
535 
536  if (ts2.getTid() == ts.getTid() && matchCxt(ts2.getContext(), ts.getContext()))
537  {
538  matched = true;
539  NodeBS lev = mhp->getInterleavingThreads(ts);
540  NodeBS lev2 = threadStmtToInterLeaving[ts2];
541  if (lev != lev2)
542  {
543  if (Options::PrintValidRes())
544  {
545  outs() << errMsg("\nValidate Interleaving: Wrong at: ") << inst->getSourceLoc() << "\n";
546  outs() << "Reason: thread interleaving on stmt is wrong\n";
547  dumpCxt(ts.getContext());
548  outs() << "Given result: \tTID " << rthdTovthd[ts.getTid()];
549  dumpInterlev(lev2);
550  outs() << "Analysis result: \tTID " << rthdTovthd[ts.getTid()];
551  dumpInterlev(lev);
552  }
553 
554  if (MTAResultValidator::INTERLEV_IMPRECISE > res)
555  res = MTAResultValidator::INTERLEV_IMPRECISE;
556 
557  if (lev.count() >= lev2.count())
558  {
559  bool findeveryelement = true;
560  for (NodeBS::iterator it = lev2.begin(), eit = lev2.end(); it != eit; it++)
561  {
562  if (!lev.test(*it))
563  {
564  findeveryelement = false;
565  break;
566  }
567  }
568  if (!findeveryelement)
569  res = MTAResultValidator::INTERLEV_UNSOUND;
570  }
571  else

573  }
574  }
575  }

579  if (Options::PrintValidRes())
580  {
581  outs() << errMsg("\nValidate Interleaving: Wrong at:") << inst->getSourceLoc() << "\n";
582  outs() << "Reason: analysis thread cxt is not matched by given thread cxt\n";

584  NodeBS lev = mhp->getInterleavingThreads(ts);
585 
586  outs() << "Analysis result: \tTID " << rthdTovthd[ts.getTid()];
587  dumpInterlev(lev);
588  }
589  res = MTAResultValidator::INTERLEV_UNSOUND;
590  }
591  }
592  }
