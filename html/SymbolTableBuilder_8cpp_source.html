75 void SymbolTableBuilder::buildMemModel(SVFModule* svfModule)
76 {
77  SVFUtil::increaseStackSize();
78 
79  symInfo->setModule(svfModule);
80 
81  // Pointer #0 always represents the null pointer.
82  assert(symInfo->totalSymNum++ == SymbolTableInfo::NullPtr && "Something changed!");
83 
84  // Pointer #1 always represents the pointer points-to black hole.
85  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlkPtr && "Something changed!");
86 
87  // Object #2 is black hole the object that may point to any object
88  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlackHole && "Something changed!");
89  createBlkObj(SymbolTableInfo::BlackHole);
90 
91  // Object #3 always represents the unique constant of a program (merging all constants if Options::ModelConsts is disabled)
92  assert(symInfo->totalSymNum++ == SymbolTableInfo::ConstantObj && "Something changed!");
93  createConstantObj(SymbolTableInfo::ConstantObj);
94 
95  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
96  {
97  // Add symbols for all the globals .
98  for (const GlobalVariable& gv : M.globals())
99  {
100  collectSym(&gv);
101  }
102 
103  // Add symbols for all the global aliases
104  for (const GlobalAlias& ga : M.aliases())
105  {
106  collectSym(&ga);
107  collectSym(ga.getAliasee());
108  }
109 
110  // Add symbols for all of the functions and the instructions in them.
111  for (const Function& fun : M.functions())
112  {
113  collectSym(&fun);
114  collectRet(&fun);
115  if (fun.getFunctionType()->isVarArg())
116  collectVararg(&fun);
117 
118  // Add symbols for all formal parameters.
119  for (const Argument& arg : fun.args())
120  {
121  collectSym(&arg);
122  }
123 
124  // collect and create symbols inside the function body
125  for (const Instruction& inst : instructions(fun))
126  {
127  collectSym(&inst);
128 
129  // initialization for some special instructions
130  //{@
131  if (const StoreInst* st = SVFUtil::dyn_cast<StoreInst>(&inst))
132  {
133  collectSym(st->getPointerOperand());
134  collectSym(st->getValueOperand());
135  }
136  else if (const LoadInst* ld =
137  SVFUtil::dyn_cast<LoadInst>(&inst))
138  {
139  collectSym(ld->getPointerOperand());
140  }
141  else if (const AllocaInst* alloc =
142  SVFUtil::dyn_cast<AllocaInst>(&inst))
143  {
144  collectSym(alloc->getArraySize());
145  }
146  else if (const PHINode* phi = SVFUtil::dyn_cast<PHINode>(&inst))
147  {
148  for (u32_t i = 0; i < phi->getNumIncomingValues(); ++i)
149  {
150  collectSym(phi->getIncomingValue(i));
151  }
152  }
153  else if (const GetElementPtrInst* gep =
154  SVFUtil::dyn_cast<GetElementPtrInst>(&inst))
155  {
156  collectSym(gep->getPointerOperand());
157  for (u32_t i = 0; i < gep->getNumOperands(); ++i)
158  {
159  collectSym(gep->getOperand(i));
160  }
161  }
162  else if (const SelectInst* sel =
163  SVFUtil::dyn_cast<SelectInst>(&inst))
164  {
165  collectSym(sel->getTrueValue());
166  collectSym(sel->getFalseValue());
167  collectSym(sel->getCondition());
168  }
169  else if (const BinaryOperator* binary =
170  SVFUtil::dyn_cast<BinaryOperator>(&inst))
171  {
172  for (u32_t i = 0; i < binary->getNumOperands(); i++)
173  collectSym(binary->getOperand(i));
174  }
175  else if (const UnaryOperator* unary =
176  SVFUtil::dyn_cast<UnaryOperator>(&inst))
177  {
178  for (u32_t i = 0; i < unary->getNumOperands(); i++)
179  collectSym(unary->getOperand(i));
180  }
181  else if (const CmpInst* cmp = SVFUtil::dyn_cast<CmpInst>(&inst))
182  {
183  for (u32_t i = 0; i < cmp->getNumOperands(); i++)
184  collectSym(cmp->getOperand(i));
185  }
186  else if (const CastInst* cast =
187  SVFUtil::dyn_cast<CastInst>(&inst))
188  {
189  collectSym(cast->getOperand(0));
190  }
191  else if (const ReturnInst* ret =
192  SVFUtil::dyn_cast<ReturnInst>(&inst))
193  {
194  if (ret->getReturnValue())
195  collectSym(ret->getReturnValue());
196  }
197  else if (const BranchInst* br =
198  SVFUtil::dyn_cast<BranchInst>(&inst))
199  {
200  Value* opnd = br->isConditional() ? br->getCondition() : br->getOperand(0);
201  collectSym(opnd);
202  }
203  else if (const SwitchInst* sw =
204  SVFUtil::dyn_cast<SwitchInst>(&inst))
205  {
206  collectSym(sw->getCondition());
207  }
208  else if (isNonInstricCallSite(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(&inst)))
209  {
210 
211  const CallBase* cs = LLVMUtil::getLLVMCallSite(&inst);
212  for (u32_t i = 0; i < cs->arg_size(); i++)
213  {
214  collectSym(cs->getArgOperand(i));
215  }
216  // Calls to inline asm need to be added as well because the
217  // callee isn't referenced anywhere else.
218  const Value* Callee = cs->getCalledOperand();
219  collectSym(Callee);
220 
221  // TODO handle inlineAsm
223  if (Options::EnableTypeCheck())
224  {
225  getTypeInference()->validateTypeCheck(cs);
226  }
227  }
229  }
230  }
231  }
232 
233  symInfo->totalSymNum = NodeIDAllocator::get()->endSymbolAllocation();
234  if (Options::SymTabPrint())
235  {
236  symInfo->dump();
237  }
238 }

289 void SymbolTableBuilder::collectVal(const Value* val)
290 {
291  // collect and record special sym here
292  if (LLVMUtil::isNullPtrSym(val) || LLVMUtil::isBlackholeSym(val))
293  {
294  return;
295  }
296  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->valSymMap.find(
297  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
298  if (iter == symInfo->valSymMap.end())
299  {
300  // create val sym and sym type
301  SVFValue* svfVal = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val);
302  SymID id = NodeIDAllocator::get()->allocateValueId();
303  symInfo->valSymMap.insert(std::make_pair(svfVal, id));
304  DBOUT(DMemModel,
305  outs() << "create a new value sym " << id << "\n");
307  if (const GlobalVariable* globalVar = SVFUtil::dyn_cast<GlobalVariable>(val))
308  handleGlobalCE(globalVar);
309  }
310 
311  if (isConstantObjSym(val))
312  collectObj(val);

318 void SymbolTableBuilder::collectObj(const Value* val)
319 {
320  val = LLVMUtil::getGlobalRep(val);
321  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->objSymMap.find(
322  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
323  if (iter == symInfo->objSymMap.end())
324  {
325  SVFValue* svfVal = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val);
326  // if the object pointed by the pointer is a constant data (e.g., i32 0) or a global constant object (e.g. string)
327  // then we treat them as one ConstantObj
328  if (isConstantObjSym(val) && !symInfo->getModelConstants())
329  {
330  symInfo->objSymMap.insert(std::make_pair(svfVal, symInfo->constantSymID()));
331  }
332  // otherwise, we will create an object for each abstract memory location
333  else
334  {
335  // create obj sym and sym type
336  SymID id = NodeIDAllocator::get()->allocateObjectId();
337  symInfo->objSymMap.insert(std::make_pair(svfVal, id));
338  DBOUT(DMemModel,
339  outs() << "create a new obj sym " << id << "\n");
340 
341  // create a memory object
342  MemObj* mem =
343  new MemObj(id, createObjTypeInfo(val),
344  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
345  assert(symInfo->objMap.find(id) == symInfo->objMap.end());
346  symInfo->objMap[id] = mem;
347  }
348  }

388 void SymbolTableBuilder::handleCE(const Value* val)
389 {
390  if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
391  {
392  if (const ConstantExpr* ce = isGepConstantExpr(ref))
393  {
394  DBOUT(DMemModelCE, outs() << "handle constant expression "
395  << LLVMModuleSet::getLLVMModuleSet()
396  ->getSVFValue(ref)
397  ->toString()
398  << "\n");
399  collectVal(ce);
400 
401  // handle the recursive constant express case
402  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
403  for (u32_t i = 0; i < ce->getNumOperands(); ++i)
404  {
405  collectVal(ce->getOperand(i));
406  handleCE(ce->getOperand(i));
407  }
408  }
409  else if (const ConstantExpr* ce = isCastConstantExpr(ref))
410  {
411  DBOUT(DMemModelCE, outs() << "handle constant expression "
412  << LLVMModuleSet::getLLVMModuleSet()
413  ->getSVFValue(ref)
414  ->toString()
415  << "\n");
416  collectVal(ce);
417  collectVal(ce->getOperand(0));
418  // handle the recursive constant express case
419  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
420  handleCE(ce->getOperand(0));
421  }
422  else if (const ConstantExpr* ce = isSelectConstantExpr(ref))
423  {
424  DBOUT(DMemModelCE, outs() << "handle constant expression "
425  << LLVMModuleSet::getLLVMModuleSet()
426  ->getSVFValue(ref)
427  ->toString()
428  << "\n");
429  collectVal(ce);
430  collectVal(ce->getOperand(0));
431  collectVal(ce->getOperand(1));
432  collectVal(ce->getOperand(2));
433  // handle the recursive constant express case
434  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
435  handleCE(ce->getOperand(0));
436  handleCE(ce->getOperand(1));
437  handleCE(ce->getOperand(2));
438  }
439  // if we meet a int2ptr, then it points-to black hole
440  else if (const ConstantExpr* int2Ptrce = isInt2PtrConstantExpr(ref))
441  {
442  collectVal(int2Ptrce);
443  }
444  else if (const ConstantExpr* ptr2Intce = isPtr2IntConstantExpr(ref))
445  {
446  collectVal(ptr2Intce);
447  const Constant* opnd = ptr2Intce->getOperand(0);
448  handleCE(opnd);
449  }
450  else if (isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
451  {
452  collectVal(ref);
453  }
454  else if (isBinaryConstantExpr(ref))
455  {
456  collectVal(ref);
457  }
458  else if (isUnaryConstantExpr(ref))
459  {
460  // we don't handle unary constant expression like fneg(x) now
461  collectVal(ref);
462  }
463  else if (SVFUtil::isa<ConstantAggregate>(ref))
464  {
465  // we don't handle constant aggregate like constant vectors
466  collectVal(ref);
467  }
468  else
469  {
470  assert(!SVFUtil::isa<ConstantExpr>(val) &&
471  "we don't handle all other constant expression for now!");
472  collectVal(ref);
473  }
474  }
475 }

480 void SymbolTableBuilder::handleGlobalCE(const GlobalVariable* G)
481 {
482  assert(G);
483 
484  //The type this global points to
485  const Type* T = G->getValueType();
486  bool is_array = 0;
487  //An array is considered a single variable of its type.
488  while (const ArrayType* AT = SVFUtil::dyn_cast<ArrayType>(T))
489  {
490  T = AT->getElementType();
491  is_array = true;
492  }
493 
494  if (SVFUtil::isa<StructType>(T))
495  {
496  //A struct may be used in constant GEP expr.
497  for (const User* user : G->users())
498  {
499  handleCE(user);
500  }
501  }
502  else if (is_array)
503  {
504  for (const User* user : G->users())
505  {
506  handleCE(user);
507  }
508  }
509 
510  if (G->hasInitializer())
511  {
512  handleGlobalInitializerCE(G->getInitializer());
513  }

588 const Type* SymbolTableBuilder::inferTypeOfHeapObjOrStaticObj(const Instruction *inst)
589 {
590  const Value* startValue = inst;
591  const PointerType *originalPType = SVFUtil::dyn_cast<PointerType>(inst->getType());
592  const Type* inferedType = nullptr;
593  assert(originalPType && "empty type?");
594  const SVFInstruction* svfinst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(inst);
595  if(SVFUtil::isHeapAllocExtCallViaRet(svfinst))
596  {
597  if(const Value* v = getFirstUseViaCastInst(inst))
598  {
599  if (const PointerType *newTy = SVFUtil::dyn_cast<PointerType>(v->getType()))
600  {
601  originalPType = newTy;
602  }
603  }
604  inferedType = inferObjType(startValue);
605  }
606  else if(SVFUtil::isHeapAllocExtCallViaArg(svfinst))
607  {
608  const CallBase* cs = LLVMUtil::getLLVMCallSite(inst);
609  int arg_pos = SVFUtil::getHeapAllocHoldingArgPosition(SVFUtil::getSVFCallSite(svfinst));
610  const Value* arg = cs->getArgOperand(arg_pos);
611  originalPType = SVFUtil::dyn_cast<PointerType>(arg->getType());
612  inferedType = inferObjType(startValue = arg);
613  }
614  else
615  {
616  assert( false && "not a heap allocation instruction?");
617  }
618 
619  getTypeInference()->typeSizeDiffTest(originalPType, inferedType, startValue);
620 
621  return inferedType;
622 }

627 ObjTypeInfo* SymbolTableBuilder::createObjTypeInfo(const Value* val)
628 {
629  const Type* objTy = nullptr;
630 
631  const Instruction* I = SVFUtil::dyn_cast<Instruction>(val);
632 
633  // We consider two types of objects:
634  // (1) A heap/static object from a callsite
635  if (I && isNonInstricCallSite(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(I)))
636  {
637  objTy = inferTypeOfHeapObjOrStaticObj(I);
638  }
639  // (2) Other objects (e.g., alloca, global, etc.)
640  else
641  {
642  if (SVFUtil::isa<PointerType>(val->getType()))
643  {
644  if (const AllocaInst *allocaInst = SVFUtil::dyn_cast<AllocaInst>(val))
645  {
646  // get the type of the allocated memory
647  // e.g., for `%retval = alloca i64, align 4`, we return i64
648  objTy = allocaInst->getAllocatedType();
649  }
650  else if (const GlobalValue *global = SVFUtil::dyn_cast<GlobalValue>(val))
651  {
652  // get the pointee type of the global pointer (begins with @ symbol in llvm)
653  objTy = global->getValueType();
654  }
655  else
656  {
657  SVFUtil::errs() << dumpValueAndDbgInfo(val) << "\n";
658  assert(false && "not an allocation or global?");
659  }
660  }
661  }
662 
663  if (objTy)
664  {
665  (void) getOrAddSVFTypeInfo(objTy);
666  ObjTypeInfo* typeInfo = new ObjTypeInfo(
667  LLVMModuleSet::getLLVMModuleSet()->getSVFType(objTy),
668  Options::MaxFieldLimit());
669  initTypeInfo(typeInfo,val, objTy);
670  return typeInfo;
671  }
672  else
673  {
674  writeWrnMsg("try to create an object with a non-pointer type.");
675  writeWrnMsg(val->getName().str());
676  writeWrnMsg("(" + LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val)->getSourceLoc() + ")");
677  if (isConstantObjSym(val))
678  {
679  ObjTypeInfo* typeInfo = new ObjTypeInfo(
680  LLVMModuleSet::getLLVMModuleSet()->getSVFType(val->getType()),
681  0);
682  initTypeInfo(typeInfo,val, val->getType());
683  return typeInfo;
684  }
685  else
686  {
687  assert(false && "Memory object must be either (1) held by a pointer-typed ref value or (2) a constant value (e.g., 10).");
688  abort();
689  }
690  }

696 void SymbolTableBuilder::analyzeObjType(ObjTypeInfo* typeinfo, const Value* val)
697 {
698  const Type *elemTy = LLVMModuleSet::getLLVMModuleSet()->getLLVMType(typeinfo->getType());
699  // Find the inter nested array element
700  while (const ArrayType* AT = SVFUtil::dyn_cast<ArrayType>(elemTy))
701  {
702  elemTy = AT->getElementType();
703  if (SVFUtil::isa<GlobalVariable>(val) &&
704  SVFUtil::cast<GlobalVariable>(val)->hasInitializer() &&

737 u32_t SymbolTableBuilder::analyzeHeapAllocByteSize(const Value* val)
738 {
739  if(const llvm::CallInst* callInst = llvm::dyn_cast<llvm::CallInst>(val))
740  {
741  if (const llvm::Function* calledFunction =
742  callInst->getCalledFunction())
743  {
744  const SVFFunction* svfFunction =
745  LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(
746  calledFunction);
747  std::vector<const Value*> args;
748  // Heap alloc functions have annoation like "AllocSize:Arg1"
749  for (std::string annotation : svfFunction->getAnnotations())
750  {
751  if (annotation.find("AllocSize:") != std::string::npos)
752  {
753  std::string allocSize = annotation.substr(10);
754  std::stringstream ss(allocSize);
755  std::string token;
756  // Analyaze annotation string and attract Arg list
757  while (std::getline(ss, token, '*'))
758  {
759  if (token.rfind("Arg", 0) == 0)
760  {
761  u32_t argIndex;
762  std::istringstream(token.substr(3)) >> argIndex;
763  if (argIndex < callInst->getNumOperands() - 1)
764  {
765  args.push_back(
766  callInst->getArgOperand(argIndex));
767  }
768  }
769  }
770  }
771  }
772  u64_t product = 1;
773  if (args.size() > 0)
774  {
775  // for annotations like "AllocSize:Arg0*Arg1"
776  for (const llvm::Value* arg : args)
777  {
778  if (const llvm::ConstantInt* constIntArg =
779  llvm::dyn_cast<llvm::ConstantInt>(arg))
780  {
781  // Multiply the constant Value if all Args are const
782  product *= constIntArg->getZExtValue();
783  }
784  else
785  {
786  // if Arg list has non-const value, return 0 to indicate it is non const byte size
787  return 0;
788  }
789  }
790  // If all the Args are const, return product
791  return product;
792  }
793  else
794  {
795  // for annotations like "AllocSize:UNKNOWN"
796  return 0;
797  }
798  }
799  }
800  // if it is not CallInst or CallInst has no CalledFunction, return 0 to indicate it is non const byte size
801  return 0;
802 }

846 void SymbolTableBuilder::initTypeInfo(ObjTypeInfo* typeinfo, const Value* val,
847  const Type* objTy)
848 {
849 
850  u32_t elemNum = 1;
851  // init byteSize = 0, If byteSize is changed in the following process,
852  // it means that ObjTypeInfo has a Constant Byte Size
853  u32_t byteSize = 0;
854  // Global variable
855  // if val is Function Obj, byteSize is not set
856  if (SVFUtil::isa<Function>(val))
857  {
858  typeinfo->setFlag(ObjTypeInfo::FUNCTION_OBJ);
859  analyzeObjType(typeinfo,val);
860  elemNum = getNumOfElements(objTy);
861  }
864  else if(const AllocaInst* allocaInst = SVFUtil::dyn_cast<AllocaInst>(val))
865  {
866  typeinfo->setFlag(ObjTypeInfo::STACK_OBJ);
867  analyzeObjType(typeinfo,val);
870  if(const ConstantInt* sz = SVFUtil::dyn_cast<ConstantInt>(allocaInst->getArraySize()))
871  {
872  elemNum = sz->getZExtValue() * getNumOfElements(objTy);
873  byteSize = sz->getZExtValue() * typeinfo->getType()->getByteSize();
874  }
876  else
877  {
878  elemNum = getNumOfElements(objTy);
879  byteSize = 0;
880  }
881  }
884  else if(SVFUtil::isa<GlobalVariable>(val))
885  {
886  typeinfo->setFlag(ObjTypeInfo::GLOBVAR_OBJ);
887  if(isConstantObjSym(val))
888  typeinfo->setFlag(ObjTypeInfo::CONST_GLOBAL_OBJ);
889  analyzeObjType(typeinfo,val);
890  elemNum = getNumOfElements(objTy);
891  byteSize = typeinfo->getType()->getByteSize();
892  }
894  else if (SVFUtil::isa<Instruction>(val) &&
895  isHeapAllocExtCall(
896  LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(
897  SVFUtil::cast<Instruction>(val))))
898  {
899  elemNum = analyzeHeapObjType(typeinfo,val);
900  // analyze heap alloc like (malloc/calloc/...), the alloc functions have
901  // annotation like "AllocSize:Arg1". Please refer to extapi.c.
902  // e.g. calloc(4, 10), annotation is "AllocSize:Arg0*Arg1",
903  // it means byteSize = 4 (Arg0) * 10 (Arg1) = 40
904  byteSize = analyzeHeapAllocByteSize(val);
905  }
906  else if(ArgInProgEntryFunction(val))
907  {
908  analyzeStaticObjType(typeinfo,val);
909  // user input data, label its field as infinite here
910  elemNum = typeinfo->getMaxFieldOffsetLimit();
911  byteSize = typeinfo->getType()->getByteSize();
912  }
913  else if(LLVMUtil::isConstDataOrAggData(val))
914  {
915  typeinfo->setFlag(ObjTypeInfo::CONST_DATA);
916  elemNum = getNumOfFlattenElements(val->getType());
917  byteSize = typeinfo->getType()->getByteSize();
918  }
919  else
920  {
921  assert("what other object do we have??");
922  abort();
923  }
924 
925  // Reset maxOffsetLimit if it is over the total fieldNum of this object
926  if(typeinfo->getMaxFieldOffsetLimit() > elemNum)
927  typeinfo->setNumOfElements(elemNum);
928 
929  // set ByteSize. If ByteSize > 0, this typeinfo has constant type.
930  // If ByteSize == 0, this typeinfo has 1) zero byte 2) non-const byte size
931  // If ByteSize>MaxFieldLimit, set MaxFieldLimit to the byteSize;
932  byteSize = Options::MaxFieldLimit() > byteSize? byteSize: Options::MaxFieldLimit();
933  typeinfo->setByteSizeOfObj(byteSize);
934 }
