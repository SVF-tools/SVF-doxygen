77 void TypeAnalysis::callGraphSolveBasedOnCHA(const CallSiteToFunPtrMap& callsites, CallEdgeMap& newEdges)
78 {
79  for(CallSiteToFunPtrMap::const_iterator iter = callsites.begin(), eiter = callsites.end(); iter!=eiter; ++iter)
80  {
81  const CallICFGNode* cbn = iter->first;
82  CallSite cs = SVFUtil::getSVFCallSite(cbn->getCallSite());
83  if (cs.isVirtualCall())
84  {
85  const SVFValue* vtbl = cs.getVtablePtr();
86  (void)vtbl; // Suppress warning of unused variable under release build
87  assert(pag->hasValueNode(vtbl));
88  VFunSet vfns;
89  getVFnsFromCHA(cbn, vfns);
90  connectVCallToVFns(cbn, vfns, newEdges);
91  }
92  }
93 }

96 void TypeAnalysis::dumpCHAStats()
97 {
98 
99  const CHGraph *chgraph = SVFUtil::dyn_cast<CHGraph>(getCHGraph());
100  if (chgraph == nullptr)
101  {
102  SVFUtil::errs() << "dumpCHAStats only implemented for standard CHGraph.\n";
103  return;
104  }
105 
106  u32_t pure_abstract_class_num = 0,
107  multi_inheritance_class_num = 0;
108  for (CHGraph::const_iterator it = chgraph->begin(), eit = chgraph->end();
109  it != eit; ++it)
110  {
111  CHNode *node = it->second;
112  outs() << "class " << node->getName() << "\n";
113  if (node->isPureAbstract())
114  pure_abstract_class_num++;
115  if (node->isMultiInheritance())
116  multi_inheritance_class_num++;
117  }
118  outs() << "class_num:\t" << chgraph->getTotalNodeNum() << '\n';
119  outs() << "pure_abstract_class_num:\t" << pure_abstract_class_num << '\n';
120  outs() << "multi_inheritance_class_num:\t" << multi_inheritance_class_num << '\n';
121 
122  /*
123  * count the following info:
124  * vtblnum
125  * total vfunction
126  * vtbl max vfunction
127  * pure abstract class
128  */
129  u32_t vtblnum = 0,
130  vfunc_total = 0,
131  vtbl_max = 0,
132  pure_abstract = 0;
133  set<const SVFFunction*> allVirtualFunctions;
134  for (CHGraph::const_iterator it = chgraph->begin(), eit = chgraph->end();
135  it != eit; ++it)
136  {
137  CHNode *node = it->second;
138  if (node->isPureAbstract())
139  pure_abstract++;
140 
141  u32_t vfuncs_size = 0;
142  const vector<CHNode::FuncVector>& vecs = node->getVirtualFunctionVectors();
143  for (vector<CHNode::FuncVector>::const_iterator vit = vecs.begin(),
144  veit = vecs.end(); vit != veit; ++vit)
145  {
146  vfuncs_size += (*vit).size();
147  for (vector<const SVFFunction*>::const_iterator fit = (*vit).begin(),
148  feit = (*vit).end(); fit != feit; ++fit)
149  {
150  const SVFFunction* func = *fit;
151  allVirtualFunctions.insert(func);
152  }
153  }
154  if (vfuncs_size > 0)
155  {
156  vtblnum++;
157  if (vfuncs_size > vtbl_max)
158  {
159  vtbl_max = vfuncs_size;
160  }
161  }
162  }
163  vfunc_total = allVirtualFunctions.size();
164 
165  outs() << "vtblnum:\t" << vtblnum << '\n';
166  outs() << "vtbl_average:\t" << (double)(vfunc_total)/vtblnum << '\n';
167  outs() << "vtbl_max:\t" << vtbl_max << '\n';
168  outs() << "pure_abstract:\t" << pure_abstract << '\n';
169 }
