385 void PointerAnalysis::resolveIndCalls(const CallICFGNode* cs, const PointsTo& target, CallEdgeMap& newEdges)
386 {
387 
388  assert(pag->isIndirectCallSites(cs) && "not an indirect callsite?");
390  for (PointsTo::iterator ii = target.begin(), ie = target.end();
391  ii != ie; ii++)
392  {
393 
394  if(getNumOfResolvedIndCallEdge() >= Options::IndirectCallLimit())
395  {
396  wrnMsg("Resolved Indirect Call Edges are Out-Of-Budget, please increase the limit");
397  return;
398  }
399 
400  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
401  {
402  const MemObj* obj = pag->getObject(objPN);
403 
404  if(obj->isFunction())
405  {
406  const SVFFunction* calleefun = SVFUtil::cast<SVFFunction>(obj->getValue());
407  const SVFFunction* callee = calleefun->getDefFunForMultipleModule();
408 
409  if(SVFUtil::matchArgs(cs->getCallSite(), callee) == false)
410  continue;
411 
412  if(0 == getIndCallMap()[cs].count(callee))
413  {
414  newEdges[cs].insert(callee);
415  getIndCallMap()[cs].insert(callee);
416 
417  ptaCallGraph->addIndirectCallGraphEdge(cs, cs->getCaller(), callee);
418  // FIXME: do we need to update llvm call graph here?
419  // The indirect call is maintained by ourself, We may update llvm's when we need to
420  //CallGraphNode* callgraphNode = callgraph->getOrInsertFunction(cs.getCaller());
421  //callgraphNode->addCalledFunction(cs,callgraph->getOrInsertFunction(callee));
422  }
423  }
424  }

466 void PointerAnalysis::connectVCallToVFns(const CallICFGNode* cs, const VFunSet &vfns, CallEdgeMap& newEdges)
467 {
469  for (VFunSet::const_iterator fit = vfns.begin(),
470  feit = vfns.end(); fit != feit; ++fit)
471  {
472  const SVFFunction* callee = *fit;
473  callee = callee->getDefFunForMultipleModule();
474  if (getIndCallMap()[cs].count(callee) > 0)
475  continue;
476  if(SVFUtil::getSVFCallSite(cs->getCallSite()).arg_size() == callee->arg_size() ||
477  (SVFUtil::getSVFCallSite(cs->getCallSite()).isVarArg() && callee->isVarArg()))
478  {
479  newEdges[cs].insert(callee);
480  getIndCallMap()[cs].insert(callee);
481  const CallICFGNode* callBlockNode = pag->getICFG()->getCallICFGNode(cs->getCallSite());
482  ptaCallGraph->addIndirectCallGraphEdge(callBlockNode, cs->getCaller(),callee);
483  }
484  }

504 void PointerAnalysis::validateSuccessTests(std::string fun)
505 {
506  // check for must alias cases, whether our alias analysis produce the correct results
507  if (const SVFFunction* checkFun = svfMod->getSVFFunction(fun))
508  {
509  if(!checkFun->isUncalledFunction())
510  outs() << "[" << this->PTAName() << "] Checking " << fun << "\n";
511 
512  for(const CallICFGNode* callNode : pag->getCallSiteSet())
513  {
514  const SVFInstruction* svfInst = callNode->getCallSite();
515  if (SVFUtil::getCallee(svfInst) == checkFun)
516  {
517 
518  CallSite cs(svfInst);
519  assert(cs.getNumArgOperands() == 2
520  && "arguments should be two pointers!!");
521  const SVFValue* V1 = cs.getArgOperand(0);
522  const SVFValue* V2 = cs.getArgOperand(1);
523  AliasResult aliasRes = alias(V1, V2);
524 
525  bool checkSuccessful = false;
526  if (fun == aliasTestMayAlias || fun == aliasTestMayAliasMangled)
527  {
528  if (aliasRes == AliasResult::MayAlias || aliasRes == AliasResult::MustAlias)
529  checkSuccessful = true;
530  }
531  else if (fun == aliasTestNoAlias || fun == aliasTestNoAliasMangled)
532  {
533  if (aliasRes == AliasResult::NoAlias)
534  checkSuccessful = true;
535  }
536  else if (fun == aliasTestMustAlias || fun == aliasTestMustAliasMangled)
537  {
538  // change to must alias when our analysis support it
539  if (aliasRes == AliasResult::MayAlias || aliasRes == AliasResult::MustAlias)
540  checkSuccessful = true;
541  }
542  else if (fun == aliasTestPartialAlias || fun == aliasTestPartialAliasMangled)
543  {
544  // change to partial alias when our analysis support it
545  if (aliasRes == AliasResult::MayAlias)
546  checkSuccessful = true;
547  }
548  else
549  assert(false && "not supported alias check!!");
550 
551  NodeID id1 = pag->getValueNode(V1);
552  NodeID id2 = pag->getValueNode(V2);
553 
554  if (checkSuccessful)
555  outs() << sucMsg("\t SUCCESS :") << fun << " check <id:" << id1 << ", id:" << id2 << "> at ("
556  << svfInst->getSourceLoc() << ")\n";
557  else
558  {
559  SVFUtil::errs() << errMsg("\t FAILURE :") << fun
560  << " check <id:" << id1 << ", id:" << id2
561  << "> at (" << svfInst->getSourceLoc() << ")\n";
562  assert(false && "test case failed!");
563  }
564  }
565  }
566  }
567 }

572 void PointerAnalysis::validateExpectedFailureTests(std::string fun)
573 {
574

576  {
577  if(!checkFun->isUncalledFunction())
578  outs() << "[" << this->PTAName() << "] Checking " << fun << "\n";
579

581  {
582  const SVFInstruction* svfInst = callNode->getCallSite();
583  if (SVFUtil::getCallee(svfInst) == checkFun)
584  {
585  CallSite call = getSVFCallSite(svfInst);
586  assert(call.arg_size() == 2

588  const SVFValue* V1 = call.getArgOperand(0);
589  const SVFValue* V2 = call.getArgOperand(1);
590  AliasResult aliasRes = alias(V1, V2);
591 
592  bool expectedFailure = false;
593  if (fun == aliasTestFailMayAlias || fun == aliasTestFailMayAliasMangled)
594  {
595  // change to must alias when our analysis support it
596  if (aliasRes == AliasResult::NoAlias)
597  expectedFailure = true;
598  }
599  else if (fun == aliasTestFailNoAlias || fun == aliasTestFailNoAliasMangled)
600  {
601  // change to partial alias when our analysis support it
602  if (aliasRes == AliasResult::MayAlias || aliasRes == AliasResult::PartialAlias || aliasRes == AliasResult::MustAlias)
603  expectedFailure = true;
604  }
605  else
606  assert(false && "not supported alias check!!");
607 
608  NodeID id1 = pag->getValueNode(V1);
609  NodeID id2 = pag->getValueNode(V2);
610 
611  if (expectedFailure)
612  outs() << sucMsg("\t EXPECTED-FAILURE :") << fun << " check <id:" << id1 << ", id:" << id2 << "> at ("
613  << call.getInstruction()->getSourceLoc() << ")\n";
614  else
615  {
616  SVFUtil::errs() << errMsg("\t UNEXPECTED FAILURE :") << fun << " check <id:" << id1 << ", id:" << id2 << "> at ("

618  assert(false && "test case failed!");
619  }
620  }
