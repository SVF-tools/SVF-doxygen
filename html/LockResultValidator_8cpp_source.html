94 bool LockResultValidator::collectLockTargets()
95 {
96  const Function* F = nullptr;
97  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
98  {
99  for(auto it = M.begin(); it != M.end(); it++)
100  {
101  const std::string fName = (*it).getName().str();
102  if(fName.find(LOCK) != std::string::npos)
103  {
104  F = &(*it);
105  break;
106  }
107  }
108  }
109  if (!F)
110  return false;
111  for(Value::const_use_iterator it = F->use_begin(), ie = F->use_end(); it!=ie; it++)
112  {
113  const Use *u = &*it;
114  const Value* user = u->getUser();
115  const Instruction* inst = SVFUtil::dyn_cast<Instruction>(user);
116  CxtLockSetStr y = getStringArg(inst, 0);
117  const Instruction* memInst = getPreviousMemoryAccessInst(inst);
118  const SVFInstruction* svfMemInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(memInst);
119 
120  instToCxtLockSet[svfMemInst] = y;
121  if(const StoreInst* store = SVFUtil::dyn_cast<StoreInst> (memInst))
122  {
123  if(const BinaryOperator* bop = SVFUtil::dyn_cast<BinaryOperator> (store->getValueOperand()))
124  {
125  const Value* v = bop->getOperand(0);
126  const Instruction* prevInst = SVFUtil::dyn_cast<LoadInst> (v);
127  const SVFInstruction* svfPrevInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(prevInst);
128  instToCxtLockSet[svfPrevInst] = y;
129  }
130  }
131  }
132  return true;

135 LockResultValidator::LOCK_FLAG LockResultValidator::validateStmtInLock()
136 {
137  SVFIR* pag = SVFIR::getPAG();
138  LockResultValidator::LOCK_FLAG res = LockResultValidator::LOCK_TRUE;
139  LockAnalysis::CxtStmtToCxtLockSet analyedLS = _la->getCSTCLS();
140  for(LockAnalysis::CxtStmtToCxtLockSet::iterator it = analyedLS.begin(),
141  eit = analyedLS.end(); it!=eit; it++)
142  {
143  const SVFInstruction* inst = ((*it).first).getStmt();
144  bool interestedInst = true;
145  for(const SVFStmt* stmt : pag->getSVFStmtList(pag->getICFG()->getICFGNode(inst)))
146  {
147  if(!SVFUtil::isa<LoadStmt>(stmt) && !SVFUtil::isa<StoreStmt>(stmt))
148  interestedInst = false;
149  }
150  if(interestedInst==false)
151  continue;
152  const SVFFunction* F = inst->getFunction();
153  if(inFilter(F->getName()))
154  continue;
155  CxtLockSetStr LS = instToCxtLockSet[inst];
156  if(LS.size() != (*it).second.size())
157  {
158  if (Options::PrintValidRes())
159  {
160  outs() << errMsg("\nValidate Stmt's Lock : Wrong at: ") << inst->toString() << "\n";
161  outs() << "Reason: The number of lock on current stmt is wrong\n";
162  outs() << "\n----Given locks:\n";
163  for (CxtLockSetStr::iterator it1 = LS.begin(),eit1 = LS.end(); it1 != eit1; it++)
164  {
165  outs() << "Lock " << *it1 << " ";
166  }
167  outs() << "\n----Analysis locks:\n";
168  for (LockAnalysis::CxtLockSet::iterator it2 = (*it).second.begin(),
169  eit2 = (*it).second.end(); it2 != eit2; ++it)
170  {
171  const SVFInstruction* call = (*it2).getStmt();
172  outs()<<"Lock " << call->toString() << " ";
173  }
174  outs() << "\n";
175  }
176  res = LockResultValidator::LOCK_UNSOUND;
177  }
178  LockAnalysis::CxtLockSet LSA = (*it).second;
179 
180  for(LockAnalysis::CxtLockSet::iterator it3 = LSA.begin(), eit3=LSA.end(); it3!=eit3; it3++)
181  {
182  const SVFInstruction* call = (*it3).getStmt();
183  if(SVFUtil::isCallSite(call) == false)
184  continue;
185  std::string lockName = SVFUtil::getSVFCallSite(call).getArgOperand(0)->getName();
186  if(!match(lockName, LS))
187  {
188  if(Options::PrintValidRes())
189  {
190  outs() << "\nValidate Stmt's Lock : Wrong at (" << inst->toString() << ")\n";
191  outs() << "Reason: The number of lock on current stmt is wrong\n";
192  outs() << "\n Lock " << lockName << " should not protect current instruction\n";
193  res = LockResultValidator::LOCK_IMPRECISE;
194  }
195  }
196  }
197  }
198  return res;
