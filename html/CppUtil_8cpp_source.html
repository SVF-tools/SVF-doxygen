44 static bool isOperOverload(const std::string& name)
45 {
46  u32_t leftnum = 0, rightnum = 0;
47  std::string subname = name;
48  size_t leftpos, rightpos;
49  leftpos = subname.find("<");
50  while (leftpos != std::string::npos)
51  {
52  subname = subname.substr(leftpos + 1);
53  leftpos = subname.find("<");
54  leftnum++;
55  }
56  subname = name;
57  rightpos = subname.find(">");
58  while (rightpos != std::string::npos)
59  {
60  subname = subname.substr(rightpos + 1);
61  rightpos = subname.find(">");
62  rightnum++;
63  }
64  return leftnum != rightnum;
65 }

67 static std::string getBeforeParenthesis(const std::string& name)
68 {
69  size_t lastRightParen = name.rfind(")");
70  assert(lastRightParen > 0);
71 
72  s32_t paren_num = 1, pos;
73  for (pos = lastRightParen - 1; pos >= 0; pos--)
74  {
75  if (name[pos] == ')')
76  paren_num++;
77  if (name[pos] == '(')
78  paren_num--;
79  if (paren_num == 0)
80  break;
81  }
82  return name.substr(0, pos);
83 }

85 std::string cppUtil::getBeforeBrackets(const std::string& name)
86 {
87  if (name.size() == 0 || name[name.size() - 1] != '>')

93  {

95  bracket_num++;
96  if (name[pos] == '<')

104 static void handleThunkFunction(cppUtil::DemangledName& dname)
105 {
106  // when handling multi-inheritance,
107  // the compiler may generate thunk functions
108  // to perform `this` pointer adjustment
109  // they are indicated with `virtual thunk to `
110  // and `nun-virtual thunk to`.
111  // if the classname starts with part of a
112  // demangled name starts with
113  // these prefixes, we need to remove the prefix
114  // to get the real class name
115 
116  static std::vector<std::string> thunkPrefixes = {VThunkFuncLabel,
117  NVThunkFunLabel
118  };
119  for (unsigned i = 0; i < thunkPrefixes.size(); i++)
120  {
121  auto prefix = thunkPrefixes[i];
122  if (dname.className.size() > prefix.size() &&
123  dname.className.compare(0, prefix.size(), prefix) == 0)
124  {
125  dname.className = dname.className.substr(prefix.size());
126  dname.isThunkFunc = true;
127  return;
128  }
129  }
130 }

153 struct cppUtil::DemangledName cppUtil::demangle(const std::string& name)
154 {
155  struct cppUtil::DemangledName dname;
156  dname.isThunkFunc = false;
157 
158  s32_t status;
159  char* realname = abi::__cxa_demangle(name.c_str(), 0, 0, &status);
160  if (realname == nullptr)
161  {
162  dname.className = "";
163  dname.funcName = "";
164  }
165  else
166  {
167  std::string realnameStr = std::string(realname);
168  std::string beforeParenthesis = getBeforeParenthesis(realnameStr);
169  if (beforeParenthesis.find("::") == std::string::npos ||
170  isOperOverload(beforeParenthesis))
171  {
172  dname.className = "";
173  dname.funcName = "";
174  }
175  else
176  {
177  std::string beforeBracket = getBeforeBrackets(beforeParenthesis);
178  size_t colon = beforeBracket.rfind("::");
179  if (colon == std::string::npos)
180  {
181  dname.className = "";
182  dname.funcName = "";
183  }
184  else
185  {
186  dname.className = beforeParenthesis.substr(0, colon);
187  dname.funcName = beforeParenthesis.substr(colon + 2);
188  }
189  }
190  std::free(realname);
191  }
192 
193  handleThunkFunction(dname);
194 
195  return dname;
196 }
