47 /*
48  * Update call graph using pointer analysis results
49  * (1) resolve function pointers for non-fork calls
50  * (2) resolve function pointers for fork sites
51  * (3) resolve function pointers for parallel_for sites
52  */
53 void ThreadCallGraph::updateCallGraph(PointerAnalysis* pta)
54 {
55 
56  PointerAnalysis::CallEdgeMap::const_iterator iter = pta->getIndCallMap().begin();
57  PointerAnalysis::CallEdgeMap::const_iterator eiter = pta->getIndCallMap().end();
58  for (; iter != eiter; iter++)
59  {
60  const CallICFGNode* cs = iter->first;
61  const PTACallGraph::FunctionSet &functions = iter->second;
62  for (PTACallGraph::FunctionSet::const_iterator func_iter =
63  functions.begin(); func_iter != functions.end(); func_iter++)
64  {
65  const SVFFunction* callee = *func_iter;
66  this->addIndirectCallGraphEdge(cs, cs->getCaller(), callee);
67  }
68  }
69 
70  // Fork sites
71  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
72  {
73  const SVFValue* forkedval = tdAPI->getForkedFun((*it)->getCallSite());
74  if(SVFUtil::dyn_cast<SVFFunction>(forkedval)==nullptr)
75  {
76  SVFIR* pag = pta->getPAG();
77  const NodeBS targets = pta->getPts(pag->getValueNode(forkedval)).toNodeBS();
78  for (NodeBS::iterator ii = targets.begin(), ie = targets.end(); ii != ie; ii++)
79  {
80  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
81  {
82  const MemObj* obj = pag->getObject(objPN);
83  if(obj->isFunction())
84  {
85  const SVFFunction* svfCallee = SVFUtil::cast<SVFFunction>(obj->getValue());
86  this->addIndirectForkEdge(*it, svfCallee);
87  }
88  }
89  }
90  }
91  }
92 
93  // parallel_for sites
94  for (CallSiteSet::const_iterator it = parForSitesBegin(), eit = parForSitesEnd(); it != eit; ++it)
95  {
96  const SVFValue* forkedval = tdAPI->getTaskFuncAtHareParForSite((*it)->getCallSite());
97  if(SVFUtil::dyn_cast<SVFFunction>(forkedval)==nullptr)
98  {
99  SVFIR* pag = pta->getPAG();
100  const NodeBS targets = pta->getPts(pag->getValueNode(forkedval)).toNodeBS();
101  for (NodeBS::iterator ii = targets.begin(), ie = targets.end(); ii != ie; ii++)
102  {
103  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
104  {
105  const MemObj* obj = pag->getObject(objPN);
106  if(obj->isFunction())
107  {
108  const SVFFunction* svfCallee = SVFUtil::cast<SVFFunction>(obj->getValue());
109  this->addIndirectForkEdge(*it, svfCallee);
110  }
111  }
112  }
113  }
114  }
115 }

121 void ThreadCallGraph::updateJoinEdge(PointerAnalysis* pta)
122 {
123 
124  for (CallSiteSet::const_iterator it = joinsitesBegin(), eit = joinsitesEnd(); it != eit; ++it)
125  {
126  const SVFValue* jointhread = tdAPI->getJoinedThread((*it)->getCallSite());
127  // find its corresponding fork sites first
128  CallSiteSet forkset;
129  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
130  {
131  const SVFValue* forkthread = tdAPI->getForkedThread((*it)->getCallSite());
132  if (pta->alias(jointhread, forkthread))
133  {
134  forkset.insert(*it);
135  }
136  }
137  assert(!forkset.empty() && "Can't find a forksite for this join!!");
138  addDirectJoinEdge(*it,forkset);
139  }

145 void ThreadCallGraph::addDirectForkEdge(const CallICFGNode* cs)
146 {
147 
148  PTACallGraphNode* caller = getCallGraphNode(cs->getCaller());
149  const SVFFunction* forkee = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun(cs->getCallSite()));
150  assert(forkee && "callee does not exist");
151  PTACallGraphNode* callee = getCallGraphNode(forkee->getDefFunForMultipleModule());
152  CallSiteID csId = addCallSite(cs, callee->getFunction());
153 
154  if (!hasGraphEdge(caller, callee, PTACallGraphEdge::TDForkEdge, csId))
155  {
156  assert(cs->getCaller() == caller->getFunction() && "callee instruction not inside caller??");
157 
158  ThreadForkEdge* edge = new ThreadForkEdge(caller, callee, csId);
159  edge->addDirectCallSite(cs);
160 
161  addEdge(edge);
162  addThreadForkEdgeSetMap(cs, edge);
163  }
164 }

194 void ThreadCallGraph::addDirectJoinEdge(const CallICFGNode* cs,const CallSiteSet& forkset)
195 {
196 
197  PTACallGraphNode* joinFunNode = getCallGraphNode(cs->getCaller());
198 
199  for (CallSiteSet::const_iterator it = forkset.begin(), eit = forkset.end(); it != eit; ++it)
200  {
201 
202  const SVFFunction* threadRoutineFun = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun((*it)->getCallSite()));
203  assert(threadRoutineFun && "thread routine function does not exist");
204  PTACallGraphNode* threadRoutineFunNode = getCallGraphNode(threadRoutineFun);
205  CallSiteID csId = addCallSite(cs, threadRoutineFun);
206 
207  if (!hasThreadJoinEdge(cs,joinFunNode,threadRoutineFunNode, csId))
208  {
209  assert(cs->getCaller() == joinFunNode->getFunction() && "callee instruction not inside caller??");
210  ThreadJoinEdge* edge = new ThreadJoinEdge(joinFunNode,threadRoutineFunNode,csId);
211  edge->addDirectCallSite(cs);
212 
213  addThreadJoinEdgeSetMap(cs, edge);
214  }
215  }
