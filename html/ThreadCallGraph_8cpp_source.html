49 /*
50  * Update call graph using pointer analysis results
51  * (1) resolve function pointers for non-fork calls
52  * (2) resolve function pointers for fork sites
53  * (3) resolve function pointers for parallel_for sites
54  */
55 void ThreadCallGraph::updateCallGraph(PointerAnalysis* pta)
56 {
57 
58  PointerAnalysis::CallEdgeMap::const_iterator iter = pta->getIndCallMap().begin();
59  PointerAnalysis::CallEdgeMap::const_iterator eiter = pta->getIndCallMap().end();
60  for (; iter != eiter; iter++)
61  {
62  const CallICFGNode* cs = iter->first;
63  const CallGraph::FunctionSet &functions = iter->second;
64  for (CallGraph::FunctionSet::const_iterator func_iter =
65  functions.begin(); func_iter != functions.end(); func_iter++)
66  {
67  const SVFFunction* callee = *func_iter;
68  this->addIndirectCallGraphEdge(cs, cs->getCaller(), callee);
69  }
70  }
71 
72  // Fork sites
73  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
74  {
75  const SVFVar* forkedval = tdAPI->getForkedFun(*it);
76  if(SVFUtil::dyn_cast<SVFFunction>(forkedval->getValue())==nullptr)
77  {
78  SVFIR* pag = pta->getPAG();
79  const NodeBS targets = pta->getPts(forkedval->getId()).toNodeBS();
80  for (NodeBS::iterator ii = targets.begin(), ie = targets.end(); ii != ie; ii++)
81  {
82  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
83  {
84  const MemObj* obj = pag->getObject(objPN);
85  if(obj->isFunction())
86  {
87  const SVFFunction* svfCallee = SVFUtil::cast<SVFFunction>(obj->getValue());
88  this->addIndirectForkEdge(*it, svfCallee);
89  }
90  }
91  }
92  }
93  }
94 }

100 void ThreadCallGraph::updateJoinEdge(PointerAnalysis* pta)
101 {
102 
103  for (CallSiteSet::const_iterator it = joinsitesBegin(), eit = joinsitesEnd(); it != eit; ++it)
104  {
105  const SVFVar* jointhread = tdAPI->getJoinedThread(*it);
106  // find its corresponding fork sites first
107  CallSiteSet forkset;
108  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
109  {
110  const SVFVar* forkthread = tdAPI->getForkedThread(*it);
111  if (pta->alias(jointhread->getId(), forkthread->getId()))
112  {
113  forkset.insert(*it);
114  }
115  }
116  assert(!forkset.empty() && "Can't find a forksite for this join!!");
117  addDirectJoinEdge(*it,forkset);
118  }
119 }

124 bool ThreadCallGraph::addDirectForkEdge(const CallICFGNode* cs)
125 {
126 
127  CallGraphNode* caller = getCallGraphNode(cs->getCaller());
128  const SVFFunction* forkee = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun(cs)->getValue());
129  assert(forkee && "callee does not exist");
130  CallGraphNode* callee = getCallGraphNode(forkee->getDefFunForMultipleModule());
131  CallSiteID csId = addCallSite(cs, callee->getFunction());
132 
133  if (!hasGraphEdge(caller, callee, CallGraphEdge::TDForkEdge, csId))
134  {
135  assert(cs->getCaller() == caller->getFunction() && "callee instruction not inside caller??");
136 
137  ThreadForkEdge* edge = new ThreadForkEdge(caller, callee, csId);
138  edge->addDirectCallSite(cs);
139 
140  addEdge(edge);
141  addThreadForkEdgeSetMap(cs, edge);
142  return true;
143  }
144  else
145  return false;
146 }

179 void ThreadCallGraph::addDirectJoinEdge(const CallICFGNode* cs,const CallSiteSet& forkset)
180 {
181 
182  CallGraphNode* joinFunNode = getCallGraphNode(cs->getCaller());
183 
184  for (CallSiteSet::const_iterator it = forkset.begin(), eit = forkset.end(); it != eit; ++it)
185  {
186 
187  const SVFFunction* threadRoutineFun = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun(*it)->getValue());
188  assert(threadRoutineFun && "thread routine function does not exist");
189  CallGraphNode* threadRoutineFunNode = getCallGraphNode(threadRoutineFun);
190  CallSiteID csId = addCallSite(cs, threadRoutineFun);
191 
192  if (!hasThreadJoinEdge(cs,joinFunNode,threadRoutineFunNode, csId))
193  {
194  assert(cs->getCaller() == joinFunNode->getFunction() && "callee instruction not inside caller??");
195  ThreadJoinEdge* edge = new ThreadJoinEdge(joinFunNode,threadRoutineFunNode,csId);
196  edge->addDirectCallSite(cs);
197 
198  addThreadJoinEdgeSetMap(cs, edge);
199  }
