60bool MTA::runOnModule(SVFIR* pag)
61{
62 DBOUT(DGENERAL, outs() << pasMsg("MTA analysis\n"));
63 DBOUT(DMTA, outs() << pasMsg("MTA analysis\n"));
64
65 PointerAnalysis* pta = AndersenWaveDiff::createAndersenWaveDiff(pag);
66 pta->getCallGraph()->dump("ptacg");
67 pag->getICFG()->updateCallGraph(pta->getCallGraph());
68
69 DBOUT(DGENERAL, outs() << pasMsg("Build TCT\n"));
70 DBOUT(DMTA, outs() << pasMsg("Build TCT\n"));
71 DOTIMESTAT(double tctStart = stat->getClk());
72 tct = std::make_unique<TCT>(pta);
73 tcg = tct->getThreadCallGraph();
74 DOTIMESTAT(double tctEnd = stat->getClk());
75 DOTIMESTAT(stat->TCTTime += (tctEnd - tctStart) / TIMEINTERVAL);
76
77 if (pta->printStat())
78 {
79 stat->performThreadCallGraphStat(tcg);
80 stat->performTCTStat(tct.get());
81 }
82
83 tcg->dump("tcg");
84
85 mhp = computeMHP(tct.get());
86 lsa = computeLocksets(tct.get());
87
88 if(Options::RaceCheck())
89 detect();
90
91 return false;
92}

121// * Check (1) write-read race
122// * (2) write-write race (optional)
123// * (3) read-read race (optional)
124// * when two memory access may-happen in parallel and are not protected by the same lock
125// * (excluding global constraints because they are initialized before running the main function)
126// */
127void MTA::detect()
128{
129
130 DBOUT(DGENERAL, outs() << pasMsg("Starting Race Detection\n"));
131
132 Set<const LoadStmt*> loads;
133 Set<const StoreStmt*> stores;
134 SVFIR* pag = SVFIR::getPAG();
135 PointerAnalysis* pta = AndersenWaveDiff::createAndersenWaveDiff(pag);
136
137 // Add symbols for all of the functions and the instructions in them.
138 for (const auto& item : *PAG::getPAG()->getCallGraph())
139 {
140 const FunObjVar* F = item.second->getFunction();
141 // collect and create symbols inside the function body
142 for (auto it : *F)
143 {
144 const SVFBasicBlock* svfbb = it.second;
145 for (const ICFGNode* icfgNode : svfbb->getICFGNodeList())
146 {
147 for(const SVFStmt* stmt : pag->getSVFStmtList(icfgNode))
148 {
149 if (const LoadStmt* l = SVFUtil::dyn_cast<LoadStmt>(stmt))
150 {
151 loads.insert(l);
152 }
153 else if (const StoreStmt* s = SVFUtil::dyn_cast<StoreStmt>(stmt))
154 {
155 stores.insert(s);
156 }
157 }
158 }
159 }
160 }
161
162 for (Set<const LoadStmt*>::const_iterator lit = loads.begin(), elit = loads.end(); lit != elit; ++lit)
163 {
164 const LoadStmt* load = *lit;
165 for (Set<const StoreStmt*>::const_iterator sit = stores.begin(), esit = stores.end(); sit != esit; ++sit)
166 {
167 const StoreStmt* store = *sit;
168 if(SVFUtil::isa<GlobalICFGNode>(load->getICFGNode()) || SVFUtil::isa<GlobalICFGNode>(store->getICFGNode()))
169 continue;
170 if(mhp->mayHappenInParallelInst(load->getICFGNode(),store->getICFGNode()) && pta->alias(load->getRHSVarID(),store->getLHSVarID()))
171 if(lsa->isProtectedByCommonLock(load->getICFGNode(),store->getICFGNode()) == false)
172 outs() << SVFUtil::bugMsg1("race pair(") << " store: " << store->toString() << ", load: " << load->toString() << SVFUtil::bugMsg1(")") << "\n";
173 }
174 }
175}
