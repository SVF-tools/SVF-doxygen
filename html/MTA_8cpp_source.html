160 // * Check (1) write-write race
161 // * (2) write-read race
162 // * (3) read-read race
163 // * when two memory access may-happen in parallel and are not protected by the same lock
164 // * (excluding global constraints because they are initialized before running the main function)
165 // */
166 void MTA::detect(SVFModule* module)
167 {
168 
169  DBOUT(DGENERAL, outs() << pasMsg("Starting Race Detection\n"));
170 
171  LoadSet loads;
172  StoreSet stores;
173  SVFIR* pag = SVFIR::getPAG();
174 
175  Set<const SVFInstruction*> needcheckinst;
176  // Add symbols for all of the functions and the instructions in them.
177  for (const SVFFunction* F : module->getFunctionSet())
178  {
179  // collect and create symbols inside the function body
180  for (const SVFBasicBlock* svfbb : F->getBasicBlockList())
181  {
182  for (const SVFInstruction* svfInst : svfbb->getInstructionList())
183  {
184 
185  for(const SVFStmt* stmt : pag->getSVFStmtList(pag->getICFG()->getICFGNode(svfInst)))
186  {
187  if (SVFUtil::isa<LoadStmt>(stmt))
188  {
189  loads.insert(svfInst);
190  }
191  else if (SVFUtil::isa<StoreStmt>(stmt))
192  {
193  stores.insert(svfInst);
194  }
195  }
196  }
197  }
198  }
199 
200  for (LoadSet::const_iterator lit = loads.begin(), elit = loads.end(); lit != elit; ++lit)
201  {
202  const SVFInstruction* load = *lit;
203  bool loadneedcheck = false;
204  for (StoreSet::const_iterator sit = stores.begin(), esit = stores.end(); sit != esit; ++sit)
205  {
206  const SVFInstruction* store = *sit;
207 
208  loadneedcheck = true;
209  needcheckinst.insert(store);
210  }
211  if (loadneedcheck)
212  needcheckinst.insert(load);
213  }
214 
215  outs() << "HP needcheck: " << needcheckinst.size() << "\n";
216 }
