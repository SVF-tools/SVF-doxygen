80 bool PTACallGraphNode::isReachableFromProgEntry() const
81 {
82  std::stack<const PTACallGraphNode*> nodeStack;
83  NodeBS visitedNodes;
84  nodeStack.push(this);
85  visitedNodes.set(getId());
86 
87  while (nodeStack.empty() == false)
88  {
89  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
90  nodeStack.pop();
91 
92  if (SVFUtil::isProgEntryFunction(node->getFunction()))
93  return true;
94 
95  for (const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
96  {
97  PTACallGraphEdge* edge = *it;
98  if (visitedNodes.test_and_set(edge->getSrcID()))
99  nodeStack.push(edge->getSrcNode());
100  }
101  }
102 
103  return false;
104 }

136 PTACallGraphEdge* PTACallGraph::hasGraphEdge(PTACallGraphNode* src, PTACallGraphNode* dst,PTACallGraphEdge::CEDGEK kind, CallSiteID csId) const
137 {
138  PTACallGraphEdge edge(src,dst,kind,csId);
139  PTACallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
140  PTACallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
141  if (outEdge && inEdge)
142  {
143  assert(outEdge == inEdge && "edges not match");
144  return outEdge;
145  }
146  else
147  return nullptr;
148 }

286 bool PTACallGraph::isReachableBetweenFunctions(const SVFFunction* srcFn, const SVFFunction* dstFn) const
287 {
288  PTACallGraphNode* dstNode = getCallGraphNode(dstFn);
289 
290  std::stack<const PTACallGraphNode*> nodeStack;
291  NodeBS visitedNodes;
292  nodeStack.push(dstNode);
293  visitedNodes.set(dstNode->getId());
294 
295  while (nodeStack.empty() == false)
296  {
297  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
298  nodeStack.pop();
299 
300  if (node->getFunction() == srcFn)
301  return true;
302 
303  for (CallGraphEdgeConstIter it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
304  {
305  PTACallGraphEdge* edge = *it;
306  if (visitedNodes.test_and_set(edge->getSrcID()))
307  nodeStack.push(edge->getSrcNode());
308  }
309  }
310 
311  return false;
312 }
