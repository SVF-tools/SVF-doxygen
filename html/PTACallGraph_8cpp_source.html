81 bool PTACallGraphNode::isReachableFromProgEntry() const
82 {
83  std::stack<const PTACallGraphNode*> nodeStack;
84  NodeBS visitedNodes;
85  nodeStack.push(this);
86  visitedNodes.set(getId());
87 
88  while (nodeStack.empty() == false)
89  {
90  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
91  nodeStack.pop();
92 
93  if (SVFUtil::isProgEntryFunction(node->getFunction()))
94  return true;
95 
96  for (const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
97  {
98  PTACallGraphEdge* edge = *it;
99  if (visitedNodes.test_and_set(edge->getSrcID()))
100  nodeStack.push(edge->getSrcNode());
101  }
102  }
103 
104  return false;
105 }

116 PTACallGraph::PTACallGraph(const CallGraph& other)
117 {
118  callGraphNodeNum = other.getTotalNodeNum();
119  numOfResolvedIndCallEdge = 0;
120  kind = NormCallGraph;
121 
123  for (const auto& item : other)
124  {
125  const CallGraphNode* cgn = item.second;
126  PTACallGraphNode* callGraphNode = new PTACallGraphNode(cgn->getId(), cgn->getFunction());
127  addGNode(cgn->getId(),callGraphNode);
128  funToCallGraphNodeMap[cgn->getFunction()] = callGraphNode;
129  }
130 
132  for (const auto& item : other.callinstToCallGraphEdgesMap)
133  {
134  const CallICFGNode* cs = item.first;
135  for (const CallGraphEdge* edge : item.second)
136  {
137  PTACallGraphNode* src = getCallGraphNode(edge->getSrcID());
138  PTACallGraphNode* dst = getCallGraphNode(edge->getDstID());
139  CallSiteID csId = addCallSite(cs, dst->getFunction());
140 
141  PTACallGraphEdge* newEdge = new PTACallGraphEdge(src,dst, PTACallGraphEdge::CallRetEdge,csId);
142  newEdge->addDirectCallSite(cs);
143  addEdge(newEdge);
144  callinstToCallGraphEdgesMap[cs].insert(newEdge);
145  }
146  }

160 PTACallGraphEdge* PTACallGraph::hasGraphEdge(PTACallGraphNode* src,
161  PTACallGraphNode* dst,
162  PTACallGraphEdge::CEDGEK kind, CallSiteID csId) const
163 {
164  PTACallGraphEdge edge(src,dst,kind,csId);
165  PTACallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
166  PTACallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
167  if (outEdge && inEdge)
168  {
169  assert(outEdge == inEdge && "edges not match");
170  return outEdge;
171  }
172  else
173  return nullptr;
174 }

275 void PTACallGraph::verifyCallGraph()
276 {
277  CallEdgeMap::const_iterator it = indirectCallMap.begin();
278  CallEdgeMap::const_iterator eit = indirectCallMap.end();
279  for (; it != eit; ++it)
280  {
281  const FunctionSet& targets = it->second;

295 bool PTACallGraph::isReachableBetweenFunctions(const SVFFunction* srcFn, const SVFFunction* dstFn) const
296 {
297  PTACallGraphNode* dstNode = getCallGraphNode(dstFn);
298 
299  std::stack<const PTACallGraphNode*> nodeStack;
300  NodeBS visitedNodes;
301  nodeStack.push(dstNode);
302  visitedNodes.set(dstNode->getId());
303 
304  while (nodeStack.empty() == false)
305  {
306  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
307  nodeStack.pop();
308 
309  if (node->getFunction() == srcFn)
310  return true;
311 
312  for (CallGraphEdgeConstIter it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
313  {
314  PTACallGraphEdge* edge = *it;
315  if (visitedNodes.test_and_set(edge->getSrcID()))
316  nodeStack.push(edge->getSrcNode());
317  }
318  }
319 
320  return false;
321 }
