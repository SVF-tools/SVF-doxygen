97 u32_t ConsExeState::hash() const
98 {
99  size_t bas = ExeState::hash();
100  size_t h = getVarToVal().size() * 2;
101  SVF::Hash<SVF::u32_t> hf;
102  for (const auto &t: getVarToVal())
103  {
104  h ^= hf(t.first) + 0x9e3779b9 + (h << 6) + (h >> 2);
105  h ^= hf(t.second.id()) + 0x9e3779b9 + (h << 6) + (h >> 2);
106  }
107 
108  size_t h2 = getVarToVal().size() * 2;
109 
110  for (const auto &t: getLocToVal())
111  {
112  h2 ^= hf(t.first) + 0x9e3779b9 + (h2 << 6) + (h2 >> 2);
113  h2 ^= hf(t.second.id()) + 0x9e3779b9 + (h2 << 6) + (h2 >> 2);
114  }
115  SVF::Hash<std::pair<SVF::u32_t, SVF::u32_t>> pairH;
116 
117  return pairH(std::make_pair(bas, pairH(std::make_pair(h, h2))));

145 bool ConsExeState::joinWith(const SVF::ConsExeState &rhs)
146 {
147  bool changed = ExeState::joinWith(rhs);
148  for (const auto &rhsItem: rhs._varToVal)
149  {
150  auto it = _varToVal.find(rhsItem.first);
151  // Intersection - lhs and rhs have the same var id
152  if (it != getVarToVal().end())
153  {
154  if (it->second.isTop() || rhsItem.second.isTop())
155  {
156  if (assign(it->second, SingleAbsValue::topConstant()))
157  changed = true;
158  }
159  else if (it->second.isBottom())
160  {
161  if (assign(it->second, rhsItem.second))
162  changed = true;
163  }
164  else if (rhsItem.second.isBottom())
165  {
166 
167  }
168  else
169  {
170  if (!eq(it->second, rhsItem.second))
171  {
172  if (assign(it->second, SingleAbsValue::topConstant()))
173  changed = true;
174  }
175  }
176  }
177  else
178  {
179  _varToVal.emplace(rhsItem.first, rhsItem.second);
180  changed = true;
181  }
182  }
183 
184  for (const auto &rhsItem: rhs._locToVal)
185  {
186  auto it = _locToVal.find(rhsItem.first);
187  // Intersection - lhs and rhs have the same var id
188  if (it != getLocToVal().end())
189  {
190  if (it->second.isTop() || rhsItem.second.isTop())
191  {
192  if (assign(it->second, SingleAbsValue::topConstant()))
193  changed = true;
194  }
195  else if (it->second.isBottom())
196  {
197  if (assign(it->second, rhsItem.second))
198  changed = true;
199  }
200  else if (rhsItem.second.isBottom())
201  {
202 
203  }
204  else
205  {

446 bool ConsExeState::applySelect(u32_t res, u32_t cond, u32_t top, u32_t fop)
447 {
448  if (inVarToValTable(top) && inVarToValTable(fop) && inVarToValTable(cond))
449  {
450  SingleAbsValue &tExpr = (*this)[top], &fExpr = (*this)[fop], &condExpr = (*this)[cond];
451 
452  return assign((*this)[res], ite(condExpr == 1, tExpr, fExpr));
453  }
454  else if (inVarToAddrsTable(top) && inVarToAddrsTable(fop) && inVarToValTable(cond))
455  {
456  SingleAbsValue &condExpr = (*this)[cond];
457  if (condExpr.is_numeral())
458  {
459  getAddrs(res) = condExpr.is_zero() ? getAddrs(fop) : getAddrs(top);
460  }
461  }
462  return false;
463 }

465 bool ConsExeState::applyPhi(u32_t res, std::vector<u32_t> &ops)
466 {
467  for (u32_t i = 0; i < ops.size(); i++)
468  {
469  NodeID curId = ops[i];
470  if (inVarToValTable(curId))
471  {
472  const SingleAbsValue &cur = (*this)[curId];
473  if (!inVarToValTable(res))
474  {
475  (*this)[res] = cur;
476  }
477  else
478  {
479  (*this)[res].join_with(cur);
480  }
481  }
482  else if (inVarToAddrsTable(curId))
483  {
484  const Addrs &cur = getAddrs(curId);
485  if (!inVarToAddrsTable(res))
486  {
487  getAddrs(res) = cur;
488  }
489  else
490  {
491  getAddrs(res).join_with(cur);
492  }
493  }
494  }
495  return true;
496 }

559 bool ConsExeState::lessThanVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
560 {
