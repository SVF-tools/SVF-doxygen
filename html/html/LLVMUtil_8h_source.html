131 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallBase* cs)
132 {
133  const PointerType *refType = nullptr;
134  const SVFInstruction* svfcall = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(cs);
135  CallSite svfcs = SVFUtil::getSVFCallSite(svfcall);
136  // Case 1: heap object held by *argument, we should get its element type.
137  if (SVFUtil::isHeapAllocExtCallViaArg(svfcs))
138  {
139  int argPos = SVFUtil::getHeapAllocHoldingArgPosition(svfcs);
140  const Value* arg = cs->getArgOperand(argPos);
141  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
142  refType = SVFUtil::dyn_cast<PointerType>(getPtrElementType(argType));
143  }
144  // Case 2: heap/static object held by return value.
145  else
146  {
147  assert((SVFUtil::isStaticExtCall(svfcs) || SVFUtil::isHeapAllocExtCallViaRet(svfcs))
148  && "Must be heap alloc via ret, or static allocation site");
149  refType = SVFUtil::dyn_cast<PointerType>(cs->getType());
150  }
151  assert(refType && "Allocated object must be held by a pointer-typed value.");
152  return refType;
153 }
