42  typedef struct Symbol
43  {
44  Kind kind: 8;
45  Attribute attribute: 16;
46  VariableAttribute variableAttribute: 8;
47 
49  Symbol() : kind(0), attribute(0), variableAttribute(0) {}
50 
52  Symbol(const u32_t& num) : kind(num & 0xFF), attribute((num >> 8 ) & 0xFFFF), variableAttribute((num >> 24)) {}
53 
55  operator u32_t()
56  {
57  static_assert(sizeof(struct Symbol)==sizeof(u32_t), "sizeof(struct Symbol)!=sizeof(u32_t)");
58  u32_t num = 0;
59  num += this->variableAttribute << 24;
60  num += this->attribute << 8;
61  num += this->kind;
62  return num;
63  }
64 
65  operator u32_t() const
66  {
67  static_assert(sizeof(struct Symbol)==sizeof(u32_t), "sizeof(struct Symbol)!=sizeof(u32_t)");

71  num += this->kind;

73  }

75  bool operator<(const Symbol& rhs)
76  {
77  return u32_t(*this) < u32_t(rhs);

79 
80  void operator=(const u32_t& i)
81  {
82  this->kind = EdgeKindMask & i;

131  struct SymbolVectorHash
132  {
133  size_t operator()(const std::vector<Symbol> &v) const
134  {
135  size_t h = v.size();
136 
137  SymbolHash hf;
138  for (const Symbol &t : v)
139  {
140  h ^= hf(t) + 0x9e3779b9 + (h << 6) + (h >> 2);
141  }
142 
143  return h;
144  }
145  };

391 template<class Data>
392 class CFLFIFOWorkList
393 {
394  typedef GrammarBase::SymbolSet<Data> DataSet;

396 public:
397  CFLFIFOWorkList() {}
398 
399  ~CFLFIFOWorkList() {}
400 
401  inline bool empty() const
402  {
403  return data_list.empty();
404  }
405 
406  inline bool find(Data data) const
407  {
408  return (data_set.find(data) == data_set.end() ? false : true);
409  }
410 
414  inline bool push(Data data)
415  {
416  if (data_set.find(data) == data_set.end())
417  {
418  data_list.push_back(data);
419  data_set.insert(data);
420  return true;
421  }
422  else
423  return false;
424  }
425 
429  inline Data pop()
430  {
431  assert(!empty() && "work list is empty");
432  Data data = data_list.front();
433  data_list.pop_front();
434  data_set.erase(data);
435  return data;
436  }
437 
441  inline void clear()
442  {
443  data_list.clear();
444  data_set.clear();
445  }
446 
447 private:
448  DataSet data_set;
449  DataDeque data_list;
