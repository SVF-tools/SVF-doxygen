82 void CFGNormalizer::ebnf_bin(CFLGrammar *grammar)
83 {
84  GrammarBase::SymbolMap<GrammarBase::Symbol, GrammarBase::Productions> new_grammar = {};
85  std::string tempStr = "";
86 
87  for(auto head : grammar->getRawProductions())
88  {
89  for(auto rule: head.second)
90  {
91 
92  GrammarBase::Production long_run = rule;
93  long_run.erase(long_run.begin());
94  auto it = grammar->getRawProductions().at(head.first).find(rule);
95  grammar->getRawProductions().at(head.first).erase(it);
96  grammar->getRawProductions()[head.first].insert(long_run);
97  }
98  }
99 
100  auto rawProductions = grammar->getRawProductions();
101 
102  for(auto itr : rawProductions)
103  {
104  auto head = *(grammar->getRawProductions().find(itr.first));
105  for(auto rule: head.second)
106  {
107  if (rule.size() < 3) continue;
108 
109  GrammarBase::Production long_run = rule;
110  GrammarBase::Symbol first = long_run[0];
111  long_run.erase(long_run.begin());
112  auto it = grammar->getRawProductions()[head.first].find(rule);
113  grammar->getRawProductions()[head.first].erase(it);
114  GrammarBase::Symbol X = check_head(new_grammar, long_run);
115  if (X == u32_t(-1))
116  {
117  X = check_head(grammar->getRawProductions(), long_run);
118  }
119  if ((X == u32_t(-1)) == false)
120  {
121  rule = {first, X};
122  grammar->getRawProductions()[head.first].insert(rule);
123  }
124  else
125  {
126  tempStr = "X";
127  std::ostringstream ss;
128  ss << grammar->num_generator();
129  tempStr.append(ss.str());
134  Set<GrammarBase::VariableAttribute> variableAttributeSet = {};
135  for (unsigned i = 0; i < long_run.size(); i++)
136  {
137  GrammarBase::VariableAttribute variableAttribute = long_run[i].variableAttribute;
138  if ( variableAttribute != 0)
139  {
140  variableAttributeSet.insert(variableAttribute);
141  }
142  }
143  if ( variableAttributeSet.size() == 1)
144  {
145  tempStr += "_";
146  tempStr += char(*variableAttributeSet.begin());
147  }
148  GrammarBase::Symbol tempSym = grammar->insertNonTerminalSymbol(tempStr);
149  rule = {first, tempSym};
150  grammar->getRawProductions()[head.first].insert(rule);
151  X = tempSym;
152  }
153  new_grammar[X] = {};
154  GrammarBase::Production temp_p = long_run;
155  GrammarBase::Symbol RHX;
156  if (long_run.size() ==2)
157  {
158  new_grammar[X].insert(temp_p);
159  long_run.clear();
160  }
161  else
162  {
163  new_grammar[X].insert(long_run);
164  RHX = X;
165  }
166  while (long_run.size() > 2)
167  {
168  first = long_run[0];
169  GrammarBase::Production prev_rule = long_run;
170  long_run.erase(long_run.begin());
171 
172  X = RHX;
173  temp_p = long_run;
174 
175  RHX = check_head(new_grammar, long_run);
176  if (RHX == u32_t(-1))
177  {
178  RHX = check_head(grammar->getRawProductions(), long_run);
179  }
180  if(RHX == u32_t(-1))
181  {
182  tempStr = "X";
183  std::ostringstream ss;
184  ss << grammar->num_generator();
185  tempStr.append(ss.str());
186  Set<GrammarBase::VariableAttribute> variableAttributeSet = {};
187  for (unsigned i = 0; i < long_run.size(); i++)
188  {
189  GrammarBase::VariableAttribute variableAttribute = long_run[i].variableAttribute;
190  if ( variableAttribute != 0)
191  {
192  variableAttributeSet.insert(variableAttribute);
193  }
194  }
195  if ( variableAttributeSet.size() == 1)
196  {
197  tempStr += "_";
198  tempStr += char(*variableAttributeSet.begin());
199  }
200  GrammarBase::Symbol tempSym = grammar->insertNonTerminalSymbol(tempStr);
201  auto it = new_grammar[X].find(prev_rule);
202  new_grammar[X].erase(it);
203  new_grammar[X].insert({first, tempSym});
204  new_grammar[tempSym].insert(long_run);
205  RHX = tempSym;
206  }
207  }
208  }
209  }
210  for (auto new_head : new_grammar)
211  {
212  for (auto prod : new_head.second)
213  {
214  auto it = grammar->getRawProductions()[new_head.first].find(prod);
215  if (it == grammar->getRawProductions()[new_head.first].end())
216  {
217  grammar->getRawProductions()[new_head.first].insert(prod);
218  }
219  }

317  GrammarBase::Productions productions;
318  //GrammarBase::Productions Originalproductions = symbolToProductionsPair.second;

343 void CFGNormalizer::ebnfSignReplace(char sign, CFLGrammar *grammar)
344 {
347  SVF::Map<std::string, std::string> newProductions;
348  std::string tempNonterminal = "X";
349 
350  for (auto &symbolToProductionsPair : grammar->getRawProductions())
351  {
352  GrammarBase::Productions productions = symbolToProductionsPair.second;
353  for (auto ebnfProduction : symbolToProductionsPair.second)
354  {
355  size_t i = 1;
356  while (i < ebnfProduction.size())
357  {
358  s32_t signGroupStart = -1;
359  if (grammar->kind2Str(ebnfProduction[i].kind) == std::string(1, sign))
360  {
362  assert(i != 1 && "sign in grammar associate with no symbol");
363  if (grammar->kind2Str(ebnfProduction[i - 1].kind) != std::string(1, ')'))
364  {
365  signGroupStart = i - 1;
366  }
368  else
369  {
370  signGroupStart = ebnfBracketMatch(ebnfProduction, i, grammar);
371  }
372  std::string groupString = "";
373  for (size_t j = signGroupStart; j < i; j++)
374  {
375  groupString.append(grammar->kind2Str(ebnfProduction[j].kind));
376  groupString.append(" ");
377  }
378  groupString.append(grammar->kind2Str(ebnfProduction[i].kind));
379  if (newProductions.find(groupString) != newProductions.end())
380  {
381  productions.erase(ebnfProduction);
382  ebnfProduction.erase(ebnfProduction.begin() + signGroupStart, ebnfProduction.begin() + i + 1);
383  ebnfProduction.insert(ebnfProduction.begin() + signGroupStart, grammar->str2Symbol(newProductions[groupString]));
384  productions.insert(ebnfProduction);
385  }
386  else if ( (signGroupStart == 1) && (i == ebnfProduction.size() -1))
387  {
388  newProductions[groupString] = grammar->kind2Str(ebnfProduction[0].kind);
389  productions.erase(ebnfProduction);
390  ebnfProduction.erase(ebnfProduction.begin() + signGroupStart, ebnfProduction.begin() + i + 1);
391 
392  }
393  else
394  {
395  tempNonterminal = "X";
396  std::ostringstream ss;
397  ss << grammar->num_generator();
398  tempNonterminal.append(ss.str());
399  GrammarBase::Symbol tempSym = grammar->insertNonTerminalSymbol(tempNonterminal);
400  productions.erase(ebnfProduction);
401  ebnfProduction.erase(ebnfProduction.begin() + signGroupStart, ebnfProduction.begin() + i + 1);
402  ebnfProduction.insert(ebnfProduction.begin() + signGroupStart, tempSym);
403  newProductions[groupString] = tempNonterminal;
404  productions.insert(ebnfProduction);
405  }
406 
407  i = signGroupStart;
408  }
409  i++;
410  }
411  }
412  symbolToProductionsPair.second = productions;
413  }
414  for(auto rep: newProductions)
415  {
417  std::string new_nonterminal = rep.second;
418  GrammarBase::Production temp_list = {grammar->str2Symbol(new_nonterminal), grammar->str2Symbol("epsilon")};
419  grammar->getRawProductions()[grammar->str2Symbol(new_nonterminal)].insert(temp_list);
421  temp_list = {grammar->str2Symbol(new_nonterminal)};
422  if (sign == '*' || sign == '?')
423  {
425  GrammarBase::Production E = strTrans(rep.first, grammar);
426  GrammarBase::Production withoutSign = {};
427  if (sign == '*')
428  {
429  for (auto &word : E)
430  {
431  if (word != grammar->str2Symbol("*") && word != grammar->str2Symbol("(") && word != grammar->str2Symbol(")"))
432  {
433  withoutSign.push_back(word);
434  }
435  }
436  withoutSign.push_back(grammar->str2Symbol(rep.second));
437  }
438  if (sign == '?')
439  {
440  for (auto &word : E)
441  {
442  if (word != grammar->str2Symbol("?") && word != grammar->str2Symbol("(") && word != grammar->str2Symbol(")"))
443  {
444  withoutSign.push_back(word);
445  }
446  }
447  }
448  temp_list.insert(temp_list.end(), withoutSign.begin(), withoutSign.end());
449  }
450  grammar->getRawProductions()[grammar->str2Symbol(new_nonterminal)].insert(temp_list);
451  }
452 }

454 GrammarBase::Production CFGNormalizer::strTrans(std::string LHS, CFLGrammar *grammar)
455 {
456  GrammarBase::Production prod = {};
457  std::smatch matches;
458  std::regex LHSReg("\\s*(.*)");
459  std::string delimiter;
460  size_t pos;
461  std::string word;
462  std::regex_search(LHS, matches, LHSReg);
463  LHS = matches.str(1);
464  delimiter = " ";
465  while ((pos = LHS.find(delimiter)) != std::string::npos)
466  {
467  word = LHS.substr(0, pos);
468  LHS.erase(0, pos + delimiter.length());
469  prod.push_back(grammar->str2Symbol(word));
470  }
471  prod.push_back(grammar->str2Symbol(LHS));
472  return prod;
473 }
