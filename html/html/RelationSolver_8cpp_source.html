33 IntervalExeState RelationSolver::bilateral(IntervalExeState domain, Z3Expr phi, u32_t descend_check)
34 {
36  IntervalExeState upper = domain.top();
37  IntervalExeState lower = domain.bottom();
38  u32_t meets_in_a_row = 0;
39  z3::solver solver = Z3Expr::getSolver();
40  z3::params p(Z3Expr::getContext());
42  p.set(":timeout", static_cast<unsigned>(1)); // in milliseconds
43  solver.set(p);
44  IntervalExeState consequence;
45 
47  while (lower != upper)
48  {
49  if (meets_in_a_row == descend_check)
50  {
51  consequence = lower;
52  }
53  else
54  {
55  consequence = abstract_consequence(lower, upper, domain);
56  }
58  Z3Expr rhs = !(gamma_hat(consequence, domain));
59  solver.push();
60  solver.add(phi.getExpr() && rhs.getExpr());
61  Map<u32_t, double> solution;
62  z3::check_result checkRes = solver.check();
64  if (checkRes == z3::sat)
65  {
66  z3::model m = solver.get_model();
67  for(u32_t i = 0; i < m.size(); i++)
68  {
69  z3::func_decl v = m[i];
70  assert(v.arity() == 0);
71  solution.emplace(std::stoi(v.name().str()), m.get_const_interp(v).get_numeral_int());
72  }
73  for(const auto &item: domain.getVarToVal())
74  {
75  if (solution.find(item.first) == solution.end())
76  {
77  solution.emplace(item.first, 0);
78  }
79  }
80  solver.pop();
81  IntervalExeState newLower = domain.bottom();
82  newLower.joinWith(lower);
83  IntervalExeState rhs = beta(solution, domain);
84  newLower.joinWith(rhs);
85  lower = newLower;
86  meets_in_a_row = 0;
87  }
88  else
89  {
90  solver.pop();
91  if (checkRes == z3::unknown)
92  {
94  if (solver.reason_unknown() == "timeout")
95  return upper;
96  }
97  IntervalExeState newUpper = domain.top();
98  newUpper.meetWith(upper);
99  newUpper.meetWith(consequence);
100  upper = newUpper;
101  meets_in_a_row += 1;
102  }
103  }
104  return upper;
105 }

107 IntervalExeState RelationSolver::RSY(IntervalExeState domain, const Z3Expr &phi)
108 {
109  IntervalExeState lower = domain.bottom();
110  z3::solver &solver = Z3Expr::getSolver();
111  z3::params p(Z3Expr::getContext());
113  p.set(":timeout", static_cast<unsigned>(600)); // in milliseconds
114  solver.set(p);
115  while (1)
116  {
117  Z3Expr rhs = !(gamma_hat(lower, domain));
118  solver.push();
119  solver.add(phi.getExpr() && rhs.getExpr());
120  Map<u32_t, double> solution;
121  z3::check_result checkRes = solver.check();
123  if (checkRes == z3::sat)
124  {
125  z3::model m = solver.get_model();
126  for(u32_t i = 0; i < m.size(); i++)
127  {
128  z3::func_decl v = m[i];
129  if (v.arity() != 0)
130  continue;
131  solution.emplace(std::stoi(v.name().str()), m.get_const_interp(v).get_numeral_int());
132  }
133  for(const auto &item: domain.getVarToVal())
134  {
135  if (solution.find(item.first) == solution.end())
136  {
137  solution.emplace(item.first, 0);
138  }
139  }
140  solver.pop();
141  IntervalExeState newLower = domain.bottom();
142  newLower.joinWith(lower);
143  IntervalExeState rhs = beta(solution, domain);
144  newLower.joinWith(rhs);
145  lower = newLower;
146  }

162 IntervalExeState RelationSolver::abstract_consequence(IntervalExeState &lower, IntervalExeState &upper, IntervalExeState &domain)
163 {
164  /*Returns the "abstract consequence" of lower and upper.
165 
166  The abstract consequence must be a superset of lower and *NOT* a
167  superset of upper.
168 
169  Note that this is a fairly "simple" abstract consequence, in that it
170  sets only one variable to a non-top interval. This improves performance
171  of the SMT solver in many cases. In certain cases, other choices for
172  the abstract consequence will lead to better algorithm performance.*/
173 
174  for (auto it = domain.getVarToVal().begin(); it != domain.getVarToVal().end(); ++it)
175  {
176  IntervalExeState proposed = domain.top();
177  proposed[it->first] = lower[it->first];
178  if (!(proposed >= upper))
180  {
181  return proposed;
182  }
183  }
184  return lower;
185 }

187 Z3Expr RelationSolver::gamma_hat(IntervalExeState &exeState)
188 {
189  Z3Expr res(Z3Expr::getContext().bool_val(true));
190  for (auto &item: exeState.getVarToVal())
191  {
192  if (item.second.isBottom()) return Z3Expr::getContext().bool_val(false);
193  if (item.second.isTop()) continue;
194  Z3Expr v = Z3Expr::getContext().int_const(std::to_string(item.first).c_str());
195  res = (res && v >= (int) item.second.lb().getNumeral() &&
196  v <= (int) item.second.ub().getNumeral()).simplify();
197  }
198  return res;
199 }

202 {

204  for (auto &item: exeState.getVarToVal())

206  IntervalValue interval = alpha[item.first];
207  if (interval.isBottom()) return Z3Expr::getContext().bool_val(false);
208  if (interval.isTop()) continue;
209  Z3Expr v = Z3Expr::getContext().int_const(std::to_string(item.first).c_str());
210  res = (res && v >= (int) interval.lb().getNumeral() &&
211  v <= (int) interval.ub().getNumeral()).simplify();
212  }
213  return res;
214 }
