42 void ICFGBuilder::build(SVFModule* svfModule)
43 {
44  DBOUT(DGENERAL, outs() << pasMsg("\t Building ICFG ...\n"));
45  // Add the unique global ICFGNode at the entry of a program (before the main method).
46  icfg->addGlobalICFGNode();
47 
48  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
49  {
50  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
51  {
52  const Function *fun = &*F;
53  const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
54  if (SVFUtil::isExtCall(svffun))
55  continue;
56  WorkList worklist;
57  processFunEntry(fun,worklist);
58  processFunBody(worklist);
59  processFunExit(fun);
60  }
61  }
62  connectGlobalToProgEntry(svfModule);
63 }

68 void ICFGBuilder::processFunEntry(const Function* fun, WorkList& worklist)
69 {
70  FunEntryICFGNode* FunEntryICFGNode = icfg->getFunEntryICFGNode(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun));
71  const Instruction* entryInst = &((fun->getEntryBlock()).front());
72  const SVFInstruction* svfentryInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(entryInst);
73 
74  InstVec insts;
75  if (isIntrinsicInst(svfentryInst))
76  getNextInsts(entryInst, insts);
77  else
78  insts.push_back(entryInst);
79  for (InstVec::const_iterator nit = insts.begin(), enit = insts.end();
80  nit != enit; ++nit)
81  {
82  ICFGNode* instNode = getOrAddBlockICFGNode(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(*nit)); //add interprocedural edge
83  icfg->addIntraEdge(FunEntryICFGNode, instNode);
84  worklist.push(*nit);
85  }
86 }

91 void ICFGBuilder::processFunBody(WorkList& worklist)
92 {
93  BBSet visited;
95  while (!worklist.empty())
96  {
97  const Instruction* inst = worklist.pop();
98  if (visited.find(inst) == visited.end())
99  {
100  visited.insert(inst);
101  const SVFInstruction* svfinst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(inst);
102  ICFGNode* srcNode = getOrAddBlockICFGNode(svfinst);
103  if (svfinst->isRetInst())
104  {
105  const SVFFunction* svfFun = svfinst->getFunction();
106  FunExitICFGNode* FunExitICFGNode = icfg->getFunExitICFGNode(svfFun);
107  icfg->addIntraEdge(srcNode, FunExitICFGNode);
108  }
109  InstVec nextInsts;
110  getNextInsts(inst, nextInsts);
111  u32_t branchID = 0;
112  for (InstVec::const_iterator nit = nextInsts.begin(), enit =
113  nextInsts.end(); nit != enit; ++nit)
114  {
115  const Instruction* succ = *nit;
116  const SVFInstruction* svfsucc = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(succ);
117  ICFGNode* dstNode = getOrAddBlockICFGNode(svfsucc);
118  if (isNonInstricCallSite(svfinst))
119  {
120  RetICFGNode* retICFGNode = getRetICFGNode(svfinst);
121  srcNode = retICFGNode;
122  }
123 
124  if (const BranchInst* br = SVFUtil::dyn_cast<BranchInst>(inst))
125  {
126  assert(branchID <= 2 && "if/else has more than two branches?");
127  if(br->isConditional())
128  icfg->addConditionalIntraEdge(srcNode, dstNode, LLVMModuleSet::getLLVMModuleSet()->getSVFValue(br->getCondition()), 1 - branchID);
129  else
130  icfg->addIntraEdge(srcNode, dstNode);
131  }
132  else if (const SwitchInst* si = SVFUtil::dyn_cast<SwitchInst>(inst))
133  {
135  const ConstantInt* condVal = const_cast<SwitchInst*>(si)->findCaseDest(const_cast<BasicBlock*>(succ->getParent()));
137  s64_t val = -1;
138  if (condVal && condVal->getBitWidth() <= 64)
139  val = condVal->getSExtValue();
140  icfg->addConditionalIntraEdge(srcNode, dstNode, LLVMModuleSet::getLLVMModuleSet()->getSVFValue(si->getCondition()),val);
141  }
142  else
143  icfg->addIntraEdge(srcNode, dstNode);
144 
145  worklist.push(succ);
146  branchID++;
147  }
148  }
149  }
150 }

157 void ICFGBuilder::processFunExit(const Function* f)
158 {
159  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(f);
160  FunExitICFGNode* FunExitICFGNode = icfg->getFunExitICFGNode(fun);
161 
162  for (const SVFBasicBlock* svfbb : fun->getBasicBlockList())
163  {
164  for (const SVFInstruction* inst : svfbb->getInstructionList())
165  {
166  if(inst->isRetInst())
167  {
168  ICFGNode* instNode = getOrAddBlockICFGNode(inst);
169  icfg->addIntraEdge(instNode, FunExitICFGNode);
170  }
171  }
172  }

182 InterICFGNode* ICFGBuilder::getOrAddInterBlockICFGNode(const SVFInstruction* inst)
183 {
184  assert(SVFUtil::isCallSite(inst) && "not a call instruction?");
185  assert(SVFUtil::isNonInstricCallSite(inst) && "associating an intrinsic debug instruction with an ICFGNode!");
186  CallICFGNode* callICFGNode = getCallICFGNode(inst);
187  addICFGInterEdges(inst, getCallee(inst)); //creating interprocedural edges
188  return callICFGNode;

194 void ICFGBuilder::addICFGInterEdges(const SVFInstruction* cs, const SVFFunction* callee)
195 {
196  CallICFGNode* callICFGNode = getCallICFGNode(cs);
197  RetICFGNode* retBlockNode = getRetICFGNode(cs);
198 
200  if(callee)
201  {
203  if (isExtCall(callee))
204  {
205  icfg->addIntraEdge(callICFGNode, retBlockNode);
206  }
208  else
209  {
210  FunEntryICFGNode* calleeEntryNode = icfg->getFunEntryICFGNode(callee);
211  FunExitICFGNode* calleeExitNode = icfg->getFunExitICFGNode(callee);
212  icfg->addCallEdge(callICFGNode, calleeEntryNode, cs);
213  icfg->addRetEdge(calleeExitNode, retBlockNode, cs);
214  }
215  }
217  else
218  {
219  icfg->addIntraEdge(callICFGNode, retBlockNode);
220  }
221 }

226 {

236  icfg->addICFGEdge(intraEdge);
