81  inline void mapValueToEdge(const Value *V, SVFStmt *edge)
82  {
83  auto inserted = valueToEdgeMap.emplace(V, SVFStmtSet{edge});
84  if (!inserted.second)
85  {
86  inserted.first->second.emplace(edge);
87  }
88  }
90  inline const MemObj* getMemObj(const Value* val) const
91  {
92  return symInfo->getObj(symInfo->getObjSym(val));
93  }

95 public:

97  {
