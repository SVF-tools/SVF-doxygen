43 // Get environment variables $SVF_DIR and "npm root" through popen() method
44 static std::string GetStdoutFromCommand(const std::string& command)
45 {
46  char buffer[128];
47  std::string result = "";
48  // Open pipe to file
49  FILE* pipe = popen(command.c_str(), "r");
50  if (!pipe)
51  {
52  return "popen failed!";
53  }
54  // read till end of process:
55  while (!feof(pipe))
56  {
57  // use buffer to read and add to result
58  if (fgets(buffer, 128, pipe) != NULL)
59  result += buffer;
60  }
61  pclose(pipe);
62  // remove "\n"
63  result.erase(remove(result.begin(), result.end(), '\n'), result.end());
64  return result;
65 }

92 static cJSON* parseJson(const std::string& path, off_t fileSize)
93 {
94  FILE* file = fopen(path.c_str(), "r");
95  if (!file)
96  {
97  return nullptr;
98  }
99 
100  // allocate memory size matched with file size
101  char* jsonStr = (char*)calloc(fileSize + 1, sizeof(char));
102 
103  // read json string from file
104  u32_t size = fread(jsonStr, sizeof(char), fileSize, file);
105  if (size == 0)
106  {
107  SVFUtil::errs() << SVFUtil::errMsg("\t Wrong ExtAPI.json path!! ")
108  << "The current ExtAPI.json path is: " << path << "\n";
109  assert(false && "Read ExtAPI.json file fails!");
110  return nullptr;
111  }
112  fclose(file);
113 
114  // convert json string to json pointer variable
115  cJSON* root = cJSON_Parse(jsonStr);
116  if (!root)
117  {
118  free(jsonStr);
119  return nullptr;
120  }
121  free(jsonStr);
122  return root;
123 }

126 {
127  if (extOp == nullptr)
128  {
129  extOp = new ExtAPI;
130  }
131  if (root == nullptr)
132  {
133  struct stat statbuf;
134 
135  // Four ways to get ExtAPI.json path
136  // 1. Explicit path provided
137  // 2. default path (get ExtAPI.json path from Util/config.h)

189  std::vector<const char*> argTypes, extType type,
190  bool overwrite_app_function)
191 {
192  assert(root && "Parse the json before adding additional entries");
193  assert(get_type(funName) == EFT_NULL && "Do not add entries that already exist");
194  auto entry = cJSON_CreateObject();
195 
196  // add the signature fields
197  auto returnTypeString = cJSON_CreateString(returnType);
198  cJSON_AddItemToObject(entry, "return", returnTypeString);
199 
200  std::stringstream ss;
201  ss << "(";
202  for (auto str : argTypes)
203  ss << str << ",";
204  std::string formattedArgs = ss.str();
205  if (formattedArgs.back() == ',')
206  formattedArgs.back() = ')';
207  else
208  formattedArgs.append(")");
209  auto argsString = cJSON_CreateString(formattedArgs.c_str());
210  cJSON_AddItemToObject(entry, "argument", argsString);
211 
212  // add the type field
213  auto typeString = cJSON_CreateString(extType_toString(type).c_str());
214  cJSON_AddItemToObject(entry, JSON_OPT_FUNCTIONTYPE, typeString);
215 
216  // add the 'overwrite_app_function' field
217  auto overwriteBool = cJSON_CreateNumber(overwrite_app_function ? 1 : 0);
218  cJSON_AddItemToObject(entry, JSON_OPT_OVERWRITE, overwriteBool);
219

221  cJSON_AddItemToObject(root, funName, entry);

351 ExtAPI::Operand ExtAPI::getBasicOperation(cJSON* obj)
352 {
353  Operand basicOp;
354  if (strstr(obj->string, "AddrStmt") != NULL)
355  basicOp.setType(ExtAPI::OperationType::Addr);
356  else if (strstr(obj->string, "CopyStmt") != NULL)
357  basicOp.setType(ExtAPI::OperationType::Copy);
358  else if (strstr(obj->string, "LoadStmt") != NULL)
359  basicOp.setType(ExtAPI::OperationType::Load);
360  else if (strstr(obj->string, "StoreStmt") != NULL)
361  basicOp.setType(ExtAPI::OperationType::Store);
362  else if (strstr(obj->string, "GepStmt") != NULL)
363  basicOp.setType(ExtAPI::OperationType::Gep);
364  else if (strstr(obj->string, "ReturnStmt") != NULL)
365  basicOp.setType(ExtAPI::OperationType::Return);
366  else if (strstr(obj->string, "Rb_tree_ops") != NULL)
367  basicOp.setType(ExtAPI::OperationType::Rb_tree_ops);
368  else if (strstr(obj->string, "memcpy_like") != NULL)
369  basicOp.setType(ExtAPI::OperationType::Memcpy_like);
370  else if (strstr(obj->string, "memset_like") != NULL)
371  basicOp.setType(ExtAPI::OperationType::Memset_like);
372  else
373  assert(false && "Unknown operation type");
374 
375  cJSON* value = obj->child;
376  while (value)
377  {
378  if (strcmp(value->string, "src") == 0)
379  basicOp.setSrcValue(value->valuestring);
380  else if (strcmp(value->string, "dst") == 0)
381  basicOp.setDstValue(value->valuestring);
382  else if (strcmp(value->string, "offset") == 0 || strcmp(value->string, "size") == 0)
383  basicOp.setOffsetOrSizeStr(value->valuestring);
384  else
385  assert(false && "Unknown operation value");
386  value = value->next;
387  }
388  return basicOp;
389 }

391 // Get all operations of an extern function
392 ExtAPI::ExtFunctionOps ExtAPI::getExtFunctionOps(const SVFFunction* extFunction)
393 {
394  ExtAPI::ExtFunctionOps extFunctionOps;
395  extFunctionOps.setExtFunName(extFunction->getName());
396  if (!is_sameSignature(extFunction))
397  return extFunctionOps;
398 
399  auto it = extFunToOps.find(extFunction->getName());
400  if (it != extFunToOps.end())
401  return it->second;
402 
403  extFunctionOps.setExtFunName(extFunction->getName());
404  cJSON* item = get_FunJson(extFunction->getName());
405  if (item != nullptr)
406  {
407  cJSON* obj = item->child;
408  // Get the first operation of the function
409  obj = obj->next->next->next->next;
410  while (obj)
411  {
412  ExtOperation operation;
413  if (obj->type == cJSON_Object || obj->type == cJSON_Array)
414  {
415  if (strstr(obj->string, "CallStmt") != NULL)
416  {
417  extFunctionOps.setCallStmtNum(extFunctionOps.getCallStmtNum() + 1);
418  operation.setCallOp(true);
419  cJSON* value = obj->child;
420  while (value)
421  {
422  if (strcmp(value->string, "callee_name") == 0)
423  operation.setCalleeName(value->valuestring);
424  else if (strcmp(value->string, "callee_return") == 0)
425  operation.setCalleeReturn(value->valuestring);
426  else if (strcmp(value->string, "callee_arguments") == 0)
427  operation.setCalleeArguments(value->valuestring);
428  else
429  operation.getCalleeOperands().push_back(getBasicOperation(value));
430  value = value->next;
431  }
432  }
433  else if (strstr(obj->string, "CondStmt") != NULL)
434  {
435  operation.setConOp(true);
436  obj = obj->child;
437  assert(strcmp(obj->string, "Condition") == 0 && "Unknown operation type");
438  operation.setConOp(obj->valuestring);
439  obj = obj->next;
440  assert(strcmp(obj->string, "TrueBranch") == 0 && "Unknown operation type");
441  cJSON* value = obj->child;
442  while (value)
443  {
444  operation.getTrueBranchOperands().push_back(getBasicOperation(value));
445  value = value->next;
446  }
447  obj = obj->next;
448  assert(strcmp(obj->string, "FalseBranch") == 0&& "Unknown operation type");
449  value = obj->child;
450  while (value)
451  {
452  operation.getFalseBranchOperands().push_back(getBasicOperation(value));
453  value = value->next;
454  }
455  }
456  else
457  operation.setBasicOp(getBasicOperation(obj));
458  }
459  obj = obj->next;
460  extFunctionOps.getOperations().push_back(operation);
461  }
462  }
463  extFunToOps[extFunction->getName()] = extFunctionOps;
464  return extFunctionOps;
465 }

625 // Does (F) have the same return type(pointer or nonpointer) and same number of
626 // arguments
627 bool ExtAPI::is_sameSignature(const SVFFunction* F)
628 {
629  cJSON* item = get_FunJson(F->getName());
630  // If return type is pointer
631  bool isPointer = false;
632  // The number of arguments
633  u32_t argNum = 0;
634  if (item != nullptr)
635  {
636  // Get the "return" attribute
637  cJSON* obj = item->child;
638  if (strlen(obj->valuestring) == 0) // e.g. "return": ""
639  assert(false && "'return' should not be empty!");
640  // If "return": "..." includes "*", the return type of extern function is a pointer
641  if (strstr(obj->valuestring, "*") != NULL)
642  isPointer = true;
643  // Get the "arguments" attribute
644  obj = obj->next;
645  if (strlen(obj->valuestring) == 0) // e.g. "arguments": "",
646  assert(false && "'arguments' should not be empty!");
647  // If "arguments": "()", the number of arguments is 0, otherwise, number >= 1;
648  if (strcmp(obj->valuestring, "()") != 0)
649  {
650  argNum++;
651  for (u32_t i = 0; i < strlen(obj->valuestring); i++)
652  if (obj->valuestring[i] == ',') // Calculate the number of arguments based on the number of ","
653  argNum++;
654  }
655  }
