56 // Get environment variables $SVF_DIR and "npm root" through popen() method
57 static std::string GetStdoutFromCommand(const std::string& command)
58 {
59  char buffer[128];
60  std::string result = "";
61  // Open pipe to file
62  FILE* pipe = popen(command.c_str(), "r");
63  if (!pipe)
64  {
65  return "popen failed!";
66  }
67  // read till end of process:
68  while (!feof(pipe))
69  {
70  // use buffer to read and add to result
71  if (fgets(buffer, 128, pipe) != NULL)
72  result += buffer;
73  }
74  pclose(pipe);
75  // remove "\n"
76  result.erase(remove(result.begin(), result.end(), '\n'), result.end());
77  return result;
78 }

192 // Should (F) be considered "external" (either not defined in the program
193 // or a user-defined version of a known alloc or no-op)?
194 bool ExtAPI::is_ext(const SVFFunction* F)
195 {
196  assert(F && "Null SVFFunction* pointer");

198  return true;
