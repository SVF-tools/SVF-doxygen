146 private:
148  int compare(const WTOCycleDepth &other) const
149  {
150  if (this == &other)
151  {
152  return 0; // 0 - equals
153  }
154 
155  auto this_it = begin();
156  auto other_it = other.begin();
157  while (this_it != end())
158  {
159  if (other_it == other.end())
160  {
161  return 1; // `this` is inside `other`
162  }
163  else if (*this_it == *other_it)
164  {
165  ++this_it;
166  ++other_it;
167  }
168  else
169  {
170  return 2; // not comparable
171  }
172  }
173  if (other_it == other.end())
174  {
175  return 0; // equals
176  }
177  else
178  {
179  return -1; // `other` is inside `this`
180  }
181  }

187  WTOCycleDepth operator^(const WTOCycleDepth &other) const
189  {
190  WTOCycleDepth res;
191  for (auto this_it = begin(), other_it = other.begin();
192  this_it != end() && other_it != other.end();
193  ++this_it, ++other_it)
194  {
195  if (*this_it == *other_it)
196  {
197  res.push_back(*this_it);
198  }
199  else
200  {
201  break;
202  }
203  }
204  return res;
205  }

768  virtual CycleDepthNumber visit(const CFBasicBlockNode *node, WTOCompRefList &partition)
769  {
770  CycleDepthNumber head(0);
771  CycleDepthNumber min(0);
772  bool loop;
773 
774  push(node);
775  _num += CycleDepthNumber(1);
776  head = _num;
777  setCDN(node, head);
778  loop = false;
779  for (auto it = node->getOutEdges().begin(), et = node->getOutEdges().end(); it != et; ++it)
780  {
781  const CFBasicBlockNode *succ = (*it)->getDstNode();
782  CycleDepthNumber succ_dfn = getCDN(succ);
783  if (succ_dfn == CycleDepthNumber(0))
784  {
785  min = visit(succ, partition);
786  }
787  else
788  {
789  min = succ_dfn;
790  }
791  if (min <= head)
792  {
793  head = min;
794  loop = true;
795  }
796  }
797  if (head == getCDN(node))
798  {
799  setCDN(node, UINT_MAX);
800  const CFBasicBlockNode *element = pop();
801  if (loop)
802  {
803  while (element != node)
804  {
805  setCDN(element, 0);
806  element = pop();
807  }
808  partition.push_front(component(node));
809  }
810  else
811  {
812  partition.push_front(newNode(node));
813  }
814  }
815  return head;
816  }
