41 SVFVar::SVFVar(const SVFValue* val, NodeID i, PNODEK k) :
42  GenericPAGNodeTy(i,k), value(val)
43 {
44  assert( ValNode <= k && k <= DummyObjNode && "new SVFIR node kind?");
45  switch (k)
46  {
47  case ValNode:
48  case GepValNode:
49  {
50  assert(val != nullptr && "value is nullptr for ValVar or GepValNode");
51  isPtr = val->getType()->isPointerTy();
52  break;
53  }
54  case RetNode:
55  {
56  assert(val != nullptr && "value is nullptr for RetNode");
57  isPtr = SVFUtil::cast<SVFFunction>(val)->getReturnType()->isPointerTy();
58  break;
59  }
60  case VarargNode:
61  case DummyValNode:
62  {
63  isPtr = true;
64  break;
65  }
66  case ObjNode:
67  case GepObjNode:
68  case FIObjNode:
69  case DummyObjNode:
70  {
71  isPtr = true;
72  if(val)
73  isPtr = val->getType()->isPointerTy();
74  break;
75  }
76  }
77 }

79 bool SVFVar::isIsolatedNode() const
80 {
81  if (getInEdges().empty() && getOutEdges().empty())
82  return true;
83  else if (isConstDataOrAggDataButNotNullPtr())
84  return true;
85  else if (value && SVFUtil::isa<SVFFunction>(value))
86  return SVFUtil::cast<SVFFunction>(value)->isIntrinsic();
87  else
88  return false;
89 }
