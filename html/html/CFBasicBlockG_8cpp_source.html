63 CFBasicBlockEdge* CFBasicBlockGraph::getCFBasicBlockEdge(const CFBasicBlockNode *src, const CFBasicBlockNode *dst)
64 {
65  CFBasicBlockEdge *edge = nullptr;
66  size_t counter = 0;
67  for (auto iter = src->OutEdgeBegin();
68  iter != src->OutEdgeEnd(); ++iter)
69  {
70  if ((*iter)->getDstID() == dst->getId())
71  {
72  counter++;
73  edge = (*iter);
74  }
75  }
76  assert(counter <= 1 && "there's more than one edge between two nodes");
77  return edge;
78 }

90 const CFBasicBlockEdge* CFBasicBlockGraph::getOrAddCFBasicBlockEdge(CFBasicBlockNode *src, CFBasicBlockNode *dst)
91 {
92  if (const CFBasicBlockEdge *edge = getCFBasicBlockEdge(src, dst)) return edge;
93  CFBasicBlockEdge *edge = new CFBasicBlockEdge(src, dst);
94  bool added1 = edge->getDstNode()->addIncomingEdge(edge);
95  bool added2 = edge->getSrcNode()->addOutgoingEdge(edge);
96  (void) added1;
97  (void) added2;
98  assert(added1 && added2 && "edge not added??");
99  _totalCFBasicBlockEdge++;
100  return edge;
