44 void MTAStat::performThreadCallGraphStat(ThreadCallGraph* tcg)
45 {
46  u32_t numOfForkEdge = 0;
47  u32_t numOfJoinEdge = 0;
48  u32_t numOfIndForksite = 0;
49  u32_t numOfIndForkEdge = 0;
50  for (ThreadCallGraph::CallSiteSet::const_iterator it = tcg->forksitesBegin(), eit = tcg->forksitesEnd(); it != eit; ++it)
51  {
52  bool indirectfork = false;
53  const SVFFunction* spawnee = SVFUtil::dyn_cast<SVFFunction>(tcg->getThreadAPI()->getForkedFun((*it)->getCallSite()));
54  if(spawnee==nullptr)
55  {
56  numOfIndForksite++;
57  indirectfork = true;
58  }
59  for (ThreadCallGraph::ForkEdgeSet::const_iterator cgIt = tcg->getForkEdgeBegin(*it), ecgIt =
60  tcg->getForkEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
61  {
62  numOfForkEdge++;
63  if(indirectfork)
64  numOfIndForkEdge++;
65  }
66  }
67 
68  for (ThreadCallGraph::CallSiteSet::const_iterator it = tcg->joinsitesBegin(), eit = tcg->joinsitesEnd(); it != eit; ++it)
69  {
70  for (ThreadCallGraph::JoinEdgeSet::const_iterator cgIt = tcg->getJoinEdgeBegin(*it), ecgIt =
71  tcg->getJoinEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
72  {
73  numOfJoinEdge++;
74  }
75  }
76 
77 
78  PTNumStatMap.clear();
79  PTNumStatMap["NumOfForkSite"] = tcg->getNumOfForksite();
80  PTNumStatMap["NumOfForkEdge"] = numOfForkEdge;
81  PTNumStatMap["NumOfJoinEdge"] = numOfJoinEdge;
82  PTNumStatMap["NumOfJoinSite"] = tcg->getNumOfJoinsite();
83  PTNumStatMap["NumOfIndForkSite"] = numOfIndForksite;
84  PTNumStatMap["NumOfIndForkEdge"] = numOfIndForkEdge;
85  PTNumStatMap["NumOfIndCallEdge"] = tcg->getNumOfResolvedIndCallEdge();
86 
87  SVFUtil::outs() << "\n****Thread Call Graph Statistics****\n";
88  PTAStat::printStat();
89 }

112 void MTAStat::performMHPPairStat(MHP* mhp, LockAnalysis* lsa)
113 {
114 
115  SVFIR* pag = SVFIR::getPAG();
116  if(Options::AllPairMHP)
117  {
118  Set<const SVFInstruction*> instSet1;
119  Set<const SVFInstruction*> instSet2;
120  SVFModule* mod = mhp->getTCT()->getSVFModule();
121  for (const SVFFunction* fun : mod->getFunctionSet())
122  {
123  if(SVFUtil::isExtCall(fun))
124  continue;
125  if(!mhp->isConnectedfromMain(fun))
126  continue;
127  for (SVFFunction::const_iterator bit = fun->begin(), ebit = fun->end(); bit != ebit; ++bit)
128  {
129  const SVFBasicBlock* bb = *bit;
130  for (SVFBasicBlock::const_iterator ii = bb->begin(), eii = bb->end(); ii != eii; ++ii)
131  {
132  const SVFInstruction* inst = *ii;
133  for(const SVFStmt* stmt : pag->getSVFStmtList(pag->getICFG()->getICFGNode(inst)))
134  {
135  if(SVFUtil::isa<LoadStmt>(stmt))
136  {
137  instSet1.insert(inst);
138  }
139  else if(SVFUtil::isa<StoreStmt>(stmt))
140  {
141  instSet1.insert(inst);
142  instSet2.insert(inst);
143  }
144  }
145 
146  }
147  }
148  }
149 
150 
151  for(Set<const SVFInstruction*>::const_iterator it1 = instSet1.begin(), eit1 = instSet1.end(); it1!=eit1; ++it1)
152  {
153  for(Set<const SVFInstruction*>::const_iterator it2 = instSet2.begin(), eit2 = instSet2.end(); it2!=eit2; ++it2)
154  {
155  mhp->mayHappenInParallel(*it1,*it2);
156  }
157  }
158  }
159 
160 
161  generalNumMap.clear();
162  PTNumStatMap.clear();
163  timeStatMap.clear();
164  PTNumStatMap["TotalMHPQueries"] = mhp->numOfTotalQueries;
165  PTNumStatMap["NumOfMHPPairs"] = mhp->numOfMHPQueries;
166  PTNumStatMap["TotalLockQueries"] = lsa->numOfTotalQueries;
167  PTNumStatMap["NumOfLockedPairs"] = lsa->numOfLockedQueries;
168  PTNumStatMap["NumOfCxtLocks"] = lsa->getNumOfCxtLocks();
169  PTNumStatMap["NumOfNewSVFGEdges"] = MTASVFGBuilder::numOfNewSVFGEdges;
170  PTNumStatMap["NumOfRemovedEdges"] = MTASVFGBuilder::numOfRemovedSVFGEdges;
171  PTNumStatMap["NumOfRemovedPTS"] = MTASVFGBuilder::numOfRemovedPTS;
172  timeStatMap["InterlevAnaTime"] = mhp->interleavingTime;
173  timeStatMap["LockAnaTime"] = lsa->lockTime;
174  timeStatMap["InterlevQueryTime"] = mhp->interleavingQueriesTime;
175  timeStatMap["LockQueryTime"] = lsa->lockQueriesTime;
176  timeStatMap["MHPAnalysisTime"] = MHPTime;
177  timeStatMap["MFSPTATime"] = FSMPTATime;
178 
179  SVFUtil::outs() << "\n****MHP Stmt Pairs Statistics****\n";
180  PTAStat::printStat();
181 }
