38 PTACallGraph* CallGraphBuilder::buildCallGraph(SVFModule* svfModule)
39 {
41  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
42  {
43  callgraph->addCallGraphNode(*F);
44  }
45 
47  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
48  {
49  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
50  {
51  for (const SVFInstruction* inst : svfbb->getInstructionList())
52  {
53  if (SVFUtil::isNonInstricCallSite(inst))
54  {
55  if(const SVFFunction* callee = getCallee(inst))
56  {
57  const CallICFGNode* callBlockNode = icfg->getCallICFGNode(inst);
58  callgraph->addDirectCallGraphEdge(callBlockNode,*F,callee);
59  }
60  }
61  }
62  }
63  }
64 
65  return callgraph;
66 }

68 PTACallGraph* ThreadCallGraphBuilder::buildThreadCallGraph(SVFModule* svfModule)
69 {
70 
71  buildCallGraph(svfModule);
72 
73  ThreadCallGraph* cg = dyn_cast<ThreadCallGraph>(callgraph);
74  assert(cg && "not a thread callgraph?");
75 
76  ThreadAPI* tdAPI = ThreadAPI::getThreadAPI();
77  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
78  {
79  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
80  {
81  for (const SVFInstruction* inst : svfbb->getInstructionList())
82  {
83  if (tdAPI->isTDFork(inst))
84  {
85  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
86  cg->addForksite(cs);
87  const SVFFunction* forkee = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun(inst));
88  if (forkee)
89  {
90  cg->addDirectForkEdge(cs);
91  }
92  // indirect call to the start routine function
93  else
94  {
95  cg->addThreadForkEdgeSetMap(cs,nullptr);
96  }
97  }
98  else if (tdAPI->isHareParFor(inst))
99  {
100  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
101  cg->addParForSite(cs);
102  const SVFFunction* taskFunc = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getTaskFuncAtHareParForSite(inst));
103  if (taskFunc)
104  {
105  cg->addDirectParForEdge(cs);
106  }
107  // indirect call to the start routine function
108  else
109  {
110  cg->addHareParForEdgeSetMap(cs,nullptr);
111  }
112  }
113  }
114  }
115  }
116  // record join sites
117  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
118  {
119  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
120  {
121  for (const SVFInstruction* inst : svfbb->getInstructionList())
122  {
123  if (tdAPI->isTDJoin(inst))
124  {
125  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
126  cg->addJoinsite(cs);
127  }
128  }
129  }
130  }
131 
132  return cg;
133 }
