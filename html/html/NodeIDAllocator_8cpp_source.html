338  }

343  // however as they will be in the one int regardless..
344  if (regionNumObjects < NATIVE_INT_SIZE)

346  for (NodeID o : regionsObjects[region]) nodeMap[o] = allocCounter++;
347  continue;
348  }

431 double *NodeIDAllocator::Clusterer::getDistanceMatrix(const std::vector<std::pair<const PointsTo *, unsigned>> pointsToSets,
432  const size_t numObjects, const Map<NodeID, unsigned> &nodeMap,
433  double &distanceMatrixTime)
434 {
435  const double clkStart = PTAStat::getClk(true);
436  size_t condensedSize = (numObjects * (numObjects - 1)) / 2;
437  double *distMatrix = new double[condensedSize];
438  for (size_t i = 0; i < condensedSize; ++i) distMatrix[i] = numObjects * numObjects;
439 
440  // TODO: maybe use machine epsilon?
441  // For reducing distance due to extra occurrences.
442  // Can differentiate ~9999 occurrences.
443  double occurrenceEpsilon = 0.0001;
444 
445  for (const std::pair<const PointsTo *, unsigned> &ptsOcc : pointsToSets)
446  {
447  const PointsTo *pts = ptsOcc.first;
448  assert(pts != nullptr);
449  const unsigned occ = ptsOcc.second;
450 
451  // Distance between each element of pts.
452  unsigned distance = requiredBits(*pts) / NATIVE_INT_SIZE;
453 
454  // Use a vector so we can index into pts.
455  std::vector<NodeID> ptsVec;
456  for (const NodeID o : *pts) ptsVec.push_back(o);
457  for (size_t i = 0; i < ptsVec.size(); ++i)
458  {
459  const NodeID oi = ptsVec[i];
460  const Map<NodeID, unsigned>::const_iterator moi = nodeMap.find(oi);
461  assert(moi != nodeMap.end());
462  for (size_t j = i + 1; j < ptsVec.size(); ++j)
463  {
464  const NodeID oj = ptsVec[j];
465  const Map<NodeID, unsigned>::const_iterator moj = nodeMap.find(oj);
466  assert(moj != nodeMap.end());
467  double &existingDistance = distMatrix[condensedIndex(numObjects, moi->second, moj->second)];
468 
469  // Subtract extra occurrenceEpsilon to make upcoming logic simpler.
470  // When existingDistance is never whole, it is always between two distances.
471  if (distance < existingDistance) existingDistance = distance - occurrenceEpsilon;
472 
473  if (distance == std::ceil(existingDistance))
474  {
475  // We have something like distance == x, existingDistance == x - e, for some e < 1
476  // (potentially even set during this iteration).
477  // So, the new distance is an occurrence the existingDistance being tracked, it just
478  // had some reductions because of multiple occurences.
479  // If there is not room within this distance to reduce more (increase priority),
480  // just ignore it. TODO: maybe warn?
481  if (existingDistance - occ * occurrenceEpsilon > std::floor(existingDistance))
482  {
483  existingDistance -= occ * occurrenceEpsilon;
484  }
485  else
486  {
487  // Reached minimum.
488  existingDistance = std::floor(existingDistance) + occurrenceEpsilon;
489  }
490  }
491  }
492  }
493 
494  }
495 
496  const double clkEnd = PTAStat::getClk(true);
497  distanceMatrixTime += (clkEnd - clkStart) / TIMEINTERVAL;
498 
499  return distMatrix;
500 }

502 void NodeIDAllocator::Clusterer::traverseDendrogram(std::vector<NodeID> &nodeMap, const int *dendrogram, const size_t numObjects, unsigned &allocCounter, Set<int> &visited, const int index, const std::vector<NodeID> &regionNodeMap)
503 {
504  if (visited.find(index) != visited.end()) return;
505  visited.insert(index);
506 
507  int left = dendrogram[index - 1];
508  if (left < 0)
509  {
510  // Reached a leaf.
511  // -1 because the items start from 1 per fastcluster (TODO).
512  nodeMap[regionNodeMap[std::abs(left) - 1]] = allocCounter;
513  ++allocCounter;
514  }
515  else
516  {
517  traverseDendrogram(nodeMap, dendrogram, numObjects, allocCounter, visited, left, regionNodeMap);
518  }
519 
520  // Repeat for the right child.
521  int right = dendrogram[(numObjects - 1) + index - 1];
522  if (right < 0)
523  {
524  nodeMap[regionNodeMap[std::abs(right) - 1]] = allocCounter;
525  ++allocCounter;
526  }
527  else
528  {
529  traverseDendrogram(nodeMap, dendrogram, numObjects, allocCounter, visited, right, regionNodeMap);
530  }

533 std::vector<NodeID> NodeIDAllocator::Clusterer::regionObjects(const Map<NodeID, Set<NodeID>> &graph, size_t numObjects, size_t &numLabels)
534 {
535  unsigned label = UINT_MAX;
536  std::vector<NodeID> labels(numObjects, UINT_MAX);
537  Set<NodeID> labelled;
538  for (const Map<NodeID, Set<NodeID>>::value_type &oos : graph)
539  {
540  const NodeID o = oos.first;
541  if (labels[o] != UINT_MAX) continue;
542  std::queue<NodeID> bfsQueue;
543  bfsQueue.push(o);
544  ++label;
545  while (!bfsQueue.empty())
546  {
547  const NodeID o = bfsQueue.front();
548  bfsQueue.pop();
549  if (labels[o] != UINT_MAX)
550  {
551  assert(labels[o] == label);
552  continue;
553  }
554 
555  labels[o] = label;
556  Map<NodeID, Set<NodeID>>::const_iterator neighboursIt = graph.find(o);
557  assert(neighboursIt != graph.end());
558  for (const NodeID neighbour : neighboursIt->second) bfsQueue.push(neighbour);
559  }
560  }
561 
562  // The remaining objects have no relation with others: they get their own label.
563  for (size_t o = 0; o < numObjects; ++o)
564  {
565  if (labels[o] == UINT_MAX) labels[o] = ++label;
566  }
567 
568  numLabels = label + 1;
569 
570  return labels;
571 }

573 void NodeIDAllocator::Clusterer::evaluate(const std::vector<NodeID> &nodeMap, const Map<PointsTo, unsigned> pointsToSets, Map<std::string, std::string> &stats, bool accountForOcc)
574 {
575  u64_t totalTheoretical = 0;
576  u64_t totalOriginalSbv = 0;
577  u64_t totalOriginalBv = 0;
578  u64_t totalNewSbv = 0;
579  u64_t totalNewBv = 0;
580 
581  for (const Map<PointsTo, unsigned>::value_type &ptsOcc : pointsToSets)
582  {
583  const PointsTo &pts = ptsOcc.first;
584  const unsigned occ = ptsOcc.second;
585  if (pts.count() == 0) continue;
586 
587  u64_t theoretical = requiredBits(pts) / NATIVE_INT_SIZE;
588  if (accountForOcc) theoretical *= occ;
589 
590  // Check number of words for original SBV.
591  Set<unsigned> words;
592  // TODO: nasty hardcoding.
593  for (const NodeID o : pts) words.insert(o / 128);
594  u64_t originalSbv = words.size() * 2;
595  if (accountForOcc) originalSbv *= occ;
596 
597  // Check number of words for original BV.
598  NodeID min = UINT_MAX;
599  NodeID max = 0;
600  for (NodeID o : pts)
601  {
602  if (o < min) min = o;
603  if (o > max) max = o;
604  }
605  words.clear();
606  for (NodeID b = min; b <= max; ++b)

651 std::pair<hclust_fast_methods, std::vector<NodeID>> NodeIDAllocator::Clusterer::determineBestMapping(
652  const std::vector<std::pair<hclust_fast_methods, std::vector<NodeID>>> &candidates,
653  Map<PointsTo, unsigned> pointsToSets, const std::string &evalSubtitle, double &evalTime)
654 {
655  // In case we're not comparing anything, set to first "candidate".
656  std::pair<hclust_fast_methods, std::vector<NodeID>> bestMapping = candidates[0];
657  // Number of bits required for the best candidate.
658  size_t bestWords = std::numeric_limits<size_t>::max();
659  if (evalSubtitle != "" || Options::ClusterMethod == HCLUST_METHOD_SVF_BEST)
660  {
661  for (const std::pair<hclust_fast_methods, std::vector<NodeID>> &candidate : candidates)
662  {
663  Map<std::string, std::string> candidateStats;
664  hclust_fast_methods candidateMethod = candidate.first;
665  std::string candidateMethodName = SVFUtil::hclustMethodToString(candidateMethod);
666  std::vector<NodeID> candidateMapping = candidate.second;
667 
668  // TODO: parameterise final arg.
669  const double clkStart = PTAStat::getClk(true);
670  evaluate(candidateMapping, pointsToSets, candidateStats, true);
671  const double clkEnd = PTAStat::getClk(true);
672  evalTime += (clkEnd - clkStart) / TIMEINTERVAL;
673  printStats(evalSubtitle + ": candidate " + candidateMethodName, candidateStats);
674 
675  size_t candidateWords = 0;
676  if (Options::PtType == PointsTo::SBV) candidateWords = std::stoull(candidateStats[NewSbvNumWords]);
677  else if (Options::PtType == PointsTo::CBV) candidateWords = std::stoull(candidateStats[NewBvNumWords]);
678  else assert(false && "Clusterer::cluster: unsupported BV type for clustering.");
679 
680  if (candidateWords < bestWords)
681  {
682  bestWords = candidateWords;
683  bestMapping = candidate;
684  }
685  }
686  }
687 
688  return bestMapping;
