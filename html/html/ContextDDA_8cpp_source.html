53 const CxtPtSet& ContextDDA::computeDDAPts(const CxtVar& var)
54 {
55 
56  resetQuery();
57  LocDPItem::setMaxBudget(Options::CxtBudget);
58 
59  NodeID id = var.get_id();
60  PAGNode* node = getPAG()->getGNode(id);
61  CxtLocDPItem dpm = getDPIm(var, getDefSVFGNode(node));
62 
63  // start DDA analysis
64  DOTIMESTAT(double start = DDAStat::getClk(true));
65  const CxtPtSet& cpts = findPT(dpm);
66  DOTIMESTAT(ddaStat->_AnaTimePerQuery = DDAStat::getClk(true) - start);
67  DOTIMESTAT(ddaStat->_TotalTimeOfQueries += ddaStat->_AnaTimePerQuery);
68 
69  if(isOutOfBudgetQuery() == false)
70  unionPts(var,cpts);
71  else
72  handleOutOfBudgetDpm(dpm);
73 
74  if (this->printStat())
75  DOSTAT(stat->performStatPerQuery(id));
76  DBOUT(DGENERAL, stat->printStatPerQuery(id,getBVPointsTo(getPts(var))));
77  return this->getPts(var);
78 }

93 void ContextDDA::handleOutOfBudgetDpm(const CxtLocDPItem& dpm)
94 {
95 
96  DBOUT(DGENERAL,outs() << "~~~Out of budget query, downgrade to flow sensitive analysis \n");
97  flowDDA->computeDDAPts(dpm.getCurNodeID());
98  const PointsTo& flowPts = flowDDA->getPts(dpm.getCurNodeID());
99  CxtPtSet cxtPts;
100  for(PointsTo::iterator it = flowPts.begin(), eit = flowPts.end(); it!=eit; ++it)
101  {
102  ContextCond cxt;

104  cxtPts.set(var);
105  }
106  updateCachedPointsTo(dpm,cxtPts);
107  unionPts(dpm.getCondVar(),cxtPts);
108  addOutOfBudgetDpm(dpm);

233 bool ContextDDA::handleBKCondition(CxtLocDPItem& dpm, const SVFGEdge* edge)
234 {
235  _client->handleStatement(edge->getSrcNode(), dpm.getCurNodeID());
236 
237  if (edge->isCallVFGEdge())
238  {
240  if(CallSiteID csId = getCSIDAtCall(dpm,edge))
241  {
242 
243  if(isEdgeInRecursion(csId))
244  {
245  DBOUT(DDDA,outs() << "\t\t call edge " << getPTACallGraph()->getCallerOfCallSite(csId)->getName() <<
246  "=>" << getPTACallGraph()->getCalleeOfCallSite(csId)->getName() << "in recursion \n");
247  popRecursiveCallSites(dpm);
248  }
249  else
250  {
251  if (dpm.matchContext(csId) == false)
252  {
253  DBOUT(DDDA, outs() << "\t\t context not match, edge "
254  << edge->getDstID() << " --| " << edge->getSrcID() << " \t");
255  DBOUT(DDDA, dumpContexts(dpm.getCond()));
256  return false;
257  }
258 
259  DBOUT(DDDA, outs() << "\t\t match contexts ");
260  DBOUT(DDDA, dumpContexts(dpm.getCond()));
261  }
262  }
263  }
264 
265  else if (edge->isRetVFGEdge())
266  {
268  if(CallSiteID csId = getCSIDAtRet(dpm,edge))
269  {
270 
271  if(isEdgeInRecursion(csId))
272  {
273  DBOUT(DDDA,outs() << "\t\t return edge " << getPTACallGraph()->getCalleeOfCallSite(csId)->getName() <<
274  "=>" << getPTACallGraph()->getCallerOfCallSite(csId)->getName() << "in recursion \n");
275  popRecursiveCallSites(dpm);
276  }
277  else
278  {
281  if (dpm.getCond().containCallStr(csId))
282  {
283  outOfBudgetQuery = true;
284  SVFUtil::writeWrnMsg("Call site ID is contained in call string. Is this a recursion?");
285  return false;
286  }
287  else
288  {
289  assert(dpm.getCond().containCallStr(csId) ==false && "contain visited call string ??");
290  if(dpm.pushContext(csId))
291  {
292  DBOUT(DDDA, outs() << "\t\t push context ");
293  DBOUT(DDDA, dumpContexts(dpm.getCond()));
294  }
295  else
296  {
297  DBOUT(DDDA, outs() << "\t\t context is full ");
298  DBOUT(DDDA, dumpContexts(dpm.getCond()));
299  }
300  }
301  }
302  }
303  }
304 
305  return true;
306 }

313 {
