42 void AndersenSCD::solveWorklist()
43 {
44  // Initialize the nodeStack via a whole SCC detection
45  // Nodes in nodeStack are in topological order by default.
46  NodeStack& nodeStack = SCCDetect();
47 
48  for (NodeID nId : sccCandidates)
49  pushIntoWorklist(nId);
50  sccCandidates.clear();
51 
52  // propagate point-to sets
53  while (!nodeStack.empty())
54  {
55  NodeID nodeId = nodeStack.top();
56  nodeStack.pop();
57 
58  if (sccRepNode(nodeId) == nodeId)
59  {
60  collapsePWCNode(nodeId);
61 
62  if (isInWorklist(nodeId))
63  // push the rep of node into worklist
64  pushIntoWorklist(nodeId);
65 
66  double propStart = stat->getClk();
67  // propagate pts through copy and gep edges
68  ConstraintNode* node = consCG->getConstraintNode(nodeId);
69  handleCopyGep(node);
70  double propEnd = stat->getClk();
71  timeOfProcessCopyGep += (propEnd - propStart) / TIMEINTERVAL;
72 
73  collapseFields();
74  }
75  }
76 
77  // New nodes will be inserted into workList during processing.
78  while (!isWorklistEmpty())
79  {
80  NodeID nodeId = popFromWorklist();
81 
82  double insertStart = stat->getClk();
83  // add copy edges via processing load or store edges
84  ConstraintNode* node = consCG->getConstraintNode(nodeId);
85  handleLoadStore(node);
86  double insertEnd = stat->getClk();
87  timeOfProcessLoadStore += (insertEnd - insertStart) / TIMEINTERVAL;
88  }
89 }

161 void AndersenSCD::processPWC(ConstraintNode* rep)
162 {
163  NodeID repId = rep->getId();
164 
165  NodeSet pwcNodes;
166  for (NodeID nId : getSCCDetector()->subNodes(repId))
167  pwcNodes.insert(nId);
168 
169  WorkList tmpWorkList;
170  for (NodeID subId : pwcNodes)
171  if (isInWorklist(subId))
172  tmpWorkList.push(subId);
173 
174  while (!tmpWorkList.empty())
175  {
176  NodeID nodeId = tmpWorkList.pop();
177  computeDiffPts(nodeId);
178 
179  if (!getDiffPts(nodeId).empty())
180  {
181  ConstraintNode *node = consCG->getConstraintNode(nodeId);
182  for (ConstraintEdge* edge : node->getCopyOutEdges())
183  {
184  bool changed = processCopy(nodeId, edge);
185  if (changed && pwcNodes.find(edge->getDstID()) != pwcNodes.end())
186  tmpWorkList.push(edge->getDstID());
187  }
188  for (ConstraintEdge* edge : node->getGepOutEdges())
189  {
190  if (GepCGEdge *gepEdge = SVFUtil::dyn_cast<GepCGEdge>(edge))
191  {
192  bool changed = processGep(nodeId, gepEdge);
193  if (changed && pwcNodes.find(edge->getDstID()) != pwcNodes.end())
194  tmpWorkList.push(edge->getDstID());
195  }
196  }
197  }
198  }
199 }

206 void AndersenSCD::handleLoadStore(ConstraintNode* node)
207 {
208  double insertStart = stat->getClk();
209 
210  NodeID nodeId = node->getId();
211  // handle load
212  for (ConstraintNode::const_iterator it = node->outgoingLoadsBegin(),
213  eit = node->outgoingLoadsEnd(); it != eit; ++it)
214  for (PointsTo::iterator piter = getPts(nodeId).begin(), epiter =
215  getPts(nodeId).end(); piter != epiter; ++piter)
216  {
217  NodeID ptd = *piter;
218  if (processLoad(ptd, *it))
219  {
220  reanalyze = true;
221  }
222  }
223 
224  // handle store
225  for (ConstraintNode::const_iterator it = node->incomingStoresBegin(),
226  eit = node->incomingStoresEnd(); it != eit; ++it)
227  for (PointsTo::iterator piter = getPts(nodeId).begin(), epiter =
228  getPts(nodeId).end(); piter != epiter; ++piter)
