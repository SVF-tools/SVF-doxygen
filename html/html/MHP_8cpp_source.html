75 void MHP::analyzeInterleaving()
76 {
77  for (const std::pair<const NodeID, TCTNode*>& tpair : *tct)
78  {
79  const CxtThread& ct = tpair.second->getCxtThread();
80  NodeID rootTid = tpair.first;
81  const SVFFunction* routine = tct->getStartRoutineOfCxtThread(ct);
82  const SVFInstruction* svfInst = routine->getEntryBlock()->front();
83  CxtThreadStmt rootcts(rootTid, ct.getContext(), svfInst);
84 
85  addInterleavingThread(rootcts, rootTid);
86  updateAncestorThreads(rootTid);
87  updateSiblingThreads(rootTid);
88 
89  while (!cxtStmtList.empty())
90  {
91  CxtThreadStmt cts = popFromCTSWorkList();
92  const SVFInstruction* curInst = cts.getStmt();
93  DBOUT(DMTA, outs() << "-----\nMHP analysis root thread: " << rootTid << " ");
94  DBOUT(DMTA, cts.dump());
95  DBOUT(DMTA, outs() << "current thread interleaving: < ");
96  DBOUT(DMTA, dumpSet(getInterleavingThreads(cts)));
97  DBOUT(DMTA, outs() << " >\n-----\n");
98 
100  if (!tct->isCandidateFun(curInst->getParent()->getParent()))
101  {
102  handleNonCandidateFun(cts);
103  }
105  else
106  {
107  if (isTDFork(curInst))
108  {
109  handleFork(cts, rootTid);
110  }
111  else if (isTDJoin(curInst))
112  {
113  handleJoin(cts, rootTid);
114  }
115  else if (SVFUtil::isCallSite(curInst) && !isExtCall(curInst))
116  {
117  handleCall(cts, rootTid);
118  PTACallGraph::FunctionSet callees;
119  if (!tct->isCandidateFun(getCallee(curInst, callees)))
120  handleIntra(cts);
121  }
122  else if (curInst->isRetInst())
123  {
124  handleRet(cts);
125  }
126  else
127  {
128  handleIntra(cts);
129  }
130  }
131  }
132  }
133 
135  updateNonCandidateFunInterleaving();
136 
137  if (Options::PrintInterLev)
138  printInterleaving();
139 }

144 void MHP::updateNonCandidateFunInterleaving()
145 {
146  SVFModule* module = tct->getSVFModule();
147  for (const SVFFunction* fun : module->getFunctionSet())
148  {
149  if (!tct->isCandidateFun(fun) && !isExtCall(fun))
150  {
151  const SVFInstruction* entryinst = fun->getEntryBlock()->front();
152  if (!hasThreadStmtSet(entryinst))
153  continue;
154 
155  const CxtThreadStmtSet& tsSet = getThreadStmtSet(entryinst);
156 
157  for (const CxtThreadStmt& cts : tsSet)
158  {
159  const CallStrCxt& curCxt = cts.getContext();
160 
161  for (const SVFBasicBlock* svfbb : fun->getBasicBlockList())
162  {
163  for (const SVFInstruction* svfInst : svfbb->getInstructionList())
164  {
165  if (svfInst == entryinst)
166  continue;
167  CxtThreadStmt newCts(cts.getTid(), curCxt, svfInst);
168  threadStmtToTheadInterLeav[newCts] |= threadStmtToTheadInterLeav[cts];
169  instToTSMap[svfInst].insert(newCts);
170  }
171  }
172  }
173  }

233 void MHP::handleJoin(const CxtThreadStmt& cts, NodeID rootTid)
234 {
235 
236  const SVFInstruction* call = cts.getStmt();
237  const CallStrCxt& curCxt = cts.getContext();
238 
239  assert(isTDJoin(call));
240 
241  NodeBS joinedTids = getDirAndIndJoinedTid(curCxt, call);
242  if (!joinedTids.empty())
243  {
244  if (fja->hasJoinLoop(call))
245  {
246  std::vector<const SVFBasicBlock*> exitbbs;
247  call->getFunction()->getExitBlocksOfLoop(call->getParent(), exitbbs);
248  while (!exitbbs.empty())
249  {
250  const SVFBasicBlock* eb = exitbbs.back();
251  exitbbs.pop_back();
252  const SVFInstruction* svfEntryInst = eb->front();
253  CxtThreadStmt newCts(cts.getTid(), curCxt, svfEntryInst);
254  addInterleavingThread(newCts, cts);
255  if (hasJoinInSymmetricLoop(curCxt, call))
256  rmInterleavingThread(newCts, joinedTids, call);
257  }
258  }
259  else
260  {
261  rmInterleavingThread(cts, joinedTids, call);
262  DBOUT(DMTA, outs() << "\n\t match join site " << call->toString() << " for thread " << rootTid << "\n");
263  }
264  }
267  else
268  {
269  if (fja->hasJoinLoop(call))
270  {
271  std::vector<const SVFBasicBlock*> exitbbs;
272  call->getFunction()->getExitBlocksOfLoop(call->getParent(), exitbbs);
273  while (!exitbbs.empty())
274  {
275  const SVFBasicBlock* eb = exitbbs.back();
276  exitbbs.pop_back();
277  const SVFInstruction* svfEntryInst = eb->front();
278  CxtThreadStmt newCts(cts.getTid(), cts.getContext(), svfEntryInst);

317 void MHP::handleRet(const CxtThreadStmt& cts)
318 {
319  PTACallGraphNode* curFunNode = tcg->getCallGraphNode(cts.getStmt()->getParent()->getParent());
320  for (PTACallGraphEdge* edge : curFunNode->getInEdges())
321  {
322  if (SVFUtil::isa<ThreadForkEdge, ThreadJoinEdge>(edge))
323  continue;
324  for (PTACallGraphEdge::CallInstSet::const_iterator cit = (edge)->directCallsBegin(),
325  ecit = (edge)->directCallsEnd();
326  cit != ecit; ++cit)
327  {
328  CallStrCxt newCxt = cts.getContext();
329  if (matchCxt(newCxt, (*cit)->getCallSite(), curFunNode->getFunction()))
330  {
331  const InstVec& nextInsts = (*cit)->getCallSite()->getSuccInstructions();
332  for (const auto& ni : nextInsts)
333  {
334  CxtThreadStmt newCts(cts.getTid(), newCxt, ni);
335  addInterleavingThread(newCts, cts);
336  }
337  }
338  }
339  for (PTACallGraphEdge::CallInstSet::const_iterator cit = (edge)->indirectCallsBegin(),
340  ecit = (edge)->indirectCallsEnd();
341  cit != ecit; ++cit)
342  {
343  CallStrCxt newCxt = cts.getContext();
344  if (matchCxt(newCxt, (*cit)->getCallSite(), curFunNode->getFunction()))
345  {
346  const InstVec& nextInsts = (*cit)->getCallSite()->getSuccInstructions();
347  for (const auto& ni: nextInsts)
348  {
349  CxtThreadStmt newCts(cts.getTid(), newCxt, ni);
350  addInterleavingThread(newCts, cts);
351  }
352  }
353  }

436 bool MHP::isRecurFullJoin(NodeID parentTid, NodeID curTid)
437 {
438  if (parentTid == curTid)
439  return true;
440 
441  const TCTNode* curNode = tct->getTCTNode(curTid);
442  FIFOWorkList<const TCTNode*> worklist;
443  worklist.push(curNode);
444  while (!worklist.empty())
445  {
446  const TCTNode* node = worklist.pop();
447  for (TCTEdge* edge : node->getInEdges())
448  {
449  NodeID srcID = edge->getSrcID();
450  if (fja->isFullJoin(srcID, node->getId()))
451  {
452  if (srcID == parentTid)
453  return true;
454  else
455  worklist.push(edge->getSrcNode());
456  }
457  else
458  {
459  return false;
460  }
461  }
462  }
463  return false;
464 }

511 {

513  FIFOWorkList<const PTACallGraphNode*> worklist;
514  TCT::PTACGNodeSet visited;
515  worklist.push(cgnode);
516  visited.insert(cgnode);
517  while (!worklist.empty())
518  {
519  const PTACallGraphNode* node = worklist.pop();
520  if ("main" == node->getFunction()->getName())
521  return true;

523  {
524  const PTACallGraphNode* srcNode = (*nit)->getSrcNode();
525  if (visited.find(srcNode) == visited.end())
526  {
527  visited.insert(srcNode);
528  worklist.push(srcNode);
529  }
530  }
531  }
532  return false;

545 bool MHP::mayHappenInParallelInst(const SVFInstruction* i1, const SVFInstruction* i2)
546 {
547 
549  if (!hasThreadStmtSet(i1) || !hasThreadStmtSet(i2))
550  return false;
551 
552  const CxtThreadStmtSet& tsSet1 = getThreadStmtSet(i1);
553  const CxtThreadStmtSet& tsSet2 = getThreadStmtSet(i2);
554  for (const CxtThreadStmt& ts1 : tsSet1)
555  {
556  NodeBS l1 = getInterleavingThreads(ts1);
557  for (const CxtThreadStmt& ts2 : tsSet2)
558  {
559  NodeBS l2 = getInterleavingThreads(ts2);
560  if (ts1.getTid() != ts2.getTid())
561  {
562  if (l1.test(ts2.getTid()) && l2.test(ts1.getTid()))
563  {
564  numOfMHPQueries++;
565  return true;
566  }
567  }
568  else
569  {
570  if (isMultiForkedThread(ts1.getTid()))
571  {
572  numOfMHPQueries++;
573  return true;
574  }
575  }
576  }
577  }
578  return false;

598  }
