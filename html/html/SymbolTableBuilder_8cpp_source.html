74 void SymbolTableBuilder::buildMemModel(SVFModule* svfModule)
75 {
76  SVFUtil::increaseStackSize();
77 
78  symInfo->setModule(svfModule);
79 
80  // Pointer #0 always represents the null pointer.
81  assert(symInfo->totalSymNum++ == SymbolTableInfo::NullPtr && "Something changed!");
82 
83  // Pointer #1 always represents the pointer points-to black hole.
84  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlkPtr && "Something changed!");
85 
86  // Object #2 is black hole the object that may point to any object
87  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlackHole && "Something changed!");
88  createBlkObj(SymbolTableInfo::BlackHole);
89 
90  // Object #3 always represents the unique constant of a program (merging all constants if Options::ModelConsts is disabled)
91  assert(symInfo->totalSymNum++ == SymbolTableInfo::ConstantObj && "Something changed!");
92  createConstantObj(SymbolTableInfo::ConstantObj);
93 
94  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
95  {
96  // Add symbols for all the globals .
97  for (const GlobalVariable& gv : M.globals())
98  {
99  collectSym(&gv);
100  }
101 
102  // Add symbols for all the global aliases
103  for (const GlobalAlias& ga : M.aliases())
104  {
105  collectSym(&ga);
106  collectSym(ga.getAliasee());
107  }
108 
109  // Add symbols for all of the functions and the instructions in them.
110  for (const Function& fun : M.functions())
111  {
112  collectSym(&fun);
113  collectRet(&fun);
114  if (fun.getFunctionType()->isVarArg())
115  collectVararg(&fun);
116 
117  // Add symbols for all formal parameters.
118  for (const Argument& arg : fun.args())
119  {
120  collectSym(&arg);
121  }
122 
123  // collect and create symbols inside the function body
124  for (const Instruction& inst : instructions(fun))
125  {
126  collectSym(&inst);
127 
128  // initialization for some special instructions
129  //{@
130  if (const StoreInst* st = SVFUtil::dyn_cast<StoreInst>(&inst))
131  {
132  collectSym(st->getPointerOperand());
133  collectSym(st->getValueOperand());
134  }
135  else if (const LoadInst* ld =
136  SVFUtil::dyn_cast<LoadInst>(&inst))
137  {
138  collectSym(ld->getPointerOperand());
139  }
140  else if (const AllocaInst* alloc =
141  SVFUtil::dyn_cast<AllocaInst>(&inst))
142  {
143  collectSym(alloc->getArraySize());
144  }
145  else if (const PHINode* phi = SVFUtil::dyn_cast<PHINode>(&inst))
146  {
147  for (u32_t i = 0; i < phi->getNumIncomingValues(); ++i)
148  {
149  collectSym(phi->getIncomingValue(i));
150  }
151  }
152  else if (const GetElementPtrInst* gep =
153  SVFUtil::dyn_cast<GetElementPtrInst>(&inst))
154  {
155  collectSym(gep->getPointerOperand());
156  for (u32_t i = 0; i < gep->getNumOperands(); ++i)
157  {
158  collectSym(gep->getOperand(i));
159  }
160  }
161  else if (const SelectInst* sel =
162  SVFUtil::dyn_cast<SelectInst>(&inst))
163  {
164  collectSym(sel->getTrueValue());
165  collectSym(sel->getFalseValue());
166  collectSym(sel->getCondition());
167  }
168  else if (const BinaryOperator* binary =
169  SVFUtil::dyn_cast<BinaryOperator>(&inst))
170  {
171  for (u32_t i = 0; i < binary->getNumOperands(); i++)
172  collectSym(binary->getOperand(i));
173  }
174  else if (const UnaryOperator* unary =
175  SVFUtil::dyn_cast<UnaryOperator>(&inst))
176  {
177  for (u32_t i = 0; i < unary->getNumOperands(); i++)
178  collectSym(unary->getOperand(i));
179  }
180  else if (const CmpInst* cmp = SVFUtil::dyn_cast<CmpInst>(&inst))
181  {
182  for (u32_t i = 0; i < cmp->getNumOperands(); i++)
183  collectSym(cmp->getOperand(i));
184  }
185  else if (const CastInst* cast =
186  SVFUtil::dyn_cast<CastInst>(&inst))
187  {
188  collectSym(cast->getOperand(0));
189  }
190  else if (const ReturnInst* ret =
191  SVFUtil::dyn_cast<ReturnInst>(&inst))
192  {
193  if (ret->getReturnValue())
194  collectSym(ret->getReturnValue());
195  }
196  else if (const BranchInst* br =
197  SVFUtil::dyn_cast<BranchInst>(&inst))
198  {
199  Value* opnd = br->isConditional() ? br->getCondition() : br->getOperand(0);
200  collectSym(opnd);
201  }
202  else if (const SwitchInst* sw =
203  SVFUtil::dyn_cast<SwitchInst>(&inst))
204  {
205  collectSym(sw->getCondition());
206  }
207  else if (isNonInstricCallSite(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(&inst)))
208  {
209 
210  const CallBase* cs = LLVMUtil::getLLVMCallSite(&inst);
211  for (u32_t i = 0; i < cs->arg_size(); i++)
212  {
213  collectSym(cs->getArgOperand(i));
214  }
215  // Calls to inline asm need to be added as well because the
216  // callee isn't referenced anywhere else.
217  const Value* Callee = cs->getCalledOperand();
218  collectSym(Callee);
219 
220  // TODO handle inlineAsm
222  }
224  }
225  }
226  }
227 
228  symInfo->totalSymNum = NodeIDAllocator::get()->endSymbolAllocation();
229  if (Options::SymTabPrint())
230  {
231  symInfo->dump();
232  }
233 }

288 void SymbolTableBuilder::collectVal(const Value* val)
289 {
290  // collect and record special sym here
291  if (LLVMUtil::isNullPtrSym(val) || LLVMUtil::isBlackholeSym(val))
292  {
293  return;
294  }
295  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->valSymMap.find(
296  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
297  if (iter == symInfo->valSymMap.end())
298  {
299  // create val sym and sym type
300  SVFValue* svfVal = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val);
301  SymID id = NodeIDAllocator::get()->allocateValueId();
302  symInfo->valSymMap.insert(std::make_pair(svfVal, id));
303  DBOUT(DMemModel,
304  outs() << "create a new value sym " << id << "\n");
306  if (const GlobalVariable* globalVar = SVFUtil::dyn_cast<GlobalVariable>(val))
307  handleGlobalCE(globalVar);
308  }
309 
310  if (isConstantObjSym(val))
311  collectObj(val);
312 }

317 void SymbolTableBuilder::collectObj(const Value* val)
318 {
319  val = LLVMUtil::getGlobalRep(val);
320  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->objSymMap.find(
321  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
322  if (iter == symInfo->objSymMap.end())
323  {
324  SVFValue* svfVal = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val);
325  // if the object pointed by the pointer is a constant data (e.g., i32 0) or a global constant object (e.g. string)
326  // then we treat them as one ConstantObj
327  if (isConstantObjSym(val) && !symInfo->getModelConstants())
328  {
329  symInfo->objSymMap.insert(std::make_pair(svfVal, symInfo->constantSymID()));
330  }
331  // otherwise, we will create an object for each abstract memory location
332  else
333  {
334  // create obj sym and sym type
335  SymID id = NodeIDAllocator::get()->allocateObjectId();
336  symInfo->objSymMap.insert(std::make_pair(svfVal, id));
337  DBOUT(DMemModel,
338  outs() << "create a new obj sym " << id << "\n");
339 
340  // create a memory object
341  MemObj* mem =
342  new MemObj(id, createObjTypeInfo(val),
343  LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val));
344  assert(symInfo->objMap.find(id) == symInfo->objMap.end());
345  symInfo->objMap[id] = mem;
346  }
347  }

387 void SymbolTableBuilder::handleCE(const Value* val)
388 {
389  if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
390  {
391  if (const ConstantExpr* ce = isGepConstantExpr(ref))
392  {
393  DBOUT(DMemModelCE, outs() << "handle constant expression "
394  << LLVMModuleSet::getLLVMModuleSet()
395  ->getSVFValue(ref)
396  ->toString()
397  << "\n");
398  collectVal(ce);
399 
400  // handle the recursive constant express case
401  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
402  for (u32_t i = 0; i < ce->getNumOperands(); ++i)
403  {
404  collectVal(ce->getOperand(i));
405  handleCE(ce->getOperand(i));
406  }
407  }
408  else if (const ConstantExpr* ce = isCastConstantExpr(ref))
409  {
410  DBOUT(DMemModelCE, outs() << "handle constant expression "
411  << LLVMModuleSet::getLLVMModuleSet()
412  ->getSVFValue(ref)
413  ->toString()
414  << "\n");
415  collectVal(ce);
416  collectVal(ce->getOperand(0));
417  // handle the recursive constant express case
418  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
419  handleCE(ce->getOperand(0));
420  }
421  else if (const ConstantExpr* ce = isSelectConstantExpr(ref))
422  {
423  DBOUT(DMemModelCE, outs() << "handle constant expression "
424  << LLVMModuleSet::getLLVMModuleSet()
425  ->getSVFValue(ref)
426  ->toString()
427  << "\n");
428  collectVal(ce);
429  collectVal(ce->getOperand(0));
430  collectVal(ce->getOperand(1));
431  collectVal(ce->getOperand(2));
432  // handle the recursive constant express case
433  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
434  handleCE(ce->getOperand(0));
435  handleCE(ce->getOperand(1));
436  handleCE(ce->getOperand(2));
437  }
438  // if we meet a int2ptr, then it points-to black hole
439  else if (const ConstantExpr* int2Ptrce = isInt2PtrConstantExpr(ref))
440  {
441  collectVal(int2Ptrce);
442  }
443  else if (const ConstantExpr* ptr2Intce = isPtr2IntConstantExpr(ref))
444  {
445  collectVal(ptr2Intce);
446  const Constant* opnd = ptr2Intce->getOperand(0);
447  handleCE(opnd);
448  }
449  else if (isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
450  {
451  collectVal(ref);
452  }
453  else if (isBinaryConstantExpr(ref))
454  {
455  collectVal(ref);
456  }
457  else if (isUnaryConstantExpr(ref))
458  {
459  // we don't handle unary constant expression like fneg(x) now
460  collectVal(ref);
461  }
462  else if (SVFUtil::isa<ConstantAggregate>(ref))
463  {
464  // we don't handle constant agrgregate like constant vectors
465  collectVal(ref);
466  }
467  else
468  {
469  assert(!SVFUtil::isa<ConstantExpr>(val) &&
470  "we don't handle all other constant expression for now!");
471  collectVal(ref);
472  }
473  }
474 }

480 {
481  assert(G);
482 
483  //The type this global points to
484  const Type* T = G->getValueType();
485  bool is_array = 0;

505  handleCE(user);

507  }

576 ObjTypeInfo* SymbolTableBuilder::createObjTypeInfo(const Value* val)
577 {
578  const PointerType* refTy = nullptr;
579 
580  const Instruction* I = SVFUtil::dyn_cast<Instruction>(val);
581 
582  // We consider two types of objects:
583  // (1) A heap/static object from a callsite
584  if (I && isNonInstricCallSite(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(I)))
585  {
586  refTy = getRefTypeOfHeapAllocOrStatic(I);
587  }
588  // (2) Other objects (e.g., alloca, global, etc.)
589  else
590  refTy = SVFUtil::dyn_cast<PointerType>(val->getType());
591 
592  if (refTy)
593  {
594  Type* objTy = getPtrElementType(refTy);
595  ObjTypeInfo* typeInfo = new ObjTypeInfo(
596  LLVMModuleSet::getLLVMModuleSet()->getSVFType(objTy),
597  Options::MaxFieldLimit());
598  initTypeInfo(typeInfo,val, objTy);
599  return typeInfo;
600  }
601  else
602  {
603  writeWrnMsg("try to create an object with a non-pointer type.");
604  writeWrnMsg(val->getName().str());
605  writeWrnMsg("(" + LLVMModuleSet::getLLVMModuleSet()->getSVFValue(val)->getSourceLoc() + ")");
606  if (isConstantObjSym(val))
607  {
608  ObjTypeInfo* typeInfo = new ObjTypeInfo(
609  LLVMModuleSet::getLLVMModuleSet()->getSVFType(val->getType()),
610  0);
611  initTypeInfo(typeInfo,val, val->getType());
612  return typeInfo;
613  }
614  else
615  {
616  assert(false && "Memory object must be either (1) held by a pointer-typed ref value or (2) a constant value (e.g., 10).");
617  abort();
618  }
619  }

625 void SymbolTableBuilder::analyzeObjType(ObjTypeInfo* typeinfo, const Value* val)
626 {
627 
628  const PointerType* refty = SVFUtil::dyn_cast<PointerType>(val->getType());
629  assert(refty && "this value should be a pointer type!");
630  Type* elemTy = getPtrElementType(refty);
631  bool isPtrObj = false;
632  // Find the inter nested array element
633  while (const ArrayType* AT = SVFUtil::dyn_cast<ArrayType>(elemTy))
634  {
635  elemTy = AT->getElementType();
636  if (elemTy->isPointerTy())
637  isPtrObj = true;
638  if (SVFUtil::isa<GlobalVariable>(val) &&
639  SVFUtil::cast<GlobalVariable>(val)->hasInitializer() &&
640  SVFUtil::isa<ConstantArray>(
641  SVFUtil::cast<GlobalVariable>(val)->getInitializer()))
642  typeinfo->setFlag(ObjTypeInfo::CONST_ARRAY_OBJ);
643  else
644  typeinfo->setFlag(ObjTypeInfo::VAR_ARRAY_OBJ);
645  }
646  if (const StructType* ST = SVFUtil::dyn_cast<StructType>(elemTy))
647  {
648  const std::vector<const SVFType*>& flattenFields =
649  getOrAddSVFTypeInfo(ST)->getFlattenFieldTypes();
650  isPtrObj |= std::any_of(flattenFields.begin(), flattenFields.end(),
651  [](const SVFType* ty)
652  {
653  return ty->isPointerTy();
654  });
655  if (SVFUtil::isa<GlobalVariable>(val) &&
656  SVFUtil::cast<GlobalVariable>(val)->hasInitializer() &&
657  SVFUtil::isa<ConstantStruct>(
658  SVFUtil::cast<GlobalVariable>(val)->getInitializer()))
659  typeinfo->setFlag(ObjTypeInfo::CONST_STRUCT_OBJ);
660  else
661  typeinfo->setFlag(ObjTypeInfo::VAR_STRUCT_OBJ);
662  }
663  else if (elemTy->isPointerTy())
664  {
665  isPtrObj = true;
666  }
667 
668  if(isPtrObj)
669  typeinfo->setFlag(ObjTypeInfo::HASPTR_OBJ);
670 }

713 void SymbolTableBuilder::initTypeInfo(ObjTypeInfo* typeinfo, const Value* val,
714  const Type* objTy)
715 {
716 
717  u32_t objSize = 1;
718  // Global variable
719  if (SVFUtil::isa<Function>(val))
720  {
721  typeinfo->setFlag(ObjTypeInfo::FUNCTION_OBJ);
722  analyzeObjType(typeinfo,val);
723  objSize = getObjSize(objTy);
724  }
725  else if(const AllocaInst* allocaInst = SVFUtil::dyn_cast<AllocaInst>(val))
726  {
727  typeinfo->setFlag(ObjTypeInfo::STACK_OBJ);
728  analyzeObjType(typeinfo,val);
731  if(const ConstantInt* sz = SVFUtil::dyn_cast<ConstantInt>(allocaInst->getArraySize()))
732  objSize = sz->getZExtValue() * getObjSize(objTy);
733  else
734  objSize = getObjSize(objTy);
735  }
736  else if(SVFUtil::isa<GlobalVariable>(val))
737  {
738  typeinfo->setFlag(ObjTypeInfo::GLOBVAR_OBJ);
739  if(isConstantObjSym(val))
740  typeinfo->setFlag(ObjTypeInfo::CONST_GLOBAL_OBJ);
741  analyzeObjType(typeinfo,val);
742  objSize = getObjSize(objTy);
743  }
744  else if (SVFUtil::isa<Instruction>(val) &&
745  isHeapAllocExtCall(
746  LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(
747  SVFUtil::cast<Instruction>(val))))
748  {
749  analyzeHeapObjType(typeinfo,val);
750  // Heap object, label its field as infinite here
751  objSize = typeinfo->getMaxFieldOffsetLimit();
752  }
753  else if (SVFUtil::isa<Instruction>(val) &&
754  isStaticExtCall(
755  LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(
756  SVFUtil::cast<Instruction>(val))))
757  {
758  analyzeStaticObjType(typeinfo,val);
759  // static object allocated before main, label its field as infinite here
760  objSize = typeinfo->getMaxFieldOffsetLimit();
761  }
762  else if(ArgInProgEntryFunction(val))
763  {
764  analyzeStaticObjType(typeinfo,val);
765  // user input data, label its field as infinite here
766  objSize = typeinfo->getMaxFieldOffsetLimit();
767  }
768  else if(LLVMUtil::isConstDataOrAggData(val))
769  {
770  typeinfo->setFlag(ObjTypeInfo::CONST_DATA);
771  objSize = getNumOfFlattenElements(val->getType());
772  }
773  else
774  {
775  assert("what other object do we have??");
776  abort();
777  }
778 
779  // Reset maxOffsetLimit if it is over the total fieldNum of this object
780  if(typeinfo->getMaxFieldOffsetLimit() > objSize)
781  typeinfo->setNumOfElements(objSize);
782 }
