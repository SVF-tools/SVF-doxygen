39 const inline std::string GrammarBuilder::parseProductionsString() const
40 {
41  std::ifstream textFile(fileName);
42  std::string lineString;
43  std::string lines = "";
44  std::string startString;
45  std::string symbolString;
46  const std::string WHITESPACE = " \n\r\t\f\v";
47  int lineNum = 0;
48  while (getline(textFile, lineString))
49  {
50  if(lineNum == 1)
51  {
52  startString = stripSpace(lineString);
53  }
54  if(lineNum == 3)
55  {
56  symbolString = lineString.substr(lineString.find_first_not_of(WHITESPACE), lineString.find_last_not_of(WHITESPACE)+1);
57  }
58 
59  lines.append(lineString.substr(lineString.find_first_not_of(WHITESPACE), lineString.find_last_not_of(WHITESPACE)+1));
60  lineNum++;
61  }
62 
63  std::regex reg("Start:([\\s\\S]*)Terminal:(.*)Productions:([\\s\\S]*)");
64  std::smatch matches;
65  if (std::regex_search(lines, matches, reg))
66  {
67  lines = matches.str(3);
68  }
69  std::string sString;
70  size_t pos = 0;
71  while ((pos = symbolString.find(" ")) != std::string::npos)
72  {
73  sString = stripSpace(symbolString.substr(0, pos));
74  symbolString.erase(0, pos + 1); //Capital is Nonterminal, Otherwise is terminal
75  grammar->insertSymbol(sString);
76  }
77  grammar->insertSymbol(symbolString);
78  grammar->setStartKind(grammar->insertSymbol(startString));
79  grammar->insertTerminalKind("epsilon");
80 
81  return lines;
82 }

110 GrammarBase* GrammarBuilder::build() const
111 {
112  std::smatch matches;
113  std::string delimiter = " ";
114  std::string delimiter1 = "->";
115  std::string word = "";
116  size_t pos;
117  GrammarBase::Production prod;
118  std::vector<std::string> wordProdVec = loadWordProductions();
119 
120  for (auto wordProd : wordProdVec)
121  {
122  if ((pos = wordProd.find(delimiter1)) != std::string::npos)
123  {
124  std::string RHS = stripSpace(wordProd.substr(0, pos));
125  std::string LHS = wordProd.substr(pos + delimiter1.size(), wordProd.size() - 1);
126  GrammarBase::Symbol RHSSymbol = grammar->insertSymbol(RHS);
127  prod.push_back(RHSSymbol);
128  if (grammar->getRawProductions().find(RHSSymbol) == grammar->getRawProductions().end()) grammar->getRawProductions().insert({RHSSymbol, {}});
129  std::regex LHSRegEx("\\s*(.*)");
130  std::regex_search(LHS, matches, LHSRegEx);
131  LHS = matches.str(1);
132  while ((pos = LHS.find(delimiter)) != std::string::npos)
133  {
134  word = LHS.substr(0, pos);
135  LHS.erase(0, pos + delimiter.length()); //Capital is Nonterminal, Otherwise is terminal
136  prod.push_back(grammar->insertSymbol(word));
137  }
138  prod.push_back(grammar->insertSymbol(LHS));
139  grammar->getRawProductions().at(RHSSymbol).insert(prod);
140  prod = {};
141  }
142  }
143 
144  return grammar;
145 };
