47 void ICFGWrapper::addICFGNodeWrapperFromICFGNode(const ICFGNode *src)
48 {
49 
50  if (!hasICFGNodeWrapper(src->getId()))
51  addICFGNodeWrapper(new ICFGNodeWrapper(src));
52  ICFGNodeWrapper *curICFGNodeWrapper = getGNode(src->getId());
53  if (isa<FunEntryICFGNode>(src))
54  {
55  _funcToFunEntry.emplace(src->getFun(), curICFGNodeWrapper);
56  }
57  else if (isa<FunExitICFGNode>(src))
58  {
59  _funcToFunExit.emplace(src->getFun(), curICFGNodeWrapper);
60  }
61  else if (const CallICFGNode *callICFGNode = dyn_cast<CallICFGNode>(src))
62  {
63  if (!hasICFGNodeWrapper(callICFGNode->getRetICFGNode()->getId()))
64  addICFGNodeWrapper(new ICFGNodeWrapper(callICFGNode->getRetICFGNode()));
65  if (!curICFGNodeWrapper->getRetICFGNodeWrapper())
66  curICFGNodeWrapper->setRetICFGNodeWrapper(getGNode(callICFGNode->getRetICFGNode()->getId()));
67  }
68  else if (const RetICFGNode *retICFGNode = dyn_cast<RetICFGNode>(src))
69  {
70  if (!hasICFGNodeWrapper(retICFGNode->getCallICFGNode()->getId()))
71  addICFGNodeWrapper(new ICFGNodeWrapper(retICFGNode->getCallICFGNode()));
72  if (!curICFGNodeWrapper->getCallICFGNodeWrapper())
73  curICFGNodeWrapper->setCallICFGNodeWrapper(getGNode(retICFGNode->getCallICFGNode()->getId()));
74  }
75  for (const auto &e: src->getOutEdges())
76  {
77  if (!hasICFGNodeWrapper(e->getDstID()))
78  addICFGNodeWrapper(new ICFGNodeWrapper(e->getDstNode()));
79  ICFGNodeWrapper *dstNodeWrapper = getGNode(e->getDstID());
80  if (!hasICFGEdgeWrapper(curICFGNodeWrapper, dstNodeWrapper, e))
81  {
82  ICFGEdgeWrapper *pEdge = new ICFGEdgeWrapper(curICFGNodeWrapper, dstNodeWrapper, e);
83  addICFGEdgeWrapper(pEdge);
84  }
85  }
86 }
