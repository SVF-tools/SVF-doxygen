102 void ThreadAPI::init()
103 {
104  set<TD_TYPE> t_seen;
105  TD_TYPE prev_t= TD_DUMMY;
106  t_seen.insert(TD_DUMMY);
107  for(const ei_pair *p= ei_pairs; p->n; ++p)
108  {
109  if(p->t != prev_t)
110  {
111  //This will detect if you move an entry to another block
112  // but forget to change the type.
113  if(t_seen.count(p->t))
114  {
115  fputs(p->n, stderr);
116  putc('\n', stderr);
117  assert(!"ei_pairs not grouped by type");
118  }
119  t_seen.insert(p->t);
120  prev_t= p->t;
121  }
122  if(tdAPIMap.count(p->n))
123  {
124  fputs(p->n, stderr);
125  putc('\n', stderr);
126  assert(!"duplicate name in ei_pairs");
127  }
128  tdAPIMap[p->n]= p->t;
129  }
130 }

156 const SVFValue* ThreadAPI::getJoinedThread(const SVFInstruction *inst) const
157 {
158  assert(isTDJoin(inst) && "not a thread join function!");
159  CallSite cs = getSVFCallSite(inst);
160  const SVFValue* join = cs.getArgument(0);
161  const SVFVar* var = PAG::getPAG()->getGNode(PAG::getPAG()->getValueNode(join));
162  for(const SVFStmt* stmt : var->getInEdges())
163  {
164  if(SVFUtil::isa<LoadStmt>(stmt))
165  return stmt->getSrcNode()->getValue();
166  }
167  if(SVFUtil::isa<SVFArgument>(join))
168  return join;
169 
170  assert(false && "the value of the first argument at join is not a load instruction?");
171  return nullptr;
172 }
