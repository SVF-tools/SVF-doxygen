101 void ThreadAPI::init()
102 {
103  set<TD_TYPE> t_seen;
104  TD_TYPE prev_t= TD_DUMMY;
105  t_seen.insert(TD_DUMMY);
106  for(const ei_pair *p= ei_pairs; p->n; ++p)
107  {
108  if(p->t != prev_t)
109  {
110  //This will detect if you move an entry to another block
111  // but forget to change the type.
112  if(t_seen.count(p->t))
113  {
114  fputs(p->n, stderr);
115  putc('\n', stderr);
116  assert(!"ei_pairs not grouped by type");
117  }
118  t_seen.insert(p->t);
119  prev_t= p->t;
120  }
121  if(tdAPIMap.count(p->n))
122  {
123  fputs(p->n, stderr);
124  putc('\n', stderr);
125  assert(!"duplicate name in ei_pairs");
126  }
127  tdAPIMap[p->n]= p->t;
128  }
129 }
