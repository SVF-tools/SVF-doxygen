223 void SVFG::addSVFGNodesForAddrTakenVars()
224 {
225 
226  // set defs for address-taken vars defined at store statements
227  PAGEdge::SVFStmtSetTy& stores = getPAGEdgeSet(PAGEdge::Store);
228  for (PAGEdge::SVFStmtSetTy::iterator iter = stores.begin(), eiter =
229  stores.end(); iter != eiter; ++iter)
230  {
231  StoreStmt* store = SVFUtil::cast<StoreStmt>(*iter);
232  const StmtSVFGNode* sNode = getStmtVFGNode(store);
233  for(CHISet::iterator pi = mssa->getCHISet(store).begin(), epi = mssa->getCHISet(store).end(); pi!=epi; ++pi)
234  setDef((*pi)->getResVer(),sNode);
235  }
236 
240  for(MemSSA::BBToPhiSetMap::iterator it = mssa->getBBToPhiSetMap().begin(),
241  eit = mssa->getBBToPhiSetMap().end(); it!=eit; ++it)
242  {
243  for(PHISet::iterator pi = it->second.begin(), epi = it->second.end(); pi!=epi; ++pi){
244  MemSSA::PHI* phi = *pi;
245  addIntraMSSAPHISVFGNode(pag->getICFG()->getBlockICFGNode(&(phi->getBasicBlock()->front())), phi->opVerBegin(), phi->opVerEnd(),phi->getResVer(), totalVFGNode++);
246  }
247  }
249  for(MemSSA::FunToEntryChiSetMap::iterator it = mssa->getFunToEntryChiSetMap().begin(),
250  eit = mssa->getFunToEntryChiSetMap().end(); it!=eit; ++it)
251  {
252  for(CHISet::iterator pi = it->second.begin(), epi = it->second.end(); pi!=epi; ++pi){
253  const MemSSA::ENTRYCHI* chi = SVFUtil::cast<ENTRYCHI>(*pi);
254  addFormalINSVFGNode(pag->getICFG()->getFunEntryBlockNode(chi->getFunction()), chi->getResVer(), totalVFGNode++);
255  }
256  }
258  for(MemSSA::FunToReturnMuSetMap::iterator it = mssa->getFunToRetMuSetMap().begin(),
259  eit = mssa->getFunToRetMuSetMap().end(); it!=eit; ++it)
260  {
261  for(MUSet::iterator pi = it->second.begin(), epi = it->second.end(); pi!=epi; ++pi){
262  const MemSSA::RETMU* mu = SVFUtil::cast<RETMU>(*pi);
263  addFormalOUTSVFGNode(pag->getICFG()->getFunExitBlockNode(mu->getFunction()), mu->getMRVer(), totalVFGNode++);
264  }
265  }
267  for(MemSSA::CallSiteToMUSetMap::iterator it = mssa->getCallSiteToMuSetMap().begin(),
268  eit = mssa->getCallSiteToMuSetMap().end();
269  it!=eit; ++it)
270  {
271  for(MUSet::iterator pi = it->second.begin(), epi = it->second.end(); pi!=epi; ++pi){
272  const MemSSA::CALLMU* mu = SVFUtil::cast<CALLMU>(*pi);
273  addActualINSVFGNode(mu->getCallSite(), mu->getMRVer(), totalVFGNode++);
274  }
275  }
277  for(MemSSA::CallSiteToCHISetMap::iterator it = mssa->getCallSiteToChiSetMap().begin(),
278  eit = mssa->getCallSiteToChiSetMap().end();
279  it!=eit; ++it)
280  {
281  for(CHISet::iterator pi = it->second.begin(), epi = it->second.end(); pi!=epi; ++pi){
282  const MemSSA::CALLCHI* chi = SVFUtil::cast<CALLCHI>(*pi);
283  addActualOUTSVFGNode(chi->getCallSite(), chi->getResVer(), totalVFGNode++);
284  }

286  }

290  * Connect def-use chains for indirect value-flow, (value-flow of address-taken variables)
291  */
292 void SVFG::connectIndirectSVFGEdges()
293 {
294 
295  for(iterator it = begin(), eit = end(); it!=eit; ++it)
296  {
297  NodeID nodeId = it->first;
298  const SVFGNode* node = it->second;
299  if(const LoadSVFGNode* loadNode = SVFUtil::dyn_cast<LoadSVFGNode>(node))
300  {
301  MUSet& muSet = mssa->getMUSet(SVFUtil::cast<LoadStmt>(loadNode->getPAGEdge()));
302  for(MUSet::iterator it = muSet.begin(), eit = muSet.end(); it!=eit; ++it)
303  {
304  if(LOADMU* mu = SVFUtil::dyn_cast<LOADMU>(*it))
305  {
306  NodeID def = getDef(mu->getMRVer());
307  addIntraIndirectVFEdge(def,nodeId, mu->getMRVer()->getMR()->getPointsTo());
308  }
309  }
310  }
311  else if(const StoreSVFGNode* storeNode = SVFUtil::dyn_cast<StoreSVFGNode>(node))
312  {
313  CHISet& chiSet = mssa->getCHISet(SVFUtil::cast<StoreStmt>(storeNode->getPAGEdge()));
314  for(CHISet::iterator it = chiSet.begin(), eit = chiSet.end(); it!=eit; ++it)
315  {
316  if(STORECHI* chi = SVFUtil::dyn_cast<STORECHI>(*it))
317  {
318  NodeID def = getDef(chi->getOpVer());
319  addIntraIndirectVFEdge(def,nodeId, chi->getOpVer()->getMR()->getPointsTo());
320  }
321  }
322  }
323  else if(const FormalINSVFGNode* formalIn = SVFUtil::dyn_cast<FormalINSVFGNode>(node))
324  {
325  PTACallGraphEdge::CallInstSet callInstSet;
326  mssa->getPTA()->getPTACallGraph()->getDirCallSitesInvokingCallee(formalIn->getFun(),callInstSet);
327  for(PTACallGraphEdge::CallInstSet::iterator it = callInstSet.begin(), eit = callInstSet.end(); it!=eit; ++it)
328  {
329  const CallBlockNode* cs = *it;
330  if(!mssa->hasMU(cs))
331  continue;
332  ActualINSVFGNodeSet& actualIns = getActualINSVFGNodes(cs);
333  for(ActualINSVFGNodeSet::iterator ait = actualIns.begin(), aeit = actualIns.end(); ait!=aeit; ++ait)
334  {
335  const ActualINSVFGNode* actualIn = SVFUtil::cast<ActualINSVFGNode>(getSVFGNode(*ait));
336  addInterIndirectVFCallEdge(actualIn,formalIn,getCallSiteID(cs, formalIn->getFun()));
337  }
338  }
339  }
340  else if(const FormalOUTSVFGNode* formalOut = SVFUtil::dyn_cast<FormalOUTSVFGNode>(node))
341  {
342  PTACallGraphEdge::CallInstSet callInstSet;
343  // const MemSSA::RETMU* retMu = formalOut->getRetMU();
344  mssa->getPTA()->getPTACallGraph()->getDirCallSitesInvokingCallee(formalOut->getFun(),callInstSet);
345  for(PTACallGraphEdge::CallInstSet::iterator it = callInstSet.begin(), eit = callInstSet.end(); it!=eit; ++it)
346  {
347  const CallBlockNode* cs = *it;
348  if(!mssa->hasCHI(cs))
349  continue;
350  ActualOUTSVFGNodeSet& actualOuts = getActualOUTSVFGNodes(cs);
351  for(ActualOUTSVFGNodeSet::iterator ait = actualOuts.begin(), aeit = actualOuts.end(); ait!=aeit; ++ait)
352  {
353  const ActualOUTSVFGNode* actualOut = SVFUtil::cast<ActualOUTSVFGNode>(getSVFGNode(*ait));
354  addInterIndirectVFRetEdge(formalOut,actualOut,getCallSiteID(cs, formalOut->getFun()));
355  }
356  }
357  NodeID def = getDef(formalOut->getMRVer());
358  addIntraIndirectVFEdge(def,nodeId, formalOut->getMRVer()->getMR()->getPointsTo());
359  }
360  else if(const ActualINSVFGNode* actualIn = SVFUtil::dyn_cast<ActualINSVFGNode>(node))
361  {
362  const MRVer* ver = actualIn->getMRVer();
363  NodeID def = getDef(ver);
364  addIntraIndirectVFEdge(def,nodeId, ver->getMR()->getPointsTo());
365  }
366  else if(SVFUtil::isa<ActualOUTSVFGNode>(node))
367  {
369  }
370  else if(const MSSAPHISVFGNode* phiNode = SVFUtil::dyn_cast<MSSAPHISVFGNode>(node))
371  {
372  for (MemSSA::PHI::OPVers::const_iterator it = phiNode->opVerBegin(), eit = phiNode->opVerEnd();
373  it != eit; it++)
374  {
375  const MRVer* op = it->second;
376  NodeID def = getDef(op);
377  addIntraIndirectVFEdge(def,nodeId, op->getMR()->getPointsTo());
378  }
379  }
380  }

383  connectFromGlobalToProgEntry();
384 }
