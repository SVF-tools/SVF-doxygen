60 void AndersenStat::collectCycleInfo(ConstraintGraph* consCG)
61 {
62  _NumOfCycles = 0;
63  _NumOfPWCCycles = 0;
64  _NumOfNodesInCycles = 0;
65  NodeSet repNodes;
66  repNodes.clear();
67  for(ConstraintGraph::iterator it = consCG->begin(), eit = consCG->end(); it!=eit; ++it)
68  {
69  // sub nodes have been removed from the constraint graph, only rep nodes are left.
70  NodeID repNode = consCG->sccRepNode(it->first);
71  NodeBS& subNodes = consCG->sccSubNodes(repNode);
72  NodeBS clone = subNodes;
73  for (NodeBS::iterator it = subNodes.begin(), eit = subNodes.end(); it != eit; ++it)
74  {
75  NodeID nodeId = *it;
76  PAGNode* pagNode = pta->getPAG()->getGNode(nodeId);
77  if (SVFUtil::isa<ObjVar>(pagNode) && pta->isFieldInsensitive(nodeId))
78  {
79  NodeID baseId = consCG->getBaseObjVar(nodeId);
80  clone.reset(nodeId);
81  clone.set(baseId);
82  }
83  }
84  u32_t num = clone.count();
85  if (num > 1)
86  {
87  if(repNodes.insert(repNode).second)
88  {
89  _NumOfNodesInCycles += num;
90  if(consCG->isPWCNode(repNode))
91  _NumOfPWCCycles ++;
92  }
93  if( num > _MaxNumOfNodesInSCC)
94  _MaxNumOfNodesInSCC = num;
95  }
96  }
97  _NumOfCycles += repNodes.size();
98 }

100 void AndersenStat::constraintGraphStat()
101 {
102 
103 
104  ConstraintGraph* consCG = pta->getConstraintGraph();
105 
106  u32_t numOfCopys = 0;
107  u32_t numOfGeps = 0;
108  // collect copy and gep edges
109  for(ConstraintEdge::ConstraintEdgeSetTy::iterator it = consCG->getDirectCGEdges().begin(),
110  eit = consCG->getDirectCGEdges().end(); it!=eit; ++it)
111  {
112  if(SVFUtil::isa<CopyCGEdge>(*it))
113  numOfCopys++;
114  else if(SVFUtil::isa<GepCGEdge>(*it))
115  numOfGeps++;
116  else
117  assert(false && "what else!!");
118  }
119 
120  u32_t totalNodeNumber = 0;
121  u32_t cgNodeNumber = 0;
122  u32_t objNodeNumber = 0;
123  u32_t addrtotalIn = 0;
124  u32_t addrtotalOut = 0;
125  u32_t addrmaxIn = 0;
126  u32_t addrmaxOut = 0;
127  u32_t copytotalIn = 0;
128  u32_t copytotalOut = 0;
129  u32_t copymaxIn = 0;
130  u32_t copymaxOut = 0;
131  u32_t loadtotalIn = 0;
132  u32_t loadtotalOut = 0;
133  u32_t loadmaxIn = 0;
134  u32_t loadmaxOut = 0;
135  u32_t storetotalIn = 0;
136  u32_t storetotalOut = 0;
137  u32_t storemaxIn = 0;
138  u32_t storemaxOut = 0;
139 
140 
141  for (ConstraintGraph::ConstraintNodeIDToNodeMapTy::iterator nodeIt = consCG->begin(), nodeEit = consCG->end();
142  nodeIt != nodeEit; nodeIt++)
143  {
144  totalNodeNumber++;
145  if(nodeIt->second->getInEdges().empty() && nodeIt->second->getOutEdges().empty())
146  continue;
147  cgNodeNumber++;
148  if(SVFUtil::isa<ObjVar>(pta->getPAG()->getGNode(nodeIt->first)))
149  objNodeNumber++;
150 
151  u32_t nCopyIn = nodeIt->second->getDirectInEdges().size();
152  if(nCopyIn > copymaxIn)
153  copymaxIn = nCopyIn;
154  copytotalIn +=nCopyIn;
155  u32_t nCopyOut = nodeIt->second->getDirectOutEdges().size();
156  if(nCopyOut > copymaxOut)
157  copymaxOut = nCopyOut;
158  copytotalOut +=nCopyOut;
159  u32_t nLoadIn = nodeIt->second->getLoadInEdges().size();
160  if(nLoadIn > loadmaxIn)
161  loadmaxIn = nLoadIn;
162  loadtotalIn +=nLoadIn;
163  u32_t nLoadOut = nodeIt->second->getLoadOutEdges().size();
164  if(nLoadOut > loadmaxOut)
165  loadmaxOut = nLoadOut;
166  loadtotalOut +=nLoadOut;
167  u32_t nStoreIn = nodeIt->second->getStoreInEdges().size();
168  if(nStoreIn > storemaxIn)
169  storemaxIn = nStoreIn;
170  storetotalIn +=nStoreIn;
171  u32_t nStoreOut = nodeIt->second->getStoreOutEdges().size();
172  if(nStoreOut > storemaxOut)
173  storemaxOut = nStoreOut;
174  storetotalOut +=nStoreOut;
175  u32_t nAddrIn = nodeIt->second->getAddrInEdges().size();
176  if(nAddrIn > addrmaxIn)
177  addrmaxIn = nAddrIn;
178  addrtotalIn +=nAddrIn;
179  u32_t nAddrOut = nodeIt->second->getAddrOutEdges().size();
180  if(nAddrOut > addrmaxOut)
181  addrmaxOut = nAddrOut;
182  addrtotalOut +=nAddrOut;
183  }
184  double storeavgIn = (double)storetotalIn/cgNodeNumber;
185  double loadavgIn = (double)loadtotalIn/cgNodeNumber;
186  double copyavgIn = (double)copytotalIn/cgNodeNumber;
187  double addravgIn = (double)addrtotalIn/cgNodeNumber;
188  double avgIn = (double)(addrtotalIn + copytotalIn + loadtotalIn + storetotalIn)/cgNodeNumber;
189 
190 
191  PTNumStatMap["NumOfCGNode"] = totalNodeNumber;
192  PTNumStatMap["TotalValidNode"] = cgNodeNumber;
193  PTNumStatMap["TotalValidObjNode"] = objNodeNumber;
194  PTNumStatMap["NumOfCGEdge"] = consCG->getLoadCGEdges().size() + consCG->getStoreCGEdges().size()
195  + numOfCopys + numOfGeps;
196  PTNumStatMap["NumOfAddrs"] = consCG->getAddrCGEdges().size();
197  PTNumStatMap["NumOfCopys"] = numOfCopys;
198  PTNumStatMap["NumOfGeps"] = numOfGeps;
199  PTNumStatMap["NumOfLoads"] = consCG->getLoadCGEdges().size();
200  PTNumStatMap["NumOfStores"] = consCG->getStoreCGEdges().size();
201  PTNumStatMap["MaxInCopyEdge"] = copymaxIn;
202  PTNumStatMap["MaxOutCopyEdge"] = copymaxOut;
203  PTNumStatMap["MaxInLoadEdge"] = loadmaxIn;
204  PTNumStatMap["MaxOutLoadEdge"] = loadmaxOut;
205  PTNumStatMap["MaxInStoreEdge"] = storemaxIn;
206  PTNumStatMap["MaxOutStoreEdge"] = storemaxOut;
207  PTNumStatMap["AvgIn/OutStoreEdge"] = storeavgIn;
208  PTNumStatMap["MaxInAddrEdge"] = addrmaxIn;
209  PTNumStatMap["MaxOutAddrEdge"] = addrmaxOut;
210  timeStatMap["AvgIn/OutCopyEdge"] = copyavgIn;
211  timeStatMap["AvgIn/OutLoadEdge"] = loadavgIn;
212  timeStatMap["AvgIn/OutAddrEdge"] = addravgIn;
213  timeStatMap["AvgIn/OutEdge"] = avgIn;
214 
215  PTAStat::printStat("Constraint Graph Stats");
216 }
220 void AndersenStat::statNullPtr()
221 {
222 
223  _NumOfNullPtr = 0;
224  for (SVFIR::iterator iter = pta->getPAG()->begin(), eiter = pta->getPAG()->end();
225  iter != eiter; ++iter)
226  {
227  NodeID pagNodeId = iter->first;
228  PAGNode* pagNode = iter->second;
229  if (SVFUtil::isa<ValVar>(pagNode) == false)
230  continue;
231  SVFStmt::SVFStmtSetTy& inComingStore = pagNode->getIncomingEdges(SVFStmt::Store);
232  SVFStmt::SVFStmtSetTy& outGoingLoad = pagNode->getOutgoingEdges(SVFStmt::Load);
233  if (inComingStore.empty()==false || outGoingLoad.empty()==false)
234  {
236  const PointsTo& pts = pta->getPts(pagNodeId);
237  if (pta->containBlackHoleNode(pts))
238  _NumOfBlackholePtr++;
239 
240  if (pta->containConstantNode(pts))
241  _NumOfConstantPtr++;
242 
243  if(pts.empty())
244  {
245  std::string str;
246  raw_string_ostream rawstr(str);
247  if (!SVFUtil::isa<DummyValVar>(pagNode) && !SVFUtil::isa<DummyObjVar>(pagNode) )
248  {
249  // if a pointer is in dead function, we do not care
250  if(isPtrInDeadFunction(pagNode->getValue()) == false)
251  {
252  _NumOfNullPtr++;
253  rawstr << "##Null Pointer : (NodeID " << pagNode->getId()
254  << ") PtrName:" << pagNode->getValue()->getName();
255  writeWrnMsg(rawstr.str());
256  //pagNode->getValue()->dump();
257  }
258  }
259  else
260  {
261  _NumOfNullPtr++;
262  rawstr << "##Null Pointer : (NodeID " << pagNode->getId() << ")";
263  writeWrnMsg(rawstr.str());
264  }
265  }
266  }
267  }

274 void AndersenStat::performStat()
275 {
276 
277  assert(SVFUtil::isa<AndersenBase>(pta) && "not an andersen pta pass!! what else??");
278  endClk();
279 
280  SVFIR* pag = pta->getPAG();
281  ConstraintGraph* consCG = pta->getConstraintGraph();
282 
283  // collect constraint graph cycles
284  collectCycleInfo(consCG);
285 
286  // stat null ptr number
287  statNullPtr();
288 
289  u32_t totalPointers = 0;
290  u32_t totalTopLevPointers = 0;
291  u32_t totalPtsSize = 0;
292  u32_t totalTopLevPtsSize = 0;
293  for (SVFIR::iterator iter = pta->getPAG()->begin(), eiter = pta->getPAG()->end();
294  iter != eiter; ++iter)
295  {
296  NodeID node = iter->first;
297  const PointsTo& pts = pta->getPts(node);
298  u32_t size = pts.count();
299  totalPointers++;
300  totalPtsSize+=size;
301 
302  if(pta->getPAG()->isValidTopLevelPtr(pta->getPAG()->getGNode(node)))
303  {
304  totalTopLevPointers++;
305  totalTopLevPtsSize+=size;
306  }
307 
308  if(size > _MaxPtsSize )
309  _MaxPtsSize = size;
310  }
311 
312 
313  PTAStat::performStat();
314 
315  constraintGraphStat();
316 
317  timeStatMap[TotalAnalysisTime] = (endTime - startTime)/TIMEINTERVAL;
318  timeStatMap[SCCDetectionTime] = Andersen::timeOfSCCDetection;
319  timeStatMap[SCCMergeTime] = Andersen::timeOfSCCMerges;
320  timeStatMap[CollapseTime] = Andersen::timeOfCollapse;
321 
322  timeStatMap[ProcessLoadStoreTime] = Andersen::timeOfProcessLoadStore;
323  timeStatMap[ProcessCopyGepTime] = Andersen::timeOfProcessCopyGep;
324  timeStatMap[UpdateCallGraphTime] = Andersen::timeOfUpdateCallGraph;
325 
326  PTNumStatMap[TotalNumOfPointers] = pag->getValueNodeNum() + pag->getFieldValNodeNum();
327  PTNumStatMap[TotalNumOfObjects] = pag->getObjectNodeNum() + pag->getFieldObjNodeNum();
328 
329 
330  PTNumStatMap[NumOfProcessedAddrs] = Andersen::numOfProcessedAddr;
331  PTNumStatMap[NumOfProcessedCopys] = Andersen::numOfProcessedCopy;
332  PTNumStatMap[NumOfProcessedGeps] = Andersen::numOfProcessedGep;
333  PTNumStatMap[NumOfProcessedLoads] = Andersen::numOfProcessedLoad;
334  PTNumStatMap[NumOfProcessedStores] = Andersen::numOfProcessedStore;
335 
336  PTNumStatMap[NumOfSfr] = Andersen::numOfSfrs;
337  PTNumStatMap[NumOfFieldExpand] = Andersen::numOfFieldExpand;
338 
339  PTNumStatMap[NumOfPointers] = pag->getValueNodeNum();
340  PTNumStatMap[NumOfMemObjects] = pag->getObjectNodeNum();
341  PTNumStatMap[NumOfGepFieldPointers] = pag->getFieldValNodeNum();
342  PTNumStatMap[NumOfGepFieldObjects] = pag->getFieldObjNodeNum();
343 
344  timeStatMap[AveragePointsToSetSize] = (double)totalPtsSize/totalPointers;;
345  timeStatMap[AverageTopLevPointsToSetSize] = (double)totalTopLevPtsSize/totalTopLevPointers;;
346 
347  PTNumStatMap[MaxPointsToSetSize] = _MaxPtsSize;
348 
349  PTNumStatMap[NumOfIterations] = pta->numOfIteration;
350 
351  PTNumStatMap[NumOfIndirectCallSites] = consCG->getIndirectCallsites().size();
352  PTNumStatMap[NumOfIndirectEdgeSolved] = pta->getNumOfResolvedIndCallEdge();
353 
354  PTNumStatMap[NumOfSCCDetection] = Andersen::numOfSCCDetection;
355  PTNumStatMap[NumOfCycles] = _NumOfCycles;
356  PTNumStatMap[NumOfPWCCycles] = _NumOfPWCCycles;
357  PTNumStatMap[NumOfNodesInCycles] = _NumOfNodesInCycles;
358  PTNumStatMap[MaxNumOfNodesInSCC] = _MaxNumOfNodesInSCC;
359  PTNumStatMap[NumOfNullPointer] = _NumOfNullPtr;
360  PTNumStatMap["PointsToConstPtr"] = _NumOfConstantPtr;
361  PTNumStatMap["PointsToBlkPtr"] = _NumOfBlackholePtr;
362 
363  PTAStat::printStat("Andersen Pointer Analysis Stats");
364 }
