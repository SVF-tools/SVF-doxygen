38 {
39  // Build CFL Graph
40  VFCFLGraphBuilder cflGraphBuilder = VFCFLGraphBuilder();
41  if (Options::CFLGraph().empty()) // built from svfir
42  {
43  PointerAnalysis::initialize();
44  AndersenWaveDiff* ander = AndersenWaveDiff::createAndersenWaveDiff(pag);
45  svfg = memSSA.buildFullSVFG(ander);
46  graph = cflGraphBuilder.buildBigraph(svfg, grammarBase->getStartKind(), grammarBase);

48  else
49  graph = cflGraphBuilder.buildFromDot(Options::CFLGraph(), grammarBase);
50 
51  // Check CFL Graph and Grammar are accordance with grammar
52  CFLGramGraphChecker cflChecker = CFLGramGraphChecker();
53  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
54 }
