341 ICFGEdge* ICFG::getICFGEdge(const ICFGNode* src, const ICFGNode* dst, ICFGEdge::ICFGEdgeK kind)
342 {
343 
344  ICFGEdge * edge = nullptr;
345  u32_t counter = 0;
346  for (ICFGEdge::ICFGEdgeSetTy::iterator iter = src->OutEdgeBegin();
347  iter != src->OutEdgeEnd(); ++iter)
348  {
349  if ((*iter)->getDstID() == dst->getId() && (*iter)->getEdgeKind() == kind)
350  {
351  counter++;
352  edge = (*iter);
353  }
354  }
355  assert(counter <= 1 && "there's more than one edge between two ICFG nodes");
356  return edge;
357 
358 }

457 void ICFG::updateCallGraph(PTACallGraph* callgraph)
458 {
459  PTACallGraph::CallEdgeMap::const_iterator iter = callgraph->getIndCallMap().begin();
460  PTACallGraph::CallEdgeMap::const_iterator eiter = callgraph->getIndCallMap().end();
461  for (; iter != eiter; iter++)
462  {
463  const CallICFGNode* callBlock = iter->first;
464  const SVFInstruction* cs = callBlock->getCallSite();
465  assert(callBlock->isIndirectCall() && "this is not an indirect call?");
466  const PTACallGraph::FunctionSet & functions = iter->second;
467  for (PTACallGraph::FunctionSet::const_iterator func_iter = functions.begin(); func_iter != functions.end(); func_iter++)
468  {
469  const SVFFunction* callee = *func_iter;
470  CallICFGNode* callBlockNode = getCallICFGNode(cs);
471  RetICFGNode* retBlockNode = getRetICFGNode(cs);
473  if (isExtCall(callee))
474  addIntraEdge(callBlockNode, retBlockNode);
475  else
476  {
477  FunEntryICFGNode* calleeEntryNode = getFunEntryBlock(callee);
478  FunExitICFGNode* calleeExitNode = getFunExitBlock(callee);
479  if(ICFGEdge* callEdge = addCallEdge(callBlockNode, calleeEntryNode, cs))
480  {
481  for (const SVFStmt *stmt : callBlockNode->getSVFStmts())
482  {
483  if(const CallPE *callPE = SVFUtil::dyn_cast<CallPE>(stmt))
484  {
485  if(callPE->getFunEntryICFGNode() == calleeEntryNode)
486  SVFUtil::cast<CallCFGEdge>(callEdge)->addCallPE(callPE);
487  }
488  }
489  }
490  if(ICFGEdge* retEdge = addRetEdge(calleeExitNode, retBlockNode, cs))
491  {
492  for (const SVFStmt *stmt : retBlockNode->getSVFStmts())
493  {
494  if(const RetPE *retPE = SVFUtil::dyn_cast<RetPE>(stmt))
495  {
496  if(retPE->getFunExitICFGNode() == calleeExitNode)
497  SVFUtil::cast<RetCFGEdge>(retEdge)->addRetPE(retPE);
498  }
499  }
500  }
501 
503  if(ICFGEdge* edge = hasIntraICFGEdge(callBlockNode,retBlockNode, ICFGEdge::IntraCF))
504  removeICFGEdge(edge);
505  }
506 
507  }
508  }
509  // dump ICFG
510  if (Options::DumpICFG())
511  {
512  dump("icfg_final");
513  }
514 }
