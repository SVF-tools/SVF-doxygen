41 void CFLAlias::onTheFlyCallGraphSolve(const CallSiteToFunPtrMap& callsites, CallEdgeMap& newEdges)
42 {
43  for(CallSiteToFunPtrMap::const_iterator iter = callsites.begin(), eiter = callsites.end(); iter!=eiter; ++iter)
44  {
45  const CallICFGNode* cs = iter->first;
46 
47  if (SVFUtil::getSVFCallSite(cs->getCallSite()).isVirtualCall())
48  {
49  const SVFValue* vtbl = SVFUtil::getSVFCallSite(cs->getCallSite()).getVtablePtr();
50  assert(pag->hasValueNode(vtbl));
51  NodeID vtblId = pag->getValueNode(vtbl);
52  resolveCPPIndCalls(cs, getCFLPts(vtblId), newEdges);
53  }
54  else
55  resolveIndCalls(iter->first,getCFLPts(iter->second),newEdges);
56  }
57 }

63 void CFLAlias::connectCaller2CalleeParams(CallSite cs, const SVFFunction* F)
64 {
65  assert(F);
66 
67  DBOUT(DAndersen, outs() << "connect parameters from indirect callsite " << cs.getInstruction()->toString() << " to callee " << *F << "\n");
68 
69  CallICFGNode* callBlockNode = svfir->getICFG()->getCallICFGNode(cs.getInstruction());
70  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
71 
72  if(SVFUtil::isHeapAllocExtFunViaRet(F) && svfir->callsiteHasRet(retBlockNode))
73  {
74  heapAllocatorViaIndCall(cs);
75  }
76 
77  if (svfir->funHasRet(F) && svfir->callsiteHasRet(retBlockNode))
78  {
79  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
80  const PAGNode* fun_return = svfir->getFunRet(F);
81  if (cs_return->isPointer() && fun_return->isPointer())
82  {
83  NodeID dstrec = cs_return->getId();
84  NodeID srcret = fun_return->getId();
85  addCopyEdge(srcret, dstrec);
86  }
87  else
88  {
89  DBOUT(DAndersen, outs() << "not a pointer ignored\n");
90  }
91  }
92 
93  if (svfir->hasCallSiteArgsMap(callBlockNode) && svfir->hasFunArgsList(F))
94  {
95 
96  // connect actual and formal param
97  const SVFIR::SVFVarList& csArgList = svfir->getCallSiteArgsList(callBlockNode);
98  const SVFIR::SVFVarList& funArgList = svfir->getFunArgsList(F);
99  //Go through the fixed parameters.
100  DBOUT(DPAGBuild, outs() << " args:");
101  SVFIR::SVFVarList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
102  SVFIR::SVFVarList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
103  for (; funArgIt != funArgEit; ++csArgIt, ++funArgIt)
104  {
105  //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
106  if (csArgIt == csArgEit)
107  {
108  DBOUT(DAndersen, outs() << " !! not enough args\n");
109  break;
110  }
111  const PAGNode *cs_arg = *csArgIt ;
112  const PAGNode *fun_arg = *funArgIt;
113 
114  if (cs_arg->isPointer() && fun_arg->isPointer())
115  {
116  DBOUT(DAndersen, outs() << "process actual parm " << cs_arg->toString() << " \n");
117  NodeID srcAA = cs_arg->getId();
118  NodeID dstFA = fun_arg->getId();
119  addCopyEdge(srcAA, dstFA);
120  }
121  }
122 
123  //Any remaining actual args must be varargs.
124  if (F->isVarArg())
125  {
126  NodeID vaF = svfir->getVarargNode(F);
127  DBOUT(DPAGBuild, outs() << "\n varargs:");
128  for (; csArgIt != csArgEit; ++csArgIt)
129  {
130  const PAGNode *cs_arg = *csArgIt;
131  if (cs_arg->isPointer())
132  {
133  NodeID vnAA = cs_arg->getId();
134  addCopyEdge(vnAA,vaF);
135  }
136  }
137  }
138  if(csArgIt != csArgEit)
139  {
140  writeWrnMsg("too many args to non-vararg func.");
141  writeWrnMsg("(" + cs.getInstruction()->getSourceLoc() + ")");
142  }
143  }
144 }

146 void CFLAlias::heapAllocatorViaIndCall(CallSite cs)
147 {
148  assert(SVFUtil::getCallee(cs) == nullptr && "not an indirect callsite?");
149  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
150  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
151  NodeID srcret;
152  CallSite2DummyValPN::const_iterator it = callsite2DummyValPN.find(cs);
153  if(it != callsite2DummyValPN.end())
154  {
155  srcret = it->second;
156  }
157  else
158  {
159  NodeID valNode = svfir->addDummyValNode();
160  NodeID objNode = svfir->addDummyObjNode(cs.getType());
161  callsite2DummyValPN.insert(std::make_pair(cs,valNode));
162  graph->addCFLNode(valNode, new CFLNode(valNode));
163  graph->addCFLNode(objNode, new CFLNode(objNode));
164  srcret = valNode;
165  }
166 
167  NodeID dstrec = cs_return->getId();
168  addCopyEdge(srcret, dstrec);
169 }

174 bool CFLAlias::updateCallGraph(const CallSiteToFunPtrMap& callsites)
175 {
176  CallEdgeMap newEdges;
177  onTheFlyCallGraphSolve(callsites,newEdges);
178  for(CallEdgeMap::iterator it = newEdges.begin(), eit = newEdges.end(); it!=eit; ++it )
179  {
180  CallSite cs = SVFUtil::getSVFCallSite(it->first->getCallSite());
181  for(FunctionSet::iterator cit = it->second.begin(), ecit = it->second.end(); cit!=ecit; ++cit)
182  {
183  connectCaller2CalleeParams(cs,*cit);
184  }
185  }
186 
187  return (!solver->isWorklistEmpty());
188 }
