226  inline Condition getVFCond(const SVFGNode* node) const
227  {
228  SVFGNodeToCondMap::const_iterator it = svfgNodeToCondMap.find(node);
229  if(it==svfgNodeToCondMap.end())
230  {
231  return getFalseCond();
232  }
233  return it->second;
234  }
235  inline bool setVFCond(const SVFGNode* node, const Condition &cond)
236  {
237  SVFGNodeToCondMap::iterator it = svfgNodeToCondMap.find(node);
238  // until a fixed-point is reached (condition is not changed)
239  if(it!=svfgNodeToCondMap.end() && isEquivalentBranchCond(it->second, cond))
240  return false;
241 
242  svfgNodeToCondMap[node] = cond;
243  return true;
244  }
