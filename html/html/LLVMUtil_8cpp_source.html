71 void LLVMUtil::getFunReachableBBs (const Function* fun, std::vector<const SVFBasicBlock*> &reachableBBs)
72 {
73  assert(!SVFUtil::isExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun)) && "The calling function cannot be an external function.");
74  //initial DominatorTree
75  DominatorTree dt;
76  dt.recalculate(const_cast<Function&>(*fun));
77 
78  Set<const BasicBlock*> visited;
79  std::vector<const BasicBlock*> bbVec;
80  bbVec.push_back(&fun->getEntryBlock());
81  while(!bbVec.empty())
82  {
83  const BasicBlock* bb = bbVec.back();
84  bbVec.pop_back();
85  const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
86  reachableBBs.push_back(svfbb);
87  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
88  {
89  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
90  DI != DE; ++DI)
91  {
92  const BasicBlock* succbb = (*DI)->getBlock();
93  if(visited.find(succbb)==visited.end())
94  visited.insert(succbb);
95  else
96  continue;
97  bbVec.push_back(succbb);
98  }
99  }
100  }
101 }

106 bool LLVMUtil::functionDoesNotRet (const Function* fun)
107 {
108  const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
109  if (SVFUtil::isExtCall(svffun))
110  {
111  return fun->getReturnType()->isVoidTy();
112  }
113  std::vector<const BasicBlock*> bbVec;
114  Set<const BasicBlock*> visited;
115  bbVec.push_back(&fun->getEntryBlock());
116  while(!bbVec.empty())
117  {
118  const BasicBlock* bb = bbVec.back();
119  bbVec.pop_back();
120  for (BasicBlock::const_iterator it = bb->begin(), eit = bb->end();
121  it != eit; ++it)
122  {
123  if(SVFUtil::isa<ReturnInst>(*it))
124  return false;
125  }
126 
127  for (succ_const_iterator sit = succ_begin(bb), esit = succ_end(bb);
128  sit != esit; ++sit)
129  {
130  const BasicBlock* succbb = (*sit);
131  if(visited.find(succbb)==visited.end())
132  visited.insert(succbb);
133  else
134  continue;
135  bbVec.push_back(succbb);
136  }
137  }
138  return true;

144 bool LLVMUtil::isUncalledFunction (const Function* fun)
145 {
146  if(fun->hasAddressTaken())
147  return false;
148  if (LLVMUtil::isProgEntryFunction(fun))
149  return false;
150  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
151  {
152  if (LLVMUtil::isCallSite(*i))
153  return false;
154  }
155  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
156  {
157  const LLVMModuleSet::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
158  for (LLVMModuleSet::FunctionSetType::const_iterator it = decls.begin(),
159  eit = decls.end(); it != eit; ++it)
160  {
161  const Function* decl = *it;
162  if(decl->hasAddressTaken())
163  return false;
164  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
165  {
166  if (LLVMUtil::isCallSite(*i))
167  return false;
168  }
169  }
170  }
171  return true;

471 const std::string LLVMUtil::getSourceLoc(const Value* val )
472 {
473  if(val==nullptr) return "{ empty val }";
474 
475  std::string str;
476  std::stringstream rawstr(str);
477  rawstr << "{ ";
478 
479  if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
480  {
481  if (SVFUtil::isa<AllocaInst>(inst))
482  {
483  for (llvm::DbgInfoIntrinsic *DII : FindDbgAddrUses(const_cast<Instruction*>(inst)))
484  {
485  if (llvm::DbgDeclareInst *DDI = SVFUtil::dyn_cast<llvm::DbgDeclareInst>(DII))
486  {
487  llvm::DIVariable *DIVar = SVFUtil::cast<llvm::DIVariable>(DDI->getVariable());
488  rawstr << "\"ln\": " << DIVar->getLine() << ", \"fl\": \"" << DIVar->getFilename().str() << "\"";
489  break;
490  }
491  }
492  }
493  else if (MDNode *N = inst->getMetadata("dbg")) // Here I is an LLVM instruction
494  {
495  llvm::DILocation* Loc = SVFUtil::cast<llvm::DILocation>(N); // DILocation is in DebugInfo.h
496  unsigned Line = Loc->getLine();
497  unsigned Column = Loc->getColumn();
498  std::string File = Loc->getFilename().str();
499  //StringRef Dir = Loc.getDirectory();
500  if(File.empty() || Line == 0)
501  {
502  auto inlineLoc = Loc->getInlinedAt();
503  if(inlineLoc)
504  {
505  Line = inlineLoc->getLine();
506  Column = inlineLoc->getColumn();
507  File = inlineLoc->getFilename().str();
508  }
509  }
510  rawstr << "\"ln\": " << Line << ", \"cl\": " << Column << ", \"fl\": \"" << File << "\"";
511  }
512  }
513  else if (const Argument* argument = SVFUtil::dyn_cast<Argument>(val))
514  {
515  if (argument->getArgNo()%10 == 1)
516  rawstr << argument->getArgNo() << "st";
517  else if (argument->getArgNo()%10 == 2)
518  rawstr << argument->getArgNo() << "nd";
519  else if (argument->getArgNo()%10 == 3)
520  rawstr << argument->getArgNo() << "rd";
521  else
522  rawstr << argument->getArgNo() << "th";
523  rawstr << " arg " << argument->getParent()->getName().str() << " "
524  << getSourceLocOfFunction(argument->getParent());
525  }
526  else if (const GlobalVariable* gvar = SVFUtil::dyn_cast<GlobalVariable>(val))
527  {
528  rawstr << "Glob ";
529  NamedMDNode* CU_Nodes = gvar->getParent()->getNamedMetadata("llvm.dbg.cu");
530  if(CU_Nodes)
531  {
532  for (unsigned i = 0, e = CU_Nodes->getNumOperands(); i != e; ++i)
533  {
534  llvm::DICompileUnit *CUNode = SVFUtil::cast<llvm::DICompileUnit>(CU_Nodes->getOperand(i));
535  for (llvm::DIGlobalVariableExpression *GV : CUNode->getGlobalVariables())
536  {
537  llvm::DIGlobalVariable * DGV = GV->getVariable();
538 
539  if(DGV->getName() == gvar->getName())
540  {
541  rawstr << "\"ln\": " << DGV->getLine() << ", \"fl\": \"" << DGV->getFilename().str() << "\"";
542  }
543 
544  }
545  }
546  }
547  }
548  else if (const Function* func = SVFUtil::dyn_cast<Function>(val))
549  {
550  rawstr << getSourceLocOfFunction(func);
551  }
552  else if (const BasicBlock* bb = SVFUtil::dyn_cast<BasicBlock>(val))
553  {
554  rawstr << "\"basic block\": " << bb->getName().str() << ", \"location\": " << getSourceLoc(bb->getFirstNonPHI());
555  }
556  else if(LLVMUtil::isConstDataOrAggData(val))
557  {
558  rawstr << "constant data";
559  }
560  else
561  {
562  rawstr << "N/A";
563  }
564  rawstr << " }";
565 
566  if(rawstr.str()=="{ }")
567  return "";
568  return rawstr.str();
569 }

654 {
655  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
656  {
657  if (LLVMUtil::isValVtbl(v))
658  return false;
659  else if (!v->hasInitializer())
660  {
661  return !v->isExternalLinkage(v->getLinkage());
662  }

664  {
665  StInfo *stInfo = LLVMModuleSet::getLLVMModuleSet()->getSVFType(v->getInitializer()->getType())->getTypeInfo();
666  const std::vector<const SVFType*> &fields = stInfo->getFlattenFieldTypes();
667  for (std::vector<const SVFType*>::const_iterator it = fields.begin(), eit = fields.end(); it != eit; ++it)
668  {
669  const SVFType* elemTy = *it;
670  assert(!SVFUtil::isa<SVFFunctionType>(elemTy) && "Initializer of a global is a function?");
671  if (SVFUtil::isa<SVFPointerType>(elemTy))
672  return false;
673  }
674

676  }
677  }
