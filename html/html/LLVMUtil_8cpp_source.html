46  createobj = true;

48  createobj = true;
49  if (SVFUtil::isa<GlobalVariable>(ref))
50  createobj = true;

60 void SVFUtil::getFunReachableBBs (const Function * fun, DominatorTree* dt, std::vector<const BasicBlock*> &reachableBBs)
61 {
62  Set<const BasicBlock*> visited;
63  std::vector<const BasicBlock*> bbVec;
64  bbVec.push_back(&fun->getEntryBlock());
65  while(!bbVec.empty())
66  {
67  const BasicBlock* bb = bbVec.back();
68  bbVec.pop_back();
69  reachableBBs.push_back(bb);
70  if(DomTreeNode *dtNode = dt->getNode(const_cast<BasicBlock*>(bb)))
71  {
72  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
73  DI != DE; ++DI)
74  {
75  const BasicBlock* succbb = (*DI)->getBlock();
76  if(visited.find(succbb)==visited.end())
77  visited.insert(succbb);
78  else
79  continue;
80  bbVec.push_back(succbb);
81  }
82  }
83  }
84 }

126 bool SVFUtil::isDeadFunction (const Function * fun)
127 {
128  if(fun->hasAddressTaken())
129  return false;
130  if(isProgEntryFunction(fun))
131  return false;
132  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
133  {
134  if (SVFUtil::isCallSite(*i))
135  return false;
136  }
137  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
138  {
139  const SVFModule::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
140  for (SVFModule::FunctionSetType::const_iterator it = decls.begin(),
141  eit = decls.end(); it != eit; ++it)
142  {
143  const Function *decl = (*it)->getLLVMFun();
144  if(decl->hasAddressTaken())
145  return false;
146  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
147  {
148  if (SVFUtil::isCallSite(*i))
149  return false;
150  }
151  }
152  }
153  return true;
154 }

214 void SVFUtil::getNextInsts(const Instruction* curInst, std::vector<const Instruction*>& instList)
215 {
216  if (!curInst->isTerminator())
217  {
218  const Instruction* nextInst = curInst->getNextNode();
219  if (isIntrinsicInst(nextInst))
220  getNextInsts(nextInst, instList);

224  else

231  if (isIntrinsicInst(nextInst))

270 const Type* SVFUtil::getTypeOfHeapAlloc(const Instruction *inst)
271 {
272  const PointerType* type = SVFUtil::dyn_cast<PointerType>(inst->getType());
273 
274  if(isHeapAllocExtCallViaRet(inst))
275  {
276  const Instruction* nextInst = inst->getNextNode();
277  if(nextInst && nextInst->getOpcode() == Instruction::BitCast)
278  // we only consider bitcast instructions and ignore others (e.g., IntToPtr and ZExt)
279  type = SVFUtil::dyn_cast<PointerType>(inst->getNextNode()->getType());
280  }
281  else if(isHeapAllocExtCallViaArg(inst))
282  {
283  CallSite cs = getLLVMCallSite(inst);
284  int arg_pos = getHeapAllocHoldingArgPosition(getCallee(cs));
285  const Value *arg = cs.getArgument(arg_pos);
286  type = SVFUtil::dyn_cast<PointerType>(arg->getType());
287  }
288  else
289  {
290  assert( false && "not a heap allocation instruction?");
291  }
292 
293  assert(type && "not a pointer type?");
294  return type->getElementType();
295 }
