60 bool LLVMUtil::isConstantObjSym(const Value *val)
61 {
62  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
63  {
64  if (cppUtil::isValVtbl(v))
65  return false;
66  else if (!v->hasInitializer())
67  {
68  if(v->isExternalLinkage(v->getLinkage()))
69  return false;
70  else
71  return true;
72  }
73  else
74  {
75  StInfo *stInfo = SymbolTableInfo::SymbolInfo()->getStructInfo(v->getInitializer()->getType());
76  const std::vector<const Type*> &fields = stInfo->getFlattenFieldTypes();
77  for (std::vector<const Type*>::const_iterator it = fields.begin(), eit = fields.end(); it != eit; ++it)
78  {
79  const Type *elemTy = *it;
80  assert(!SVFUtil::isa<FunctionType>(elemTy) && "Initializer of a global is a function?");
81  if (SVFUtil::isa<PointerType>(elemTy))
82  return false;
83  }
84 
85  return v->isConstant();
86  }
87  }
88  return SVFUtil::isConstantData(val);
89 }

94 void LLVMUtil::getFunReachableBBs (const SVFFunction* svfFun, std::vector<const BasicBlock*> &reachableBBs)
95 {
96  assert(!SVFUtil::isExtCall(svfFun) && "The calling function cannot be an external function.");
97  //initial DominatorTree
98  DominatorTree dt;
99  dt.recalculate(*svfFun->getLLVMFun());
100 
101  Set<const BasicBlock*> visited;
102  std::vector<const BasicBlock*> bbVec;
103  bbVec.push_back(&svfFun->getLLVMFun()->getEntryBlock());
104  while(!bbVec.empty())
105  {
106  const BasicBlock* bb = bbVec.back();
107  bbVec.pop_back();
108  reachableBBs.push_back(bb);
109  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
110  {
111  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
112  DI != DE; ++DI)
113  {
114  const BasicBlock* succbb = (*DI)->getBlock();
115  if(visited.find(succbb)==visited.end())
116  visited.insert(succbb);
117  else
118  continue;
119  bbVec.push_back(succbb);
120  }
121  }
122  }
123 }

128 bool LLVMUtil::functionDoesNotRet (const Function * fun)
129 {
130  const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
131  if (SVFUtil::isExtCall(svffun))
132  {
133  return fun->getReturnType()->isVoidTy();
134  }

136  Set<const BasicBlock*> visited;
137  bbVec.push_back(&fun->getEntryBlock());
138  while(!bbVec.empty())
139  {

141  bbVec.pop_back();
142  for (BasicBlock::const_iterator it = bb->begin(), eit = bb->end();
143  it != eit; ++it)
144  {
145  if(SVFUtil::isa<ReturnInst>(*it))
146  return false;
147  }

150  sit != esit; ++sit)

152  const BasicBlock* succbb = (*sit);
153  if(visited.find(succbb)==visited.end())
154  visited.insert(succbb);
155  else
156  continue;
157  bbVec.push_back(succbb);

162 // }

169 bool LLVMUtil::isUncalledFunction (const Function * fun)
170 {
171  if(fun->hasAddressTaken())
172  return false;
173  if(SVFUtil::isProgEntryFunction(fun))
174  return false;
175  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
176  {
177  if (SVFUtil::isCallSite(*i))
178  return false;
179  }
180  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
181  {
182  const SVFModule::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
183  for (SVFModule::FunctionSetType::const_iterator it = decls.begin(),
184  eit = decls.end(); it != eit; ++it)
185  {
186  const Function *decl = (*it)->getLLVMFun();
187  if(decl->hasAddressTaken())
188  return false;
189  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
190  {
191  if (SVFUtil::isCallSite(*i))
192  return false;
193  }
194  }
195  }
196  return true;
197 }

259 void LLVMUtil::getNextInsts(const Instruction* curInst, std::vector<const Instruction*>& instList)
260 {
261  if (!curInst->isTerminator())
262  {
263  const Instruction* nextInst = curInst->getNextNode();
264  if (SVFUtil::isIntrinsicInst(nextInst))
265  getNextInsts(nextInst, instList);
266  else
267  instList.push_back(nextInst);
268  }
269  else
270  {
271  const BasicBlock *BB = curInst->getParent();
272  // Visit all successors of BB in the CFG
273  for (succ_const_iterator it = succ_begin(BB), ie = succ_end(BB); it != ie; ++it)
274  {
275  const Instruction* nextInst = &((*it)->front());
276  if (SVFUtil::isIntrinsicInst(nextInst))
277  getNextInsts(nextInst, instList);
278  else
279  instList.push_back(nextInst);
280  }
281  }
282 }

286 void LLVMUtil::getPrevInsts(const Instruction* curInst, std::vector<const Instruction*>& instList)
287 {
288  if (curInst != &(curInst->getParent()->front()))
289  {
290  const Instruction* prevInst = curInst->getPrevNode();
291  if (SVFUtil::isIntrinsicInst(prevInst))
292  getPrevInsts(prevInst, instList);
293  else
294  instList.push_back(prevInst);
295  }
296  else
297  {
298  const BasicBlock *BB = curInst->getParent();
299  // Visit all successors of BB in the CFG
300  for (const_pred_iterator it = pred_begin(BB), ie = pred_end(BB); it != ie; ++it)

302  const Instruction* prevInst = &((*it)->back());
303  if (SVFUtil::isIntrinsicInst(prevInst))
304  getPrevInsts(prevInst, instList);

312  * Get the first dominated cast instruction for heap allocations since they typically come from void* (i8*)
313  * for example, %4 = call align 16 i8* @malloc(i64 10); %5 = bitcast i8* %4 to i32*
314  * return %5 whose type is i32* but not %4 whose type is i8*
315  */
316 const Value* LLVMUtil::getUniqueUseViaCastInst(const Value* val)
317 {
318  const PointerType * type = SVFUtil::dyn_cast<PointerType>(val->getType());
319  assert(type && "this value should be a pointer type!");
321  if (IntegerType *IT = SVFUtil::dyn_cast<IntegerType>(getPtrElementType(type)))
322  {
323  if (IT->getBitWidth() == 8 && val->getNumUses()==1)
324  {
325  const Use *u = &*val->use_begin();
326  return SVFUtil::dyn_cast<BitCastInst>(u->getUser());
327  }
328  }
329  return nullptr;
330 }

335 const Type* LLVMUtil::getTypeOfHeapAlloc(const Instruction *inst)
336 {
337  const PointerType* type = SVFUtil::dyn_cast<PointerType>(inst->getType());
338 
339  if(SVFUtil::isHeapAllocExtCallViaRet(inst))
340  {
341  if(const Value* v = getUniqueUseViaCastInst(inst))
342  {
343  if(const PointerType* newTy = SVFUtil::dyn_cast<PointerType>(v->getType()))
344  type = newTy;
345  }
346  }
347  else if(SVFUtil::isHeapAllocExtCallViaArg(inst))
348  {
349  CallSite cs = SVFUtil::getLLVMCallSite(inst);
350  int arg_pos = SVFUtil::getHeapAllocHoldingArgPosition(SVFUtil::getCallee(cs));
351  const Value *arg = cs.getArgument(arg_pos);
352  type = SVFUtil::dyn_cast<PointerType>(arg->getType());
353  }
354  else
355  {
356  assert( false && "not a heap allocation instruction?");
357  }
358 
359  assert(type && "not a pointer type?");
360  return getPtrElementType(type);
361 }
