77 bool PTACallGraphNode::isReachableFromProgEntry() const
78 {
79  std::stack<const PTACallGraphNode*> nodeStack;
80  NodeBS visitedNodes;
81  nodeStack.push(this);
82  visitedNodes.set(getId());
83 
84  while (nodeStack.empty() == false)
85  {
86  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
87  nodeStack.pop();
88 
89  if (SVFUtil::isProgEntryFunction(node->getFunction()))
90  return true;
91 
92  for (const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
93  {
94  PTACallGraphEdge* edge = *it;
95  if (visitedNodes.test_and_set(edge->getSrcID()))
96  nodeStack.push(edge->getSrcNode());
97  }
98  }
99 
100  return false;
101 }

133 PTACallGraphEdge* PTACallGraph::hasGraphEdge(PTACallGraphNode* src, PTACallGraphNode* dst,PTACallGraphEdge::CEDGEK kind, CallSiteID csId) const
134 {
135  PTACallGraphEdge edge(src,dst,kind,csId);
136  PTACallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
137  PTACallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
138  if (outEdge && inEdge)
139  {
140  assert(outEdge == inEdge && "edges not match");
141  return outEdge;
142  }
143  else
144  return nullptr;
145 }

263 void PTACallGraph::verifyCallGraph()
264 {
265  CallEdgeMap::const_iterator it = indirectCallMap.begin();
266  CallEdgeMap::const_iterator eit = indirectCallMap.end();
267  for (; it != eit; ++it)
268  {
269  const FunctionSet& targets = it->second;
270  if (targets.empty() == false)
271  {
272  const CallBlockNode* cs = it->first;
273  const SVFFunction* func = cs->getCaller();
274  if (getCallGraphNode(func)->isReachableFromProgEntry() == false)
275  writeWrnMsg(func->getName().str() + " has indirect call site but not reachable from main");
276  }

283 bool PTACallGraph::isReachableBetweenFunctions(const SVFFunction* srcFn, const SVFFunction* dstFn) const
284 {
285  PTACallGraphNode* dstNode = getCallGraphNode(dstFn);
286 
287  std::stack<const PTACallGraphNode*> nodeStack;
288  NodeBS visitedNodes;
289  nodeStack.push(dstNode);
290  visitedNodes.set(dstNode->getId());
291 
292  while (nodeStack.empty() == false)
293  {
294  PTACallGraphNode* node = const_cast<PTACallGraphNode*>(nodeStack.top());
295  nodeStack.pop();
296 
297  if (node->getFunction() == srcFn)
298  return true;
299 
300  for (CallGraphEdgeConstIter it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
301  {
302  PTACallGraphEdge* edge = *it;
303  if (visitedNodes.test_and_set(edge->getSrcID()))
304  nodeStack.push(edge->getSrcNode());
305  }
306  }
307 
308  return false;
309 }
