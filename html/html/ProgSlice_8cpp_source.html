43 bool ProgSlice::AllPathReachableSolve()
44 {
45  const SVFGNode* source = getSource();
46  VFWorkList worklist;
47  worklist.push(source);
49  setVFCond(source,getTrueCond());
50 
51  while(!worklist.empty())
52  {
53  const SVFGNode* node = worklist.pop();
54  setCurSVFGNode(node);
55  Condition cond = getVFCond(node);
56  for(SVFGNode::const_iterator it = node->OutEdgeBegin(), eit = node->OutEdgeEnd(); it!=eit; ++it)
57  {
58  const SVFGEdge* edge = (*it);
59  const SVFGNode* succ = edge->getDstNode();
60  if(inBackwardSlice(succ))
61  {
62  Condition vfCond;
63  const SVFBasicBlock* nodeBB = getSVFGNodeBB(node);
64  const SVFBasicBlock* succBB = getSVFGNodeBB(succ);
66  clearCFCond();
67 
68  if(edge->isCallVFGEdge())
69  {
70  vfCond = ComputeInterCallVFGGuard(nodeBB,succBB, getCallSite(edge)->getParent());
71  }
72  else if(edge->isRetVFGEdge())
73  {
74  vfCond = ComputeInterRetVFGGuard(nodeBB,succBB, getRetSite(edge)->getParent());
75  }
76  else
77  vfCond = ComputeIntraVFGGuard(nodeBB,succBB);
78 
79  Condition succPathCond = condAnd(cond, vfCond);
80  if(setVFCond(succ, condOr(getVFCond(succ), succPathCond) ))
81  worklist.push(succ);
82  }
83 
84  DBOUT(DSaber, outs() << " node (" << node->getId() <<
85  ") --> " << "succ (" << succ->getId() << ") condition: " << getVFCond(succ) << "\n");
86  }
87  }
88 
89  return isSatisfiableForAll();
90 }

111 bool ProgSlice::isSatisfiableForPairs()
112 {
113 
114  for(SVFGNodeSetIter it = sinksBegin(), eit = sinksEnd(); it!=eit; ++it)
115  {
116  for(SVFGNodeSetIter sit = it, esit = sinksEnd(); sit!=esit; ++sit)
117  {
118  if(*it == *sit)
119  continue;
120  Condition guard = condAnd(getVFCond(*sit),getVFCond(*it));
121  if(!isEquivalentBranchCond(guard, getFalseCond()))
122  {
123  setFinalCond(guard);
124  return false;
125  }
126  }
127  }
128 
129  return true;
130 }

149 void ProgSlice::evalFinalCond2Event(GenericBug::EventStack &eventStack) const
150 {

152  for(NodeBS::iterator it = elems.begin(), eit = elems.end(); it!=eit; ++it)
153  {

156  eventStack.push_back(SVFBugEvent(
157  SVFBugEvent::Branch|((((u32_t)false) << 4) & BRANCHFLAGMASK), tinst));
