107 inline bool cmpPts (const PointsTo& lpts,const PointsTo& rpts)
108 {
109  if (lpts.count() != rpts.count())
110  return (lpts.count() < rpts.count());
111  else
112  {
113  PointsTo::iterator bit = lpts.begin(), eit = lpts.end();
114  PointsTo::iterator rbit = rpts.begin(), reit = rpts.end();
115  for (; bit != eit && rbit != reit; bit++, rbit++)
116  {
117  if (*bit != *rbit)
118  return (*bit < *rbit);
119  }
120 
121  return false;
122  }
123 }

218 inline std::vector<std::string> split(const std::string& s, char separator)
219 {
220  std::vector<std::string> output;
221  std::string::size_type prev_pos = 0, pos = 0;
222  while ((pos = s.find(separator, pos)) != std::string::npos)
223  {
224  std::string substring(s.substr(prev_pos, pos - prev_pos));
225  if (!substring.empty())
226  {
227  output.push_back(substring);
228  }
229  prev_pos = ++pos;
230  }
231  std::string lastSubstring(s.substr(prev_pos, pos - prev_pos));
232  if (!lastSubstring.empty())
233  {
234  output.push_back(lastSubstring);
235  }
236  return output;
237 }
