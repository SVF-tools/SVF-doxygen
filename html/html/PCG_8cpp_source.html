63 bool PCG::mayHappenInParallelBetweenFunctions(const SVFFunction* fun1, const SVFFunction* fun2) const
64 {
65  // if neither of functions are spawnees, then they won't happen in parallel
66  if (isSpawneeFun(fun1) == false && isSpawneeFun(fun2) == false)
67  return false;
68  // if there exit one of the function are not spawner, spawnee or follower, then they won't happen in parallel
69  if (isSpawnerFun(fun1) == false && isSpawneeFun(fun1) == false && isFollowerFun(fun1) == false)
70  return false;
71  if (isSpawnerFun(fun2) == false && isSpawneeFun(fun2) == false && isFollowerFun(fun2) == false)
72  return false;
73 
74  return true;
75 }

140 void PCG::collectSpawners()
141 {
142 
144  FunWorkList worklist;
145  for (FunSet::iterator it = spawners.begin(), eit = spawners.end(); it != eit; ++it)
146  {
147  worklist.push(*it);
148  }
149  while (!worklist.empty())
150  {
151  const SVFFunction* svffun = worklist.pop();
152  PTACallGraphNode* funNode = callgraph->getCallGraphNode(svffun);
153  for (PTACallGraphNode::const_iterator it = funNode->InEdgeBegin(), eit = funNode->InEdgeEnd(); it != eit;
154  ++it)
155  {
156  PTACallGraphEdge* callEdge = (*it);
157  const SVFFunction* caller = callEdge->getSrcNode()->getFunction();
158  if (isSpawnerFun(caller) == false)
159  {
160  worklist.push(caller);
161  addSpawnerFun(caller);
162  }
164  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->directCallsBegin(), deit =
165  callEdge->directCallsEnd(); dit != deit; ++dit)
166  {
167  addSpawnsite((*dit)->getCallSite());
168  }
169  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->indirectCallsBegin(), deit =
170  callEdge->indirectCallsEnd(); dit != deit; ++dit)
171  {
172  addSpawnsite((*dit)->getCallSite());
173  }
174  }

211 void PCG::identifyFollowers()
212 {
213 
214  for (CallInstSet::const_iterator sit = spawnSitesBegin(), esit = spawnSitesEnd(); sit != esit; ++sit)
215  {
216  const SVFInstruction* inst = *sit;
217  BBWorkList bb_worklist;
218  Set<const SVFBasicBlock*> visitedBBs;
219  bb_worklist.push(inst->getParent());
220  while (!bb_worklist.empty())
221  {
222  const SVFBasicBlock* bb = bb_worklist.pop();
223  for (SVFBasicBlock::const_iterator it = bb->begin(), eit = bb->end(); it != eit; ++it)
224  {
225  const SVFInstruction* inst = *it;
226  // mark the callee of this callsite as follower
227  // if this is an call/invoke instruction but not a spawn site
228  if ((SVFUtil::isCallSite(inst)) && !isSpawnsite(inst))
229  {
230  CallICFGNode* cbn = getCallICFGNode(inst);
231  if (callgraph->hasCallGraphEdge(cbn))
232  {
233  for (PTACallGraph::CallGraphEdgeSet::const_iterator cgIt = callgraph->getCallEdgeBegin(cbn),
234  ecgIt = callgraph->getCallEdgeEnd(cbn); cgIt != ecgIt; ++cgIt)
235  {
236  const PTACallGraphEdge* edge = *cgIt;
237  addFollowerFun(edge->getDstNode()->getFunction());
238  }
239  }
240  }
241  }
242  for (const SVFBasicBlock* svf_scc_bb : bb->getSuccessors())
243  {
244  if (visitedBBs.count(svf_scc_bb) == 0)
245  {
246  visitedBBs.insert(svf_scc_bb);
247  bb_worklist.push(svf_scc_bb);
248  }
249  }
250  }
251  }
