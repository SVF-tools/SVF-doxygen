61 {
62 
63  initialize(module);
64 
65  ContextCond::setMaxCxtLen(Options::CxtLimit());
66 
67  for (SVFGNodeSetIter iter = sourcesBegin(), eiter = sourcesEnd();
68  iter != eiter; ++iter)
69  {
70  setCurSlice(*iter);
71 
72  DBOUT(DGENERAL, outs() << "Analysing slice:" << (*iter)->getId() << ")\n");
73  ContextCond cxt;
74  DPIm item((*iter)->getId(),cxt);
75  forwardTraverse(item);
76 
79  if (getCurSlice()->isReachGlobal())
80  {
81  DBOUT(DSaber, outs() << "Forward analysis reaches globals for slice:" << (*iter)->getId() << ")\n");
82  }
83  else
84  {
85  DBOUT(DSaber, outs() << "Forward process for slice:" << (*iter)->getId() << " (size = " << getCurSlice()->getForwardSliceSize() << ")\n");
86 
87  for (SVFGNodeSetIter sit = getCurSlice()->sinksBegin(), esit =
88  getCurSlice()->sinksEnd(); sit != esit; ++sit)
89  {
90  ContextCond cxt;
91  DPIm item((*sit)->getId(),cxt);
92  backwardTraverse(item);
93  }
94 
95  DBOUT(DSaber, outs() << "Backward process for slice:" << (*iter)->getId() << " (size = " << getCurSlice()->getBackwardSliceSize() << ")\n");
96 
97  if(Options::DumpSlice())
98  annotateSlice(_curSlice);
99 
100  if(_curSlice->AllPathReachableSolve())
101  _curSlice->setAllReachable();
102 
103  DBOUT(DSaber, outs() << "Guard computation for slice:" << (*iter)->getId() << ")\n");
104  }
105 
106  reportBug(getCurSlice());
107  }
108  finalize();

117 bool SrcSnkDDA::isInAWrapper(const SVFGNode* src, CallSiteSet& csIdSet)
118 {
119 
120  bool reachFunExit = false;
121 
122  WorkList worklist;
123  worklist.push(src);
124  SVFGNodeBS visited;
125  u32_t step = 0;
126  while (!worklist.empty())
127  {
128  const SVFGNode* node = worklist.pop();
129 
130  if(visited.test(node->getId())==0)
131  visited.set(node->getId());
132  else
133  continue;
134  // reaching maximum steps when traversing on SVFG to identify a memory allocation wrapper
135  if (step++ > Options::MaxStepInWrapper())
136  return false;
137 
138  for (SVFGNode::const_iterator it = node->OutEdgeBegin(), eit =
139  node->OutEdgeEnd(); it != eit; ++it)
140  {
141  const SVFGEdge* edge = (*it);
142  //assert(edge->isDirectVFGEdge() && "the edge should always be direct VF");
143  // if this is a call edge
144  if(edge->isCallDirectVFGEdge())
145  {
146  return false;
147  }
148  // if this is a return edge
149  else if(edge->isRetDirectVFGEdge())
150  {
151  reachFunExit = true;
152  csIdSet.insert(getSVFG()->getCallSite(SVFUtil::cast<RetDirSVFGEdge>(edge)->getCallSiteId()));
153  }
154  // (1) an intra direct edge, we will keep tracking
155  // (2) an intra indirect edge, we only track if the succ SVFGNode is a load, which means we only track one level store-load pair .
156  // (3) do not track for all other interprocedural edges.
157  else
158  {
159  const SVFGNode* succ = edge->getDstNode();
160  if(SVFUtil::isa<IntraDirSVFGEdge>(edge))
161  {
162  if (SVFUtil::isa<CopySVFGNode, GepSVFGNode, PHISVFGNode,
163  FormalRetSVFGNode, ActualRetSVFGNode,
164  StoreSVFGNode>(succ))
165  {
166  worklist.push(succ);
167  }
168  }
169  else if(SVFUtil::isa<IntraIndSVFGEdge>(edge))
170  {
171  if(SVFUtil::isa<LoadSVFGNode, IntraMSSAPHISVFGNode>(succ))
172  {
173  worklist.push(succ);
174  }
175  }
176  else
177  return false;
178  }
179  }
180  }
181  if(reachFunExit)
182  return true;
183  else
184  return false;
185 }

191 void SrcSnkDDA::FWProcessOutgoingEdge(const DPIm& item, SVFGEdge* edge)
192 {
193  DBOUT(DSaber,outs() << "\n##processing source: " << getCurSlice()->getSource()->getId() <<" forward propagate from (" << edge->getSrcID());
194 
195  // for indirect SVFGEdge, the propagation should follow the def-use chains
196  // points-to on the edge indicate whether the object of source node can be propagated
197 
198  const SVFGNode* dstNode = edge->getDstNode();
199  DPIm newItem(dstNode->getId(),item.getContexts());
200 
202  if(isGlobalSVFGNode(dstNode) || getCurSlice()->isReachGlobal())
203  {
204  getCurSlice()->setReachGlobal();
205  return;
206  }
207 
208 
210  // push context for calling
211  if (edge->isCallVFGEdge())
212  {
213  CallSiteID csId = 0;
214  if(const CallDirSVFGEdge* callEdge = SVFUtil::dyn_cast<CallDirSVFGEdge>(edge))
215  csId = callEdge->getCallSiteId();
216  else
217  csId = SVFUtil::cast<CallIndSVFGEdge>(edge)->getCallSiteId();
218 
219  newItem.pushContext(csId);
220  DBOUT(DSaber, outs() << " push cxt [" << csId << "] ");
221  }
222  // match context for return
223  else if (edge->isRetVFGEdge())
224  {
225  CallSiteID csId = 0;
226  if(const RetDirSVFGEdge* callEdge = SVFUtil::dyn_cast<RetDirSVFGEdge>(edge))
227  csId = callEdge->getCallSiteId();
228  else
229  csId = SVFUtil::cast<RetIndSVFGEdge>(edge)->getCallSiteId();
230 
231  if (newItem.matchContext(csId) == false)
232  {
233  DBOUT(DSaber, outs() << "-|-\n");
234  return;
235  }
236  DBOUT(DSaber, outs() << " pop cxt [" << csId << "] ");
237  }
238 
240  if(forwardVisited(dstNode,newItem))
241  {
242  DBOUT(DSaber,outs() << " node "<< dstNode->getId() <<" has been visited\n");
243  return;
244  }
245  else
246  addForwardVisited(dstNode, newItem);
247 
248  if(pushIntoWorklist(newItem))
249  DBOUT(DSaber,outs() << " --> " << edge->getDstID() << ", cxt size: " << newItem.getContexts().cxtSize() <<")\n");
250 
251 }
