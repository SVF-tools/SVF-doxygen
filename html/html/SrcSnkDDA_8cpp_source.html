62 void SrcSnkDDA::analyze(SVFModule* module)
63 {
64 
65  initialize(module);
66 
67  ContextCond::setMaxCxtLen(Options::CxtLimit);
68 
69  for (SVFGNodeSetIter iter = sourcesBegin(), eiter = sourcesEnd();
70  iter != eiter; ++iter)
71  {
72  setCurSlice(*iter);
73 
74  DBOUT(DGENERAL, outs() << "Analysing slice:" << (*iter)->getId() << ")\n");
75  ContextCond cxt;
76  DPIm item((*iter)->getId(),cxt);
77  forwardTraverse(item);
78 
81  if (getCurSlice()->isReachGlobal())
82  {
83  DBOUT(DSaber, outs() << "Forward analysis reaches globals for slice:" << (*iter)->getId() << ")\n");
84  }
85  else
86  {
87  DBOUT(DSaber, outs() << "Forward process for slice:" << (*iter)->getId() << " (size = " << getCurSlice()->getForwardSliceSize() << ")\n");
88 
89  for (SVFGNodeSetIter sit = getCurSlice()->sinksBegin(), esit =
90  getCurSlice()->sinksEnd(); sit != esit; ++sit)
91  {
92  ContextCond cxt;
93  DPIm item((*sit)->getId(),cxt);
94  backwardTraverse(item);
95  }
96 
97  DBOUT(DSaber, outs() << "Backward process for slice:" << (*iter)->getId() << " (size = " << getCurSlice()->getBackwardSliceSize() << ")\n");
98 
99  if(Options::DumpSlice)
100  annotateSlice(_curSlice);
101 
102  if(_curSlice->AllPathReachableSolve())
103  _curSlice->setAllReachable();
104 
105  DBOUT(DSaber, outs() << "Guard computation for slice:" << (*iter)->getId() << ")\n");
106  }
107 
108  reportBug(getCurSlice());
109  }
110 
111  finalize();
112

120 bool SrcSnkDDA::isInAWrapper(const SVFGNode* src, CallSiteSet& csIdSet)
121 {
122 
123  bool reachFunExit = false;
124 
125  WorkList worklist;
126  worklist.push(src);
127  SVFGNodeBS visited;
128  u32_t step = 0;
129  while (!worklist.empty())
130  {
131  const SVFGNode* node = worklist.pop();
132 
133  if(visited.test(node->getId())==0)
134  visited.set(node->getId());
135  else
136  continue;
137  // reaching maximum steps when traversing on SVFG to identify a memory allocation wrapper
138  if (step++ > Options::MaxStepInWrapper)
139  return false;
140 
141  for (SVFGNode::const_iterator it = node->OutEdgeBegin(), eit =
142  node->OutEdgeEnd(); it != eit; ++it)
143  {
144  const SVFGEdge* edge = (*it);
145  //assert(edge->isDirectVFGEdge() && "the edge should always be direct VF");
146  // if this is a call edge
147  if(edge->isCallDirectVFGEdge())
148  {
149  return false;
150  }
151  // if this is a return edge
152  else if(edge->isRetDirectVFGEdge())
153  {
154  reachFunExit = true;
155  csIdSet.insert(getSVFG()->getCallSite(SVFUtil::cast<RetDirSVFGEdge>(edge)->getCallSiteId()));
156  }
157  // (1) an intra direct edge, we will keep tracking
158  // (2) an intra indirect edge, we only track if the succ SVFGNode is a load, which means we only track one level store-load pair .
159  // (3) do not track for all other interprocedural edges.
160  else
161  {
162  const SVFGNode* succ = edge->getDstNode();
163  if(SVFUtil::isa<IntraDirSVFGEdge>(edge))
164  {
165  if (SVFUtil::isa<CopySVFGNode>(succ) || SVFUtil::isa<GepSVFGNode>(succ)
166  || SVFUtil::isa<PHISVFGNode>(succ) || SVFUtil::isa<FormalRetSVFGNode>(succ)
167  || SVFUtil::isa<ActualRetSVFGNode>(succ) || SVFUtil::isa<StoreSVFGNode>(succ))
168  {
169  worklist.push(succ);
170  }
171  }
172  else if(SVFUtil::isa<IntraIndSVFGEdge>(edge))
173  {
174  if(SVFUtil::isa<LoadSVFGNode>(succ))
175  {
176  worklist.push(succ);
177  }
178  }
179  else
180  return false;
181  }
182  }
183  }
184  if(reachFunExit)
185  return true;
186  else
187  return false;
188 }

194 void SrcSnkDDA::FWProcessOutgoingEdge(const DPIm& item, SVFGEdge* edge)
195 {
196  DBOUT(DSaber,outs() << "\n##processing source: " << getCurSlice()->getSource()->getId() <<" forward propagate from (" << edge->getSrcID());
197 
198  // for indirect SVFGEdge, the propagation should follow the def-use chains
199  // points-to on the edge indicate whether the object of source node can be propagated
200 
201  const SVFGNode* dstNode = edge->getDstNode();
202  DPIm newItem(dstNode->getId(),item.getContexts());
203 
205  if(isGlobalSVFGNode(dstNode) || getCurSlice()->isReachGlobal())
206  {
207  getCurSlice()->setReachGlobal();
208  return;
209  }
210 
211 
213  // push context for calling
214  if (edge->isCallVFGEdge())
215  {
216  CallSiteID csId = 0;
217  if(const CallDirSVFGEdge* callEdge = SVFUtil::dyn_cast<CallDirSVFGEdge>(edge))
218  csId = callEdge->getCallSiteId();
219  else
220  csId = SVFUtil::cast<CallIndSVFGEdge>(edge)->getCallSiteId();
221 
222  newItem.pushContext(csId);
223  DBOUT(DSaber, outs() << " push cxt [" << csId << "] ");
224  }
225  // match context for return
226  else if (edge->isRetVFGEdge())
227  {
228  CallSiteID csId = 0;
229  if(const RetDirSVFGEdge* callEdge = SVFUtil::dyn_cast<RetDirSVFGEdge>(edge))
230  csId = callEdge->getCallSiteId();
231  else
232  csId = SVFUtil::cast<RetIndSVFGEdge>(edge)->getCallSiteId();
233 
234  if (newItem.matchContext(csId) == false)
235  {
236  DBOUT(DSaber, outs() << "-|-\n");
237  return;
238  }
239  DBOUT(DSaber, outs() << " pop cxt [" << csId << "] ");
240  }
241 
243  if(forwardVisited(dstNode,newItem))
244  {
245  DBOUT(DSaber,outs() << " node "<< dstNode->getId() <<" has been visited\n");
246  return;
247  }
248  else
249  addForwardVisited(dstNode, newItem);
250 
251  if(pushIntoWorklist(newItem))
252  DBOUT(DSaber,outs() << " --> " << edge->getDstID() << ", cxt size: " << newItem.getContexts().cxtSize() <<")\n");
253 
254 }
