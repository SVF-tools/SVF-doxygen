341  void sanitizePts()
342  {
343  for(ConstraintGraph::iterator it = consCG->begin(), eit = consCG->end(); it!=eit; ++it)
344  {
345  const PointsTo& pts = getPts(it->first);
346  NodeBS fldInsenObjs;
347 
348  for (NodeID o : pts)
349  {
350  if(isFieldInsensitive(o))
351  fldInsenObjs.set(o);
352  }
353 
354  for (NodeID o : fldInsenObjs)
355  {
356  const NodeBS &allFields = consCG->getAllFieldsObjVars(o);
357  for (NodeID f : allFields) addPts(it->first, f);
358  }
359  }
360  }
