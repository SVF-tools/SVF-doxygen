90 void SVFGOPT::handleInterValueFlow()
91 {
92  SVFGNodeSet candidates;
93  for (SVFGNodeIDToNodeMapTy::iterator it = SVFG::begin(), eit = SVFG::end();
94  it!=eit; ++it)
95  {
96  SVFGNode* node = it->second;
97  if (SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<ActualRetSVFGNode>(node)
98  || SVFUtil::isa<FormalParmSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node)
99  || SVFUtil::isa<ActualINSVFGNode>(node) || SVFUtil::isa<ActualOUTSVFGNode>(node)
100  || SVFUtil::isa<FormalINSVFGNode>(node) || SVFUtil::isa<FormalOUTSVFGNode>(node))
101  candidates.insert(node);
102  }
103 
104  SVFGNodeSet nodesToBeDeleted;
105  for (SVFGNodeSet::const_iterator it = candidates.begin(), eit = candidates.end();
106  it!=eit; ++it)
107  {
108  SVFGNode* node = *it;
109  if (FormalParmSVFGNode* fp = SVFUtil::dyn_cast<FormalParmSVFGNode>(node))
110  {
111  replaceFParamARetWithPHI(addInterPHIForFP(fp), fp);
112  nodesToBeDeleted.insert(fp);
113  }
114  else if (ActualRetSVFGNode* ar = SVFUtil::dyn_cast<ActualRetSVFGNode>(node))
115  {
116  replaceFParamARetWithPHI(addInterPHIForAR(ar), ar);
117  nodesToBeDeleted.insert(ar);
118  }
119  else if (SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node))
120  {
121  nodesToBeDeleted.insert(node);
122  }
123  else if (SVFUtil::isa<ActualINSVFGNode>(node) || SVFUtil::isa<FormalOUTSVFGNode>(node))
124  {
125  retargetEdgesOfAInFOut(node);
126  nodesToBeDeleted.insert(node);
127  }
128  else if (SVFUtil::isa<ActualOUTSVFGNode>(node) || SVFUtil::isa<FormalINSVFGNode>(node))
129  {
130  if(keepActualOutFormalIn == false)
131  nodesToBeDeleted.insert(node);
132  }
133  }
134 
135  for (SVFGNodeSet::iterator it = nodesToBeDeleted.begin(), eit = nodesToBeDeleted.end(); it != eit; ++it)
136  {
137  SVFGNode* node = *it;
138  if (canBeRemoved(node))
139  {
140  if (SVFUtil::isa<ActualOUTSVFGNode>(node) || SVFUtil::isa<FormalINSVFGNode>(node))
141  retargetEdgesOfAOutFIn(node);
142 
143  removeAllEdges(node);
144  removeSVFGNode(node);
145  }
146  }

152 void SVFGOPT::replaceFParamARetWithPHI(PHISVFGNode* phi, SVFGNode* svfgNode)
153 {
154  assert((SVFUtil::isa<FormalParmSVFGNode>(svfgNode) || SVFUtil::isa<ActualRetSVFGNode>(svfgNode))
155  && "expecting a formal param or actual ret svfg node");
156 
158  NodeID phiId = phi->getId();
160  for (SVFGNode::const_iterator it = svfgNode->OutEdgeBegin(), eit = svfgNode->OutEdgeEnd();
161  it != eit; ++it)
162  {
163  const SVFGEdge* outEdge = *it;
164  SVFGNode* dstNode = outEdge->getDstNode();
165  NodeID dstId = dstNode->getId();
166  if (const CallDirSVFGEdge* callEdge = SVFUtil::dyn_cast<CallDirSVFGEdge>(outEdge))
167  addCallEdge(phiId, dstId, callEdge->getCallSiteId());
168  else if (const RetDirSVFGEdge* retEdge = SVFUtil::dyn_cast<RetDirSVFGEdge>(outEdge))
169  addRetEdge(phiId, dstId, retEdge->getCallSiteId());
170  else
171  addIntraDirectVFEdge(phiId, dstId);
172  }
173 
175  if (FormalParmSVFGNode* fp = SVFUtil::dyn_cast<FormalParmSVFGNode>(svfgNode))
176  {
177  for (SVFGNode::iterator it = svfgNode->InEdgeBegin(), eit = svfgNode->InEdgeEnd();
178  it != eit; ++it)
179  {
180  ActualParmSVFGNode* ap = SVFUtil::cast<ActualParmSVFGNode>((*it)->getSrcNode());
181  addInterPHIOperands(phi, ap->getParam());
182  // connect actual param's def node to phi node
183  addCallEdge(getDef(ap->getParam()), phiId, getCallSiteID(ap->getCallSite(), fp->getFun()));
184  }
185  }
186  else if (ActualRetSVFGNode* ar = SVFUtil::dyn_cast<ActualRetSVFGNode>(svfgNode))
187  {
188  for (SVFGNode::iterator it = svfgNode->InEdgeBegin(), eit = svfgNode->InEdgeEnd();
189  it != eit; ++it)
190  {
191  FormalRetSVFGNode* fr = SVFUtil::cast<FormalRetSVFGNode>((*it)->getSrcNode());
192  addInterPHIOperands(phi, fr->getRet());
193  // connect formal return's def node to phi node
194  addRetEdge(getDef(fr->getRet()), phiId, getCallSiteID(ar->getCallSite(), fr->getFun()));
195  }
196  }
197 
198  removeAllEdges(svfgNode);
199 }

205 void SVFGOPT::retargetEdgesOfAInFOut(SVFGNode* node)
206 {
207  assert(node->getInEdges().size() == 1 && "actual-in/formal-out can only have one incoming edge as its def size");
208 
209  SVFGNode* def = nullptr;
210  PointsTo inPointsTo;
211 
212  SVFGNode::const_iterator it = node->InEdgeBegin();
213  SVFGNode::const_iterator eit = node->InEdgeEnd();
214  for (; it != eit; ++it)
215  {
216  const IndirectSVFGEdge* inEdge = SVFUtil::cast<IndirectSVFGEdge>(*it);
217  inPointsTo = inEdge->getPointsTo();
218 
219  def = inEdge->getSrcNode();
220  if (SVFUtil::isa<ActualINSVFGNode>(node))
221  setActualINDef(node->getId(), def->getId());
222  else if (SVFUtil::isa<FormalOUTSVFGNode>(node))
223  setFormalOUTDef(node->getId(), def->getId());
224  }
225 
226  it = node->OutEdgeBegin(), eit = node->OutEdgeEnd();
227  for (; it != eit; ++it)
228  {
229  const IndirectSVFGEdge* outEdge = SVFUtil::cast<IndirectSVFGEdge>(*it);
230  PointsTo intersection = inPointsTo;
231  intersection &= outEdge->getPointsTo();
232 
233  if (intersection.empty())
234  continue;
235 
236  SVFGNode* dstNode = outEdge->getDstNode();
237  if (const CallIndSVFGEdge* callEdge = SVFUtil::dyn_cast<CallIndSVFGEdge>(outEdge))
238  addCallIndirectSVFGEdge(def->getId(), dstNode->getId(), callEdge->getCallSiteId(), intersection);
239  else if (const RetIndSVFGEdge* retEdge = SVFUtil::dyn_cast<RetIndSVFGEdge>(outEdge))
240  addRetIndirectSVFGEdge(def->getId(), dstNode->getId(), retEdge->getCallSiteId(), intersection);
241  else
242  assert(false && "expecting an inter-procedural SVFG edge");
243  }
244 
245  removeAllEdges(node);
246 }

251 void SVFGOPT::retargetEdgesOfAOutFIn(SVFGNode* node)
252 {
253  SVFGNode::const_iterator inIt = node->InEdgeBegin();
254  SVFGNode::const_iterator inEit = node->InEdgeEnd();
255  for (; inIt != inEit; ++inIt)
256  {
257  const IndirectSVFGEdge* inEdge = SVFUtil::cast<IndirectSVFGEdge>(*inIt);
258  NodeID srcId = inEdge->getSrcID();
259 
260  SVFGNode::const_iterator outIt = node->OutEdgeBegin();
261  SVFGNode::const_iterator outEit = node->OutEdgeEnd();
262  for (; outIt != outEit; ++outIt)
263  {
264  const IndirectSVFGEdge* outEdge = SVFUtil::cast<IndirectSVFGEdge>(*outIt);
265 
266  PointsTo intersection = inEdge->getPointsTo();
267  intersection &= outEdge->getPointsTo();
268  if (intersection.empty())
269  continue;
270 
271  NodeID dstId = outEdge->getDstID();
272  if (const RetIndSVFGEdge* retEdge = SVFUtil::dyn_cast<RetIndSVFGEdge>(inEdge))
273  {
274  addRetIndirectSVFGEdge(srcId, dstId, retEdge->getCallSiteId(), intersection);
275  }
276  else if (const CallIndSVFGEdge* callEdge = SVFUtil::dyn_cast<CallIndSVFGEdge>(inEdge))
277  {
278  addCallIndirectSVFGEdge(srcId, dstId, callEdge->getCallSiteId(), intersection);
279  }
280  else
281  {
282  addIntraIndirectVFEdge(srcId, dstId, intersection);
283  }
284  }
285  }

293 bool SVFGOPT::isConnectingTwoCallSites(const SVFGNode* node) const
294 {
295  bool hasInCallRet = false;
296  bool hasOutCallRet = false;
297 
298  SVFGNode::const_iterator edgeIt = node->InEdgeBegin();
299  SVFGNode::const_iterator edgeEit = node->InEdgeEnd();
300  for (; edgeIt != edgeEit; ++edgeIt)
301  {
302  if (SVFUtil::isa<CallIndSVFGEdge>(*edgeIt) || SVFUtil::isa<RetIndSVFGEdge>(*edgeIt))
303  {
304  hasInCallRet = true;
305  break;
306  }
307  }
308 
309  edgeIt = node->OutEdgeBegin();
310  edgeEit = node->OutEdgeEnd();
311  for (; edgeIt != edgeEit; ++edgeIt)
312  {
313  if (SVFUtil::isa<CallIndSVFGEdge>(*edgeIt) || SVFUtil::isa<RetIndSVFGEdge>(*edgeIt))
314  {
315  hasOutCallRet = true;
316  break;
317  }
318  }
319 
320  if (hasInCallRet && hasOutCallRet)
321  return true;
322  else
323  return false;

335 bool SVFGOPT::canBeRemoved(const SVFGNode * node)
336 {
337  if (SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalParmSVFGNode>(node)
338  || SVFUtil::isa<ActualRetSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node))
339  return true;
340  else if (SVFUtil::isa<ActualINSVFGNode>(node) || SVFUtil::isa<ActualOUTSVFGNode>(node)
341  || SVFUtil::isa<FormalINSVFGNode>(node) || SVFUtil::isa<FormalOUTSVFGNode>(node)
342  || SVFUtil::isa<MSSAPHISVFGNode>(node))
343  {
347  if (isConnectingTwoCallSites(node))
348  return false;
349 
350  if (const ActualINSVFGNode* ai = SVFUtil::dyn_cast<ActualINSVFGNode>(node))
351  {
352  return (actualInOfIndCS(ai) == false);
353  }
354  else if (const ActualOUTSVFGNode* ao = SVFUtil::dyn_cast<ActualOUTSVFGNode>(node))
355  {
356  return (actualOutOfIndCS(ao) == false && isDefOfAInFOut(node) == false);
357  }
358  else if (const FormalINSVFGNode* fi = SVFUtil::dyn_cast<FormalINSVFGNode>(node))
359  {
360  return (formalInOfAddressTakenFunc(fi) == false && isDefOfAInFOut(node) == false);
361  }
362  else if (const FormalOUTSVFGNode* fo = SVFUtil::dyn_cast<FormalOUTSVFGNode>(node))
363  {
364  return (formalOutOfAddressTakenFunc(fo) == false);
365  }
366  }
367 
368  return false;
369 }

374 void SVFGOPT::parseSelfCycleHandleOption()
375 {
376  std::string choice = (Options::SelfCycle.getValue().empty()) ? "" : Options::SelfCycle.getValue();
377  if (choice.empty() || choice == KeepAllSelfCycle)
378  keepAllSelfCycle = true;
379  else if (choice == KeepContextSelfCycle)
380  keepContextSelfCycle = true;
381  else if (choice == KeepNoneSelfCycle)
382  keepAllSelfCycle = keepContextSelfCycle = false;
383  else
384  {
385  SVFUtil::writeWrnMsg("Unrecognised option. All self cycle edges will be kept.");
386  keepAllSelfCycle = true;

393 void SVFGOPT::handleIntraValueFlow()
394 {
395  parseSelfCycleHandleOption();
396 
397  initialWorkList();
398 
399  while (!worklist.empty())
400  {
401  const MSSAPHISVFGNode* node = worklist.pop();
402 
404  if (checkSelfCycleEdges(node))
405  continue;
406 
407  if (node->hasOutgoingEdge() && node->hasIncomingEdge())
408  bypassMSSAPHINode(node);
409 
411  if (node->hasIncomingEdge() && node->hasOutgoingEdge() == false)
412  {
414  SVFGNode::const_iterator edgeIt = node->InEdgeBegin();
415  SVFGNode::const_iterator edgeEit = node->InEdgeEnd();
416  for (; edgeIt != edgeEit; ++edgeIt)
417  addIntoWorklist((*edgeIt)->getSrcNode());
418 
419  removeInEdges(node);
420  }
421  else if (node->hasOutgoingEdge() && node->hasIncomingEdge() == false)
422  {
424  SVFGNode::const_iterator edgeIt = node->OutEdgeBegin();
425  SVFGNode::const_iterator edgeEit = node->OutEdgeEnd();
426  for (; edgeIt != edgeEit; ++edgeIt)
427  addIntoWorklist((*edgeIt)->getDstNode());
428 
429  removeOutEdges(node);
430  }
431 
433  if (node->hasIncomingEdge() == false && node->hasOutgoingEdge() == false)
434  removeSVFGNode(const_cast<MSSAPHISVFGNode*>(node));
435  }
436 }

444 bool SVFGOPT::checkSelfCycleEdges(const MSSAPHISVFGNode* node)
445 {
446  bool hasSelfCycle = false;
447 
448  SVFGEdge::SVFGEdgeSetTy inEdges = node->getInEdges();
449  SVFGNode::const_iterator inEdgeIt = inEdges.begin();
450  SVFGNode::const_iterator inEdgeEit = inEdges.end();
451  for (; inEdgeIt != inEdgeEit; ++inEdgeIt)
452  {
453  SVFGEdge* preEdge = *inEdgeIt;
454 
455  if (preEdge->getSrcID() == preEdge->getDstID())
456  {
457  if (keepAllSelfCycle)
458  {
459  hasSelfCycle = true;
460  break;
461  }
462  else if (keepContextSelfCycle &&
463  (SVFUtil::isa<CallIndSVFGEdge>(preEdge) || SVFUtil::isa<RetIndSVFGEdge>(preEdge)))
464  {
465  hasSelfCycle = true;
466  continue;
467  }
468  else
469  {
470  assert(SVFUtil::isa<IndirectSVFGEdge>(preEdge) && "can only remove indirect SVFG edge");
471  removeSVFGEdge(preEdge);
472  }
473  }
474  }
475 
476  return hasSelfCycle;
477 }

482 void SVFGOPT::bypassMSSAPHINode(const MSSAPHISVFGNode* node)
483 {
484  SVFGNode::const_iterator inEdgeIt = node->InEdgeBegin();
485  SVFGNode::const_iterator inEdgeEit = node->InEdgeEnd();
486  for (; inEdgeIt != inEdgeEit; ++inEdgeIt)
487  {
488  const SVFGEdge* preEdge = *inEdgeIt;
489  const SVFGNode* srcNode = preEdge->getSrcNode();
490 
491  bool added = false;
493  SVFGNode::const_iterator outEdgeIt = node->OutEdgeBegin();
494  SVFGNode::const_iterator outEdgeEit = node->OutEdgeEnd();
495  for (; outEdgeIt != outEdgeEit; ++outEdgeIt)
496  {
497  const SVFGEdge* succEdge = *outEdgeIt;
498  const SVFGNode* dstNode = (*outEdgeIt)->getDstNode();
499  if (srcNode->getId() != dstNode->getId()
500  && addNewSVFGEdge(srcNode->getId(), dstNode->getId(), preEdge, succEdge))
501  added = true;
502  else
503  {
506  addIntoWorklist(dstNode);
507  }
508  }
509 
510  if (added == false)
511  {
514  addIntoWorklist(srcNode);
515  }
516  }
517 
518  removeAllEdges(node);
519 }

525 bool SVFGOPT::addNewSVFGEdge(NodeID srcId, NodeID dstId, const SVFGEdge* preEdge, const SVFGEdge* succEdge)
526 {
527  assert(SVFUtil::isa<IndirectSVFGEdge>(preEdge) && SVFUtil::isa<IndirectSVFGEdge>(succEdge)
528  && "either pre or succ edge is not indirect SVFG edge");
529 
530  const IndirectSVFGEdge* preIndEdge = SVFUtil::cast<IndirectSVFGEdge>(preEdge);
531  const IndirectSVFGEdge* succIndEdge = SVFUtil::cast<IndirectSVFGEdge>(succEdge);
532 
533  PointsTo intersection = preIndEdge->getPointsTo();
534  intersection &= succIndEdge->getPointsTo();
535 
536  if (intersection.empty())
537  return false;
538 
539  assert(bothInterEdges(preEdge, succEdge) == false && "both edges are inter edges");
540 
541  if (const CallIndSVFGEdge* preCallEdge = SVFUtil::dyn_cast<CallIndSVFGEdge>(preEdge))
542  {
543  return addCallIndirectSVFGEdge(srcId, dstId, preCallEdge->getCallSiteId(), intersection);
544  }
545  else if (const CallIndSVFGEdge* succCallEdge = SVFUtil::dyn_cast<CallIndSVFGEdge>(succEdge))
546  {
547  return addCallIndirectSVFGEdge(srcId, dstId, succCallEdge->getCallSiteId(), intersection);
548  }
549  else if (const RetIndSVFGEdge* preRetEdge = SVFUtil::dyn_cast<RetIndSVFGEdge>(preEdge))
550  {
551  return addRetIndirectSVFGEdge(srcId, dstId, preRetEdge->getCallSiteId(), intersection);
552  }
553  else if (const RetIndSVFGEdge* succRetEdge = SVFUtil::dyn_cast<RetIndSVFGEdge>(succEdge))
554  {
555  return addRetIndirectSVFGEdge(srcId, dstId, succRetEdge->getCallSiteId(), intersection);
556  }
557  else
558  {
559  return addIntraIndirectVFEdge(srcId, dstId, intersection);
560  }
561 
562  return false;
