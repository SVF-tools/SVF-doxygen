128  inline bool hasEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)
129  {
130  ConstraintEdge edge(src,dst,kind);
131  if(kind == ConstraintEdge::Copy ||
132  kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
133  return directEdgeSet.find(&edge) != directEdgeSet.end();
134  else if(kind == ConstraintEdge::Addr)
135  return AddrCGEdgeSet.find(&edge) != AddrCGEdgeSet.end();
136  else if(kind == ConstraintEdge::Store)
137  return StoreCGEdgeSet.find(&edge) != StoreCGEdgeSet.end();
138  else if(kind == ConstraintEdge::Load)
139  return LoadCGEdgeSet.find(&edge) != LoadCGEdgeSet.end();
140  else
141  assert(false && "no other kind!");
142  return false;
143  }
144 
146  inline ConstraintEdge* getEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)
147  {
148  ConstraintEdge edge(src,dst,kind);
149  if(kind == ConstraintEdge::Copy || kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
150  {
151  auto eit = directEdgeSet.find(&edge);
152  return *eit;
153  }
154  else if(kind == ConstraintEdge::Addr)
155  {
156  auto eit = AddrCGEdgeSet.find(&edge);
157  return *eit;
158  }
159  else if(kind == ConstraintEdge::Store)
160  {
161  auto eit = StoreCGEdgeSet.find(&edge);
162  return *eit;
163  }
164  else if(kind == ConstraintEdge::Load)
165  {
166  auto eit = LoadCGEdgeSet.find(&edge);
167  return *eit;
168  }
169  else
170  {
171  assert(false && "no other kind!");
172  return nullptr;
173  }
174  }
