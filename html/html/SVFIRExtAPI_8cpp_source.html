41 void SVFIRBuilder::addComplexConsForExt(const SVFValue* D, const SVFValue* S, const SVFValue* szValue)
42 {
43  assert(D && S);
44  NodeID vnD= pag->getValueNode(D), vnS= pag->getValueNode(S);
45  if(!vnD || !vnS)
46  return;
47 
48  std::vector<AccessPath> fields;
49 
50  //Get the max possible size of the copy, unless it was provided.
51  const SVFType* stype = pag->getTypeLocSetsMap(vnS).first;
52  const SVFType* dtype = pag->getTypeLocSetsMap(vnD).first;
53  std::vector<AccessPath> srcFields = pag->getTypeLocSetsMap(vnS).second;
54  std::vector<AccessPath> dstFields = pag->getTypeLocSetsMap(vnD).second;
55 
56  if(srcFields.size() > dstFields.size())
57  fields = dstFields;
58  else
59  fields = srcFields;
60 
62  u32_t sz = fields.size();
63  if (szValue && SVFUtil::dyn_cast<SVFConstantInt>(szValue))
64  {
65  const SVFConstantInt* arg2 = SVFUtil::dyn_cast<SVFConstantInt>(szValue);
66  sz = (fields.size() > static_cast<u32_t>(arg2->getSExtValue())) ? arg2->getSExtValue() : fields.size();
67  }
68 
69  if (fields.size() == 1 && (SVFUtil::isa<SVFConstantData>(D) || SVFUtil::isa<SVFConstantData>(S)))
70  {
71  NodeID dummy = pag->addDummyValNode();
72  addLoadEdge(vnD,dummy);
73  addStoreEdge(dummy,vnS);
74  return;
75  }
76 
77  //For each field (i), add (Ti = *S + i) and (*D + i = Ti).
78  for (u32_t index = 0; index < sz; index++)
79  {
80  const SVFType* dElementType = pag->getSymbolInfo()->getFlatternedElemType(dtype,
81  fields[index].getConstantFieldIdx());
82  const SVFType* sElementType = pag->getSymbolInfo()->getFlatternedElemType(stype,
83  fields[index].getConstantFieldIdx());
84  NodeID dField = getGepValVar(D,fields[index],dElementType);
85  NodeID sField = getGepValVar(S,fields[index],sElementType);
86  NodeID dummy = pag->addDummyValNode();
87  addLoadEdge(sField,dummy);
88  addStoreEdge(dummy,dField);
89  }
90 }

92 NodeID SVFIRBuilder::getExtID(ExtAPI::OperationType operationType, const std::string &s, const SVFCallInst* svfCall)
93 {
94  s32_t nodeIDType = ExtAPI::getExtAPI()->getNodeIDType(s);
95 
96  // return value >= 0 is an argument node
97  if (nodeIDType >= 0)
98  {
99  assert(svfCall->arg_size() > (u32_t) nodeIDType && "Argument out of bounds!");
100  if (operationType == ExtAPI::OperationType::Memcpy_like || operationType == ExtAPI::OperationType::Memset_like)
101  return nodeIDType;
102  else
103  return pag->getValueNode(svfCall->getArgOperand(nodeIDType));
104  }
105  // return value = -1 is an inst node
106  else if (nodeIDType == -1)
107  return pag->getValueNode(svfCall);
108  // return value = -2 is a Dummy node
109  else if (nodeIDType == -2)
110  return pag->addDummyValNode();
111  // return value = -3 is an object node
112  else if (nodeIDType == -3)
113  {
114  assert(svfCall->getType()->isPointerTy() && "The operand should be a pointer type!");
115  NodeID objId;
116  // Indirect call
117  if (getCallee(svfCall) == nullptr)
118  objId = pag->addDummyObjNode(svfCall->getType());
119  else // Direct call
120  objId = pag->getObjectNode(svfCall);
121  return objId;
122  }
123  // return value = -4 is a nullptr node
124  else if (nodeIDType == -4)
125  return pag->getNullPtr();
126  // return value = -5 is an offset
127  else if (nodeIDType == -5)
128  {
129  for (char const &c : s)
130  {
131  if (std::isdigit(c) == 0)
132  assert(false && "Invalid offset!");
133  }
134  return atoi(s.c_str());
135  }
136  // return value = -6 is an illegal operand format
137  else
138  {
139  assert(false && "The operand format of function operation is illegal!");
140  abort();
141  }
142 }

144 void SVFIRBuilder::parseAtomaticOp(SVF::ExtAPI::Operand &atomaticOp, const SVFCallInst* svfCall, std::map<std::string, NodeID> &nodeIDMap)
145 {
146  // Skip Rb_tree operation, which is handled in extFuncAtomaticOperation()
147  if (atomaticOp.getType() == ExtAPI::OperationType::Rb_tree_ops)
148  return;
149  // Get src and dst node ID
150  if (!atomaticOp.getSrcValue().empty())
151  {
152  const std::string s = atomaticOp.getSrcValue();
153  if (nodeIDMap.find(s) != nodeIDMap.end())
154  atomaticOp.setSrcID(nodeIDMap[s]);
155  else
156  {
157  NodeID srcId = getExtID(atomaticOp.getType(), atomaticOp.getSrcValue(), svfCall);
158  atomaticOp.setSrcID(srcId);
159  nodeIDMap[s] = srcId;
160  }
161  }
162  else
163  assert(false && "The 'src' operand cannot be empty.");
164 
165  if (!atomaticOp.getDstValue().empty())
166  {
167  const std::string s = atomaticOp.getDstValue();
168  if (nodeIDMap.find(s) != nodeIDMap.end())
169  atomaticOp.setDstID(nodeIDMap[s]);
170  else
171  {
172  NodeID dstId = getExtID(atomaticOp.getType(), atomaticOp.getDstValue(), svfCall);
173  atomaticOp.setDstID(dstId);
174  nodeIDMap[s] = dstId;
175  }
176  }
177  else
178  assert(false && "The 'dst' operand cannot be empty.");
179 
180  // Get offset or size
181  if (!atomaticOp.getOffsetOrSizeStr().empty())
182  {
183  std::string s = atomaticOp.getOffsetOrSizeStr();
184  if (nodeIDMap.find(s) != nodeIDMap.end())
185  atomaticOp.setOffsetOrSize(nodeIDMap[s]);
186  else
187  {
188  NodeID offsetOrSize = getExtID(atomaticOp.getType(), atomaticOp.getOffsetOrSizeStr(), svfCall);
189  atomaticOp.setOffsetOrSize(offsetOrSize);
190  nodeIDMap[s] = offsetOrSize;
191  }
192  }
193 }

195 void SVFIRBuilder::parseExtFunctionOps(ExtAPI::ExtFunctionOps &extFunctionOps, const SVFCallInst* svfCall)
196 {
197  // CallStmt operation
198  if (extFunctionOps.getCallStmtNum() != 0)
199  handleExtCallStat(extFunctionOps, svfCall);
200  // Record all dummy nodes
201  std::map<std::string, NodeID> nodeIDMap;
202  for (ExtAPI::ExtOperation& extOperation : extFunctionOps.getOperations())
203  {
204  // CondStmt operation
205  if(extOperation.isConOp())
206  {
207  for (ExtAPI::Operand &atomaticOp : extOperation.getTrueBranchOperands())
208  parseAtomaticOp(atomaticOp, svfCall, nodeIDMap);
209  for (ExtAPI::Operand &atomaticOp : extOperation.getFalseBranchOperands())
210  parseAtomaticOp(atomaticOp, svfCall, nodeIDMap);
211  }
212  // General operation, e.g., "AddrStmt", "CopyStmt", ....
213  else if (!extOperation.isCallOp())
214  parseAtomaticOp(extOperation.getBasicOp(), svfCall, nodeIDMap);
215  }

218 SVFCallInst* SVFIRBuilder::addSVFExtCallInst(const SVFCallInst* svfInst, SVFBasicBlock* svfBB, const SVFFunction* svfCaller, const SVFFunction* svfCallee)
219 {
220  SVFCallInst* svfCall = new SVFCallInst(svfCallee->getFunctionType(), svfBB, false, false);
221  svfCall->setName("ext_inst");
222  LLVMModuleSet::getLLVMModuleSet()->SVFValue2LLVMValue[svfCall] = nullptr;
223  svfCall->setCalledOperand(svfCallee);
224  setCurrentLocation(svfCall, svfBB);
225  SymbolTableInfo::ValueToIDMapTy::iterator iter = pag->getSymbolInfo()->valSyms().find(svfCall);
226  if (iter == pag->getSymbolInfo()->valSyms().end())
227  {
228  SymID id = NodeIDAllocator::get()->allocateValueId();
229  pag->getSymbolInfo()->valSyms().insert(std::make_pair(svfCall, id));
230  DBOUT(DMemModel, outs() << "create a new value sym " << id << "\n");
231  pag->addValNode(svfCall, id);
232  }
233  svfCall->setName(svfCall->getName() + "_" + std::to_string(pag->getValueNode(svfCall)));
234 
235  for(u32_t i = 0; i < svfCallee->arg_size(); i++)
236  {
237  SVFInstruction* svfArg = addSVFExtInst("ext_inst", svfInst, svfBB, ExtAPI::OperationType::Other, svfCallee->getArg(i)->getType());
238  svfArg->setName(svfArg->getName() + "_" + std::to_string(pag->getValueNode(svfArg)));
239  svfCall->addArgument(svfArg);
240  }
241 
242  svfBB->addInstruction(svfCall);
243 
244  CallICFGNode* callBlockNode = pag->getICFG()->getCallICFGNode(svfCall);
245  pag->addCallSite(callBlockNode);
246 
247  if (!svfCallee->isNotRetFunction() && !isExtCall(svfCallee))
248  {
249  NodeID srcret = getReturnNode(svfCallee);
250  NodeID dstrec = pag->getValueNode(svfCall);
251  CallICFGNode* callICFGNode = pag->getICFG()->getCallICFGNode(svfCall);
252  FunExitICFGNode* exitICFGNode = pag->getICFG()->getFunExitICFGNode(svfCallee);
253  RetICFGNode* retBlockNode = pag->getICFG()->getRetICFGNode(svfCall);
254  addRetEdge(srcret, dstrec,callICFGNode, exitICFGNode);
255  pag->addCallSiteRets(retBlockNode,pag->getGNode(pag->getValueNode(svfCall)));
256  }
257 
258  return svfCall;
259 }

274 SVFInstruction* SVFIRBuilder::addSVFExtInst(const std::string& instName, const SVFCallInst* svfInst, SVFBasicBlock* svfBB, SVF::ExtAPI::OperationType opType, const SVFType* svfType)
275 {
276  // Get new SVFInstruction type;
277  const SVFType* ptType = nullptr;
278  if (svfType != nullptr)
279  ptType = svfType;
280  else
281  {
282  if (opType == ExtAPI::OperationType::Addr
283  || opType == ExtAPI::OperationType::Copy
284  || opType == ExtAPI::OperationType::Load
285  || opType == ExtAPI::OperationType::Gep)
286  {
287  for (u32_t i = 0; i < svfInst->arg_size(); ++i)
288  if (svfInst->getArgOperand(i)->getType()->isPointerTy())
289  {
290  ptType = svfInst->getArgOperand(i)->getType();
291  break;
292  }
293  }
294  else
295  ptType = svfInst->getParent()->getType();
296  }
297 
298  assert(ptType != nullptr && "At least one argument of an external call is of pointer type!");
299  SVFInstruction* inst = new SVFInstruction(ptType, svfBB, false, false);
300  inst->setName(instName);
301  LLVMModuleSet::getLLVMModuleSet()->SVFValue2LLVMValue[inst] = nullptr;
302  svfBB->addInstruction(inst);
303  SymbolTableInfo::ValueToIDMapTy::iterator iter = pag->getSymbolInfo()->valSyms().find(inst);
304  if (iter == pag->getSymbolInfo()->valSyms().end())
305  {
306  SymID id = NodeIDAllocator::get()->allocateValueId();
307  pag->getSymbolInfo()->valSyms().insert(std::make_pair(inst, id));
308  DBOUT(DMemModel, outs() << "create a new value sym " << id << "\n");
309  pag->addValNode(inst, id);
310  }
311  inst->setName(inst->getName() + "_" + std::to_string(pag->getValueNode(inst)));
312  return inst;

315 SVFBasicBlock* SVFIRBuilder::extFuncInitialization(const SVFCallInst* svfInst, SVFFunction* svfCaller)
316 {
317  // Initialization, linking actual parameters with formal parameters,
318  // adding basic blocks for external functions,
319  // and creating return edges (if the external function has a return value)
320  CallICFGNode* callSiteICFGNode = pag->getICFG()->getCallICFGNode(svfInst);
321  FunEntryICFGNode* funEntryICFGNode = pag->getICFG()->getFunEntryICFGNode(svfCaller);
322  FunExitICFGNode* funExitICFGNode = pag->getICFG()->getFunExitICFGNode(svfCaller);
323  for(u32_t i = 0; i < svfCaller->arg_size(); i++)
324  {
325  const SVFValue *AA = svfInst->getArgOperand(i);
326  const SVFValue *FA = svfCaller->getArg(i);
327  NodeID srcAA = pag->getValueNode(AA);
328  NodeID dstFA = pag->getValueNode(FA);
329  addCallEdge(srcAA, dstFA, callSiteICFGNode, funEntryICFGNode);
330  pag->addFunArgs(svfCaller,pag->getGNode(dstFA));
331  }
332 
333  SVFBasicBlock* svfBB = new SVFBasicBlock(svfInst->getParent()->getType(), svfCaller);
334  svfBB->setName("ext_bb");
335  LLVMModuleSet::getLLVMModuleSet()->SVFValue2LLVMValue[svfBB] = nullptr;
336 
337  if (!svfCaller->isNotRetFunction())
338  {
339  NodeID srcret = getReturnNode(svfCaller);
340  NodeID dstrec = pag->getValueNode(svfInst);
341  addRetEdge(srcret, dstrec,callSiteICFGNode, funExitICFGNode);
342  pag->addFunRet(svfCaller,pag->getGNode(srcret));
343  }
344  return svfBB;
345 }

347 void SVFIRBuilder::handleExtCallStat(ExtAPI::ExtFunctionOps &extFunctionOps, const SVFCallInst* svfInst)
348 {
349  SVFFunction* svfCaller = const_cast<SVFFunction *>(svfModule->getSVFFunction(svfInst->getCalledFunction()->getName()));
350  SVFBasicBlock* svfBB = extFuncInitialization(svfInst, svfCaller);
351  // Map an operand to its new created SVFInstruction
352  std::map<std::string, SVFValue*> operandToSVFValueMap;
353  for (ExtAPI::ExtOperation& extOperation : extFunctionOps.getOperations())
354  {
355  if (extOperation.isCallOp())
356  {
357  // To create a CallInst for the callee
358  const SVFFunction* svfCallee = svfModule->getSVFFunction(extOperation.getCalleeName());
359  SVFCallInst* svfCall = addSVFExtCallInst(svfInst, svfBB, svfCaller, svfCallee);
360  setCurrentLocation(svfCall, getCurrentBB());
361  operandToSVFValueMap[svfCallee->getName()] = svfCall;
362  CallICFGNode* callBlockNode = pag->getICFG()->getCallICFGNode(svfCall);
363 
364  assert(extOperation.getCalleeOperands().size() >= svfCallee->arg_size()
365  && "Number of arguments set in CallStmt in ExtAPI.json is inconsistent with the number of arguments required by the Callee?");
366  // To parse the operations contained in ‘CallStmt’, obtain the NodeID, and add the callEdge
367  for (ExtAPI::Operand &operand : extOperation.getCalleeOperands())
368  {
369  std::string src = operand.getSrcValue();
370  std::string dst = operand.getDstValue();
371  // ReturnStmt
372  if (operand.getType() == ExtAPI::OperationType::Return && !svfCaller->isNotRetFunction())
373  {
374  addSVFExtRetInst(svfCall, svfBB, svfCaller);
375  continue;
376  }
377 
378  auto getCallStmtOperands = [](const std::string& str) -> std::pair<std::string, std::string>
379  {
380  size_t pos = str.find('_');
381  assert(pos != std::string::npos && "The operand format in CallStmt is incorrect! It should be either 'funName_Argi' or 'funName_Ret'!");
382  return std::make_pair(str.substr(0, pos), str.substr(pos + 1));
383  };
384  // 'src' operand
385  if (src.find('_') != std::string::npos)
386  {
387  std::pair<std::string, std::string> srcRet = getCallStmtOperands(src);
388  s32_t argPos = ExtAPI::getExtAPI()->getNodeIDType(srcRet.second);
389  // operand like "caller_Argi"
390  if (svfCaller->getName() == srcRet.first)
391  {
392  assert(argPos >= 0 && static_cast<u32_t>(argPos) < svfCaller->arg_size() && "The argument index is out of bounds in CallStmt?");
393  operand.setSrcID(pag->getValueNode(svfCaller->getArg(argPos)));
394  }
395  // operand like "callee_Ret"
396  else
397  {
398  assert(argPos == -1 && "The operand format in CallStmt is incorrect! It should be either 'funName_Argi' or 'funName_Ret'!");
399  assert(operandToSVFValueMap.find(srcRet.first) != operandToSVFValueMap.end() && "No created SVFCallInst in external functions?");
400  operand.setSrcID(pag->getValueNode(operandToSVFValueMap[srcRet.first]));
401  }
402  }
403  // operand like self-defined "x", which should be created beforehand
404  else
405  {
406  assert(operandToSVFValueMap.find(src) != operandToSVFValueMap.end()
407  && "Cannot find manual create ext inst, incorrect invocation order for external functions?");
408  operand.setSrcID(pag->getValueNode(operandToSVFValueMap[src]));
409  }
410  // 'dst' operand
411  if (dst.find('_') != std::string::npos)
412  {
413  // operand like "callee_Argi"
414  std::pair<std::string, std::string> dstRet = getCallStmtOperands(dst);
415  assert(svfCallee->getName() == dstRet.first && "The operand format of 'dst' in external CallStmt is illegal!");
416  s32_t argPos = ExtAPI::getExtAPI()->getNodeIDType(dstRet.second);
417  assert(argPos >= 0 && static_cast<u32_t>(argPos) < svfCallee->arg_size() && "The argument index is out of bounds of callee in CallStmt?");
418  // Create an new SVFInstruction for "callee_Argi".
419  if(operandToSVFValueMap.find(dst) == operandToSVFValueMap.end())
420  {
421  SVFInstruction* inst = addSVFExtInst("ext_inst", svfInst, svfBB, operand.getType(), nullptr);
422  operand.setDstID(pag->getValueNode(inst));
423  operandToSVFValueMap[dst] = inst;
424  pag->addCallSiteArgs(callBlockNode,pag->getGNode(pag->getValueNode(inst)));
425  }
426  CallICFGNode* icfgNode = pag->getICFG()->getCallICFGNode(svfCall);
427  FunEntryICFGNode* entry = pag->getICFG()->getFunEntryICFGNode(svfCallee);
428  addCallEdge(operand.getDstID(), pag->getValueNode(svfCallee->getArg(argPos)), icfgNode, entry);
429  }
430  else
431  {
432  // operand like self-defined "x", if there are no SVFInstructions for 'x', create an new SVFInstruction.
433  SVFInstruction* inst = addSVFExtInst(dst, svfInst, svfBB, operand.getType(), nullptr);
434  operand.setDstID(pag->getValueNode(inst));
435  operandToSVFValueMap[dst] = inst;
436  }
437  }
438  }
439  }
440  svfCaller->addBasicBlock(svfBB);
441 }

443 void SVFIRBuilder::extFuncAtomaticOperation(ExtAPI::Operand& atomicOp, const SVFCallInst* svfCall)
444 {
445  if (atomicOp.getType() == ExtAPI::OperationType::Addr)
446  {
447  if (!atomicOp.getSrcValue().empty() && !atomicOp.getDstValue().empty())
448  addAddrEdge(atomicOp.getSrcID(), atomicOp.getDstID());
449  else
450  writeWrnMsg("We need two valid NodeIDs to add an Addr edge");
451  }
452  else if (atomicOp.getType() == ExtAPI::OperationType::Copy)
453  {
454  if (!atomicOp.getSrcValue().empty() && !atomicOp.getDstValue().empty())
455  addCopyEdge(atomicOp.getSrcID(), atomicOp.getDstID());
456  else
457  writeWrnMsg("We need two valid NodeIDs to add a Copy edge");
458  }
459  else if (atomicOp.getType() == ExtAPI::OperationType::Load)
460  {
461  if (!atomicOp.getSrcValue().empty() && !atomicOp.getDstValue().empty())
462  addLoadEdge(atomicOp.getSrcID(), atomicOp.getDstID());
463  else
464  writeWrnMsg("We need two valid NodeIDs to add a Load edge");
465  }
466  else if (atomicOp.getType() == ExtAPI::OperationType::Store)
467  {
468  if (!atomicOp.getSrcValue().empty() && !atomicOp.getDstValue().empty())
469  addStoreEdge(atomicOp.getSrcID(), atomicOp.getDstID());
470  else
471  writeWrnMsg("We need two valid NodeIDs to add a Store edge");
472  }
473  else if (atomicOp.getType() == ExtAPI::OperationType::Gep)
474  {
475  if (!atomicOp.getSrcValue().empty() && !atomicOp.getDstValue().empty() && !atomicOp.getOffsetOrSizeStr().empty())
476  {
477  AccessPath ap(atomicOp.getOffsetOrSize());
478  addNormalGepEdge(atomicOp.getSrcID(), atomicOp.getDstID(), ap);
479  }
480  else
481  writeWrnMsg("We need two valid NodeIDs and an offset to add a Gep edge");
482  }
483  else if (atomicOp.getType() == ExtAPI::OperationType::Return)
484  return;
485  else if (atomicOp.getType() == ExtAPI::OperationType::Memset_like)
486  {
487  // this is for memset(void *str, int c, size_t n)
488  // which copies the character c (an unsigned char) to the first n characters of the string pointed to, by the argument str
489  // const SVFConstantInt* arg2 = SVFUtil::dyn_cast<SVFConstantInt>(svfCall->getArgOperand(op.getOperands()[2]));
490  NodeID argId = pag->getValueNode(svfCall->getArgOperand(0));
491  std::vector<AccessPath> dstFields = pag->getTypeLocSetsMap(argId).second;
492  u32_t sz = dstFields.size();
493  if (const SVFConstantInt* arg2 = SVFUtil::dyn_cast<SVFConstantInt>(svfCall->getArgOperand(2)))
494  sz = (dstFields.size() > static_cast<u32_t>(arg2->getSExtValue())) ? arg2->getSExtValue() : dstFields.size();
495  //For each field (i), add store edge *(arg0 + i) = arg1
496  for (u32_t index = 0; index < sz; index++)
497  {
498  const SVFType* dElementType = pag->getSymbolInfo()->getFlatternedElemType(pag->getTypeLocSetsMap(argId).first, dstFields[index].getConstantFieldIdx());
499  NodeID dField = getGepValVar(svfCall->getArgOperand(0), dstFields[index], dElementType);
500  addStoreEdge(pag->getValueNode(svfCall->getArgOperand(1)),dField);
501  }
502  if(svfCall->getType()->isPointerTy())
503  addCopyEdge(pag->getValueNode(svfCall->getArgOperand(0)), pag->getValueNode(svfCall));
504  }
505  else if (atomicOp.getType() == ExtAPI::OperationType::Memcpy_like)
506  {
507  if(svfCall->arg_size() == 3)
508  addComplexConsForExt(svfCall->getArgOperand(0), svfCall->getArgOperand(1), svfCall->getArgOperand(2));
509  else

549 void SVFIRBuilder::handleExtCall(const SVFInstruction* svfInst, const SVFFunction* svfCallee)
550 {
551  const SVFCallInst* svfCall = SVFUtil::cast<SVFCallInst>(svfInst);
552 
553  if (isHeapAllocOrStaticExtCall(svfInst))
554  {
555  // case 1: ret = new obj
556  if (isHeapAllocExtCallViaRet(svfInst) || isStaticExtCall(svfInst))
557  {
558  NodeID val = pag->getValueNode(svfInst);
559  NodeID obj = pag->getObjectNode(svfInst);
560  addAddrEdge(obj, val);
561  }
562  // case 2: *arg = new obj
563  else
564  {
565  assert(isHeapAllocExtCallViaArg(svfInst) && "Must be heap alloc call via arg.");
566  u32_t arg_pos = getHeapAllocHoldingArgPosition(svfCallee);
567  const SVFValue* arg = svfCall->getArgOperand(arg_pos);
568  if (arg->getType()->isPointerTy())
569  {
570  NodeID vnArg = pag->getValueNode(arg);
571  NodeID dummy = pag->addDummyValNode();
572  NodeID obj = pag->addDummyObjNode(arg->getType());
573  if (vnArg && dummy && obj)
574  {
575  addAddrEdge(obj, dummy);
576  addStoreEdge(dummy, vnArg);
577  }
578  }
579  else
580  {
581  writeWrnMsg("Arg receiving new object must be pointer type");
582  }
583  }
584  }
585  else
586  {
587  if (isExtCall(svfCallee))
588  {
589  ExtAPI::ExtFunctionOps extFunctionOps = ExtAPI::getExtAPI()->getExtFunctionOps(svfCallee);
590  if (extFunctionOps.getOperations().size() == 0)
591  {
592  std::string str;
593  std::stringstream rawstr(str);
594  rawstr << "function " << svfCallee->getName() << " not in the external function summary ExtAPI.json file";
595  writeWrnMsg(rawstr.str());
596  }
597  else
598  {
599  parseExtFunctionOps(extFunctionOps, svfCall);
600  for (ExtAPI::ExtOperation op : extFunctionOps.getOperations())
601  {
602  if (op.isCallOp())
603  for (ExtAPI::Operand atomicOp : op.getCalleeOperands())
604  extFuncAtomaticOperation(atomicOp, svfCall);
605  else if(op.isConOp())
606  {
607  for (ExtAPI::Operand atomicOp : op.getTrueBranchOperands())
608  extFuncAtomaticOperation(atomicOp, svfCall);
609  for (ExtAPI::Operand atomicOp : op.getFalseBranchOperands())
610  extFuncAtomaticOperation(atomicOp, svfCall);
611  }
612  else
613  {
614  ExtAPI::Operand atomicOp = op.getBasicOp();
615  extFuncAtomaticOperation(atomicOp, svfCall);
616  }
617  }
618  }
619  }
620 
622  if (isThreadForkCall(svfInst))
623  {
624  if (const SVFFunction* forkedFun = SVFUtil::dyn_cast<SVFFunction>(getForkedFun(svfInst)))
625  {
626  forkedFun = forkedFun->getDefFunForMultipleModule();
627  const SVFValue* actualParm = getActualParmAtForkSite(svfInst);
630  assert((forkedFun->arg_size() <= 2) && "Size of formal parameter of start routine should be one");
631  if (forkedFun->arg_size() <= 2 && forkedFun->arg_size() >= 1)
632  {

635  if (actualParm->getType()->isPointerTy() && formalParm->getType()->isPointerTy())
636  {

638  FunEntryICFGNode *entry = pag->getICFG()->getFunEntryICFGNode(forkedFun);

640  }
641  }
642  }

653  }

659  {

669  addThreadForkEdge(pag->getValueNode(actualParm), pag->getValueNode(formalParm), icfgNode, entry);
