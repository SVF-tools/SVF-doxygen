38 {
39 
40  for (SVFModule::iterator F = mod->begin(), E = mod->end(); F != E; ++F)
41  {
42  if (SVFUtil::isExtCall(*F))
43  continue;

45  {
46  const Instruction *inst = &*II;
47  if (SVFUtil::isa<LoadInst>(inst))
48  {

51  else if (SVFUtil::isa<StoreInst>(inst))
52  {
53  storeset.insert(inst);
54  }
55  else if (isMemset(inst))
56  {
57  storeset.insert(inst);
58  }

113 void MTAAnnotator::pruneThreadLocal(PointerAnalysis* pta)
114 {
115  bool AnnoLocal = Options::AnnoFlag & ANNO_LOCAL;
116  if (!AnnoLocal)
117  return;
118 
119  DBOUT(DGENERAL, outs() << pasMsg("Run annotator prune thread local pairs\n"));
120  SVFIR* pag = pta->getPAG();
121  PointsTo nonlocalobjs;
122  PointsTo worklist;
123 
125  const SVFStmt::SVFStmtSetTy& forkedges = pag->getPTASVFStmtSet(SVFStmt::ThreadFork);
126  for (SVFStmt::SVFStmtSetTy::const_iterator it = forkedges.begin(), eit = forkedges.end(); it != eit; ++it)
127  {
128  PAGEdge* edge = *it;
129  worklist |= pta->getPts(edge->getDstID());
130  worklist |= pta->getPts(edge->getSrcID());
131  }
132 
134  const SVFIR::SVFStmtSet& globaledges = pag->getGlobalSVFStmtSet();
135  for (SVFIR::SVFStmtSet::const_iterator it = globaledges.begin(), eit = globaledges.end(); it != eit; ++it)
136  {
137  const PAGEdge* edge = *it;
138  if (edge->getEdgeKind() == SVFStmt::Addr)
139  {
140  worklist.set(edge->getSrcID());
141  }
142  }
143 
145  while (!worklist.empty())
146  {
147  NodeID obj = worklist.find_first();
148  nonlocalobjs.set(obj);
149  worklist.reset(obj);
150  PointsTo pts = pta->getPts(obj);
151  for (PointsTo::iterator pit = pts.begin(), epit = pts.end(); pit != epit; ++pit)
152  {
153  if (!nonlocalobjs.test(*pit))
154  worklist.set(*pit);
155  }
156  NodeBS fields = pag->getAllFieldsObjVars(obj);
157  for (NodeBS::iterator pit = fields.begin(), epit = fields.end(); pit != epit; ++pit)
158  {
159  if (!nonlocalobjs.test(*pit))
160  worklist.set(*pit);
161  }
162  }
163 
165  InstSet needannost;
166  InstSet needannold;
167  for (InstSet::iterator it = storeset.begin(), eit = storeset.end(); it != eit; ++it)
168  {
169  PointsTo pts = pta->getPts(pag->getValueNode(getStoreOperand(*it)));
170  for (PointsTo::iterator pit = pts.begin(), epit = pts.end(); pit != epit; ++pit)
171  {
172  if (nonlocalobjs.test(*pit))
173  {
174  needannost.insert(*it);
175  break;
176  }
177  }
178  }
179 
180  for (InstSet::iterator it = loadset.begin(), eit = loadset.end(); it != eit; ++it)
181  {
182  PointsTo pts = pta->getPts(pag->getValueNode(getLoadOperand(*it)));
183  for (PointsTo::iterator pit = pts.begin(), epit = pts.end(); pit != epit; ++pit)
184  {
185  if (nonlocalobjs.test(*pit))
186  {
187  needannold.insert(*it);
188  break;
189  }
190  }
191  }
192 
193  storeset = needannost;
194  loadset = needannold;
195 
196  numOfNonLocalSt = storeset.size();
197  numOfNonLocalLd = loadset.size();
198 }
