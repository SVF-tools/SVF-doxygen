150  {
151  CFBasicBlockEdge edge(src, dst);
152  CFBasicBlockEdge *outEdge = src->hasOutgoingEdge(&edge);
153  CFBasicBlockEdge *inEdge = dst->hasIncomingEdge(&edge);
154  if (outEdge && inEdge)

157  return true;
