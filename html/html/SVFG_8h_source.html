358  inline void setDef(const MRVer* mvar, const SVFGNode* node)
359  {
360  MSSAVarToDefMapTy::iterator it = MSSAVarToDefMap.find(mvar);
361  if(it==MSSAVarToDefMap.end())
362  {
363  MSSAVarToDefMap[mvar] = node->getId();
364  assert(hasSVFGNode(node->getId()) && "not in the map!!");
365  }
366  else
367  {
368  assert((it->second == node->getId()) && "a SVFIR node can only have unique definition ");
369  }
370  }
