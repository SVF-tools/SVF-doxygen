88 {

427 NodeID SVFIR::getGepObjVar(const MemObj* obj, const LocationSet& ls)
428 {
429  NodeID base = obj->getId();
430 
432  if (obj->isFieldInsensitive())
433  return getFIObjVar(obj);
434 
435  LocationSet newLS = SymbolTableInfo::SymbolInfo()->getModulusOffset(obj,ls);
436 
437  // Base and first field are the same memory location.
438  if (Options::FirstFieldEqBase && newLS.accumulateConstantFieldIdx() == 0) return base;
439 
440  NodeLocationSetMap::iterator iter = GepObjVarMap.find(std::make_pair(base, newLS));
441  if (iter == GepObjVarMap.end())
442  return addGepObjNode(obj, newLS);
443  else
444  return iter->second;
445 
446 }

694 bool SVFIR::isNonPointerObj(NodeID id) const
695 {
696  SVFVar* node = getGNode(id);
697  if (const FIObjVar* fiNode = SVFUtil::dyn_cast<FIObjVar>(node))
698  {
699  return (fiNode->getMemObj()->hasPtrObj()==false);
700  }
701  else if (const GepObjVar* gepNode = SVFUtil::dyn_cast<GepObjVar>(node))
702  {
703  return (gepNode->getMemObj()->isNonPtrFieldObj(gepNode->getLocationSet()));
704  }
705  else if (const DummyObjVar* dummyNode = SVFUtil::dyn_cast<DummyObjVar>(node))
706  {
707  return (dummyNode->getMemObj()->hasPtrObj()==false);
708  }
