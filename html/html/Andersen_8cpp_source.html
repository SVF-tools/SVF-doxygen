392 bool Andersen::processGepPts(const PointsTo& pts, const GepCGEdge* edge)
393 {
394  numOfProcessedGep++;
395 
396  PointsTo tmpDstPts;
397  if (SVFUtil::isa<VariantGepCGEdge>(edge))
398  {
399  // If a pointer is connected by a variant gep edge,
400  // then set this memory object to be field insensitive,
401  // unless the object is a black hole/constant.
402  for (NodeID o : pts)
403  {
404  if (consCG->isBlkObjOrConstantObj(o))
405  {
406  tmpDstPts.set(o);
407  continue;
408  }
409 
410  if (!isFieldInsensitive(o))
411  {
412  setObjFieldInsensitive(o);
413  consCG->addNodeToBeCollapsed(consCG->getBaseObjVar(o));
414  }
415 
416  // Add the field-insensitive node into pts.
417  NodeID baseId = consCG->getFIObjVar(o);
418  tmpDstPts.set(baseId);
419  }
420  }
421  else if (const NormalGepCGEdge* normalGepEdge = SVFUtil::dyn_cast<NormalGepCGEdge>(edge))
422  {
423  // TODO: after the node is set to field insensitive, handling invariant
424  // gep edge may lose precision because offsets here are ignored, and the
425  // base object is always returned.
426  for (NodeID o : pts)
427  {
428  if (consCG->isBlkObjOrConstantObj(o) || isFieldInsensitive(o))
429  {
430  tmpDstPts.set(o);
431  continue;
432  }
433 
434  if (!matchType(edge->getSrcID(), o, normalGepEdge)) continue;
435 
436  NodeID fieldSrcPtdNode = consCG->getGepObjVar(o, normalGepEdge->getLocationSet());
437  tmpDstPts.set(fieldSrcPtdNode);
438  addTypeForGepObjNode(fieldSrcPtdNode, normalGepEdge);
439  }
440  }
441  else
442  {
443  assert(false && "Andersen::processGepPts: New type GEP edge type?");
444  }
445 
446  NodeID dstId = edge->getDstID();
447  if (unionPts(dstId, tmpDstPts))
448  {
449  pushIntoWorklist(dstId);
450  return true;
451  }
452 
453  return false;
454 }

483 void Andersen::mergeSccCycle()
484 {
485  NodeStack revTopoOrder;
486  NodeStack & topoOrder = getSCCDetector()->topoNodeStack();
487  while (!topoOrder.empty())
488  {
489  NodeID repNodeId = topoOrder.top();
490  topoOrder.pop();
491  revTopoOrder.push(repNodeId);
492  const NodeBS& subNodes = getSCCDetector()->subNodes(repNodeId);
493  // merge sub nodes to rep node
494  mergeSccNodes(repNodeId, subNodes);
495  }
496 
497  // restore the topological order for later solving.
498  while (!revTopoOrder.empty())
499  {
500  NodeID nodeId = revTopoOrder.top();
501  revTopoOrder.pop();
502  topoOrder.push(nodeId);
503  }
504 }

546 bool Andersen::collapseField(NodeID nodeId)
547 {
552  if (consCG->isBlkObjOrConstantObj(nodeId))
553  return false;
554 
555  bool changed = false;
556 
557  double start = stat->getClk();
558 
559  // set base node field-insensitive.
560  setObjFieldInsensitive(nodeId);
561 
562  // replace all occurrences of each field with the field-insensitive node
563  NodeID baseId = consCG->getFIObjVar(nodeId);
564  NodeID baseRepNodeId = consCG->sccRepNode(baseId);
565  NodeBS & allFields = consCG->getAllFieldsObjVars(baseId);
566  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end(); fieldIt != fieldEit; fieldIt++)
567  {
568  NodeID fieldId = *fieldIt;
569  if (fieldId != baseId)
570  {
571  // use the reverse pts of this field node to find all pointers point to it
572  const NodeSet revPts = getRevPts(fieldId);
573  for (const NodeID o : revPts)
574  {
575  // change the points-to target from field to base node
576  clearPts(o, fieldId);
577  addPts(o, baseId);
578  pushIntoWorklist(o);
579 
580  changed = true;
581  }
582  // merge field node into base node, including edges and pts.
583  NodeID fieldRepNodeId = consCG->sccRepNode(fieldId);
584  mergeNodeToRep(fieldRepNodeId, baseRepNodeId);
585  if (fieldId != baseRepNodeId)
586  {
587  // gep node fieldId becomes redundant if it is merged to its base node who is set as field-insensitive
588  // two node IDs should be different otherwise this field is actually the base and should not be removed.
589  redundantGepNodes.set(fieldId);
590  }
591  }
592  }
593 
594  if (consCG->isPWCNode(baseRepNodeId))
595  if (collapseNodePts(baseRepNodeId))
596  changed = true;
597 
598  double end = stat->getClk();
599  timeOfCollapse += (end - start) / TIMEINTERVAL;
600 
601  return changed;
602 }

688 void Andersen::connectCaller2CalleeParams(CallSite cs, const SVFFunction* F, NodePairSet &cpySrcNodes)
689 {
690  assert(F);
691 
692  DBOUT(DAndersen, outs() << "connect parameters from indirect callsite " << SVFUtil::value2String(cs.getInstruction()) << " to callee " << *F << "\n");
693 
694  CallICFGNode* callBlockNode = pag->getICFG()->getCallICFGNode(cs.getInstruction());
695  RetICFGNode* retBlockNode = pag->getICFG()->getRetICFGNode(cs.getInstruction());
696 
697  if(SVFUtil::isHeapAllocExtFunViaRet(F) && pag->callsiteHasRet(retBlockNode))
698  {
699  heapAllocatorViaIndCall(cs,cpySrcNodes);
700  }
701 
702  if (pag->funHasRet(F) && pag->callsiteHasRet(retBlockNode))
703  {
704  const PAGNode* cs_return = pag->getCallSiteRet(retBlockNode);
705  const PAGNode* fun_return = pag->getFunRet(F);
706  if (cs_return->isPointer() && fun_return->isPointer())
707  {
708  NodeID dstrec = sccRepNode(cs_return->getId());
709  NodeID srcret = sccRepNode(fun_return->getId());
710  if(addCopyEdge(srcret, dstrec))
711  {
712  cpySrcNodes.insert(std::make_pair(srcret,dstrec));
713  }
714  }
715  else
716  {
717  DBOUT(DAndersen, outs() << "not a pointer ignored\n");
718  }
719  }
720 
721  if (pag->hasCallSiteArgsMap(callBlockNode) && pag->hasFunArgsList(F))
722  {
723 
724  // connect actual and formal param
725  const SVFIR::SVFVarList& csArgList = pag->getCallSiteArgsList(callBlockNode);
726  const SVFIR::SVFVarList& funArgList = pag->getFunArgsList(F);
727  //Go through the fixed parameters.
728  DBOUT(DPAGBuild, outs() << " args:");
729  SVFIR::SVFVarList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
730  SVFIR::SVFVarList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
731  for (; funArgIt != funArgEit; ++csArgIt, ++funArgIt)
732  {
733  //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
734  if (csArgIt == csArgEit)
735  {
736  DBOUT(DAndersen, outs() << " !! not enough args\n");
737  break;
738  }
739  const PAGNode *cs_arg = *csArgIt ;
740  const PAGNode *fun_arg = *funArgIt;
741 
742  if (cs_arg->isPointer() && fun_arg->isPointer())
743  {
744  DBOUT(DAndersen, outs() << "process actual parm " << cs_arg->toString() << " \n");
745  NodeID srcAA = sccRepNode(cs_arg->getId());
746  NodeID dstFA = sccRepNode(fun_arg->getId());
747  if(addCopyEdge(srcAA, dstFA))
748  {
749  cpySrcNodes.insert(std::make_pair(srcAA,dstFA));
750  }
751  }
752  }
753 
754  //Any remaining actual args must be varargs.
755  if (F->isVarArg())
756  {
757  NodeID vaF = sccRepNode(pag->getVarargNode(F));
758  DBOUT(DPAGBuild, outs() << "\n varargs:");
759  for (; csArgIt != csArgEit; ++csArgIt)
760  {
761  const PAGNode *cs_arg = *csArgIt;
762  if (cs_arg->isPointer())
763  {
764  NodeID vnAA = sccRepNode(cs_arg->getId());
765  if (addCopyEdge(vnAA,vaF))
766  {
767  cpySrcNodes.insert(std::make_pair(vnAA,vaF));
768  }
769  }
770  }
771  }
