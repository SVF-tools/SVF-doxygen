45 bool TCT::isInLoopInstruction(const SVFInstruction* inst)
46 {
47  assert(inst && "null value instruction!!");
48 
49  InstSet insts;
50  FIFOWorkList<const SVFInstruction*> worklist;
51  worklist.push(inst);
52 
53  while(!worklist.empty())
54  {
55  const SVFInstruction* inst = worklist.pop();
56  insts.insert(inst);
57  PTACallGraphNode* cgnode = tcg->getCallGraphNode(inst->getFunction());
58  for(PTACallGraphNode::const_iterator nit = cgnode->InEdgeBegin(), neit = cgnode->InEdgeEnd(); nit!=neit; nit++)
59  {
60  for(PTACallGraphEdge::CallInstSet::const_iterator cit = (*nit)->directCallsBegin(),
61  ecit = (*nit)->directCallsEnd(); cit!=ecit; ++cit)
62  {
63  if(insts.insert((*cit)->getCallSite()).second)
64  worklist.push((*cit)->getCallSite());
65  }
66  for(PTACallGraphEdge::CallInstSet::const_iterator cit = (*nit)->indirectCallsBegin(),
67  ecit = (*nit)->indirectCallsEnd(); cit!=ecit; ++cit)
68  {
69  if(insts.insert((*cit)->getCallSite()).second)
70  worklist.push((*cit)->getCallSite());
71  }
72  }
73  }
74 
75  for(InstSet::const_iterator it = insts.begin(), eit = insts.end(); it!=eit; ++it)
76  {
77  const SVFInstruction* i = *it;
78  if(i->getFunction()->hasLoopInfo(i->getParent()))
79  return true;
80  }
81 
82 
83  return false;
84 }

91 bool TCT::isInRecursion(const SVFInstruction* inst) const
92 {
93  const SVFFunction* f = inst->getFunction();
94  FIFOWorkList<const SVFFunction*> worklist;
95  Set<const SVFFunction*> visits;
96  worklist.push(f);
97 
98  while(!worklist.empty())
99  {
100  const SVFFunction* svffun = worklist.pop();
101  visits.insert(svffun);
102  if(tcgSCC->isInCycle(tcg->getCallGraphNode(svffun)->getId()))
103  return true;
104 
105  const PTACallGraphNode* cgnode = tcg->getCallGraphNode(svffun);
106 
107  for(PTACallGraphNode::const_iterator nit = cgnode->InEdgeBegin(), neit = cgnode->InEdgeEnd(); nit!=neit; nit++)
108  {
109  for(PTACallGraphEdge::CallInstSet::const_iterator cit = (*nit)->directCallsBegin(),
110  ecit = (*nit)->directCallsEnd(); cit!=ecit; ++cit)
111  {
112  const SVFFunction* caller = (*cit)->getCallSite()->getFunction();
113  if(visits.find(caller)==visits.end())
114  worklist.push(caller);
115  }
116  for(PTACallGraphEdge::CallInstSet::const_iterator cit = (*nit)->indirectCallsBegin(),
117  ecit = (*nit)->indirectCallsEnd(); cit!=ecit; ++cit)
118  {
119  const SVFFunction* caller = (*cit)->getCallSite()->getFunction();
120  if(visits.find(caller)==visits.end())
121  worklist.push(caller);
122  }
123  }
124  }
125 
126  return false;

290 {

292  const SVFFunction* svffun = join->getFunction();
293  const SVFBasicBlock* loopheadbb = svffun->getLoopHeader(lp);
294  const SVFBasicBlock* joinbb = join->getParent();
295  assert(loopheadbb->getParent()==joinbb->getParent() && "should inside same function");
296 
297  for (const SVFBasicBlock* svf_scc_bb : loopheadbb->getSuccessors())
298  {
299  if(svffun->loopContainsBB(lp,svf_scc_bb))

302  return false;
303  }
304  }
305 
306  return true;

382 void TCT::build()
383 {
384 
385  markRelProcs();
386 
387  collectLoopInfoForJoin();
388 
389  // the fork site of main function is initialized with nullptr.
390  // the context of main is initialized with empty
391  // start routine is empty
392 
393  collectEntryFunInCallGraph();
394  for (FunSet::iterator it=entryFuncSet.begin(), eit=entryFuncSet.end(); it!=eit; ++it)
395  {
396  if (!isCandidateFun(*it))
397  continue;
398  CallStrCxt cxt;
399  TCTNode* mainTCTNode = getOrCreateTCTNode(cxt, nullptr, cxt, *it);
400  CxtThreadProc t(mainTCTNode->getId(), cxt, *it);
401  pushToCTPWorkList(t);
402  }
403 
404  while(!ctpList.empty())
405  {
406  CxtThreadProc ctp = popFromCTPWorkList();
407  PTACallGraphNode* cgNode = tcg->getCallGraphNode(ctp.getProc());
408  if(isCandidateFun(cgNode->getFunction()) == false)
409  continue;
410 
411  for(PTACallGraphNode::const_iterator nit = cgNode->OutEdgeBegin(), neit = cgNode->OutEdgeEnd(); nit!=neit; nit++)
412  {
413  const PTACallGraphEdge* cgEdge = (*nit);
414 
415  for(PTACallGraphEdge::CallInstSet::const_iterator cit = cgEdge->directCallsBegin(),
416  ecit = cgEdge->directCallsEnd(); cit!=ecit; ++cit)
417  {
418  DBOUT(DMTA,outs() << "\nTCT handling direct call:" << **cit << "\t" << cgEdge->getSrcNode()->getFunction()->getName() << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
419  handleCallRelation(ctp,cgEdge,getSVFCallSite((*cit)->getCallSite()));
420  }
421  for(PTACallGraphEdge::CallInstSet::const_iterator ind = cgEdge->indirectCallsBegin(),
422  eind = cgEdge->indirectCallsEnd(); ind!=eind; ++ind)
423  {
424  DBOUT(DMTA,outs() << "\nTCT handling indirect call:" << **ind << "\t" << cgEdge->getSrcNode()->getFunction()->getName() << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
425  handleCallRelation(ctp,cgEdge,getSVFCallSite((*ind)->getCallSite()));
426  }
427  }
428  }
429 
430  collectMultiForkedThreads();
431 
432  if (Options::TCTDotGraph)
433  {
434  print();
435  dump("tct");
436  }
437 
438 }
