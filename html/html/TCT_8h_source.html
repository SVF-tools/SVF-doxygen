302  const NodeBS getAncestorThread(NodeID tid) const
303  {
304  NodeBS tds;
305  if(hasParentThread(tid) == false)
306  return tds;
307 
308  FIFOWorkList<NodeID> worklist;
309  worklist.push(getParentThread(tid));
310 
311  while(!worklist.empty())
312  {
313  NodeID t = worklist.pop();
314  if(tds.test_and_set(t))
315  {
316  if(hasParentThread(t))
317  worklist.push(getParentThread(t));
318  }
319  }
