301  base = getCanonicalType(base);
302  if (base == nullptr)
303  {
304  // Conservative; the base object is untyped, sadly.
305  return nullptr;
306  }
307 
308  // For TBHC this is conservative because the union type is lower in the DCHG

312  return base;
313  }

319  assert(cbase && "DCHG: bad DIComposite case");
320  return cbase->getBaseType();
321  }
322 
323  if (!(base->getTag() == dwarf::DW_TAG_class_type ||
324  base->getTag() == dwarf::DW_TAG_structure_type))

326  return nullptr;
