91  initialize();

131 void FlowSensitive::finalize()
132 {
133  if(Options::DumpVFG)
134  svfg->dump("fs_solved", true);
135 
136  NodeStack& nodeStack = WPASolver<SVFG*>::SCCDetect();
137  while (nodeStack.empty() == false)
138  {
139  NodeID rep = nodeStack.top();
140  nodeStack.pop();
141  const NodeBS& subNodes = getSCCDetector()->subNodes(rep);
142  if (subNodes.count() > maxSCCSize)
143  maxSCCSize = subNodes.count();
144  if (subNodes.count() > 1)
145  {
146  numOfNodesInSCC += subNodes.count();
147  numOfSCC++;
148  }
149  }
150 
151  // TODO: check -stat too.
152  if (Options::ClusterFs)
153  {
154  Map<std::string, std::string> stats;
155  const PTDataTy *ptd = getPTDataTy();
156  // TODO: should we use liveOnly?
157  Map<PointsTo, unsigned> allPts = ptd->getAllPts(true);
158  // TODO: parameterise final arg.
159  NodeIDAllocator::Clusterer::evaluate(*PointsTo::getCurrentBestNodeMapping(), allPts, stats, true);
160  NodeIDAllocator::Clusterer::printStats("post-main: best", stats);
161 
162  // Do the same for the candidates. TODO: probably temporary for eval. purposes.
163  for (std::pair<hclust_fast_methods, std::vector<NodeID>> &candidate : candidateMappings)
164  {
165  // Can reuse stats, since we're always filling it with `evaluate`, it will always be overwritten.
166  NodeIDAllocator::Clusterer::evaluate(candidate.second, allPts, stats, true);
167  NodeIDAllocator::Clusterer::printStats("post-main: candidate " + SVFUtil::hclustMethodToString(candidate.first), stats);
168  }
169  }

171  BVDataPTAImpl::finalize();
172 }

202 {

212  {
213  numOfProcessedCopy++;

219  numOfProcessedGep++;

224  {
225  numOfProcessedLoad++;
226  if(processLoad(load))
227  changed = true;
228  }
229  else if(StoreSVFGNode* store = SVFUtil::dyn_cast<StoreSVFGNode>(node))
230  {
231  numOfProcessedStore++;
232  if(processStore(store))
233  changed = true;
234  }
235  else if(PHISVFGNode* phi = SVFUtil::dyn_cast<PHISVFGNode>(node))
236  {
237  numOfProcessedPhi++;
238  if (processPhi(phi))
239  changed = true;
240  }
241  else if(SVFUtil::isa<MSSAPHISVFGNode>(node) || SVFUtil::isa<FormalINSVFGNode>(node)
242  || SVFUtil::isa<FormalOUTSVFGNode>(node) || SVFUtil::isa<ActualINSVFGNode>(node)
243  || SVFUtil::isa<ActualOUTSVFGNode>(node))
244  {
245  numOfProcessedMSSANode++;
246  changed = true;

252  changed = true;
253  }
254  else if(SVFUtil::isa<CmpVFGNode>(node) || SVFUtil::isa<BinaryOPVFGNode>(node) || SVFUtil::dyn_cast<UnaryOPVFGNode>(node))
255  {
256 
257  }
258  else
259  assert(false && "unexpected kind of SVFG nodes");
260 
261  double end = stat->getClk();

295 bool FlowSensitive::propAlongDirectEdge(const DirectSVFGEdge* edge)
296 {
297  double start = stat->getClk();
298  bool changed = false;
299 
300  SVFGNode* src = edge->getSrcNode();
301  SVFGNode* dst = edge->getDstNode();
302  // If this is an actual-param or formal-ret, top-level pointer's pts must be
303  // propagated from src to dst.
304  if (ActualParmSVFGNode* ap = SVFUtil::dyn_cast<ActualParmSVFGNode>(src))
305  changed = propagateFromAPToFP(ap, dst);
306  else if (FormalRetSVFGNode* fp = SVFUtil::dyn_cast<FormalRetSVFGNode>(src))
307  changed = propagateFromFRToAR(fp, dst);
308  else
309  {
310  // Direct SVFG edge links between def and use of a top-level pointer.
311  // There's no points-to information propagated along direct edge.
312  // Since the top-level pointer's value has been changed at src node,
313  // return TRUE to put dst node into the work list.
314  changed = true;
315  }
316 
317  double end = stat->getClk();
318  directPropaTime += (end - start) / TIMEINTERVAL;
319  return changed;
320 }

357 bool FlowSensitive::propAlongIndirectEdge(const IndirectSVFGEdge* edge)
358 {
359  double start = stat->getClk();
360 
361  SVFGNode* src = edge->getSrcNode();
362  SVFGNode* dst = edge->getDstNode();
363 
364  bool changed = false;
365 
366  // Get points-to targets may be used by next SVFG node.
367  // Propagate points-to set for node used in dst.
368  const NodeBS& pts = edge->getPointsTo();
369  for (NodeBS::iterator ptdIt = pts.begin(), ptdEit = pts.end(); ptdIt != ptdEit; ++ptdIt)
370  {
371  NodeID ptd = *ptdIt;
372 
373  if (propVarPtsFromSrcToDst(ptd, src, dst))
374  changed = true;
375 
376  if (isFieldInsensitive(ptd))
377  {
379  const NodeBS& allFields = getAllFieldsObjVars(ptd);
380  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end();
381  fieldIt != fieldEit; ++fieldIt)
382  {
383  if (propVarPtsFromSrcToDst(*fieldIt, src, dst))
384  changed = true;
385  }
386  }
387  }
388 
389  double end = stat->getClk();
390  indirectPropaTime += (end - start) / TIMEINTERVAL;
391  return changed;

648 bool FlowSensitive::updateCallGraph(const CallSiteToFunPtrMap& callsites)
649 {
650  double start = stat->getClk();
651  CallEdgeMap newEdges;
652  onTheFlyCallGraphSolve(callsites, newEdges);
653 
654  // Bound the new edges by the Andersen's call graph.
655  // TODO: we want this to be an assertion eventually.
656  const CallEdgeMap &andersCallEdgeMap = ander->getIndCallMap();
657  for (typename CallEdgeMap::value_type &csfs : newEdges)
658  {
659  const CallICFGNode *potentialCallSite = csfs.first;
660  FunctionSet &potentialFunctionSet = csfs.second;
661 
662  // Check this callsite even calls anything per Andersen's.
663  typename CallEdgeMap::const_iterator andersFunctionSetIt
664  = andersCallEdgeMap.find(potentialCallSite);
665  if (andersFunctionSetIt == andersCallEdgeMap.end())
666  {
667  potentialFunctionSet.clear();
668  }
669 
670  const FunctionSet &andersFunctionSet = andersFunctionSetIt->second;
671  for (FunctionSet::iterator potentialFunctionIt = potentialFunctionSet.begin();
672  potentialFunctionIt != potentialFunctionSet.end(); )
673  {
674  const SVFFunction *potentialFunction = *potentialFunctionIt;
675  if (andersFunctionSet.find(potentialFunction) == andersFunctionSet.end())
676  {
677  // potentialFunction is not in the Andersen's call graph -- remove it.
678  potentialFunctionIt = potentialFunctionSet.erase(potentialFunctionIt);
679  }
680  else
681  {
682  // potentialFunction is in the Andersen's call graph -- keep it..
683  ++potentialFunctionIt;
684  }
685  }
686  }
687 
688  SVFGEdgeSetTy svfgEdges;
689  connectCallerAndCallee(newEdges, svfgEdges);
690 
691  updateConnectedNodes(svfgEdges);
692 
693  double end = stat->getClk();
694  updateCallGraphTime += (end - start) / TIMEINTERVAL;
695  return (!newEdges.empty());
696 }

721 void FlowSensitive::updateConnectedNodes(const SVFGEdgeSetTy& edges)
722 {
723  for (SVFGEdgeSetTy::const_iterator it = edges.begin(), eit = edges.end();
724  it != eit; ++it)
725  {
726  const SVFGEdge* edge = *it;
727  SVFGNode* dstNode = edge->getDstNode();
728  if (SVFUtil::isa<PHISVFGNode>(dstNode))
729  {
732  pushIntoWorklist(dstNode->getId());
733  }
734  else if (SVFUtil::isa<FormalINSVFGNode>(dstNode) || SVFUtil::isa<ActualOUTSVFGNode>(dstNode))
735  {
738  bool changed = false;
739 
740  SVFGNode* srcNode = edge->getSrcNode();
741 
742  const NodeBS& pts = SVFUtil::cast<IndirectSVFGEdge>(edge)->getPointsTo();
743  for (NodeBS::iterator ptdIt = pts.begin(), ptdEit = pts.end(); ptdIt != ptdEit; ++ptdIt)
744  {
745  NodeID ptd = *ptdIt;
746 
747  if (propVarPtsAfterCGUpdated(ptd, srcNode, dstNode))
748  changed = true;
749 
750  if (isFieldInsensitive(ptd))
751  {
753  const NodeBS& allFields = getAllFieldsObjVars(ptd);
754  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end();
755  fieldIt != fieldEit; ++fieldIt)
756  {
757  if (propVarPtsAfterCGUpdated(*fieldIt, srcNode, dstNode))
758  changed = true;
759  }
760  }
761  }
762 
763  if (changed)
764  pushIntoWorklist(dstNode->getId());
765  }
766  }

822 
823  // < SVFG node ID (loc), SVFIR node of interest (top-level pointer) >.
824  Set<std::pair<NodeID, NodeID>> cmpLocs;
825  for (SVFG::iterator npair = svfg->begin(); npair != svfg->end(); ++npair)
826  {
827  NodeID loc = npair->first;
828  SVFGNode *node = npair->second;
829 
830  // Only care about loads, stores, and GEPs.
831  if (StmtSVFGNode *stmt = SVFUtil::dyn_cast<StmtSVFGNode>(node))

835  {
836  continue;
837  }

840  {
841  continue;
842  }
843

845  if (SVFUtil::isa<LoadSVFGNode>(stmt))
846  {
847  p = stmt->getPAGSrcNodeID();
848  }
849  else if (SVFUtil::isa<StoreSVFGNode>(stmt))
850  {

854  {
855  p = stmt->getPAGSrcNodeID();
856  }
857  else
858  {

863  cmpLocs.insert(std::make_pair(loc, p));
864  }
865  }

868  countAliases(cmpLocs, &mayAliases, &noAliases);

886  {

899  break;

901  ++(*mayAliases);
902  break;
