152 void VersionedFlowSensitiveStat::versionStat(void)
153 {
154  // TODO! Need to merge yield/consume.
155  _NumSingleVersion = 0;
156  _MaxVersions = 0;
157 
158  u32_t totalVersionPtsSize = 0;
159  for (const VersionedFlowSensitive::LocVersionMap *lvm : { &vfspta->consume, &vfspta->yield })
160  {
161  for (const VersionedFlowSensitive::ObjToVersionMap &lov : *lvm)
162  {
163  for (const VersionedFlowSensitive::ObjToVersionMap::value_type &ov : lov)
164  {
165  const NodeID o = ov.first;
166  const Version v = ov.second;
167 
168  ++_NumVersions;
169 
170  // If the version was just over-approximate and never accessed, ignore.
171  // TODO: with vPtD changed there is no interface to check if the PTS
172  // exists; an emptiness check is *not* an existence check.
173  if (vfspta->vPtD->getPts(vfspta->atKey(o, v)).empty()) continue;
174 
175  const PointsTo &ovPts = vfspta->vPtD->getPts(vfspta->atKey(o, v));
176  if (!ovPts.empty()) ++_NumNonEmptyVersions;
177  else ++_NumEmptyVersions;
178 
179  _TotalPtsSize += ovPts.count();
180  totalVersionPtsSize += ovPts.count();
181  if (ovPts.count() > _MaxVersionPtsSize) _MaxVersionPtsSize = ovPts.count();
182  }
183  }
184  }
185 
186  _NumUsedVersions = _NumNonEmptyVersions + _NumEmptyVersions;
187 
188  if (_NumNonEmptyVersions != 0) _AvgVersionPtsSize = (double)totalVersionPtsSize / (double)_NumUsedVersions;
189 
190  _TotalPtsSize += totalVersionPtsSize;
191 }

193 void VersionedFlowSensitiveStat::ptsSizeStat()
194 {
195  u32_t totalValidTopLvlPointers = 0;
196  u32_t totalTopLvlPtsSize = 0;
197  for (SVFIR::iterator it = vfspta->getPAG()->begin(); it != vfspta->getPAG()->end(); ++it)
198  {
199  if (!vfspta->getPAG()->isValidTopLevelPtr(it->second)) continue;
200 
201  NodeID p = it->first;
202 
203  totalValidTopLvlPointers++;
204 
205  u32_t size = vfspta->getPts(p).count();
206  totalTopLvlPtsSize += size;
207  if (size > _MaxTopLvlPtsSize) _MaxTopLvlPtsSize = size;
208  }
209 
210  if (totalValidTopLvlPointers != 0) _AvgTopLvlPtsSize = (double)totalTopLvlPtsSize / (double)totalValidTopLvlPointers;
211 
212  _TotalPtsSize += totalTopLvlPtsSize;
213 
214  if (_NumNonEmptyVersions + totalValidTopLvlPointers != 0)
215  {
216  _AvgPtsSize = (double)_TotalPtsSize / (double)(_NumNonEmptyVersions + totalValidTopLvlPointers);
217  }
218 
219  _MaxPtsSize = _MaxVersionPtsSize > _MaxTopLvlPtsSize ? _MaxVersionPtsSize : _MaxTopLvlPtsSize;
220 }
