237  inline const SVFFunction* getSVFFunction(std::string name)
238  {
239  Function* fun = nullptr;
240 
241  for (u32_t i = 0; i < llvmModuleSet->getModuleNum(); ++i)
242  {
243  Module* mod = llvmModuleSet->getModule(i);
244  fun = mod->getFunction(name);
245  if (fun)
246  {
247  return llvmModuleSet->getSVFFunction(fun);
248  }
249  }
250  return nullptr;
251  }

260  const Function* getDefinition(const Function* fun) const
261  {
262  assert(fun->isDeclaration() && "not a function declaration?");
263  FunDeclToDefMapTy::const_iterator it = FunDeclToDefMap.find(fun);
264  assert(it != FunDeclToDefMap.end() && "has no definition?");
265  return it->second;
266  }
267 
268  bool hasDeclaration(const Function* fun) const
269  {
270  if(fun->isDeclaration() && !hasDefinition(fun))
271  return false;
272 
273  const Function* funDef = fun;
274  if(fun->isDeclaration() && hasDefinition(fun))
275  funDef = getDefinition(fun);
276 
277  FunDefToDeclsMapTy::const_iterator it = FunDefToDeclsMap.find(funDef);
278  return it != FunDefToDeclsMap.end();
279  }
280 
281  const FunctionSetType& getDeclaration(const Function* fun) const
282  {
283  const Function* funDef = fun;
284  if(fun->isDeclaration() && hasDefinition(fun))
285  funDef = getDefinition(fun);
286 
287  FunDefToDeclsMapTy::const_iterator it = FunDefToDeclsMap.find(funDef);
288  assert(it != FunDefToDeclsMap.end() && "does not have a function definition (body)?");
289  return it->second;
