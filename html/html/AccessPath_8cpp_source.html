118 APOffset AccessPath::computeConstantOffset() const
119 {
120 
121  assert(isConstantOffset() && "not a constant offset");
122 
123  if(offsetVarAndGepTypePairs.empty())
124  return getConstantFieldIdx();
125 
126  APOffset totalConstOffset = 0;
127  for(int i = offsetVarAndGepTypePairs.size() - 1; i >= 0; i--)
128  {
129  const SVFValue* value = offsetVarAndGepTypePairs[i].first->getValue();
130  const SVFType* type = offsetVarAndGepTypePairs[i].second;
131  const SVFConstantInt* op = SVFUtil::dyn_cast<SVFConstantInt>(value);
132  assert(op && "not a constant offset?");
133  if(type==nullptr)
134  {
135  totalConstOffset += op->getSExtValue();
136  continue;
137  }
138 
139  if(const SVFPointerType* pty = SVFUtil::dyn_cast<SVFPointerType>(type))
140  totalConstOffset += op->getSExtValue() * getElementNum(pty->getPtrElementType());
141  else
142  {
143  APOffset offset = op->getSExtValue();
144  if (offset >= 0)
145  {
146  u32_t flattenOffset =
147  SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(type,
148  offset);
149  totalConstOffset += flattenOffset;
150  }
151  else
152  totalConstOffset += offset;
153  }
154  }
155  return totalConstOffset;
156 }

177 bool AccessPath::operator< (const AccessPath& rhs) const
178 {
179  if (fldIdx != rhs.fldIdx)
180  return (fldIdx < rhs.fldIdx);
181  else
182  {
183  const OffsetVarAndGepTypePairs& pairVec = getOffsetVarAndGepTypePairVec();
184  const OffsetVarAndGepTypePairs& rhsPairVec = rhs.getOffsetVarAndGepTypePairVec();
185  if (pairVec.size() != rhsPairVec.size())
186  return (pairVec.size() < rhsPairVec.size());
187  else
188  {
189  OffsetVarAndGepTypePairs::const_iterator it = pairVec.begin();
190  OffsetVarAndGepTypePairs::const_iterator rhsIt = rhsPairVec.begin();
191  for (; it != pairVec.end() && rhsIt != rhsPairVec.end(); ++it, ++rhsIt)
192  {
193  return (*it) < (*rhsIt);
194  }
195 
196  return false;
197  }
198  }

201 SVF::AccessPath::LSRelation AccessPath::checkRelation(const AccessPath& LHS, const AccessPath& RHS)
202 {
203  NodeBS lhsLocations = LHS.computeAllLocations();
204  NodeBS rhsLocations = RHS.computeAllLocations();
205  if (lhsLocations.intersects(rhsLocations))
206  {
207  if (lhsLocations == rhsLocations)
208  return Same;
209  else if (lhsLocations.contains(rhsLocations))
210  return Superset;
211  else if (rhsLocations.contains(lhsLocations))
212  return Subset;
213  else
214  return Overlap;
215  }
216  else
217  {
218  return NonOverlap;
219  }
220 }
