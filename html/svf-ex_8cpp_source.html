74void traverseOnICFG(ICFG* icfg, const ICFGNode* iNode)
75{
76 FIFOWorkList<const ICFGNode*> worklist;
77 Set<const ICFGNode*> visited;
78 worklist.push(iNode);
79
81 while (!worklist.empty())
82 {
83 const ICFGNode* vNode = worklist.pop();
84 for (ICFGNode::const_iterator it = vNode->OutEdgeBegin(), eit =
85 vNode->OutEdgeEnd(); it != eit; ++it)
86 {
87 ICFGEdge* edge = *it;
88 ICFGNode* succNode = edge->getDstNode();
89 if (visited.find(succNode) == visited.end())
90 {
91 visited.insert(succNode);
92 worklist.push(succNode);
93 }
94 }
95 }

105void traverseOnVFG(const SVFG* vfg, const SVFVar* svfval)
106{
107 if (!vfg->hasDefSVFGNode(svfval))
108 return;
109 const VFGNode* vNode = vfg->getDefSVFGNode(svfval);
110 FIFOWorkList<const VFGNode*> worklist;
111 Set<const VFGNode*> visited;

118 for (VFGNode::const_iterator it = vNode->OutEdgeBegin(), eit =
119 vNode->OutEdgeEnd(); it != eit; ++it)
120 {
121 VFGEdge* edge = *it;
122 VFGNode* succNode = edge->getDstNode();
123 if (visited.find(succNode) == visited.end())
124 {
125 visited.insert(succNode);
126 worklist.push(succNode);

142int main(int argc, char ** argv)
143{
144
145 std::vector<std::string> moduleNameVec;
146 moduleNameVec = OptionBase::parseOptions(
147 argc, argv, "Whole Program Points-to Analysis", "[options] <input-bitcode...>"
148 );
149
150 LLVMModuleSet::preProcessBCs(moduleNameVec);
151 LLVMModuleSet::buildSVFModule(moduleNameVec);
152
154 SVFIRBuilder builder;
155 SVFIR* pag = builder.build();
156
158 Andersen* ander = AndersenWaveDiff::createAndersenWaveDiff(pag);
159
160
162 CallGraph* callgraph = ander->getCallGraph();
163
165 ICFG* icfg = pag->getICFG();
166
168 VFG* vfg = new VFG(callgraph);
169
171 SVFGBuilder svfBuilder;
172 SVFG* svfg = svfBuilder.buildFullSVFG(ander);
173
175 if (Options::PTSPrint())
176 {
177 for (const auto& it : *svfg)
178 {
179 const SVFGNode* node = it.second;
180 if (node->getValue())
181 {
182 traverseOnVFG(svfg, node->getValue());
184 printPts(ander, node->getValue());
185 for (const SVFGEdge* edge : node->getOutEdges())
186 {
187 const SVFGNode* node2 = edge->getDstNode();
188 if (node2->getValue())
189 aliasQuery(ander, node->getValue(), node2->getValue());
190 }
191 }
192 }
193 }
194
196 if (Options::PTSPrint())
197 {
198 for (const auto& it : *icfg)
199 {
200 const ICFGNode* node = it.second;
201 traverseOnICFG(icfg, node);
202 }
203 }
204
205 // clean up memory
206 delete vfg;
207 AndersenWaveDiff::releaseAndersenWaveDiff();
208 SVFIR::releaseSVFIR();
209
210 LLVMModuleSet::getLLVMModuleSet()->dumpModulesToFile(".svf.bc");
211 SVF::LLVMModuleSet::releaseLLVMModuleSet();
212 llvm::llvm_shutdown();
213 return 0;
214}
