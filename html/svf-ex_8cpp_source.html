77 void traverseOnSVFStmt(const ICFGNode* node)
78 {
79  SVFIR2ItvExeState* svfir2ExeState = new SVFIR2ItvExeState(SVFIR::getPAG());
80  for (const SVFStmt* stmt: node->getSVFStmts())
81  {
82  if (const AddrStmt *addr = SVFUtil::dyn_cast<AddrStmt>(stmt))
83  {
84  svfir2ExeState->translateAddr(addr);
85  }
86  else if (const BinaryOPStmt *binary = SVFUtil::dyn_cast<BinaryOPStmt>(stmt))
87  {
88  svfir2ExeState->translateBinary(binary);
89  }
90  else if (const CmpStmt *cmp = SVFUtil::dyn_cast<CmpStmt>(stmt))
91  {
92  svfir2ExeState->translateCmp(cmp);
93  }
94  else if (const LoadStmt *load = SVFUtil::dyn_cast<LoadStmt>(stmt))
95  {
96  svfir2ExeState->translateLoad(load);
97  }
98  else if (const StoreStmt *store = SVFUtil::dyn_cast<StoreStmt>(stmt))
99  {
100  svfir2ExeState->translateStore(store);
101  }
102  else if (const CopyStmt *copy = SVFUtil::dyn_cast<CopyStmt>(stmt))
103  {
104  svfir2ExeState->translateCopy(copy);
105  }
106  else if (const GepStmt *gep = SVFUtil::dyn_cast<GepStmt>(stmt))
107  {
108  if (gep->isConstantOffset())
109  {
110  gep->accumulateConstantByteOffset();
111  gep->accumulateConstantOffset();
112  }
113  svfir2ExeState->translateGep(gep);
114  }
115  else if (const SelectStmt *select = SVFUtil::dyn_cast<SelectStmt>(stmt))
116  {
117  svfir2ExeState->translateSelect(select);
118  }
119  else if (const PhiStmt *phi = SVFUtil::dyn_cast<PhiStmt>(stmt))
120  {
121  svfir2ExeState->translatePhi(phi);
122  }
123  else if (const CallPE *callPE = SVFUtil::dyn_cast<CallPE>(stmt))
124  {
125  // To handle Call Edge
126  svfir2ExeState->translateCall(callPE);
127  }
128  else if (const RetPE *retPE = SVFUtil::dyn_cast<RetPE>(stmt))
129  {
130  svfir2ExeState->translateRet(retPE);
131  }
132  else
133  assert(false && "implement this part");
134  }

141 void traverseOnICFG(ICFG* icfg, const ICFGNode* iNode)
142 {
143  FIFOWorkList<const ICFGNode*> worklist;
144  Set<const ICFGNode*> visited;
145  worklist.push(iNode);
146 
148  while (!worklist.empty())
149  {
150  const ICFGNode* vNode = worklist.pop();
151  for (ICFGNode::const_iterator it = vNode->OutEdgeBegin(), eit =
152  vNode->OutEdgeEnd(); it != eit; ++it)
153  {
154  ICFGEdge* edge = *it;
155  ICFGNode* succNode = edge->getDstNode();
156  if (visited.find(succNode) == visited.end())
157  {
158  visited.insert(succNode);
159  worklist.push(succNode);
160  }
161  }
162  }
163 }
