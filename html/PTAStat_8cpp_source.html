64 void PTAStat::callgraphStat()

68  PointerAnalysis::CallGraphSCC* callgraphSCC = new PointerAnalysis::CallGraphSCC(graph);

80  PTACallGraph::iterator eit = graph->end();
81  for (; it != eit; ++it)
82  {
83  totalNode++;
84  if(callgraphSCC->isInCycle(it->first))
85  {
86  sccRepNodeSet.insert(callgraphSCC->repNode(it->first));
87  nodeInCycle++;
88  const NodeBS& subNodes = callgraphSCC->subNodes(it->first);
89  if(subNodes.count() > maxNodeInCycle)
90  maxNodeInCycle = subNodes.count();
91  }
92 
93  PTACallGraphNode::const_iterator edgeIt = it->second->InEdgeBegin();
94  PTACallGraphNode::const_iterator edgeEit = it->second->InEdgeEnd();
95  for (; edgeIt != edgeEit; ++edgeIt)
96  {
97  PTACallGraphEdge *edge = *edgeIt;
98  totalEdge+= edge->getDirectCalls().size() + edge->getIndirectCalls().size();
99  if(callgraphSCC->repNode(edge->getSrcID()) == callgraphSCC->repNode(edge->getDstID()))
100  {
101  edgeInCycle+=edge->getDirectCalls().size() + edge->getIndirectCalls().size();
102  }
103  }
104  }
105 
106  totalCycle = sccRepNodeSet.size();
107 
108  PTNumStatMap["TotalNode"] = totalNode;
109  PTNumStatMap["TotalCycle"] = totalCycle;
110  PTNumStatMap["NodeInCycle"] = nodeInCycle;
111  PTNumStatMap["MaxNodeInCycle"] = maxNodeInCycle;
112  PTNumStatMap["TotalEdge"] = totalEdge;
113  PTNumStatMap["CalRetPairInCycle"] = edgeInCycle;
114 
115  SVFStat::printStat("CallGraph Stats");

118 }
