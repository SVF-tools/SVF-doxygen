37 std::string IntervalToIntStr(const IntervalValue& inv)
38 {
39  if (inv.is_infinite())
40  {
41  return inv.toString();
42  }
43  else
44  {
45  s64_t lb_val = inv.lb().getNumeral();
46  s64_t ub_val = inv.ub().getNumeral();
47 
48  // check lb
49  s32_t lb_s32 = (lb_val < static_cast<s64_t>(INT_MIN)) ? INT_MIN :
50  (lb_val > static_cast<s64_t>(INT_MAX)) ? INT_MAX :
51  static_cast<s32_t>(lb_val);
52 
53  // check ub
54  s32_t ub_s32 = (ub_val < static_cast<s64_t>(INT_MIN)) ? INT_MIN :
55  (ub_val > static_cast<s64_t>(INT_MAX)) ? INT_MAX :
56  static_cast<s32_t>(ub_val);
57 
58  return "[" + std::to_string(lb_s32) + ", " + std::to_string(ub_s32) + "]";
59  }

372 bool BufOverflowChecker::detectStrcat(const CallICFGNode *call)
373 {
374  const SVFFunction *fun = SVFUtil::getCallee(call->getCallSite());
375  // check the arg size
376  // if it is strcat group, we need to check the length of string,
377  // e.g. strcat(str1, str2); which checks AllocSize(str1) >= Strlen(str1) + Strlen(str2);
378  // if it is strncat group, we do not need to check the length of string,
379  // e.g. strncat(str1, str2, n); which checks AllocSize(str1) >= Strlen(str1) + n;
380 
381  const std::vector<std::string> strcatGroup = {"__strcat_chk", "strcat", "__wcscat_chk", "wcscat"};
382  const std::vector<std::string> strncatGroup = {"__strncat_chk", "strncat", "__wcsncat_chk", "wcsncat"};
383  if (std::find(strcatGroup.begin(), strcatGroup.end(), fun->getName()) != strcatGroup.end())
384  {
385  CallSite cs = SVFUtil::getSVFCallSite(call->getCallSite());
386  const SVFValue* arg0Val = cs.getArgument(0);
387  const SVFValue* arg1Val = cs.getArgument(1);
388  IntervalValue strLen0 = getStrlen(arg0Val);
389  IntervalValue strLen1 = getStrlen(arg1Val);
390  IntervalValue totalLen = strLen0 + strLen1;
391  return canSafelyAccessMemory(arg0Val, totalLen, call);
392  }
393  else if (std::find(strncatGroup.begin(), strncatGroup.end(), fun->getName()) != strncatGroup.end())
394  {
395  CallSite cs = SVFUtil::getSVFCallSite(call->getCallSite());
396  const SVFValue* arg0Val = cs.getArgument(0);
397  const SVFValue* arg2Val = cs.getArgument(2);
398  IntervalValue arg2Num = _svfir2ExeState->getEs()[_svfir->getValueNode(arg2Val)];
399  IntervalValue strLen0 = getStrlen(arg0Val);
400  IntervalValue totalLen = strLen0 + arg2Num;
401  return canSafelyAccessMemory(arg0Val, totalLen, call);
402  }
403  else
404  {
405  assert(false && "unknown strcat function, please add it to strcatGroup or strncatGroup");
406  abort();
407  }
408 }

410 void BufOverflowChecker::handleExtAPI(const CallICFGNode *call)
411 {
412  AbstractExecution::handleExtAPI(call);
413  const SVFFunction *fun = SVFUtil::getCallee(call->getCallSite());
414  assert(fun && "SVFFunction* is nullptr");
415  CallSite cs = SVFUtil::getSVFCallSite(call->getCallSite());
416  // check the type of mem api,
417  // MEMCPY: like memcpy, memcpy_chk, llvm.memcpy etc.
418  // MEMSET: like memset, memset_chk, llvm.memset etc.
419  // STRCPY: like strcpy, strcpy_chk, wcscpy etc.
420  // STRCAT: like strcat, strcat_chk, wcscat etc.
421  // for other ext api like printf, scanf, etc., they have their own handlers
422  ExtAPIType extType = UNCLASSIFIED;
423  // get type of mem api
424  for (const std::string &annotation: fun->getAnnotations())
425  {
426  if (annotation.find("MEMCPY") != std::string::npos)
427  extType = MEMCPY;
428  if (annotation.find("MEMSET") != std::string::npos)
429  extType = MEMSET;
430  if (annotation.find("STRCPY") != std::string::npos)
431  extType = STRCPY;
432  if (annotation.find("STRCAT") != std::string::npos)
433  extType = STRCAT;
434  }
435  // 1. memcpy functions like memcpy_chk, strncpy, annotate("MEMCPY"), annotate("BUF_CHECK:Arg0, Arg2"), annotate("BUF_CHECK:Arg1, Arg2")
436  if (extType == MEMCPY)
437  {
438  if (_extAPIBufOverflowCheckRules.count(fun->getName()) == 0)
439  {
440  // if it is not in the rules, we do not check it
441  SVFUtil::errs() << "Warning: " << fun->getName() << " is not in the rules, please implement it\n";
442  return;
443  }
444  // call parseMemcpyBufferCheckArgs to parse the BUF_CHECK annotation
445  std::vector<std::pair<u32_t, u32_t>> args = _extAPIBufOverflowCheckRules.at(fun->getName());
446  // loop the args and check the offset
447  for (auto arg: args)
448  {
449  IntervalValue offset = _svfir2ExeState->getEs()[_svfir->getValueNode(cs.getArgument(arg.second))] - IntervalValue(1);
450  canSafelyAccessMemory(cs.getArgument(arg.first), offset, call);
451  }
452  }
453  // 2. memset functions like memset, memset_chk, annotate("MEMSET"), annotate("BUF_CHECK:Arg0, Arg2")
454  else if (extType == MEMSET)
455  {
456  if (_extAPIBufOverflowCheckRules.count(fun->getName()) == 0)
457  {
458  // if it is not in the rules, we do not check it
459  SVFUtil::errs() << "Warning: " << fun->getName() << " is not in the rules, please implement it\n";
460  return;
461  }
462  std::vector<std::pair<u32_t, u32_t>> args = _extAPIBufOverflowCheckRules.at(fun->getName());
463  // loop the args and check the offset
464  for (auto arg: args)
465  {
466  IntervalValue offset = _svfir2ExeState->getEs()[_svfir->getValueNode(cs.getArgument(arg.second))] - IntervalValue(1);
467  canSafelyAccessMemory(cs.getArgument(arg.first), offset, call);
468  }
469  }
470  else if (extType == STRCPY)
471  {
472  detectStrcpy(call);
473  }
474  else if (extType == STRCAT)
475  {
476  detectStrcat(call);
477  }
478  else
479  {
480 
481  }

486 {

497  Set<const SVFValue *> visited;
498  visited.insert(value);
499  Map<const ICFGNode *, IntervalValue> gep_offsets;
500  IntervalValue total_bytes = len;
501  worklist.push(value);
502  std::vector<const CallICFGNode *> callstack = _callSiteStack;
503  while (!worklist.empty())
504  {
505  value = worklist.pop();
506  if (const SVFInstruction *ins = SVFUtil::dyn_cast<SVFInstruction>(value))
507  {
508  const ICFGNode *node = _svfir->getICFG()->getICFGNode(ins);
509  if (const CallICFGNode *callnode = SVFUtil::dyn_cast<CallICFGNode>(node))
510  {
511  AccessMemoryViaRetNode(callnode, worklist, visited);
512  }
513  for (const SVFStmt *stmt: node->getSVFStmts())
514  {
515  if (const CopyStmt *copy = SVFUtil::dyn_cast<CopyStmt>(stmt))
516  {
517  AccessMemoryViaCopyStmt(copy, worklist, visited);
518  }
519  else if (const LoadStmt *load = SVFUtil::dyn_cast<LoadStmt>(stmt))
520  {
521  AccessMemoryViaLoadStmt(load, worklist, visited);
522  }
523  else if (const GepStmt *gep = SVFUtil::dyn_cast<GepStmt>(stmt))
524  {
525  // there are 3 type of gepStmt
526  // 1. ptr get offset
527  // 2. struct get field
528  // 3. array get element
529  // for array gep, there are two kind of overflow checking
530  // Arr [Struct.C * 10] arr, Struct.C {i32 a, i32 b}
531  // arr[11].a = **, it is "lhs = gep *arr, 0 (ptr), 11 (arrIdx), 0 (ptr), 0(struct field)"
532  // 1) in this case arrIdx 11 is overflow.
533  // Other case,
534  // Struct.C {i32 a, [i32*10] b, i32 c}, C.b[11] = 1
535  // it is "lhs - gep *C, 0(ptr), 1(struct field), 0(ptr), 11(arrIdx)"
536  // 2) in this case arrIdx 11 is larger than its getOffsetVar.Type Array([i32*10])
537 
538  // therefore, if last getOffsetVar.Type is not the Array, just check the overall offset and its
539  // gep source type size (together with totalOffset along the value flow).
540  // so if curgepOffset + totalOffset >= gepSrc (overflow)
541  // else totalOffset += curgepOffset
542 
543  // otherwise, if last getOffsetVar.Type is the Array, check the last idx and array. (just offset,
544  // not with totalOffset during check)
545  // so if getOffsetVarVal > getOffsetVar.TypeSize (overflow)
546  // else safe and return.
547  IntervalValue byteOffset;
548  if (gep->isConstantOffset())
549  {
550  byteOffset = IntervalValue(gep->accumulateConstantByteOffset());
551  }
552  else
553  {
554  byteOffset = _svfir2ExeState->getByteOffset(gep);
555  }
556  // for variable offset, join with accumulate gep offset
557  gep_offsets[gep->getICFGNode()] = byteOffset;
558  if (byteOffset.ub().getNumeral() >= Options::MaxFieldLimit() && Options::GepUnknownIdx())
559  {
560  return true;
561  }
562 
563  if (gep->getOffsetVarAndGepTypePairVec().size() > 0)
564  {
565  const SVFVar *gepVal = gep->getOffsetVarAndGepTypePairVec().back().first;
566  const SVFType *gepType = gep->getOffsetVarAndGepTypePairVec().back().second;
567 
568  if (gepType->isArrayTy())
569  {
570  const SVFArrayType *gepArrType = SVFUtil::dyn_cast<SVFArrayType>(gepType);
571  IntervalValue gepArrTotalByte(0);
572  const SVFValue *idxValue = gepVal->getValue();
573  u32_t arrElemSize = gepArrType->getTypeOfElement()->getByteSize();
574  if (const SVFConstantInt *op = SVFUtil::dyn_cast<SVFConstantInt>(idxValue))
575  {
576  u32_t lb = (double) Options::MaxFieldLimit() / arrElemSize >= op->getSExtValue() ?
577  op->getSExtValue() * arrElemSize : Options::MaxFieldLimit();
578  gepArrTotalByte = gepArrTotalByte + IntervalValue(lb, lb);
579  }
580  else
581  {
582  u32_t idx = _svfir->getValueNode(idxValue);
583  IntervalValue idxVal = _svfir2ExeState->getEs()[idx];
584  if (idxVal.isBottom())
585  {
586  gepArrTotalByte = gepArrTotalByte + IntervalValue(0, 0);
587  }
588  else
589  {
590  u32_t ub = (idxVal.ub().getNumeral() < 0) ? 0 :
591  (double) Options::MaxFieldLimit() / arrElemSize >=
592  idxVal.ub().getNumeral() ?
593  arrElemSize * idxVal.ub().getNumeral() : Options::MaxFieldLimit();
594  u32_t lb = (idxVal.lb().getNumeral() < 0) ? 0 :
595  ((double) Options::MaxFieldLimit() / arrElemSize >=
596  idxVal.lb().getNumeral()) ?
597  arrElemSize * idxVal.lb().getNumeral() : Options::MaxFieldLimit();
598  gepArrTotalByte = gepArrTotalByte + IntervalValue(lb, ub);
599  }
600  }
601  total_bytes = total_bytes + gepArrTotalByte;
602  if (total_bytes.ub().getNumeral() >= gepArrType->getByteSize())
603  {
604  std::string msg =
605  "Buffer overflow!! Accessing buffer range: " +
606  IntervalToIntStr(total_bytes) +
607  "\nAllocated Gep buffer size: " +
608  std::to_string(gepArrType->getByteSize()) + "\n";
609  msg += "Position: " + firstValue->toString() + "\n";
610  msg += " The following is the value flow. [[\n";
611  for (auto it = gep_offsets.begin(); it != gep_offsets.end(); ++it)
612  {
613  msg += it->first->toString() + ", Offset: " + IntervalToIntStr(it->second) +
614  "\n";
615  }
616  msg += "]].\nAlloc Site: " + gep->toString() + "\n";
617 
618  BufOverflowException bug(SVFUtil::errMsg(msg), gepArrType->getByteSize(),
619  gepArrType->getByteSize(),
620  total_bytes.lb().getNumeral(), total_bytes.ub().getNumeral(),
621  firstValue);
622  addBugToRecoder(bug, curNode);
623  return false;
624  }
625  else
626  {
627  // for gep last index's type is arr, stop here.
628  return true;
629  }
630  }
631  else
632  {
633  total_bytes = total_bytes + byteOffset;
634  }
635 
636  }
637  if (!visited.count(gep->getRHSVar()->getValue()))
638  {
639  visited.insert(gep->getRHSVar()->getValue());
640  worklist.push(gep->getRHSVar()->getValue());
641  }
642  }
643  else if (const AddrStmt *addr = SVFUtil::dyn_cast<AddrStmt>(stmt))
644  {

651  "Buffer overflow!! Accessing buffer range: " + IntervalToIntStr(total_bytes) +
652  "\nAllocated buffer size: " + std::to_string(arr_type_size) + "\n";
653  msg += "Position: " + firstValue->toString() + "\n";
654  msg += " The following is the value flow. [[\n";
655  for (auto it = gep_offsets.begin(); it != gep_offsets.end(); ++it)
656  {
657  msg += it->first->toString() + ", Offset: " + IntervalToIntStr(it->second) + "\n";

661  total_bytes.lb().getNumeral(), total_bytes.ub().getNumeral(),
662  firstValue);

665  }
