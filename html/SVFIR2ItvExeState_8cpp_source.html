49 AbstractValue SVFIR2ItvExeState::getRangeLimitFromType(const SVFType* type)
50 {
51  if (const SVFIntegerType* intType = SVFUtil::dyn_cast<SVFIntegerType>(type))
52  {
53  u32_t bits = type->getByteSize() * 8;
54  s64_t ub = 0;
55  s64_t lb = 0;
56  if (bits >= 32)
57  {
58  if (intType->isSigned())
59  {
60  ub = static_cast<s64_t>(std::numeric_limits<s32_t>::max());
61  lb = static_cast<s64_t>(std::numeric_limits<s32_t>::min());
62  }
63  else
64  {
65  ub = static_cast<s64_t>(std::numeric_limits<u32_t>::max());
66  lb = static_cast<s64_t>(std::numeric_limits<u32_t>::min());
67  }
68  }
69  else if (bits == 16)
70  {
71  if (intType->isSigned())
72  {
73  ub = static_cast<s64_t>(std::numeric_limits<s16_t>::max());
74  lb = static_cast<s64_t>(std::numeric_limits<s16_t>::min());
75  }
76  else
77  {
78  ub = static_cast<s64_t>(std::numeric_limits<u16_t>::max());
79  lb = static_cast<s64_t>(std::numeric_limits<u16_t>::min());
80  }
81  }
82  else if (bits == 8)
83  {
84  if (intType->isSigned())
85  {
86  ub = static_cast<s64_t>(std::numeric_limits<int8_t>::max());
87  lb = static_cast<s64_t>(std::numeric_limits<int8_t>::min());
88  }
89  else
90  {
91  ub = static_cast<s64_t>(std::numeric_limits<u_int8_t>::max());
92  lb = static_cast<s64_t>(std::numeric_limits<u_int8_t>::min());
93  }
94  }
95  return IntervalValue(lb, ub);
96  }
97  else if (SVFUtil::isa<SVFOtherType>(type))
98  {
99  // handle other type like float double, set s32_t as the range
100  s64_t ub = static_cast<s64_t>(std::numeric_limits<s32_t>::max());
101  s64_t lb = static_cast<s64_t>(std::numeric_limits<s32_t>::min());
102  return IntervalValue(lb, ub);
103  }
104  else
105  {
106  return IntervalValue::top();
107  // other types, return top interval
108  }

432 AbstractValue SVFIR2ItvExeState::getByteOffset(const GepStmt *gep)
433 {
434  if (gep->isConstantOffset())
435  return IntervalValue((s64_t)gep->accumulateConstantByteOffset());
436  AbstractValue res = IntervalValue(0); // Initialize the result interval 'res' to 0.
437  // Loop through the offsetVarAndGepTypePairVec in reverse order.
438  for (int i = gep->getOffsetVarAndGepTypePairVec().size() - 1; i >= 0; i--)
439  {
440  const SVFVar* idxOperandVar =
441  gep->getOffsetVarAndGepTypePairVec()[i].first;
442  const SVFType* idxOperandType =
443  gep->getOffsetVarAndGepTypePairVec()[i].second;
444  // calculating Array/Ptr by elemByteSize * indexOperand
445  if (SVFUtil::isa<SVFArrayType>(idxOperandType) || SVFUtil::isa<SVFPointerType>(idxOperandType))
446  {
447  u32_t elemByteSize = 1;
448  if (const SVFArrayType* arrOperandType = SVFUtil::dyn_cast<SVFArrayType>(idxOperandType))
449  elemByteSize = arrOperandType->getTypeOfElement()->getByteSize();
450  else if (SVFUtil::isa<SVFPointerType>(idxOperandType))
451  elemByteSize = gep->getAccessPath().gepSrcPointeeType()->getByteSize();
452  else
453  assert(false && "idxOperandType must be ArrType or PtrType");
454  if (const SVFConstantInt *op = SVFUtil::dyn_cast<SVFConstantInt>(idxOperandVar->getValue()))
455  {
456  s64_t lb = (double)Options::MaxFieldLimit() / elemByteSize >= op->getSExtValue() ?op->getSExtValue() * elemByteSize
457  : Options::MaxFieldLimit();
458  res = res + IntervalValue(lb, lb);
459  }
460  else
461  {
462  u32_t idx = _svfir->getValueNode(idxOperandVar->getValue());
463  IntervalValue idxVal = _es[idx].getInterval();
464  if (idxVal.isBottom())
465  res = res + IntervalValue(0, 0);
466  else
467  {
468  // if lb or ub is negative number, set 0.
469  // if lb or ub is positive number, guarantee lb/ub * elemByteSize <= MaxFieldLimit
470  s64_t ub = (idxVal.ub().getIntNumeral() < 0) ? 0 :
471  (double)Options::MaxFieldLimit() /
472  elemByteSize >= idxVal.ub().getIntNumeral() ? elemByteSize * idxVal.ub().getIntNumeral(): Options::MaxFieldLimit();
473  s64_t lb = (idxVal.lb().getIntNumeral() < 0) ? 0 :
474  ((double)Options::MaxFieldLimit() /
475  elemByteSize >= idxVal.lb().getIntNumeral()) ? elemByteSize * idxVal.lb().getIntNumeral() : Options::MaxFieldLimit();
476  res = res + IntervalValue(lb, ub);
477  }
478  }
479  }
480  // Process struct subtype by calculating the byte offset from beginning to the field of struct
481  else if (const SVFStructType* structOperandType = SVFUtil::dyn_cast<SVFStructType>(idxOperandType))
482  {
483  res = res + IntervalValue(gep->getAccessPath().getStructFieldOffset(
484  idxOperandVar, structOperandType));
485  }
486  else
487  {
488  assert(false && "gep type pair only support arr/ptr/struct");
489  }
490  }
491  return res; // Return the resulting byte offset as an IntervalValue.
492 }

502 AbstractValue SVFIR2ItvExeState::getItvOfFlattenedElemIndex(const GepStmt *gep)
503 {
504  if (gep->isConstantOffset())
505  return IntervalValue((s64_t)gep->accumulateConstantOffset());
506  AbstractValue res = IntervalValue(0);
507  for (int i = gep->getOffsetVarAndGepTypePairVec().size() - 1; i >= 0; i--)
508  {
509  AccessPath::IdxOperandPair IdxVarAndType =
510  gep->getOffsetVarAndGepTypePairVec()[i];
511  const SVFValue *value =
512  gep->getOffsetVarAndGepTypePairVec()[i].first->getValue();
513  const SVFType *type = IdxVarAndType.second;
514  // idxLb/Ub is the flattened offset generated by the current OffsetVarAndGepTypePair
515  s64_t idxLb;
516  s64_t idxUb;
517  // get lb and ub of the index value
518  if (const SVFConstantInt* constInt = SVFUtil::dyn_cast<SVFConstantInt>(value))
519  idxLb = idxUb = constInt->getSExtValue();
520  else
521  {
522  IntervalValue idxItv = _es[_svfir->getValueNode(value)].getInterval();
523  if (idxItv.isBottom())
524  idxLb = idxUb = 0;
525  else
526  {
527  idxLb = idxItv.lb().getIntNumeral();
528  idxUb = idxItv.ub().getIntNumeral();
529  }
530  }
531  // for pointer type, flattened index = elemNum * idx
532  if (SVFUtil::isa<SVFPointerType>(type))
533  {
534  u32_t elemNum = gep->getAccessPath().getElementNum(gep->getAccessPath().gepSrcPointeeType());
535  idxLb = (double)Options::MaxFieldLimit() / elemNum < idxLb? Options::MaxFieldLimit(): idxLb * elemNum;
536  idxUb = (double)Options::MaxFieldLimit() / elemNum < idxUb? Options::MaxFieldLimit(): idxUb * elemNum;
537  }
538  // for array or struct, get flattened index from SymbolTable Info
539  else
540  {
541  if(Options::ModelArrays())
542  {
543  const std::vector<u32_t>& so = SymbolTableInfo::SymbolInfo()
544  ->getTypeInfo(type)
545  ->getFlattenedElemIdxVec();
546  if (so.empty() || idxUb >= (APOffset)so.size() || idxLb < 0)
547  {
548  idxLb = idxUb = 0;
549  }
550  else
551  {
552  idxLb = SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(
553  type, idxLb);
554  idxUb = SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(
555  type, idxUb);
556  }
557  }
558  else
559  idxLb = idxUb = 0;
560  }
561  res = res + IntervalValue(idxLb, idxUb);
562  }
563  res.meet_with(IntervalValue((s64_t)0, (s64_t)Options::MaxFieldLimit()));
564  if (res.isBottom())
565  {
566  res = IntervalValue(0);
567  }
568  return res;
569 }
