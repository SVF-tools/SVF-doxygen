226  inline const Value* getJoinedThread(const Instruction *inst) const
229  {
230  assert(isTDJoin(inst) && "not a thread join function!");
231  CallSite cs = getLLVMCallSite(inst);
232  Value* join = cs.getArgument(0);
233  if(SVFUtil::isa<LoadInst>(join))
234  return SVFUtil::cast<LoadInst>(join)->getPointerOperand();
235  else if(SVFUtil::isa<Argument>(join))
236  return join;
237  assert(false && "the value of the first argument at join is not a load instruction?");
238  return NULL;
239  }
