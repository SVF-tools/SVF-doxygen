98CallGraphEdge* CallGraph::hasGraphEdge(CallGraphNode* src,
99 CallGraphNode* dst,
100 const CallICFGNode* callIcfgNode) const
101{
102 CallGraphEdge edge(src,dst,callIcfgNode);
103 CallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
104 CallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
105 if (outEdge && inEdge)
106 {
107 assert(outEdge == inEdge && "edges not match");
108 return outEdge;
109 }
110 else
111 return nullptr;
112}

117void CallGraph::addDirectCallGraphEdge(const CallICFGNode* cs,const SVFFunction* callerFun, const SVFFunction* calleeFun)
118{
119
120 CallGraphNode* caller = getCallGraphNode(callerFun);
121 CallGraphNode* callee = getCallGraphNode(calleeFun);
122
123
124 if(!hasGraphEdge(caller,callee, cs))
125 {
126 CallGraphEdge* edge = new CallGraphEdge(caller,callee, cs);
127 edge->addDirectCallSite(cs);
128 addEdge(edge);
129 callinstToCallGraphEdgesMap[cs].insert(edge);
130 }
131}
