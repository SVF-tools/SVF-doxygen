80 bool CallGraphNode::isReachableFromProgEntry() const
81 {
82  std::stack<const CallGraphNode*> nodeStack;
83  NodeBS visitedNodes;
84  nodeStack.push(this);
85  visitedNodes.set(getId());
86 
87  while (nodeStack.empty() == false)
88  {
89  CallGraphNode* node = const_cast<CallGraphNode*>(nodeStack.top());
90  nodeStack.pop();
91 
92  if (SVFUtil::isProgEntryFunction(node->getFunction()))
93  return true;
94 
95  for (const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
96  {
97  CallGraphEdge* edge = *it;
98  if (visitedNodes.test_and_set(edge->getSrcID()))
99  nodeStack.push(edge->getSrcNode());
100  }
101  }
102 
103  return false;
104 }

136 CallGraphEdge* CallGraph::hasGraphEdge(CallGraphNode* src, CallGraphNode* dst,CallGraphEdge::CEDGEK kind, CallSiteID csId) const
137 {
138  CallGraphEdge edge(src,dst,kind,csId);
139  CallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
140  CallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
141  if (outEdge && inEdge)
142  {
143  assert(outEdge == inEdge && "edges not match");
144  return outEdge;
145  }
146  else
147  return nullptr;
148 }

266 void CallGraph::verifyCallGraph()
267 {
268  CallEdgeMap::const_iterator it = indirectCallMap.begin();
269  CallEdgeMap::const_iterator eit = indirectCallMap.end();
270  for (; it != eit; ++it)
271  {
272  const FunctionSet& targets = it->second;
273  if (targets.empty() == false)
274  {
275  const CallICFGNode* cs = it->first;
276  const SVFFunction* func = cs->getCaller();
277  if (getCallGraphNode(func)->isReachableFromProgEntry() == false)
278  writeWrnMsg(func->getName() + " has indirect call site but not reachable from main");
279  }

286 bool CallGraph::isReachableBetweenFunctions(const SVFFunction* srcFn, const SVFFunction* dstFn) const
287 {
288  CallGraphNode* dstNode = getCallGraphNode(dstFn);
289 
290  std::stack<const CallGraphNode*> nodeStack;
291  NodeBS visitedNodes;
292  nodeStack.push(dstNode);
293  visitedNodes.set(dstNode->getId());
294 
295  while (nodeStack.empty() == false)
296  {
297  CallGraphNode* node = const_cast<CallGraphNode*>(nodeStack.top());
298  nodeStack.pop();
299 
300  if (node->getFunction() == srcFn)
301  return true;
302 
303  for (CallGraphEdgeConstIter it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
304  {
305  CallGraphEdge* edge = *it;
306  if (visitedNodes.test_and_set(edge->getSrcID()))
307  nodeStack.push(edge->getSrcNode());
308  }
309  }
310 
311  return false;
312 }
