92bool CallGraphNode::isReachableFromProgEntry() const
93{
94 std::stack<const CallGraphNode*> nodeStack;
95 NodeBS visitedNodes;
96 nodeStack.push(this);
97 visitedNodes.set(getId());
98
99 while (nodeStack.empty() == false)
100 {
101 CallGraphNode* node = const_cast<CallGraphNode*>(nodeStack.top());
102 nodeStack.pop();
103
104 if (SVFUtil::isProgEntryFunction(node->getFunction()))
105 return true;
106
107 for (const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
108 {
109 CallGraphEdge* edge = *it;
110 if (visitedNodes.test_and_set(edge->getSrcID()))
111 nodeStack.push(edge->getSrcNode());
112 }
113 }
114
115 return false;
116}

127CallGraph::CallGraph(const CallGraph& other)
128{
129 callGraphNodeNum = other.getTotalNodeNum();
130 numOfResolvedIndCallEdge = 0;
131 kind = NormCallGraph;
132
134 for (const auto& item : other)
135 {
136 const CallGraphNode* cgn = item.second;
137 CallGraphNode* callGraphNode = new CallGraphNode(cgn->getId(), cgn->getFunction());
138 addGNode(cgn->getId(),callGraphNode);
139 funToCallGraphNodeMap[cgn->getFunction()] = callGraphNode;
140 }
141
143 for (const auto& item : other.callinstToCallGraphEdgesMap)
144 {
145 const CallICFGNode* cs = item.first;
146 for (const CallGraphEdge* edge : item.second)
147 {
148 CallGraphNode* src = getCallGraphNode(edge->getSrcID());
149 CallGraphNode* dst = getCallGraphNode(edge->getDstID());
150 CallSiteID csId = addCallSite(cs, dst->getFunction());
151
152 CallGraphEdge* newEdge = new CallGraphEdge(src,dst, CallGraphEdge::CallRetEdge,csId);
153 newEdge->addDirectCallSite(cs);
154 addEdge(newEdge);
155 callinstToCallGraphEdgesMap[cs].insert(newEdge);
156 }
157 }

171CallGraphEdge* CallGraph::hasGraphEdge(CallGraphNode* src,
172 CallGraphNode* dst,
173 CallGraphEdge::CEDGEK kind, CallSiteID csId) const
174{
175 CallGraphEdge edge(src,dst,kind,csId);
176 CallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
177 CallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
178 if (outEdge && inEdge)
179 {
180 assert(outEdge == inEdge && "edges not match");
181 return outEdge;
182 }
183 else
184 return nullptr;
185}

190CallGraphEdge* CallGraph::getGraphEdge(CallGraphNode* src,
191 CallGraphNode* dst,
192 CallGraphEdge::CEDGEK kind, CallSiteID)
193{
194 for (CallGraphEdge::CallGraphEdgeSet::iterator iter = src->OutEdgeBegin();
195 iter != src->OutEdgeEnd(); ++iter)
196 {
197 CallGraphEdge* edge = (*iter);
198 if (edge->getEdgeKind() == kind && edge->getDstID() == dst->getId())
199 return edge;
200 }
201 return nullptr;
202}

208void CallGraph::addIndirectCallGraphEdge(const CallICFGNode* cs,const FunObjVar* callerFun, const FunObjVar* calleeFun)
209{
210

212 CallGraphNode* callee = getCallGraphNode(calleeFun);
213
214 numOfResolvedIndCallEdge++;
215
216 CallSiteID csId = addCallSite(cs, callee->getFunction());
217
218 if(!hasGraphEdge(caller,callee, CallGraphEdge::CallRetEdge,csId))
219 {
220 CallGraphEdge* edge = new CallGraphEdge(caller,callee, CallGraphEdge::CallRetEdge, csId);
221 edge->addInDirectCallSite(cs);
222 addEdge(edge);
223 callinstToCallGraphEdgesMap[cs].insert(edge);
224 }

286void CallGraph::verifyCallGraph()
287{
288 CallEdgeMap::const_iterator it = indirectCallMap.begin();
289 CallEdgeMap::const_iterator eit = indirectCallMap.end();
290 for (; it != eit; ++it)
291 {
292 const FunctionSet& targets = it->second;
293 if (targets.empty() == false)
294 {
295 const CallICFGNode* cs = it->first;
296 const FunObjVar* func = cs->getCaller();
297 if (getCallGraphNode(func)->isReachableFromProgEntry() == false)
298 writeWrnMsg(func->getName() + " has indirect call site but not reachable from main");
299 }
300 }

306bool CallGraph::isReachableBetweenFunctions(const FunObjVar* srcFn, const FunObjVar* dstFn) const
307{
308 CallGraphNode* dstNode = getCallGraphNode(dstFn);
309
310 std::stack<const CallGraphNode*> nodeStack;
311 NodeBS visitedNodes;
312 nodeStack.push(dstNode);
313 visitedNodes.set(dstNode->getId());
314
315 while (nodeStack.empty() == false)
316 {
317 CallGraphNode* node = const_cast<CallGraphNode*>(nodeStack.top());
318 nodeStack.pop();
319
320 if (node->getFunction() == srcFn)
321 return true;
322
323 for (CallGraphEdgeConstIter it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit; ++it)
324 {
325 CallGraphEdge* edge = *it;
326 if (visitedNodes.test_and_set(edge->getSrcID()))
327 nodeStack.push(edge->getSrcNode());
328 }
329 }
330
331 return false;
332}
