98 CallGraphEdge* CallGraph::hasGraphEdge(CallGraphNode* src,
99  CallGraphNode* dst,
100  const CallICFGNode* callIcfgNode) const
101 {
102  CallGraphEdge edge(src,dst,callIcfgNode);
103  CallGraphEdge* outEdge = src->hasOutgoingEdge(&edge);
104  CallGraphEdge* inEdge = dst->hasIncomingEdge(&edge);
105  if (outEdge && inEdge)
106  {
107  assert(outEdge == inEdge && "edges not match");
108  return outEdge;
109  }
110  else
111  return nullptr;
112 }
