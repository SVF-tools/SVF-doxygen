170  virtual void handleSingleStatement(const DPIm& dpm, CPtSet& pts)
171  {
173  resolveFunPtr(dpm);
174 
175  const SVFGNode* node = dpm.getLoc();
176  if(SVFUtil::isa<AddrSVFGNode>(node))
177  {
178  handleAddr(pts,dpm,SVFUtil::cast<AddrSVFGNode>(node));
179  }
180  else if (SVFUtil::isa<CopySVFGNode, PHISVFGNode, ActualParmSVFGNode,
181  FormalParmSVFGNode, ActualRetSVFGNode,
182  FormalRetSVFGNode, NullPtrSVFGNode>(node))
183  {
184  backtraceAlongDirectVF(pts,dpm);
185  }
186  else if(SVFUtil::isa<GepSVFGNode>(node))
187  {
188  CPtSet gepPts;
189  backtraceAlongDirectVF(gepPts,dpm);
190  unionDDAPts(pts, processGepPts(SVFUtil::cast<GepSVFGNode>(node),gepPts));
191  }
192  else if(SVFUtil::isa<LoadSVFGNode>(node))
193  {
194  CPtSet loadpts;
195  startNewPTCompFromLoadSrc(loadpts,dpm);
196  for(typename CPtSet::iterator it = loadpts.begin(), eit = loadpts.end(); it!=eit; ++it)
197  {
198  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
199  }
200  }
201  else if(SVFUtil::isa<StoreSVFGNode>(node))
202  {
203  if(isMustAlias(getLoadDpm(dpm),dpm))
204  {
205  DBOUT(DDDA, SVFUtil::outs() << "+++must alias for load and store:");
206  DBOUT(DDDA, getLoadDpm(dpm).dump());
207  DBOUT(DDDA, dpm.dump());
208  DBOUT(DDDA, SVFUtil::outs() << "+++\n");
209  DOSTAT(ddaStat->_NumOfMustAliases++);
210  backtraceToStoreSrc(pts,dpm);
211  }
212  else
213  {
214  CPtSet storepts;
215  startNewPTCompFromStoreDst(storepts,dpm);
216  for(typename CPtSet::iterator it = storepts.begin(), eit = storepts.end(); it!=eit; ++it)
217  {
218  if(propagateViaObj(*it,getLoadCVar(dpm)))
219  {
220  backtraceToStoreSrc(pts,getDPImWithOldCond(dpm,*it,node));
221 
222  if(isStrongUpdate(storepts,SVFUtil::cast<StoreSVFGNode>(node)))
223  {
224  DBOUT(DDDA, SVFUtil::outs() << "backward strong update for obj " << dpm.getCurNodeID() << "\n");
225  DOSTAT(addSUStat(dpm,node);)
226  }
227  else
228  {
229  DOSTAT(rmSUStat(dpm,node);)
230  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
231  }
232  }
233  else
234  {
235  backtraceAlongIndirectVF(pts,dpm);
236  }
237  }
238  }
239  }
240  else if(SVFUtil::isa<MRSVFGNode>(node))
241  {
242  backtraceAlongIndirectVF(pts,dpm);
243  }
244  else
245  assert(false && "unexpected kind of SVFG nodes");
246  }

249  void reCompute(const DPIm& dpm)
250  {
252  SVFGEdgeSet newIndirectEdges;
253  if(_pag->isFunPtr(dpm.getCurNodeID()))
254  {
255  const CallSiteSet& csSet = _pag->getIndCallSites(dpm.getCurNodeID());
256  for(CallSiteSet::const_iterator it = csSet.begin(), eit = csSet.end(); it!=eit; ++it)
257  updateCallGraphAndSVFG(dpm, (*it),newIndirectEdges);
258  }
260  if(!newIndirectEdges.empty())
261  _callGraphSCC->find();
262  reComputeForEdges(dpm,newIndirectEdges,true);
263 
265  SVFGEdgeSet edgeSet(dpm.getLoc()->getOutEdges());
266  reComputeForEdges(dpm,edgeSet,false);
267  }

270  void reComputeForEdges(const DPIm& dpm, const SVFGEdgeSet& edgeSet, bool indirectCall = false)
271  {
272  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
273  {
274  const SVFGEdge* edge = *it;
275  const SVFGNode* dst = edge->getDstNode();
276  typename LocToDPMVecMap::const_iterator locIt = getLocToDPMVecMap().find(dst->getId());
278  if (locIt == getLocToDPMVecMap().end())
279  continue;
280  DPTItemSet dpmSet(locIt->second.begin(), locIt->second.end());
281  for(typename DPTItemSet::const_iterator it = dpmSet.begin(),eit = dpmSet.end(); it!=eit; ++it)
282  {
283  const DPIm& dstDpm = *it;
284  if(!indirectCall && SVFUtil::isa<IndirectSVFGEdge>(edge) && !SVFUtil::isa<LoadSVFGNode>(edge->getDstNode()))
285  {
286  if(dstDpm.getCurNodeID() == dpm.getCurNodeID())
287  {
288  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute, forward from :");
289  DBOUT(DDDA, dpm.dump());
290  DOSTAT(ddaStat->_NumOfStepInCycle++);
291  clearbkVisited(dstDpm);
292  findPT(dstDpm);
293  }
294  }
295  else
296  {
297  if(indirectCall)
298  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute for indirect call from :");
299  else
300  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute forward from :");
301  DBOUT(DDDA, dpm.dump());
302  DOSTAT(ddaStat->_NumOfStepInCycle++);
303  clearbkVisited(dstDpm);
304  findPT(dstDpm);
305  }
306  }
307  }

346  void backtraceAlongIndirectVF(CPtSet& pts, const DPIm& oldDpm)
347  {
348  const SVFGNode* node = oldDpm.getLoc();
349  NodeID obj = oldDpm.getCurNodeID();
350  if (_pag->isConstantObj(obj) || _pag->isNonPointerObj(obj))
351  return;
352  const SVFGEdgeSet edgeSet(node->getInEdges());
353  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
354  {
355  if(const IndirectSVFGEdge* indirEdge = SVFUtil::dyn_cast<IndirectSVFGEdge>(*it))
356  {
357  const NodeBS& guard = indirEdge->getPointsTo();
358  if(guard.test(obj))
359  {
360  DBOUT(DDDA, SVFUtil::outs() << "\t\t==backtrace indirectVF svfgNode " <<
361  indirEdge->getDstID() << " --> " << indirEdge->getSrcID() << "\n");
362  backwardPropDpm(pts,oldDpm.getCurNodeID(),oldDpm,indirEdge);
363  }
364  }
365  }
366  }
368  void backtraceAlongDirectVF(CPtSet& pts, const DPIm& oldDpm)
369  {
370  const SVFGNode* node = oldDpm.getLoc();
371  const SVFGEdgeSet edgeSet(node->getInEdges());
372  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
373  {
374  if(const DirectSVFGEdge* dirEdge = SVFUtil::dyn_cast<DirectSVFGEdge>(*it))
375  {
376  DBOUT(DDDA, SVFUtil::outs() << "\t\t==backtrace directVF svfgNode " <<
377  dirEdge->getDstID() << " --> " << dirEdge->getSrcID() << "\n");
378  const SVFGNode* srcNode = dirEdge->getSrcNode();
379  backwardPropDpm(pts,getSVFG()->getLHSTopLevPtr(srcNode)->getId(),oldDpm,dirEdge);
380  }
381  }

420  virtual void backwardPropDpm(CPtSet& pts, NodeID ptr,const DPIm& oldDpm,const SVFGEdge* edge)
421  {
422  DPIm dpm(oldDpm);
423  dpm.setLocVar(edge->getSrcNode(),ptr);
424  DOTIMESTAT(double start = DDAStat::getClk(true));
426  if(handleBKCondition(dpm,edge)==false)

428  DOTIMESTAT(ddaStat->_TotalTimeOfBKCondition += DDAStat::getClk(true) - start);

432  }
433

436  addLoadDpmAndCVar(dpm,getLoadDpm(oldDpm),getLoadCVar(oldDpm));

438  DOSTAT(ddaStat->_NumOfDPM++);
