232 bool AbstractInterpretation::isCmpBranchFeasible(const CmpStmt* cmpStmt, s64_t succ,
233  AbstractState& as)
234 {
235  AbstractState new_es = as;
236  // get cmp stmt's op0, op1, and predicate
237  NodeID op0 = cmpStmt->getOpVarID(0);
238  NodeID op1 = cmpStmt->getOpVarID(1);
239  NodeID res_id = cmpStmt->getResID();
240  s32_t predicate = cmpStmt->getPredicate();
241 
242  // if op0 or op1 is undefined, return;
243  // skip address compare
244  if (new_es.inVarToAddrsTable(op0) || new_es.inVarToAddrsTable(op1))
245  {
246  as = new_es;
247  return true;
248  }
249  const LoadStmt *load_op0 = nullptr;
250  const LoadStmt *load_op1 = nullptr;
251  // get '%1 = load i32 s', and load inst may not exist
252  SVFVar* loadVar0 = svfir->getGNode(op0);
253  if (!loadVar0->getInEdges().empty())
254  {
255  SVFStmt *loadVar0InStmt = *loadVar0->getInEdges().begin();
256  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar0InStmt))
257  {
258  load_op0 = loadStmt;
259  }
260  else if (const CopyStmt *copyStmt = SVFUtil::dyn_cast<CopyStmt>(loadVar0InStmt))
261  {
262  loadVar0 = svfir->getGNode(copyStmt->getRHSVarID());
263  if (!loadVar0->getInEdges().empty())
264  {
265  SVFStmt *loadVar0InStmt2 = *loadVar0->getInEdges().begin();
266  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar0InStmt2))
267  {
268  load_op0 = loadStmt;
269  }
270  }
271  }
272  }
273 
274  SVFVar* loadVar1 = svfir->getGNode(op1);
275  if (!loadVar1->getInEdges().empty())
276  {
277  SVFStmt *loadVar1InStmt = *loadVar1->getInEdges().begin();
278  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar1InStmt))
279  {
280  load_op1 = loadStmt;
281  }
282  else if (const CopyStmt *copyStmt = SVFUtil::dyn_cast<CopyStmt>(loadVar1InStmt))
283  {
284  loadVar1 = svfir->getGNode(copyStmt->getRHSVarID());
285  if (!loadVar1->getInEdges().empty())
286  {
287  SVFStmt *loadVar1InStmt2 = *loadVar1->getInEdges().begin();
288  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar1InStmt2))
289  {
290  load_op1 = loadStmt;
291  }
292  }
293  }
294  }
295  // for const X const, we may get concrete resVal instantly
296  // for var X const, we may get [0,1] if the intersection of var and const is not empty set
297  IntervalValue resVal = new_es[res_id].getInterval();
298  resVal.meet_with(IntervalValue((s64_t) succ, succ));
299  // If Var X const generates bottom value, it means this branch path is not feasible.
300  if (resVal.isBottom())
301  {
302  return false;
303  }
304 
305  bool b0 = new_es[op0].getInterval().is_numeral();
306  bool b1 = new_es[op1].getInterval().is_numeral();
307 
308  // if const X var, we should reverse op0 and op1.
309  if (b0 && !b1)
310  {
311  std::swap(op0, op1);
312  std::swap(load_op0, load_op1);
313  predicate = _switch_lhsrhs_predicate[predicate];
314  }
315  else
316  {
317  // if var X var, we cannot preset the branch condition to infer the intervals of var0,var1
318  if (!b0 && !b1)
319  {
320  as = new_es;
321  return true;
322  }
323  // if const X const, we can instantly get the resVal
324  else if (b0 && b1)
325  {
326  as = new_es;
327  return true;
328  }
329  }
330  // if cmp is 'var X const == false', we should reverse predicate 'var X' const == true'
331  // X' is reverse predicate of X
332  if (succ == 0)
333  {
334  predicate = _reverse_predicate[predicate];
335  }
336  else {}
337  // change interval range according to the compare predicate
338  AddressValue addrs;
339  if(load_op0 && new_es.inVarToAddrsTable(load_op0->getRHSVarID()))
340  addrs = new_es[load_op0->getRHSVarID()].getAddrs();
341 
342  IntervalValue &lhs = new_es[op0].getInterval(), &rhs = new_es[op1].getInterval();
343  switch (predicate)
344  {
345  case CmpStmt::Predicate::ICMP_EQ:
346  case CmpStmt::Predicate::FCMP_OEQ:
347  case CmpStmt::Predicate::FCMP_UEQ:
348  {
349  // Var == Const, so [var.lb, var.ub].meet_with(const)
350  lhs.meet_with(rhs);
351  // if lhs is register value, we should also change its mem obj
352  for (const auto &addr: addrs)
353  {
354  NodeID objId = new_es.getInternalID(addr);
355  if (new_es.inAddrToValTable(objId))
356  {
357  new_es.load(addr).meet_with(rhs);
358  }
359  }
360  break;
361  }
362  case CmpStmt::Predicate::ICMP_NE:
363  case CmpStmt::Predicate::FCMP_ONE:
364  case CmpStmt::Predicate::FCMP_UNE:
365  // Compliment set
366  break;
367  case CmpStmt::Predicate::ICMP_UGT:
368  case CmpStmt::Predicate::ICMP_SGT:
369  case CmpStmt::Predicate::FCMP_OGT:
370  case CmpStmt::Predicate::FCMP_UGT:
371  // Var > Const, so [var.lb, var.ub].meet_with([Const+1, +INF])
372  lhs.meet_with(IntervalValue(rhs.lb() + 1, IntervalValue::plus_infinity()));
373  // if lhs is register value, we should also change its mem obj
374  for (const auto &addr: addrs)
375  {
376  NodeID objId = new_es.getInternalID(addr);
377  if (new_es.inAddrToValTable(objId))
378  {
379  new_es.load(addr).meet_with(
380  IntervalValue(rhs.lb() + 1, IntervalValue::plus_infinity()));
381  }
382  }
383  break;
384  case CmpStmt::Predicate::ICMP_UGE:
385  case CmpStmt::Predicate::ICMP_SGE:
386  case CmpStmt::Predicate::FCMP_OGE:
387  case CmpStmt::Predicate::FCMP_UGE:
388  {
389  // Var >= Const, so [var.lb, var.ub].meet_with([Const, +INF])
390  lhs.meet_with(IntervalValue(rhs.lb(), IntervalValue::plus_infinity()));
391  // if lhs is register value, we should also change its mem obj
392  for (const auto &addr: addrs)
393  {
394  NodeID objId = new_es.getInternalID(addr);
395  if (new_es.inAddrToValTable(objId))
396  {
397  new_es.load(addr).meet_with(
398  IntervalValue(rhs.lb(), IntervalValue::plus_infinity()));
399  }
400  }
401  break;
402  }
403  case CmpStmt::Predicate::ICMP_ULT:
404  case CmpStmt::Predicate::ICMP_SLT:
405  case CmpStmt::Predicate::FCMP_OLT:
406  case CmpStmt::Predicate::FCMP_ULT:
407  {
408  // Var < Const, so [var.lb, var.ub].meet_with([-INF, const.ub-1])
409  lhs.meet_with(IntervalValue(IntervalValue::minus_infinity(), rhs.ub() - 1));
410  // if lhs is register value, we should also change its mem obj
411  for (const auto &addr: addrs)
412  {
413  NodeID objId = new_es.getInternalID(addr);
414  if (new_es.inAddrToValTable(objId))
415  {
416  new_es.load(addr).meet_with(
417  IntervalValue(IntervalValue::minus_infinity(), rhs.ub() - 1));
418  }
419  }
420  break;
421  }
422  case CmpStmt::Predicate::ICMP_ULE:
423  case CmpStmt::Predicate::ICMP_SLE:
424  case CmpStmt::Predicate::FCMP_OLE:
425  case CmpStmt::Predicate::FCMP_ULE:
426  {
427  // Var <= Const, so [var.lb, var.ub].meet_with([-INF, const.ub])
428  lhs.meet_with(IntervalValue(IntervalValue::minus_infinity(), rhs.ub()));
429  // if lhs is register value, we should also change its mem obj
430  for (const auto &addr: addrs)
431  {
432  NodeID objId = new_es.getInternalID(addr);
433  if (new_es.inAddrToValTable(objId))
434  {
435  new_es.load(addr).meet_with(
436  IntervalValue(IntervalValue::minus_infinity(), rhs.ub()));
437  }
438  }
439  break;
440  }
441  case CmpStmt::Predicate::FCMP_FALSE:
442  break;
443  case CmpStmt::Predicate::FCMP_TRUE:
444  break;
445  default:
446  assert(false && "implement this part");
447  abort();
448  }
449  as = new_es;
450  return true;
451 }

453 bool AbstractInterpretation::isSwitchBranchFeasible(const SVFVar* var, s64_t succ,
454  AbstractState& as)
455 {
456  AbstractState new_es = as;
457  IntervalValue& switch_cond = new_es[var->getId()].getInterval();
458  s64_t value = succ;
459  FIFOWorkList<const SVFStmt*> workList;
460  for (SVFStmt *cmpVarInStmt: var->getInEdges())
461  {
462  workList.push(cmpVarInStmt);
463  }
464  switch_cond.meet_with(IntervalValue(value, value));
465  if (switch_cond.isBottom())
466  {
467  return false;
468  }
469  while(!workList.empty())
470  {
471  const SVFStmt* stmt = workList.pop();
472  if (SVFUtil::isa<CopyStmt>(stmt))
473  {
474  IntervalValue& copy_cond = new_es[var->getId()].getInterval();
475  copy_cond.meet_with(IntervalValue(value, value));
476  }
477  else if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(stmt))
478  {
479  if (new_es.inVarToAddrsTable(load->getRHSVarID()))
480  {
481  AddressValue &addrs = new_es[load->getRHSVarID()].getAddrs();
482  for (const auto &addr: addrs)
483  {
484  NodeID objId = new_es.getInternalID(addr);
485  if (new_es.inAddrToValTable(objId))
486  {
487  new_es.load(addr).meet_with(switch_cond);
488  }
489  }
490  }
491  }
492  }
493  as = new_es;
494  return true;
495 }

710 void AbstractInterpretation::handleCycleWTO(const ICFGCycleWTO*cycle)
711 {
712  const ICFGNode* cycle_head = cycle->head()->getICFGNode();
713  // Flag to indicate if we are in the increasing phase
714  bool increasing = true;
715  // Infinite loop until a fixpoint is reached,
716  for (u32_t cur_iter = 0;; cur_iter++)
717  {
718  // Start widening or narrowing if cur_iter >= widen threshold (widen delay)
719  if (cur_iter >= Options::WidenDelay())
720  {
721  // Widen or narrow after processing cycle head node
722  AbstractState prev_head_state = abstractTrace[cycle_head];
723  handleWTOComponent(cycle->head());
724  AbstractState cur_head_state = abstractTrace[cycle_head];
725  if (increasing)
726  {
727  // Widening phase
728  abstractTrace[cycle_head] = prev_head_state.widening(cur_head_state);
729  if (abstractTrace[cycle_head] == prev_head_state)
730  {
731  increasing = false;
732  continue;
733  }
734  }
735  else
736  {
737  // Widening's fixpoint reached in the widening phase, switch to narrowing
738  abstractTrace[cycle_head] = prev_head_state.narrowing(cur_head_state);
739  if (abstractTrace[cycle_head] == prev_head_state)
740  {
741  // Narrowing's fixpoint reached in the narrowing phase, exit loop
742  break;
743  }
744  }
745  }
746  else
747  {
748  // Handle the cycle head
749  handleSingletonWTO(cycle->head());
750  }
751  // Handle the cycle body
752  handleWTOComponents(cycle->getWTOComponents());
753  }
