225 bool AbstractInterpretation::isCmpBranchFeasible(const CmpStmt* cmpStmt, s64_t succ,
226  AbstractState& as)
227 {
228  AbstractState new_es = as;
229  // get cmp stmt's op0, op1, and predicate
230  NodeID op0 = cmpStmt->getOpVarID(0);
231  NodeID op1 = cmpStmt->getOpVarID(1);
232  NodeID res_id = cmpStmt->getResID();
233  s32_t predicate = cmpStmt->getPredicate();
234 
235  // if op0 or op1 is undefined, return;
236  // skip address compare
237  if (new_es.inVarToAddrsTable(op0) || new_es.inVarToAddrsTable(op1))
238  {
239  as = new_es;
240  return true;
241  }
242  const LoadStmt *load_op0 = nullptr;
243  const LoadStmt *load_op1 = nullptr;
244  // get '%1 = load i32 s', and load inst may not exist
245  SVFVar* loadVar0 = _svfir->getGNode(op0);
246  if (!loadVar0->getInEdges().empty())
247  {
248  SVFStmt *loadVar0InStmt = *loadVar0->getInEdges().begin();
249  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar0InStmt))
250  {
251  load_op0 = loadStmt;
252  }
253  else if (const CopyStmt *copyStmt = SVFUtil::dyn_cast<CopyStmt>(loadVar0InStmt))
254  {
255  loadVar0 = _svfir->getGNode(copyStmt->getRHSVarID());
256  if (!loadVar0->getInEdges().empty())
257  {
258  SVFStmt *loadVar0InStmt2 = *loadVar0->getInEdges().begin();
259  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar0InStmt2))
260  {
261  load_op0 = loadStmt;
262  }
263  }
264  }
265  }
266 
267  SVFVar* loadVar1 = _svfir->getGNode(op1);
268  if (!loadVar1->getInEdges().empty())
269  {
270  SVFStmt *loadVar1InStmt = *loadVar1->getInEdges().begin();
271  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar1InStmt))
272  {
273  load_op1 = loadStmt;
274  }
275  else if (const CopyStmt *copyStmt = SVFUtil::dyn_cast<CopyStmt>(loadVar1InStmt))
276  {
277  loadVar1 = _svfir->getGNode(copyStmt->getRHSVarID());
278  if (!loadVar1->getInEdges().empty())
279  {
280  SVFStmt *loadVar1InStmt2 = *loadVar1->getInEdges().begin();
281  if (const LoadStmt *loadStmt = SVFUtil::dyn_cast<LoadStmt>(loadVar1InStmt2))
282  {
283  load_op1 = loadStmt;
284  }
285  }
286  }
287  }
288  // for const X const, we may get concrete resVal instantly
289  // for var X const, we may get [0,1] if the intersection of var and const is not empty set
290  IntervalValue resVal = new_es[res_id].getInterval();
291  resVal.meet_with(IntervalValue((s64_t) succ, succ));
292  // If Var X const generates bottom value, it means this branch path is not feasible.
293  if (resVal.isBottom())
294  {
295  return false;
296  }
297 
298  bool b0 = new_es[op0].getInterval().is_numeral();
299  bool b1 = new_es[op1].getInterval().is_numeral();
300 
301  // if const X var, we should reverse op0 and op1.
302  if (b0 && !b1)
303  {
304  std::swap(op0, op1);
305  std::swap(load_op0, load_op1);
306  predicate = _switch_lhsrhs_predicate[predicate];
307  }
308  else
309  {
310  // if var X var, we cannot preset the branch condition to infer the intervals of var0,var1
311  if (!b0 && !b1)
312  {
313  as = new_es;
314  return true;
315  }
316  // if const X const, we can instantly get the resVal
317  else if (b0 && b1)
318  {
319  as = new_es;
320  return true;
321  }
322  }
323  // if cmp is 'var X const == false', we should reverse predicate 'var X' const == true'
324  // X' is reverse predicate of X
325  if (succ == 0)
326  {
327  predicate = _reverse_predicate[predicate];
328  }
329  else {}
330  // change interval range according to the compare predicate
331  AddressValue addrs;
332  if(load_op0 && new_es.inVarToAddrsTable(load_op0->getRHSVarID()))
333  addrs = new_es.getAddrs(load_op0->getRHSVarID()).getAddrs();
334 
335  IntervalValue &lhs = new_es[op0].getInterval(), &rhs = new_es[op1].getInterval();
336  switch (predicate)
337  {
338  case CmpStmt::Predicate::ICMP_EQ:
339  case CmpStmt::Predicate::FCMP_OEQ:
340  case CmpStmt::Predicate::FCMP_UEQ:
341  {
342  // Var == Const, so [var.lb, var.ub].meet_with(const)
343  lhs.meet_with(rhs);
344  // if lhs is register value, we should also change its mem obj
345  for (const auto &addr: addrs)
346  {
347  NodeID objId = new_es.getInternalID(addr);
348  if (new_es.inLocToValTable(objId))
349  {
350  new_es.load(addr).meet_with(rhs);
351  }
352  }
353  break;
354  }
355  case CmpStmt::Predicate::ICMP_NE:
356  case CmpStmt::Predicate::FCMP_ONE:
357  case CmpStmt::Predicate::FCMP_UNE:
358  // Compliment set
359  break;
360  case CmpStmt::Predicate::ICMP_UGT:
361  case CmpStmt::Predicate::ICMP_SGT:
362  case CmpStmt::Predicate::FCMP_OGT:
363  case CmpStmt::Predicate::FCMP_UGT:
364  // Var > Const, so [var.lb, var.ub].meet_with([Const+1, +INF])
365  lhs.meet_with(IntervalValue(rhs.lb() + 1, IntervalValue::plus_infinity()));
366  // if lhs is register value, we should also change its mem obj
367  for (const auto &addr: addrs)
368  {
369  NodeID objId = new_es.getInternalID(addr);
370  if (new_es.inLocToValTable(objId))
371  {
372  new_es.load(addr).meet_with(
373  IntervalValue(rhs.lb() + 1, IntervalValue::plus_infinity()));
374  }
375  }
376  break;
377  case CmpStmt::Predicate::ICMP_UGE:
378  case CmpStmt::Predicate::ICMP_SGE:
379  case CmpStmt::Predicate::FCMP_OGE:
380  case CmpStmt::Predicate::FCMP_UGE:
381  {
382  // Var >= Const, so [var.lb, var.ub].meet_with([Const, +INF])
383  lhs.meet_with(IntervalValue(rhs.lb(), IntervalValue::plus_infinity()));
384  // if lhs is register value, we should also change its mem obj
385  for (const auto &addr: addrs)
386  {
387  NodeID objId = new_es.getInternalID(addr);
388  if (new_es.inLocToValTable(objId))
389  {
390  new_es.load(addr).meet_with(
391  IntervalValue(rhs.lb(), IntervalValue::plus_infinity()));
392  }
393  }
394  break;
395  }
396  case CmpStmt::Predicate::ICMP_ULT:
397  case CmpStmt::Predicate::ICMP_SLT:
398  case CmpStmt::Predicate::FCMP_OLT:
399  case CmpStmt::Predicate::FCMP_ULT:
400  {
401  // Var < Const, so [var.lb, var.ub].meet_with([-INF, const.ub-1])
402  lhs.meet_with(IntervalValue(IntervalValue::minus_infinity(), rhs.ub() - 1));
403  // if lhs is register value, we should also change its mem obj
404  for (const auto &addr: addrs)
405  {
406  NodeID objId = new_es.getInternalID(addr);
407  if (new_es.inLocToValTable(objId))
408  {
409  new_es.load(addr).meet_with(
410  IntervalValue(IntervalValue::minus_infinity(), rhs.ub() - 1));
411  }
412  }
413  break;
414  }
415  case CmpStmt::Predicate::ICMP_ULE:
416  case CmpStmt::Predicate::ICMP_SLE:
417  case CmpStmt::Predicate::FCMP_OLE:
418  case CmpStmt::Predicate::FCMP_ULE:
419  {
420  // Var <= Const, so [var.lb, var.ub].meet_with([-INF, const.ub])
421  lhs.meet_with(IntervalValue(IntervalValue::minus_infinity(), rhs.ub()));
422  // if lhs is register value, we should also change its mem obj
423  for (const auto &addr: addrs)
424  {
425  NodeID objId = new_es.getInternalID(addr);
426  if (new_es.inLocToValTable(objId))
427  {
428  new_es.load(addr).meet_with(
429  IntervalValue(IntervalValue::minus_infinity(), rhs.ub()));
430  }
431  }
432  break;
433  }
434  case CmpStmt::Predicate::FCMP_FALSE:
435  break;
436  case CmpStmt::Predicate::FCMP_TRUE:
437  break;
438  default:
439  assert(false && "implement this part");
440  abort();
441  }
442  as = new_es;
443  return true;
444 }

446 bool AbstractInterpretation::isSwitchBranchFeasible(const SVFVar* var, s64_t succ,
447  AbstractState& as)
448 {
449  AbstractState new_es = as;
450  IntervalValue& switch_cond = new_es[var->getId()].getInterval();
451  s64_t value = succ;
452  FIFOWorkList<const SVFStmt*> workList;
453  for (SVFStmt *cmpVarInStmt: var->getInEdges())
454  {
455  workList.push(cmpVarInStmt);
456  }
457  switch_cond.meet_with(IntervalValue(value, value));
458  if (switch_cond.isBottom())
459  {
460  return false;
461  }
462  while(!workList.empty())
463  {
464  const SVFStmt* stmt = workList.pop();
465  if (SVFUtil::isa<CopyStmt>(stmt))
466  {
467  IntervalValue& copy_cond = new_es[var->getId()].getInterval();
468  copy_cond.meet_with(IntervalValue(value, value));
469  }
470  else if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(stmt))
471  {
472  if (new_es.inVarToAddrsTable(load->getRHSVarID()))
473  {
474  AddressValue &addrs = new_es.getAddrs(load->getRHSVarID()).getAddrs();
475  for (const auto &addr: addrs)
476  {
477  NodeID objId = new_es.getInternalID(addr);
478  if (new_es.inLocToValTable(objId))
479  {
480  new_es.load(addr).meet_with(switch_cond);
481  }
482  }
483  }
484  }
485  }
486  as = new_es;
487  return true;
488 }

692 void AbstractInterpretation::handleCycle(const ICFGWTOCycle *cycle)
693 {
694  // Get execution states from in edges
695  if (!propagateStateIfFeasible(cycle->head()))
696  {
697  // No ES on the in edges - Infeasible block
698  return;
699  }
700  AbstractState pre_es = _preAbsTrace[cycle->head()];
701  // set -widen-delay
702  s32_t widen_delay = Options::WidenDelay();
703  bool incresing = true;
704  for (int i = 0; ; i++)
705  {
706  const ICFGNode* cycle_head = cycle->head();
707  // handle cycle head
708  handleWTONode(cycle_head);
709  if (i < widen_delay)
710  {
711  if (i> 0 && pre_es >= _postAbsTrace[cycle_head])
712  {
713  break;
714  }
715  pre_es = _postAbsTrace[cycle_head];
716  }
717  else
718  {
719  if (i >= widen_delay)
720  {
721  if (incresing)
722  {
723  bool is_fixpoint =
724  isFixPointAfterWidening(cycle_head, pre_es);
725  if (is_fixpoint)
726  {
727  incresing = false;
728  continue;
729  }
730  }
731  else if (!incresing)
732  {
733  bool is_fixpoint =
734  isFixPointAfterNarrowing(cycle_head, pre_es);
735  if (is_fixpoint)
736  break;
737  }
738  }
739  }
740  for (auto it = cycle->begin(); it != cycle->end(); ++it)
741  {
742  const ICFGWTOComp* cur = *it;
743  if (const ICFGWTONode* vertex = SVFUtil::dyn_cast<ICFGWTONode>(cur))
744  {
745  handleWTONode(vertex->node());
746  }
747  else if (const ICFGWTOCycle* cycle2 = SVFUtil::dyn_cast<ICFGWTOCycle>(cur))
748  {
749  handleCycle(cycle2);

754  }
755  }
