227  static double safeAdd(double lhs, double rhs)
228  {
229  if ((lhs == std::numeric_limits<double>::infinity() &&
230  rhs == -std::numeric_limits<double>::infinity()) ||
231  (lhs == -std::numeric_limits<double>::infinity() &&
232  rhs == std::numeric_limits<double>::infinity()))
233  {
234  assert(false && "invalid add");
235  }
236  double res =
237  lhs + rhs; // Perform the addition and store the result in 'res'
238 
239  // Check if the result is positive infinity due to overflow
240  if (res == std::numeric_limits<double>::infinity())
241  {
242  return res; // Positive overflow has occurred, return positive
243  // infinity
244  }
245 
246  // Check if the result is negative infinity, which can indicate a large
247  // negative overflow
248  if (res == -std::numeric_limits<double>::infinity())
249  {
250  return res; // Negative "overflow", effectively an underflow to
251  // negative infinity
252  }
253 
254  // Check for positive overflow: verify if both operands are positive and
255  // their sum exceeds the maximum double value
256  if (lhs > 0 && rhs > 0 &&
257  (std::numeric_limits<double>::max() - lhs) < rhs)
258  {
259  res =
260  std::numeric_limits<double>::infinity(); // Set result to
261  // positive infinity to
262  // indicate overflow
263  return res;
264  }
265 
266  // Check for an underflow scenario: both numbers are negative and their
267  // sum is more negative than what double can represent
268  if (lhs < 0 && rhs < 0 &&
269  (-std::numeric_limits<double>::max() - lhs) > rhs)
270  {
271  res = -std::numeric_limits<
272  double>::infinity(); // Set result to negative infinity to
273  // clarify extreme negative sum
274  return res;
275  }
276 
277  // If none of the above conditions are met, return the result of the
278  // addition
279  return res;
280  }

306  static double safeMul(double lhs, double rhs)
307  {
308  if (doubleEqual(lhs, 0.0f) || doubleEqual(rhs, 0.0f))
309  return 0.0f;
310  double res = lhs * rhs;
311  // Check if the result is positive infinity due to overflow
312  if (res == std::numeric_limits<double>::infinity())
313  {
314  return res; // Positive overflow has occurred, return positive
315  // infinity
316  }
317 
318  // Check if the result is negative infinity, which can indicate a large
319  // negative overflow
320  if (res == -std::numeric_limits<double>::infinity())
321  {
322  return res; // Negative "overflow", effectively an underflow to
323  // negative infinity
324  }
325  // Check for overflow scenarios
326  if (lhs > 0 && rhs > 0 && lhs > std::numeric_limits<double>::max() / rhs)
327  {
328  return std::numeric_limits<double>::infinity();
329  }
330  if (lhs < 0 && rhs < 0 && lhs < std::numeric_limits<double>::max() / rhs)
331  {
332  return std::numeric_limits<double>::infinity();
333  }
334 
335  // Check for "underflow" scenarios (negative overflow)
336  if (lhs > 0 && rhs < 0 &&
337  rhs < std::numeric_limits<double>::lowest() / lhs)
338  {
339  return -std::numeric_limits<double>::infinity();
340  }
341  if (lhs < 0 && rhs > 0 &&
342  lhs < std::numeric_limits<double>::lowest() / rhs)
343  {
344  return -std::numeric_limits<double>::infinity();
345  }
346 
347  return res; // If no overflow or underflow, return the product
348  }

364  static double safeDiv(double lhs, double rhs)
365  {
366  // Check for division by zero
367  if (doubleEqual(rhs, 0.0f))
368  {
369  return (lhs >= 0.0f) ? std::numeric_limits<double>::infinity()
370  : -std::numeric_limits<double>::infinity();
371  }
372  double res = lhs / rhs;
373  // Check if the result is positive infinity due to overflow
374  if (res == std::numeric_limits<double>::infinity())
375  {
376  return res; // Positive overflow has occurred, return positive
377  // infinity
378  }
379 
380  // Check if the result is negative infinity, which can indicate a large
381  // negative overflow
382  if (res == -std::numeric_limits<double>::infinity())
383  {
384  return res; // Negative "overflow", effectively an underflow to
385  // negative infinity
386  }
387 
388  // Check for overflow when dividing small numbers
389  if (rhs > 0 && rhs < std::numeric_limits<double>::min() &&
390  lhs > std::numeric_limits<double>::max() * rhs)
391  {
392  return std::numeric_limits<double>::infinity();
393  }
394  if (rhs < 0 && rhs > -std::numeric_limits<double>::min() &&
395  lhs > std::numeric_limits<double>::max() * rhs)
396  {
397  return -std::numeric_limits<double>::infinity();
398  }
399 
400  return res; // If no special cases, return the quotient
401  }
