42 SVFVar::SVFVar(const SVFValue* val, NodeID i, PNODEK k) :
43  GenericPAGNodeTy(i,k), value(val), func(nullptr)
44 {
45  assert( ValNode <= k && k <= DummyObjNode && "new SVFIR node kind?");
46  switch (k)
47  {
48  case ValNode:
49  case GepValNode:
50  {
51  assert(val != nullptr && "value is nullptr for ValVar or GepValNode");
52  isPtr = val->getType()->isPointerTy();
53  break;
54  }
55  case FunValNode:
56  case VarargNode:
57  case DummyValNode:
58  {
59  isPtr = true;
60  break;
61  }
62  case ObjNode:
63  case GepObjNode:
64  case BaseObjNode:
65  case DummyObjNode:
66  {
67  isPtr = true;
68  if(val)
69  isPtr = val->getType()->isPointerTy();
70  break;
71  }
72  case RetNode:
73  case FunObjNode:
74  case HeapObjNode:
75  case StackObjNode:
76  {
77  // to be completed in derived class
78  break;
79  }
80  default:
81  assert(false && "var not handled");
82  break;
83  }
84 }

86 bool SVFVar::isIsolatedNode() const
87 {
88  if (getInEdges().empty() && getOutEdges().empty())
89  return true;
90  else if (isConstDataOrAggDataButNotNullPtr())
91  return true;
92  else
93  return false;
