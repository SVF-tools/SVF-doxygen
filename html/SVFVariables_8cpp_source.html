42SVFVar::SVFVar(const SVFValue* val, NodeID i, PNODEK k) :
43 GenericPAGNodeTy(i,k), value(val), func(nullptr)
44{
45 assert( ValNode <= k && k <= DummyObjNode && "new SVFIR node kind?");
46 switch (k)
47 {
48 case ValNode:
49 case ArgNode:
50 case ConstantDataValNode:
51 case GlobalValNode:
52 case BlackHoleNode:
53 case ConstantFPValNode:
54 case ConstantIntValNode:
55 case ConstantNullptrValNode:
56 case GepValNode:
57 {
58 assert(val != nullptr && "value is nullptr for ValVar or GepValNode");
59 isPtr = val->getType()->isPointerTy();
60 break;
61 }
62 case FunValNode:
63 case VarargNode:
64 case DummyValNode:
65 {
66 isPtr = true;
67 break;
68 }
69 case ObjNode:
70 case GepObjNode:
71 case BaseObjNode:
72 case ConstantDataObjNode:
73 case GlobalObjNode:
74 case ConstantFPObjNode:
75 case ConstantIntObjNode:
76 case ConstantNullptrObjNode:
77 case DummyObjNode:
78 {
79 isPtr = true;
80 if(val)
81 isPtr = val->getType()->isPointerTy();
82 break;
83 }
84 case RetNode:
85 case FunObjNode:
86 case HeapObjNode:
87 case StackObjNode:
88 {
89 // to be completed in derived class
90 break;
91 }
92 default:
93 assert(false && "var not handled");
94 break;
95 }
96}

98bool SVFVar::isIsolatedNode() const
99{
100 if (getInEdges().empty() && getOutEdges().empty())
101 return true;
102 else if (isConstDataOrAggDataButNotNullPtr())
103 return true;
104 else
105 return false;
106}
