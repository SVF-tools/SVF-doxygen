42 void CFBasicBlockGBuilder::initCFBasicBlockGNodes(ICFG *icfg,
43  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
44 {
45  for (const auto &func : *PAG::getPAG()->getModule())
46  {
47  for(const auto& bb: *func)
48  {
49  for(const auto& inst: *bb)
50  {
51  if(SVFUtil::isIntrinsicInst(inst)) continue;
52  const ICFGNode* icfgNode = icfg->getICFGNode(inst);
53  if (const CallICFGNode *callNode = SVFUtil::dyn_cast<CallICFGNode>(icfgNode))
54  {
55  // Create a new CFBasicBlockNode for the CallICFGNode
56  CFBasicBlockNode* pNode = new CFBasicBlockNode({callNode});
57  bbToNodes[bb].push_back(pNode);
58  _CFBasicBlockG->addCFBBNode(pNode);
59 
60  // Create a new CFBasicBlockNode for the corresponding RetICFGNode
61  auto *retNode = new CFBasicBlockNode({callNode->getRetICFGNode()});
62  bbToNodes[bb].push_back(retNode);
63  _CFBasicBlockG->addCFBBNode(retNode);
64  }
65  else
66  {
67  if (bbToNodes.find(bb) == bbToNodes.end())
68  {
69  // Create a new CFBasicBlockNode for the non-CallICFGNode
70  CFBasicBlockNode* pNode = new CFBasicBlockNode({icfgNode});
71  bbToNodes[bb] = {pNode};
72  _CFBasicBlockG->addCFBBNode(pNode);
73  }
74  else
75  {
76  CFBasicBlockNode* pNode = bbToNodes[bb].back();
77  if (!SVFUtil::isa<RetICFGNode>(pNode->getICFGNodes()[0]))
78  {
79  // Add the non-CallICFGNode to the existing CFBasicBlockNode
80  pNode->addNode(icfgNode);
81  }
82  else
83  {
84  // Create a new CFBasicBlockNode for the non-CallICFGNode
85  pNode = new CFBasicBlockNode({icfgNode});
86  bbToNodes[bb].push_back(pNode);
87  _CFBasicBlockG->addCFBBNode(pNode);
88  }
89  }
90  }
91  }
92  }
93 
94  if(const FunEntryICFGNode* funEntryNode = icfg->getFunEntryICFGNode(func))
95  {
96  if(const SVFBasicBlock* bb = funEntryNode->getBB())
97  {
98  std::vector<CFBasicBlockNode *>& nodes = bbToNodes[bb];
99  CFBasicBlockNode* pNode = new CFBasicBlockNode({funEntryNode});
100  nodes.insert(nodes.begin(), pNode);
101  _CFBasicBlockG->addCFBBNode(pNode);
102  }
103  }
104 
105  if(const FunExitICFGNode* funExitNode = icfg->getFunExitICFGNode(func))
106  {
107  if(const SVFBasicBlock* bb = funExitNode->getBB())
108  {
109  std::vector<CFBasicBlockNode *>& nodes = bbToNodes[bb];
110  CFBasicBlockNode* pNode = new CFBasicBlockNode({funExitNode});
111  nodes.push_back(pNode);
112  _CFBasicBlockG->addCFBBNode(pNode);
113  }
114  }
115  }
116 }

125 void CFBasicBlockGBuilder::addInterBBEdge(ICFG *icfg,
126  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
127 {
128  // Connect inter-BB BBNodes
129  for (const auto &node : *icfg)
130  {
131  for (const auto &succ : node.second->getOutEdges())
132  {
133  // Check if it's an intraCFGEdge in case of recursive functions
134  // if recursive functions, node_fun == succ_fun but they are different call context
135  // edges related to recursive functions would be added in addInterProceduralEdge()
136  if (succ->isIntraCFGEdge())
137  {
138  const SVFFunction *node_fun = node.second->getFun();
139  const SVFFunction *succ_fun = succ->getDstNode()->getFun();
140  const SVFBasicBlock *node_bb = node.second->getBB();
141  const SVFBasicBlock *succ_bb = succ->getDstNode()->getBB();
142  if (node_fun == succ_fun)
143  {
144  if (node_bb != succ_bb)
145  {
146  // Create a new CFBasicBlockEdge connecting the last node of the source BB
147  // and the first node of the destination BB
148  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(
149  bbToNodes[node_bb].back(),
150  bbToNodes[succ_bb].front(), succ);
151  _CFBasicBlockG->addCFBBEdge(pEdge);
152  }
153  }
154  }
155  }
156  }

165 void CFBasicBlockGBuilder::addIntraBBEdge(ICFG *icfg,
166  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
167 {
168  // Connect intra-BB BBNodes
169  for (const auto &bbNodes : bbToNodes)
170  {
171  for (u32_t i = 0; i < bbNodes.second.size() - 1; ++i)
172  {
173  // Check if an intraCFGEdge exists between the last node of the source BB and the first node of the destination BB
174  if (ICFGEdge *icfgEdge = icfg->getICFGEdge(
175  const_cast<ICFGNode *>(bbNodes.second[i]->getICFGNodes().back()),
176  const_cast<ICFGNode *>(bbNodes.second[i + 1]->getICFGNodes().front()),
177  ICFGEdge::IntraCF))
178  {
179  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbNodes.second[i], bbNodes.second[i + 1], icfgEdge);
180  _CFBasicBlockG->addCFBBEdge(pEdge);
181  }
182  else
183  {
184  // No intra-procedural edge found, possibly an external API call
185  }
186  }
187  }

196 void CFBasicBlockGBuilder::addInterProceduralEdge(ICFG *icfg,
197  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
198 {
199  // Connect inter-procedural BBNodes
200  for (const auto &bbNodes : bbToNodes)
201  {
202  for (u32_t i = 0; i < bbNodes.second.size(); ++i)
203  {
204  if (const CallICFGNode *callICFGNode = SVFUtil::dyn_cast<CallICFGNode>(
205  bbNodes.second[i]->getICFGNodes().front()))
206  {
207  for (const auto &icfgEdge : callICFGNode->getOutEdges())
208  {
209  if (const CallCFGEdge *callEdge = SVFUtil::dyn_cast<CallCFGEdge>(icfgEdge))
210  {
211  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbNodes.second[i],
212  bbToNodes[callEdge->getDstNode()->getBB()].front(),
213  callEdge);
214  _CFBasicBlockG->addCFBBEdge(pEdge);
215  }
216  }
217  }
218  else if (const RetICFGNode *retICFGNode = SVFUtil::dyn_cast<RetICFGNode>(
219  bbNodes.second[i]->getICFGNodes().front()))
220  {
221  for (const auto &icfgEdge : retICFGNode->getInEdges())
222  {
223  if (const RetCFGEdge *retEdge = SVFUtil::dyn_cast<RetCFGEdge>(icfgEdge))
224  {
225  // no return function
226  if(!retEdge->getSrcNode()->getFun()->hasReturn()) continue;
227  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbToNodes[retEdge->getSrcNode()->getBB()].back(),
228  bbNodes.second[i],
229  retEdge);
230  _CFBasicBlockG->addCFBBEdge(pEdge);
231  }
232  }
233  }
234  else
235  {
236  // Other nodes are intra-procedural
237  }
238  }
239  }
