34 void CFBasicBlockGBuilder::initCFBasicBlockGNodes(ICFG *icfg,
35  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
36 {
37  for (const auto &node: *icfg)
38  {
39  CFBasicBlockNode *pNode;
40  if (const SVFBasicBlock *bb = node.second->getBB())
41  {
42  if (const CallICFGNode *callNode = SVFUtil::dyn_cast<CallICFGNode>(node.second))
43  {
44  pNode = new CFBasicBlockNode({callNode});
45  bbToNodes[bb].push_back(pNode);
46  _CFBasicBlockG->addCFBBNode(pNode);
47 
48  auto *retNode = new CFBasicBlockNode({callNode->getRetICFGNode()});
49  bbToNodes[bb].push_back(retNode);
50  _CFBasicBlockG->addCFBBNode(retNode);
51 
52  }
53  else if (!SVFUtil::isa<RetICFGNode>(node.second))
54  {
55  if (bbToNodes.find(bb) == bbToNodes.end())
56  {
57  pNode = new CFBasicBlockNode({node.second});
58  bbToNodes[node.second->getBB()] = {pNode};
59  _CFBasicBlockG->addCFBBNode(pNode);
60  }
61  else
62  {
63  pNode = bbToNodes[node.second->getBB()].back();
64  if (!SVFUtil::isa<RetICFGNode>(pNode->getICFGNodes()[0]))
65  {
66  pNode->addNode(node.second);
67  }
68  else
69  {
70  pNode = new CFBasicBlockNode({node.second});
71  bbToNodes[node.second->getBB()].push_back(pNode);
72  _CFBasicBlockG->addCFBBNode(pNode);
73  }
74  }
75  }
76  }
77  }

80 void CFBasicBlockGBuilder::addInterBBEdge(ICFG *icfg,
81  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)
82 {
83  // connect inter-BB BBNodes
84  for (const auto &node: *icfg)
85  {
86  for (const auto &succ: node.second->getOutEdges())
87  {
88  const SVFFunction *node_fun = node.second->getFun();
89  const SVFFunction *succ_fun = succ->getDstNode()->getFun();
90  const SVFBasicBlock *node_bb = node.second->getBB();
91  const SVFBasicBlock *succ_bb = succ->getDstNode()->getBB();
92  if (node_fun == succ_fun)
93  {
94  if (node_bb != succ_bb)
95  {
96  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbToNodes[node_bb].back(),
97  bbToNodes[succ_bb].front(), succ);
98  _CFBasicBlockG->addCFBBEdge(pEdge);
99  }
100  }
101  }
102  }

116  ICFGEdge::IntraCF))
117  {
118  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbNodes.second[i], bbNodes.second[i + 1], icfgEdge);
119  _CFBasicBlockG->addCFBBEdge(pEdge);
120  }

129 void CFBasicBlockGBuilder::addInterProceduralEdge(ICFG *icfg,
130  Map<const SVFBasicBlock *, std::vector<CFBasicBlockNode *>> &bbToNodes)

134  {
135  for (u32_t i = 0; i < bbNodes.second.size(); ++i)
136  {
137  if (const CallICFGNode *callICFGNode = SVFUtil::dyn_cast<CallICFGNode>(
138  bbNodes.second[i]->getICFGNodes().front()))
139  {
140  for (const auto &icfgEdge: callICFGNode->getOutEdges())
141  {
142  if (const CallCFGEdge *callEdge = SVFUtil::dyn_cast<CallCFGEdge>(icfgEdge))
143  {
144  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbNodes.second[i],
145  bbToNodes[callEdge->getDstNode()->getBB()].front(),
146  callEdge);
147  _CFBasicBlockG->addCFBBEdge(pEdge);
148  }
149  }
150  }
151  else if (const RetICFGNode *retICFGNode = SVFUtil::dyn_cast<RetICFGNode>(
152  bbNodes.second[i]->getICFGNodes().front()))
153  {
154  for (const auto &icfgEdge: retICFGNode->getInEdges())
155  {
156  if (const RetCFGEdge *retEdge = SVFUtil::dyn_cast<RetCFGEdge>(icfgEdge))
157  {
158  CFBasicBlockEdge *pEdge = new CFBasicBlockEdge(bbToNodes[retEdge->getSrcNode()->getBB()].back(),
159  bbNodes.second[i],
160  retEdge);
161  _CFBasicBlockG->addCFBBEdge(pEdge);
162  }
163  }
164  }
165  else
166  {
167  // other nodes are intra-procedural
168  }
169  }
170  }
