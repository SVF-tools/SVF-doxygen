38 CallGraph* CallGraphBuilder::buildCallGraph(SVFModule* svfModule)
39 {
41  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
42  {
43  callgraph->addCallGraphNode(*F);
44  }
45 
47  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
48  {
49  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
50  {
51  for (const ICFGNode* inst : svfbb->getICFGNodeList())
52  {
53  if (SVFUtil::isNonInstricCallSite(inst))
54  {
55  if(const SVFFunction* callee = getCallee(inst))
56  {
57  const CallICFGNode* callBlockNode = cast<CallICFGNode>(inst);
58  callgraph->addDirectCallGraphEdge(callBlockNode,*F,callee);
59  }
60  }
61  }
62  }
63  }
64 
65  return callgraph;
66 }

68 CallGraph* ThreadCallGraphBuilder::buildThreadCallGraph(SVFModule* svfModule)
69 {
70 
71  buildCallGraph(svfModule);
72 
73  ThreadCallGraph* cg = dyn_cast<ThreadCallGraph>(callgraph);
74  assert(cg && "not a thread callgraph?");
75 
76  ThreadAPI* tdAPI = ThreadAPI::getThreadAPI();
77  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
78  {
79  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
80  {
81  for (const ICFGNode* inst : svfbb->getICFGNodeList())
82  {
83  if (tdAPI->isTDFork(inst))
84  {
85  const CallICFGNode* cs = cast<CallICFGNode>(inst);
86  cg->addForksite(cs);
87  const SVFFunction* forkee = SVFUtil::dyn_cast<SVFFunction>(tdAPI->getForkedFun(inst));
88  if (forkee)
89  {
90  cg->addDirectForkEdge(cs);
91  }
92  // indirect call to the start routine function
93  else
94  {
95  cg->addThreadForkEdgeSetMap(cs,nullptr);
96  }
97  }
98  }
99  }
100  }
101  // record join sites
102  for (SVFModule::const_iterator F = svfModule->begin(), E = svfModule->end(); F != E; ++F)
103  {
104  for (const SVFBasicBlock* svfbb : (*F)->getBasicBlockList())
105  {
106  for (const ICFGNode* node : svfbb->getICFGNodeList())
107  {
108  if (tdAPI->isTDJoin(node))
109  {
110  const CallICFGNode* cs = SVFUtil::cast<CallICFGNode>(node);
111  cg->addJoinsite(cs);
112  }
113  }
114  }
115  }
116 
117  return cg;
118 }
