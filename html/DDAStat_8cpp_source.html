107 void DDAStat::performStatPerQuery(NodeID ptr)
108 {
109 
110  u32_t NumOfDPM = 0;
111  u32_t NumOfLoc = 0;
112  u32_t maxNumOfDPMPerLoc = 0;
113  u32_t cptsSize = 0;
114  PointsTo pts;
115  if(flowDDA)
116  {
117  for(FlowDDA::LocToDPMVecMap::const_iterator it = flowDDA->getLocToDPMVecMap().begin(),
118  eit = flowDDA->getLocToDPMVecMap().end(); it!=eit; ++it)
119  {
120  NumOfLoc++;
121  u32_t num = it->second.size();
122  NumOfDPM += num;
123  if(num > maxNumOfDPMPerLoc)
124  maxNumOfDPMPerLoc = num;
125  }
126  cptsSize = flowDDA->getPts(ptr).count();
127  pts = flowDDA->getPts(ptr);
128  }
129  else if(contextDDA)
130  {
131  for(ContextDDA::LocToDPMVecMap::const_iterator it = contextDDA->getLocToDPMVecMap().begin(),
132  eit = contextDDA->getLocToDPMVecMap().end(); it!=eit; ++it)
133  {
134  NumOfLoc++;
135  u32_t num = it->second.size();
136  NumOfDPM += num;
137  if(num > maxNumOfDPMPerLoc)
138  maxNumOfDPMPerLoc = num;
139  }
140  ContextCond cxt;
141  CxtVar var(cxt,ptr);
142  cptsSize = contextDDA->getPts(var).count();
143  pts = contextDDA->getBVPointsTo(contextDDA->getPts(var));
144  }
145  u32_t ptsSize = pts.count();
146 
147  double avgDPMAtLoc = NumOfLoc!=0 ? (double)NumOfDPM/NumOfLoc : 0;
148  _AvgNumOfDPMAtSVFGNode += avgDPMAtLoc;
149  if(maxNumOfDPMPerLoc > _MaxNumOfDPMAtSVFGNode)
150  _MaxNumOfDPMAtSVFGNode = maxNumOfDPMPerLoc;
151 
152  _TotalCPtsSize += cptsSize;
153  if (_MaxCPtsSize < cptsSize)
154  _MaxCPtsSize = cptsSize;
155 
156  _TotalPtsSize += ptsSize;
157  if(_MaxPtsSize < ptsSize)
158  _MaxPtsSize = ptsSize;
159 
160  if(cptsSize == 0)
161  _NumOfNullPtr++;
162 
163  if(getPTA()->containBlackHoleNode(pts))
164  {
165  _NumOfConstantPtr++;
166  }
167  if(getPTA()->containConstantNode(pts))
168  {
169  _NumOfBlackholePtr++;
170  }
171 
172  _TotalNumOfQuery++;
173  _TotalNumOfDPM += _NumOfDPM;
174  _TotalNumOfStrongUpdates += _NumOfStrongUpdates;
175  _TotalNumOfMustAliases += _NumOfMustAliases;
176  _TotalNumOfInfeasiblePath += _NumOfInfeasiblePath;
177 
178  _TotalNumOfStep += _NumOfStep;
179  _TotalNumOfStepInCycle += _NumOfStepInCycle;
180 
181  timeStatMap.clear();
182  NumPerQueryStatMap.clear();
183 
184  timeStatMap["TimePerQuery"] = _AnaTimePerQuery/TIMEINTERVAL;
185  timeStatMap["CyleTimePerQuery"] = _AnaTimeCyclePerQuery/TIMEINTERVAL;
186 
187  NumPerQueryStatMap["CPtsSize"] = cptsSize;
188  NumPerQueryStatMap["PtsSize"] = ptsSize;
189  NumPerQueryStatMap["NumOfStep"] = _NumOfStep;
190  NumPerQueryStatMap["NumOfStepInCycle"] = _NumOfStepInCycle;
191  NumPerQueryStatMap["NumOfDPM"] = _NumOfDPM;
192  NumPerQueryStatMap["NumOfSU"] = _NumOfStrongUpdates;
193  NumPerQueryStatMap["IndEdgeResolved"] = getPTA()->getNumOfResolvedIndCallEdge() - _NumOfIndCallEdgeSolved;
194  NumPerQueryStatMap["AvgDPMAtLoc"] = avgDPMAtLoc;
195  NumPerQueryStatMap["MaxDPMAtLoc"] = maxNumOfDPMPerLoc;
196  NumPerQueryStatMap["MaxPathPerQuery"] = ContextCond::maximumPath;
197  NumPerQueryStatMap["MaxCxtPerQuery"] = ContextCond::maximumCxt;
198  NumPerQueryStatMap["NumOfMustAA"] = _NumOfMustAliases;
199  NumPerQueryStatMap["NumOfInfePath"] = _NumOfInfeasiblePath;
200 
202  _NumOfStep = 0;
203  _NumOfStepInCycle = 0;
204  _NumOfDPM = 0;
205  _NumOfStrongUpdates = 0;
206  _NumOfMustAliases = 0;
207  _NumOfInfeasiblePath = 0;
208  _AnaTimeCyclePerQuery = 0;
209  _NumOfIndCallEdgeSolved = getPTA()->getNumOfResolvedIndCallEdge();
210 }

220 void DDAStat::performStat()
221 {
222 
223  generalNumMap.clear();
224  PTNumStatMap.clear();
225  timeStatMap.clear();
226 
227  callgraphStat();
228 
229  getNumOfOOBQuery();
230 
231  for (SVFIR::const_iterator nodeIt = SVFIR::getPAG()->begin(), nodeEit = SVFIR::getPAG()->end(); nodeIt != nodeEit; nodeIt++)
232  {
233  PAGNode* pagNode = nodeIt->second;
234  if(SVFUtil::isa<ObjVar>(pagNode))
235  {
236  if(getPTA()->isLocalVarInRecursiveFun(nodeIt->first))
237  {
238  localVarInRecursion.set(nodeIt->first);
239  }
240  }
241  }
242 
243  timeStatMap["TotalQueryTime"] = _TotalTimeOfQueries/TIMEINTERVAL;
244  timeStatMap["AvgTimePerQuery"] = (_TotalTimeOfQueries/TIMEINTERVAL)/_TotalNumOfQuery;
245  timeStatMap["TotalBKCondTime"] = (_TotalTimeOfBKCondition/TIMEINTERVAL);
246 
247  PTNumStatMap["NumOfQuery"] = _TotalNumOfQuery;
248  PTNumStatMap["NumOfOOBQuery"] = _TotalNumOfOutOfBudgetQuery;
249  PTNumStatMap["NumOfDPM"] = _TotalNumOfDPM;
250  PTNumStatMap["NumOfSU"] = _TotalNumOfStrongUpdates;
251  PTNumStatMap["NumOfStoreSU"] = _StrongUpdateStores.count();
252  PTNumStatMap["NumOfStep"] = _TotalNumOfStep;
253  PTNumStatMap["NumOfStepInCycle"] = _TotalNumOfStepInCycle;
254  timeStatMap["AvgDPMAtLoc"] = (double)_AvgNumOfDPMAtSVFGNode/_TotalNumOfQuery;
255  PTNumStatMap["MaxDPMAtLoc"] = _MaxNumOfDPMAtSVFGNode;
256  PTNumStatMap["MaxPathPerQuery"] = ContextCond::maximumPath;
257  PTNumStatMap["MaxCxtPerQuery"] = ContextCond::maximumCxt;
258  PTNumStatMap["MaxCPtsSize"] = _MaxCPtsSize;
259  PTNumStatMap["MaxPtsSize"] = _MaxPtsSize;
260  timeStatMap["AvgCPtsSize"] = (double)_TotalCPtsSize/_TotalNumOfQuery;
261  timeStatMap["AvgPtsSize"] = (double)_TotalPtsSize/_TotalNumOfQuery;
262  PTNumStatMap["IndEdgeSolved"] = getPTA()->getNumOfResolvedIndCallEdge();
263  PTNumStatMap["NumOfNullPtr"] = _NumOfNullPtr;
264  PTNumStatMap["PointsToConstPtr"] = _NumOfConstantPtr;
265  PTNumStatMap["PointsToBlkPtr"] = _NumOfBlackholePtr;
266  PTNumStatMap["NumOfMustAA"] = _TotalNumOfMustAliases;
267  PTNumStatMap["NumOfInfePath"] = _TotalNumOfInfeasiblePath;
268  PTNumStatMap["NumOfStore"] = SVFIR::getPAG()->getPTASVFStmtSet(SVFStmt::Store).size();
269  PTNumStatMap["MemoryUsageVmrss"] = _vmrssUsageAfter - _vmrssUsageBefore;
270  PTNumStatMap["MemoryUsageVmsize"] = _vmsizeUsageAfter - _vmsizeUsageBefore;
271 
272  printStat();
273 }

276 {
