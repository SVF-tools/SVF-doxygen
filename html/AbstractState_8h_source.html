355  static bool eqVarToValMap(const VarToAbsValMap&lhs, const VarToAbsValMap&rhs)
356  {
357  if (lhs.size() != rhs.size()) return false;
358  for (const auto &item: lhs)
359  {
360  auto it = rhs.find(item.first);
361  if (it == rhs.end())
362  return false;
363  if (item.second.getType() == it->second.getType())
364  {
365  if (item.second.isInterval())
366  {
367  if (!item.second.getInterval().equals(it->second.getInterval()))
368  {
369  return false;
370  }
371  }
372  else if (item.second.isAddr())
373  {
374  if (!item.second.getAddrs().equals(it->second.getAddrs()))
375  {
376  return false;
377  }
378  }
379  }
380  else
381  {
382  return false;
383  }
384  }
385  return true;
386  }

389  static bool lessThanVarToValMap(const VarToAbsValMap&lhs, const VarToAbsValMap&rhs)
390  {
391  if (lhs.empty()) return !rhs.empty();
392  for (const auto &item: lhs)
393  {
394  auto it = rhs.find(item.first);
395  if (it == rhs.end()) return false;
396  // judge from expr id
397  if (item.second.getInterval().geq(it->second.getInterval())) return false;
398  }
399  return true;
400  }
401 
402  // lhs >= rhs
403  static bool geqVarToValMap(const VarToAbsValMap&lhs, const VarToAbsValMap&rhs)
404  {
405  if (rhs.empty()) return true;
406  for (const auto &item: rhs)
407  {
408  auto it = lhs.find(item.first);
409  if (it == lhs.end()) return false;
410  // judge from expr id
411  if (it->second.isInterval() && item.second.isInterval())
412  {
413  if (!it->second.getInterval().geq(item.second.getInterval()))
414  return false;
415  }
416 
417  }
418  return true;
419  }
