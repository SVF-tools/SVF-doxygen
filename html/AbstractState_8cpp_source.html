41 u32_t AbstractState::hash() const
42 {
43  size_t h = getVarToVal().size() * 2;
44  Hash<u32_t> hf;
45  for (const auto &t: getVarToVal())
46  {
47  h ^= hf(t.first) + 0x9e3779b9 + (h << 6) + (h >> 2);
48  }
49  size_t h2 = getLocToVal().size() * 2;
50  for (const auto &t: getLocToVal())
51  {
52  h2 ^= hf(t.first) + 0x9e3779b9 + (h2 << 6) + (h2 >> 2);
53  }
54  Hash<std::pair<u32_t, u32_t>> pairH;
55  return pairH({h, h2});
56 }
