34 void ICFGSimplification::mergeAdjacentNodes(ICFG* icfg)
35 {
36  Map<const SVFBasicBlock*, std::vector<const ICFGNode*>> bbToNodes;
37  Set<const ICFGNode*> subNodes;
38  for (const auto& func : *PAG::getPAG()->getModule())
39  {
40  for (const auto& bb : *func)
41  {
42  for (const auto& inst : *bb)
43  {
44  if (SVFUtil::isIntrinsicInst(inst))
45  continue;
46  const ICFGNode* icfgNode = icfg->getICFGNode(inst);
47  // e.g.
48  // block: entry
49  // insts: call = i32 %fun(i32 %arg)
50  // We put the callnode in an icfgnode alone, and the retnode is similar.
51  if (const CallICFGNode* callNode =
52  SVFUtil::dyn_cast<CallICFGNode>(icfgNode))
53  {
54  bbToNodes[bb].push_back(callNode);
55  subNodes.insert(callNode);
56 
57  const RetICFGNode* retNode = callNode->getRetICFGNode();
58  bbToNodes[bb].push_back(retNode);
59  subNodes.insert(retNode);
60  }
61  else
62  {
63  // For ordinary instructions, we put multiple instructions in an ICFGNode
70  if (bbToNodes.find(bb) == bbToNodes.end())
71  {
72  // branch 1, for the first node of basic block
73  bbToNodes[bb] = {icfgNode};
74  subNodes.insert(icfgNode);
75  }
76  else
77  {
78  const ICFGNode* pNode = bbToNodes[bb].back();
79  if (!SVFUtil::isa<RetICFGNode>(pNode))
80  {
81  // branch 2, for the middle node of basic block
82  // do nothing if it is not ret node
83  }
84  else
85  {
86  // branch 3, for the node after ret node
87  bbToNodes[bb].push_back(icfgNode);
88  subNodes.insert(icfgNode);
89  }
90  }
91  }
92  }
93  }
94 
95  if (const FunEntryICFGNode* funEntryNode = icfg->getFunEntryICFGNode(func))
96  {
97  if (const SVFBasicBlock* bb = funEntryNode->getBB())
98  {
99  std::vector<const ICFGNode*>& nodes = bbToNodes[bb];
100  nodes.insert(nodes.begin(), funEntryNode);
101  subNodes.insert(funEntryNode);
102  }
103  }
104  if (const FunExitICFGNode* funExitNode = icfg->getFunExitICFGNode(func))
105  {
106  if (const SVFBasicBlock* bb = funExitNode->getBB())
107  {
108  std::vector<const ICFGNode*>& nodes = bbToNodes[bb];
109  nodes.push_back(funExitNode);
110  subNodes.insert(funExitNode);
111  }
112  }
113  }
114 
115  for (auto &it: subNodes)
116  {
117  ICFGNode* head = const_cast<ICFGNode*>(it);
118  if (head->getOutEdges().size() != 1)
119  {
120  // if head has more than one out edges, we don't merge any following nodes.
121  continue;
122  }
123  else
124  {
125  ICFGNode* next = (*head->getOutEdges().begin())->getDstNode();
126  // merge the following nodes, until the next subnode
127  while (subNodes.find(next) == subNodes.end())
128  {
129  assert(icfg->getRepNode(next) != head && "should not find a circle here");
130  icfg->removeICFGEdge(*head->getOutEdges().begin());
131  std::vector<ICFGEdge*> rm_edges;
132  // Step 1: merge the out edges of next to head
133  for (ICFGEdge* outEdge: next->getOutEdges())
134  {
135  rm_edges.push_back(outEdge);
136  ICFGNode* post = outEdge->getDstNode();
137  if (outEdge->isIntraCFGEdge())
138  {
139  IntraCFGEdge* intraEdge = SVFUtil::dyn_cast<IntraCFGEdge>(outEdge);
140  if (intraEdge->getCondition())
141  {
142  icfg->addConditionalIntraEdge(head, post, intraEdge->getCondition(), intraEdge->getSuccessorCondValue());
143  }
144  else
145  {
146  icfg->addIntraEdge(head, post);
147  }
148  }
149  }
150  // Step 2: update the sub node map and rep node map
151  icfg->updateSubAndRep(next, head);
152  if (next->getOutEdges().size() == 1)
153  {
154  // Step 3: remove the edges from next to its next, since next has been merged to head
155  // if only one out edge, we may continue to merge the next node if it is not a subnode
156  next = (*next->getOutEdges().begin())->getDstNode();
157  for (ICFGEdge* edge: rm_edges)
158  icfg->removeICFGEdge(edge);
159 
160  }
161  else
162  {
163  // if more than one out edges, we don't merge any following nodes.
164  for (ICFGEdge* edge: rm_edges)
165  icfg->removeICFGEdge(edge);
166  break;
167  }
168  }
169  }
170  }
171 }
