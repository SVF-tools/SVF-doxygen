160  bool leq(const NumericLiteral &rhs) const
161  {
162  if (is_infinity() ^ rhs.is_infinity())
163  {
164  if (is_infinity())
165  {
166  return is_minus_infinity();
167  }
168  else
169  {
170  return rhs.is_plus_infinity();
171  }
172  }
173  if(is_infinity() && rhs.is_infinity())
174  {
175  if(is_minus_infinity()) return true;
176  else return rhs.is_plus_infinity();
177  }
178  else
179  return _n.leq(rhs._n).simplify().is_true();
