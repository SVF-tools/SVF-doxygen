58  virtual inline OrderedNodeSet& collectCandidateQueries(SVFIR* p)
59  {
60  setPAG(p);
61  if (solveAll)
62  candidateQueries = pag->getAllValidPtrs();
63  else
64  {
65  for (OrderedNodeSet::iterator it = userInput.begin(), eit = userInput.end(); it != eit; ++it)
66  addCandidate(*it);
67  }
68  return candidateQueries;
69  }
