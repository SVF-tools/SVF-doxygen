114 void MemSSA::createMUCHI(const SVFFunction& fun)
115 {
116 
117  SVFIR* pag = pta->getPAG();
118 
119  DBOUT(DMSSA,
120  outs() << "\t creating mu chi for function " << fun.getName()
121  << "\n");
122  // 1. create mu/chi
123  // insert a set of mus for memory regions at each load
124  // inset a set of chis for memory regions at each store
125 
126  // 2. find global names (region name before renaming) of each memory region,
127  // collect used mrs in usedRegs, and collect its def basic block in reg2BBMap
128  // in the form of mu(r) and r = chi (r)
129  // a) mu(r):
130  // if(r \not\in varKills) global = global \cup r
131  // b) r = chi(r):
132  // if(r \not\in varKills) global = global \cup r
133  // varKills = varKills \cup r
134  // block(r) = block(r) \cup bb_{chi}
135 
138  BBList reachableBBs = fun.getReachableBBs();
139 
140  for (BBList::const_iterator iter = reachableBBs.begin(), eiter = reachableBBs.end();
141  iter != eiter; ++iter)
142  {
143  const SVFBasicBlock* bb = *iter;
144  varKills.clear();
145  for (SVFBasicBlock::const_iterator it = bb->begin(), eit = bb->end(); it != eit; ++it)
146  {
147  const SVFInstruction* inst = *it;
148  if(mrGen->hasSVFStmtList(inst))
149  {
150  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(inst);
151  for (SVFStmtList::const_iterator bit = pagEdgeList.begin(),
152  ebit = pagEdgeList.end(); bit != ebit; ++bit)
153  {
154  const PAGEdge* inst = *bit;
155  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(inst))
156  AddLoadMU(bb, load, mrGen->getLoadMRSet(load));
157  else if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(inst))
158  AddStoreCHI(bb, store, mrGen->getStoreMRSet(store));
159  }
160  }
161  if (isNonInstricCallSite(inst))
162  {
163  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(inst);
164  if(mrGen->hasRefMRSet(cs))
165  AddCallSiteMU(cs,mrGen->getCallSiteRefMRSet(cs));
166 
167  if(mrGen->hasModMRSet(cs))
168  AddCallSiteCHI(cs,mrGen->getCallSiteModMRSet(cs));
169  }
170  }
171  }
172 
173  // create entry chi for this function including all memory regions
174  // initialize them with version 0 and 1 r_1 = chi (r_0)
175  for (MRSet::iterator iter = usedRegs.begin(), eiter = usedRegs.end();
176  iter != eiter; ++iter)
177  {
178  const MemRegion* mr = *iter;
179  // initialize mem region version and stack for renaming phase
180  mr2CounterMap[mr] = 0;
181  mr2VerStackMap[mr].clear();
182  ENTRYCHI* chi = new ENTRYCHI(&fun, mr);
183  chi->setOpVer(newSSAName(mr,chi));
184  chi->setResVer(newSSAName(mr,chi));
185  funToEntryChiSetMap[&fun].insert(chi);
186 
189  if(fun.hasReturn())
190  {
191  RETMU* mu = new RETMU(&fun, mr);
192  funToReturnMuSetMap[&fun].insert(mu);
193  }
194 
195  }
196 
197 }

202 void MemSSA::insertPHI(const SVFFunction& fun)
203 {
204 
205  DBOUT(DMSSA,
206  outs() << "\t insert phi for function " << fun.getName() << "\n");
207 
208  const Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>>& df = fun.getDomFrontierMap();
209  // record whether a phi of mr has already been inserted into the bb.
210  BBToMRSetMap bb2MRSetMap;
211 
212  // start inserting phi node
213  for (MRSet::iterator iter = usedRegs.begin(), eiter = usedRegs.end();
214  iter != eiter; ++iter)
215  {
216  const MemRegion* mr = *iter;
217 
218  BBList bbs = reg2BBMap[mr];
219  while (!bbs.empty())
220  {
221  const SVFBasicBlock* bb = bbs.back();
222  bbs.pop_back();
223  Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>>::const_iterator it = df.find(bb);
224  if(it == df.end())
225  {
226  writeWrnMsg("bb not in the dominance frontier map??");
227  continue;
228  }
229  const Set<const SVFBasicBlock*>& domSet = it->second;
230  for (const SVFBasicBlock* pbb : domSet)
231  {
232  // if we never insert this phi node before
233  if (0 == bb2MRSetMap[pbb].count(mr))
234  {
235  bb2MRSetMap[pbb].insert(mr);
236  // insert phi node
237  AddMSSAPHI(pbb,mr);
238  // continue to insert phi in its iterative dominate frontiers
239  bbs.push_back(pbb);
240  }
241  }
242  }
243  }

263 void MemSSA::SSARenameBB(const SVFBasicBlock& bb)
264 {
265 
266  SVFIR* pag = pta->getPAG();
267  // record which mem region needs to pop stack
268  MRVector memRegs;
269 
270  // rename phi result op
271  // for each r = phi (...)
272  // rewrite r as new name
273  if (hasPHISet(&bb))
274  RenamePhiRes(getPHISet(&bb),memRegs);
275 
276 
277  // process mu and chi
278  // for each mu(r)
279  // rewrite r with top mrver of stack(r)
280  // for each r = chi(r')
281  // rewrite r' with top mrver of stack(r)
282  // rewrite r with new name
283 
284  for (SVFBasicBlock::const_iterator it = bb.begin(), eit = bb.end();
285  it != eit; ++it)
286  {
287  const SVFInstruction* inst = *it;
288  if(mrGen->hasSVFStmtList(inst))
289  {
290  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(inst);
291  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
292  bit!=ebit; ++bit)
293  {
294  const PAGEdge* inst = *bit;
295  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(inst))
296  RenameMuSet(getMUSet(load));
297 
298  else if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(inst))
299  RenameChiSet(getCHISet(store),memRegs);
300 
301  }
302  }
303  if (isNonInstricCallSite(inst))
304  {
305  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(inst);
306  if(mrGen->hasRefMRSet(cs))
307  RenameMuSet(getMUSet(cs));
308 
309  if(mrGen->hasModMRSet(cs))
310  RenameChiSet(getCHISet(cs),memRegs);
311  }
312  else if(inst->isRetInst())
313  {
314  const SVFFunction* fun = bb.getParent();
315  RenameMuSet(getReturnMuSet(fun));
316  }
317  }
318 
319 
320  // fill phi operands of succ basic blocks
321  for (const SVFBasicBlock* succ : bb.getSuccessors())
322  {
323  u32_t pos = bb.getBBPredecessorPos(succ);
324  if (hasPHISet(succ))
325  RenamePhiOps(getPHISet(succ),pos,memRegs);
326  }
327 
328  // for succ basic block in dominator tree
329  const SVFFunction* fun = bb.getParent();
330  const Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>>& dtBBsMap = fun->getDomTreeMap();
331  Map<const SVFBasicBlock*,Set<const SVFBasicBlock*>>::const_iterator mapIter = dtBBsMap.find(&bb);
332  if (mapIter != dtBBsMap.end())
333  {
334  const Set<const SVFBasicBlock*>& dtBBs = mapIter->second;
335  for (const SVFBasicBlock* dtbb : dtBBs)
336  {
337  SSARenameBB(*dtbb);
338  }
339  }
340  // for each r = chi(..), and r = phi(..)
341  // pop ver stack(r)
342  while (!memRegs.empty())
343  {
344  const MemRegion* mr = memRegs.back();
345  memRegs.pop_back();
346  mr2VerStackMap[mr].pop_back();
347  }
348

351 MRVer* MemSSA::newSSAName(const MemRegion* mr, MSSADEF* def)
352 {
353  assert(0 != mr2CounterMap.count(mr)
354  && "did not find initial version in map? ");
355  assert(0 != mr2VerStackMap.count(mr)
356  && "did not find initial stack in map? ");
357 
358  MRVERSION version = mr2CounterMap[mr];
359  mr2CounterMap[mr] = version + 1;
360  auto mrVer = std::make_unique<MRVer>(mr, version, def);
361  auto mrVerPtr = mrVer.get();
362  mr2VerStackMap[mr].push_back(mrVerPtr);
363  usedMRVers.push_back(std::move(mrVer));
364  return mrVerPtr;

580 void MemSSA::dumpMSSA(OutStream& Out)
581 {
582  SVFIR* pag = pta->getPAG();
583 
584  for (SVFModule::iterator fit = pta->getModule()->begin(), efit = pta->getModule()->end();
585  fit != efit; ++fit)
586  {
587  const SVFFunction* fun = *fit;
588  if(Options::MSSAFun()!="" && Options::MSSAFun()!=fun->getName())
589  continue;
590 
591  Out << "==========FUNCTION: " << fun->getName() << "==========\n";
592  // dump function entry chi nodes
593  if (hasFuncEntryChi(fun))
594  {
595  CHISet & entry_chis = getFuncEntryChiSet(fun);
596  for (CHISet::iterator chi_it = entry_chis.begin(); chi_it != entry_chis.end(); chi_it++)
597  {
598  (*chi_it)->dump();
599  }
600  }
601 
602  for (SVFFunction::const_iterator bit = fun->begin(), ebit = fun->end();
603  bit != ebit; ++bit)
604  {
605  const SVFBasicBlock* bb = *bit;
606  Out << bb->getName() << "\n";
607  PHISet& phiSet = getPHISet(bb);
608  for(PHISet::iterator pi = phiSet.begin(), epi = phiSet.end(); pi !=epi; ++pi)
609  {
610  (*pi)->dump();
611  }
612 
613  bool last_is_chi = false;
614  for (SVFBasicBlock::const_iterator it = bb->begin(), eit = bb->end();
615  it != eit; ++it)
616  {
617  const SVFInstruction* inst = *it;
618  bool isAppCall = isNonInstricCallSite(inst) && !isExtCall(inst);
619  if (isAppCall || isHeapAllocExtCall(inst))
620  {
621  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(inst);
622  if(hasMU(cs))
623  {
624  if (!last_is_chi)
625  {
626  Out << "\n";
627  }
628  for (MUSet::iterator mit = getMUSet(cs).begin(), emit = getMUSet(cs).end();
629  mit != emit; ++mit)
630  {
631  (*mit)->dump();
632  }
633  }
634 
635  Out << inst->toString() << "\n";
636 
637  if(hasCHI(cs))
638  {
639  for (CHISet::iterator cit = getCHISet(cs).begin(), ecit = getCHISet(cs).end();
640  cit != ecit; ++cit)
641  {
642  (*cit)->dump();
643  }
644  Out << "\n";
645  last_is_chi = true;
646  }
647  else
648  last_is_chi = false;
649  }
650  else
651  {
652  bool dump_preamble = false;
653  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(inst);
654  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
655  bit!=ebit; ++bit)
656  {
657  const PAGEdge* edge = *bit;
658  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(edge))
659  {
660  MUSet& muSet = getMUSet(load);
661  for(MUSet::iterator it = muSet.begin(), eit = muSet.end(); it!=eit; ++it)
662  {
663  if (!dump_preamble && !last_is_chi)
664  {
665  Out << "\n";
666  dump_preamble = true;
667  }
668  (*it)->dump();
669  }
670  }
671  }
672 
673  Out << inst->toString() << "\n";
674 
675  bool has_chi = false;
676  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
677  bit!=ebit; ++bit)
678  {
679  const PAGEdge* edge = *bit;
680  if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(edge))
681  {
682  CHISet& chiSet = getCHISet(store);
683  for(CHISet::iterator it = chiSet.begin(), eit = chiSet.end(); it!=eit; ++it)
684  {
685  has_chi = true;
686  (*it)->dump();
687  }
688  }
689  }
690  if (has_chi)
691  {
692  Out << "\n";
693  last_is_chi = true;
694  }
695  else
696  last_is_chi = false;
697  }
698  }
699  }
700 
701  // dump return mu nodes
702  if (hasReturnMu(fun))
703  {
704  MUSet & return_mus = getReturnMuSet(fun);
705  for (MUSet::iterator mu_it = return_mus.begin(); mu_it != return_mus.end(); mu_it++)
706  {
707  (*mu_it)->dump();
708  }
709  }
