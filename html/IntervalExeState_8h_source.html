382  static bool eqVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
383  {
384  if (lhs.size() != rhs.size()) return false;
385  for (const auto &item: lhs)
386  {
387  auto it = rhs.find(item.first);
388  if (it == rhs.end())
389  return false;
390  if (!item.second.equals(it->second))
391  {
392  return false;
393  }
394  }
395  return true;
396  }
397 
398 
399  static bool lessThanVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
400  {
401  if (lhs.empty()) return !rhs.empty();
402  for (const auto &item: lhs)
403  {
404  auto it = rhs.find(item.first);
405  if (it == rhs.end()) return false;
406  // judge from expr id
407  if (item.second.geq(it->second)) return false;
408  }
409  return true;
410  }
411 
412  // lhs >= rhs
413  static bool geqVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
414  {
415  if (rhs.empty()) return true;
416  for (const auto &item: rhs)
417  {
418  auto it = lhs.find(item.first);
419  if (it == lhs.end()) return false;
420  // judge from expr id
421  if (!it->second.geq(item.second)) return false;
422  }
423  return true;
424  }
