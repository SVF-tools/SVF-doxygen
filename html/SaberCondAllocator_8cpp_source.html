58 void SaberCondAllocator::allocate(const SVFModule *M)
59 {
60  DBOUT(DGENERAL, outs() << pasMsg("path condition allocation starts\n"));
61 
62  for (const auto &func: *M)
63  {
64  if (!SVFUtil::isExtCall(func))
65  {
66  // Allocate conditions for a program.

86 void SaberCondAllocator::allocateForBB(const SVFBasicBlock &bb)
87 {
88 
89  u32_t succ_number = bb.getNumSuccessors();
90 
91  // if successor number greater than 1, allocate new decision variable for successors
92  if (succ_number > 1)
93  {
94 
95  //allocate log2(num_succ) decision variables
96  double num = log(succ_number) / log(2);
97  u32_t bit_num = (u32_t) ceil(num);
98  u32_t succ_index = 0;
99  std::vector<Condition> condVec;
100  for (u32_t i = 0; i < bit_num; i++)
101  {
102  const SVFInstruction* svfInst = bb.getTerminator();
103  condVec.push_back(newCond(svfInst));
104  }
105 
106  // iterate each successor
107  for (const SVFBasicBlock* svf_succ_bb : bb.getSuccessors())
108  {
109  Condition path_cond = getTrueCond();
110 
112 
113  // for each successor decide its bit representation
114  // decide whether each bit of succ_index is 1 or 0, if (three successor) succ_index is 000 then use C1^C2^C3
115  // if 001 use C1^C2^negC3
116  for (u32_t j = 0; j < bit_num; j++)
117  {
118  //test each bit of this successor's index (binary representation)
119  u32_t tool = 0x01 << j;
120  if (tool & succ_index)
121  {
122  path_cond = condAnd(path_cond, (condNeg(condVec.at(j))));
123  }
124  else
125  {
126  path_cond = condAnd(path_cond, condVec.at(j));
127  }
128  }
129  setBranchCond(&bb, svf_succ_bb, path_cond);
130 
131  succ_index++;
132  }
133 
134  }
135 }

184 SaberCondAllocator::evaluateTestNullLikeExpr(const BranchStmt *branchStmt, const SVFBasicBlock* succ)
185 {
186

213 SaberCondAllocator::Condition SaberCondAllocator::evaluateProgExit(const BranchStmt *branchStmt, const SVFBasicBlock* succ)
214 {
215  const SVFBasicBlock* succ1 = branchStmt->getSuccessor(0)->getBB();
216  const SVFBasicBlock* succ2 = branchStmt->getSuccessor(1)->getBB();
217 
218  bool branch1 = isBBCallsProgExit(succ1);
219  bool branch2 = isBBCallsProgExit(succ2);
220 
222  if (branch1 && !branch2)
223  {
224  // succ is then branch
225  if (succ1 == succ)
226  return getFalseCond();
227  // succ is else branch
228  else
229  return getTrueCond();
230  }
232  else if (!branch1 && branch2)
233  {
234  // succ is else branch
235  if (succ2 == succ)
236  return getFalseCond();
237  // succ is then branch
238  else
239  return getTrueCond();
240  }
241  // two branches both call program exit
242  else if (branch1 && branch2)
243  {
244  return getFalseCond();
245  }
247  else
248  return Condition::nullExpr();
249

257 SaberCondAllocator::Condition SaberCondAllocator::evaluateLoopExitBranch(const SVFBasicBlock* bb, const SVFBasicBlock* dst)
258 {
259  const SVFFunction* svffun = bb->getParent();
260  assert(svffun == dst->getParent() && "two basic blocks should be in the same function");
261 
262  if (svffun->isLoopHeader(bb))
263  {
264  Set<const SVFBasicBlock* > filteredbbs;
265  std::vector<const SVFBasicBlock*> exitbbs;
266  svffun->getExitBlocksOfLoop(bb,exitbbs);
268  for(const SVFBasicBlock* eb : exitbbs)
269  {
270  if(!isBBCallsProgExit(eb))
271  filteredbbs.insert(eb);
272  }
273 
275  bool allPDT = true;
276  for (const auto &filteredbb: filteredbbs)
277  {
278  if (!postDominate(dst, filteredbb))
279  allPDT = false;
280  }
281 
282  if (allPDT)
283  return getTrueCond();
284  }
285  return Condition::nullExpr();

293 SaberCondAllocator::Condition SaberCondAllocator::evaluateBranchCond(const SVFBasicBlock* bb, const SVFBasicBlock* succ)
294 {
295  if(bb->getNumSuccessors() == 1)
296  {
297  return getTrueCond();
298  }
299 
300  const SVFInstruction* svfInst = bb->getTerminator();
301  if (ICFGNode *icfgNode = getICFG()->getICFGNode(svfInst))
302  {
303  for (const auto &svfStmt: icfgNode->getSVFStmts())
304  {
305  if (const BranchStmt *branchStmt = SVFUtil::dyn_cast<BranchStmt>(svfStmt))
306  {
307  if (branchStmt->getNumSuccessors() == 2)
308  {
309  const SVFBasicBlock* succ1 = branchStmt->getSuccessor(0)->getBB();
310  const SVFBasicBlock* succ2 = branchStmt->getSuccessor(1)->getBB();
311  bool is_succ = (succ1 == succ || succ2 == succ);
312  (void)is_succ; // Suppress warning of unused variable under release build
313  assert(is_succ && "not a successor??");
314  Condition evalLoopExit = evaluateLoopExitBranch(bb, succ);
315  if (!eq(evalLoopExit, Condition::nullExpr()))
316  return evalLoopExit;
317 
318  Condition evalProgExit = evaluateProgExit(branchStmt, succ);
319  if (!eq(evalProgExit, Condition::nullExpr()))
320  return evalProgExit;
321 
322  Condition evalTestNullLike = evaluateTestNullLikeExpr(branchStmt, succ);
323  if (!eq(evalTestNullLike, Condition::nullExpr()))
324  return evalTestNullLike;
325  break;
326  }
327  }
328  }
329  }
330 
331  return getBranchCond(bb, succ);
332 }

516 SaberCondAllocator::Condition SaberCondAllocator::ComputeIntraVFGGuard(const SVFBasicBlock* srcBB, const SVFBasicBlock* dstBB)
517 {
518 
519  assert(srcBB->getParent() == dstBB->getParent() && "two basic blocks are not in the same function??");
520 
521  if (postDominate(dstBB, srcBB))
522  return getTrueCond();
523 
524  CFWorkList worklist;
525  worklist.push(srcBB);
526  setCFCond(srcBB, getTrueCond());
527 
528  while (!worklist.empty())
529  {
530  const SVFBasicBlock* bb = worklist.pop();
531  Condition cond = getCFCond(bb);
532 
535  Condition loopExitCond = evaluateLoopExitBranch(bb, dstBB);
536  if (!eq(loopExitCond, Condition::nullExpr()))
537  return condAnd(cond, loopExitCond);
538 
539  for (const SVFBasicBlock* succ : bb->getSuccessors())
540  {
545  Condition brCond;
546  if (postDominate(succ, bb))
547  brCond = getTrueCond();
548  else
549  brCond = getEvalBrCond(bb, succ);
550 
551  DBOUT(DSaber, outs() << " bb (" << bb->getName() <<
552  ") --> " << "succ_bb (" << succ->getName() << ") condition: " << brCond << "\n");
553  Condition succPathCond = condAnd(cond, brCond);
554  if (setCFCond(succ, condOr(getCFCond(succ), succPathCond)))
555  worklist.push(succ);
556  }
557  }
558 
559  DBOUT(DSaber, outs() << " src_bb (" << srcBB->getName() <<
560  ") --> " << "dst_bb (" << dstBB->getName() << ") condition: " << getCFCond(dstBB)
561  << "\n");
562 
563  return getCFCond(dstBB);
564 }
