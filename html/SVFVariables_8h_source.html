129 virtual inline const SVFFunction* getFunction() const
130 {
131 // Return cached function if available
132 if(func) return func;
133
134 // If we have an associated LLVM value, check its parent function
135 if (value)
136 {
137 // For instructions, return the function containing the parent basic block
138 if (auto inst = SVFUtil::dyn_cast<SVFInstruction>(value))
139 {
140 return inst->getParent()->getParent();
141 }
142 }
143
144 // Return nullptr for globals/constants with no parent function
145 return nullptr;
146 }

149 inline SVFStmt::SVFStmtSetTy& getIncomingEdges(SVFStmt::PEDGEK kind)
150 {
151 return InEdgeKindToSetMap[kind];

159 inline bool hasIncomingEdges(SVFStmt::PEDGEK kind) const
160 {
161 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
162 if (it != InEdgeKindToSetMap.end())
163 return (!it->second.empty());
164 else
165 return false;
166 }
168 inline bool hasOutgoingEdges(SVFStmt::PEDGEK kind) const
169 {
170 SVFStmt::KindToSVFStmtMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
171 if (it != OutEdgeKindToSetMap.end())
172 return (!it->second.empty());
173 else
174 return false;
175 }

178 inline SVFStmt::SVFStmtSetTy::iterator getIncomingEdgesBegin(SVFStmt::PEDGEK kind) const
179 {
180 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
181 assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
182 return it->second.begin();
183 }
184
186 inline SVFStmt::SVFStmtSetTy::iterator getIncomingEdgesEnd(SVFStmt::PEDGEK kind) const
187 {
188 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
189 assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
190 return it->second.end();
191 }

194 inline SVFStmt::SVFStmtSetTy::iterator getOutgoingEdgesBegin(SVFStmt::PEDGEK kind) const
195 {
196 SVFStmt::KindToSVFStmtMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
197 assert(it!=OutEdgeKindToSetMap.end() && "The node does not have such kind of edge");

256 OutEdgeKindToSetMap[kind].insert(outEdge);
257 addOutgoingEdge(outEdge);

261 {

264 {
265 for(auto gep : it->second)
266 {
267 if(SVFUtil::cast<GepStmt>(gep)->isVariantFieldGep())
268 return true;
269 }
270 }
271 return false;
272 }
