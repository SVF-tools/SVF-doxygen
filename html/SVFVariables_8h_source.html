84 inline const SVFValue* getValue() const
85 {
86 assert(this->getNodeKind() != DummyValNode &&
87 this->getNodeKind() != DummyObjNode &&
88 "dummy node do not have value!");
89 assert(!SymbolTableInfo::isBlkObjOrConstantObj(this->getId()) &&
90 "blackhole and constant obj do not have value");
91 assert(value && "value is null (GepObjNode whose basenode is a DummyObj?)");

123 virtual inline const SVFFunction* getFunction() const
124 {
125 // Return cached function if available
126 if(func) return func;
127
128 // If we have an associated LLVM value, check its parent function
129 if (value)
130 {
131 // For instructions, return the function containing the parent basic block
132 if (auto inst = SVFUtil::dyn_cast<SVFInstruction>(value))
133 {
134 return inst->getParent()->getParent();
135 }
136 }
137
138 // Return nullptr for globals/constants with no parent function
139 return nullptr;
140 }

143 inline SVFStmt::SVFStmtSetTy& getIncomingEdges(SVFStmt::PEDGEK kind)
144 {
145 return InEdgeKindToSetMap[kind];
146 }
148 inline SVFStmt::SVFStmtSetTy& getOutgoingEdges(SVFStmt::PEDGEK kind)
149 {
150 return OutEdgeKindToSetMap[kind];

153 inline bool hasIncomingEdges(SVFStmt::PEDGEK kind) const
154 {
155 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
156 if (it != InEdgeKindToSetMap.end())
157 return (!it->second.empty());
158 else
159 return false;
160 }
162 inline bool hasOutgoingEdges(SVFStmt::PEDGEK kind) const
163 {
164 SVFStmt::KindToSVFStmtMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
165 if (it != OutEdgeKindToSetMap.end())
166 return (!it->second.empty());
167 else
168 return false;
169 }
170
172 inline SVFStmt::SVFStmtSetTy::iterator getIncomingEdgesBegin(SVFStmt::PEDGEK kind) const
173 {
174 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
175 assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
176 return it->second.begin();
177 }
178
180 inline SVFStmt::SVFStmtSetTy::iterator getIncomingEdgesEnd(SVFStmt::PEDGEK kind) const
181 {
182 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
183 assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
184 return it->second.end();
185 }
186
188 inline SVFStmt::SVFStmtSetTy::iterator getOutgoingEdgesBegin(SVFStmt::PEDGEK kind) const
189 {
190 SVFStmt::KindToSVFStmtMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
191 assert(it!=OutEdgeKindToSetMap.end() && "The node does not have such kind of edge");
192 return it->second.begin();
193 }
194
196 inline SVFStmt::SVFStmtSetTy::iterator getOutgoingEdgesEnd(SVFStmt::PEDGEK kind) const
197 {
198 SVFStmt::KindToSVFStmtMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
199 assert(it!=OutEdgeKindToSetMap.end() && "The node does not have such kind of edge");
200 return it->second.end();

222 inline void addInEdge(SVFStmt* inEdge)
223 {
224 GEdgeKind kind = inEdge->getEdgeKind();
225 InEdgeKindToSetMap[kind].insert(inEdge);
226 addIncomingEdge(inEdge);
227 }
228
229 inline void addOutEdge(SVFStmt* outEdge)
230 {
231 GEdgeKind kind = outEdge->getEdgeKind();
232 OutEdgeKindToSetMap[kind].insert(outEdge);
233 addOutgoingEdge(outEdge);
234 }
236 inline bool hasIncomingVariantGepEdge() const
237 {
238 SVFStmt::KindToSVFStmtMapTy::const_iterator it = InEdgeKindToSetMap.find(SVFStmt::Gep);
239 if (it != InEdgeKindToSetMap.end())
240 {
241 for(auto gep : it->second)
242 {
243 if(SVFUtil::cast<GepStmt>(gep)->isVariantFieldGep())
244 return true;
245 }
246 }
247 return false;
248 }
