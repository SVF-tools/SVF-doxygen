86 s64_t CDGBuilder::getBBSuccessorBranchID(const SVFBasicBlock *BB, const SVFBasicBlock *Succ)
87 {
88  ICFG *icfg = PAG::getPAG()->getICFG();
89  const ICFGNode *pred = icfg->getICFGNode(BB->getTerminator());
90  const ICFGEdge *edge = nullptr;
91  for (const auto &inst: Succ->getInstructionList())
92  {
93  if (const ICFGEdge *e = icfg->getICFGEdge(pred, icfg->getICFGNode(inst), ICFGEdge::ICFGEdgeK::IntraCF))
94  {
95  edge = e;
96  break;
97  }
98  }
99  if (const IntraCFGEdge *intraEdge = SVFUtil::dyn_cast<IntraCFGEdge>(edge))
100  {
101  if(intraEdge->getCondition())
102  return intraEdge->getSuccessorCondValue();
103  else
104  return 0;
105  }
106  else
107  {
108  assert(false && "not intra edge?");
109  abort();
110  }
111 }

123 void CDGBuilder::buildControlDependence(const SVFModule *svfgModule)
124 {
125  for (const auto &svfFun: *svfgModule)
126  {
127  if (SVFUtil::isExtCall(svfFun)) continue;
128  // extract basic block edges to be processed
129  Map<const SVFBasicBlock *, std::vector<const SVFBasicBlock *>> BBS;
130  extractBBS(svfFun, BBS);
131 
132  for (const auto &item: BBS)
133  {
134  const SVFBasicBlock *pred = item.first;
135  // for each bb pair
136  for (const SVFBasicBlock *succ: item.second)
137  {
138  const SVFBasicBlock *SVFLCA = const_cast<SVFFunction *>(svfFun)->
139  getLoopAndDomInfo()->findNearestCommonPDominator(pred, succ);
140  std::vector<const SVFBasicBlock *> tgtNodes;
141  // no common ancestor, may be exit()
142  if (SVFLCA == NULL)
143  tgtNodes.push_back(succ);
144  else
145  {
146  if (SVFLCA == pred) tgtNodes.push_back(SVFLCA);
147  // from succ to LCA
148  extractNodesBetweenPdomNodes(succ, SVFLCA, tgtNodes);
149  }
150 
151  s64_t pos = getBBSuccessorBranchID(pred, succ);
152  for (const SVFBasicBlock *bb: tgtNodes)
153  {
154  updateMap(pred, bb, pos);
155  }
156  }
157  }
158  }
159 }

196  if (!terminator) continue;

198  if (!controlNode) continue;
199  // controlNode control at pos

201  {
202  const ICFGNode *controllee = icfg->getICFGNode(inst);
203  _nodeControlMap[controlNode][controllee].insert(it2.second.begin(), it2.second.end());
204  _nodeDependentOnMap[controllee][controlNode].insert(it2.second.begin(), it2.second.end());
