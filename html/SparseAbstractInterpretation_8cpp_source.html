109AbstractState SemiSparseAbstractInterpretation::getFullCycleHeadState(
110 const ICFGCycleWTO* cycle)
111{
112 // Start from the dense snapshot (ObjVars + any ValVars that happen to
113 // be cached at cycle_head's trace entry).
114 AbstractState snap = AbstractInterpretation::getFullCycleHeadState(cycle);
115
116 const Set<const ValVar*>& valVars = preAnalysis->getCycleValVars(cycle);
117 if (valVars.empty())
118 return snap; // no cycle ValVars known: nothing to pull
119
120 // Drop stale ValVar entries and pull each cycle ValVar from its
121 // def-site. ValVars without a genuine stored value are skipped to
122 // avoid getAbsValue's top-fallback contaminating body def-sites on
123 // the subsequent widen/narrow scatter.
124 snap.clearValVars();
125 for (const ValVar* v : valVars)
126 {
127 const ICFGNode* defSite = v->getICFGNode();
128 if (!defSite || !hasAbsValue(v, defSite))
129 continue;
130 snap[v->getId()] = getAbsValue(v, defSite);
131 }
132 return snap;
133}

135bool SemiSparseAbstractInterpretation::widenCycleState(
136 const AbstractState& prev, const AbstractState& cur, const ICFGCycleWTO* cycle)
137{
138 // Base widens, writes trace[cycle_head], and returns fixpoint bool.
139 bool fixpoint = AbstractInterpretation::widenCycleState(prev, cur, cycle);
140
141 // Scatter the widened ValVars back to their def-sites so body nodes
142 // observe the widened values on the next iteration. Matches the
143 // pre-refactor semantics: scatter unconditionally, including at
144 // widening fixpoint (see the narrowing-starts-with-stale-body issue
145 // fixed by always writing widened state back).
146 const ICFGNode* cycle_head = cycle->head()->getICFGNode();
147 const AbstractState& next = abstractTrace[cycle_head];
148 for (const auto& [id, val] : next.getVarToVal())
149 updateAbsValue(svfir->getSVFVar(id), val, cycle_head);
150 return fixpoint;
151}

153bool SemiSparseAbstractInterpretation::narrowCycleState(
154 const AbstractState& prev, const AbstractState& cur, const ICFGCycleWTO* cycle)
155{
156 // Delegate to base. It returns true on the two non-scatter cases
157 // (narrowing disabled, or narrow fixpoint); we preserve the original
158 // "skip scatter at fixpoint" semantics by bailing early here.
159 bool fixpoint = AbstractInterpretation::narrowCycleState(prev, cur, cycle);
160 if (fixpoint)
161 return true;
162
163 // Non-fixpoint: base wrote the narrowed state to trace. Scatter the
164 // narrowed ValVars back to def-sites.
165 const ICFGNode* cycle_head = cycle->head()->getICFGNode();
166 const AbstractState& next = abstractTrace[cycle_head];
167 for (const auto& [id, val] : next.getVarToVal())
168 updateAbsValue(svfir->getSVFVar(id), val, cycle_head);
169 return false;
170}
