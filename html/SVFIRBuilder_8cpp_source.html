54SVFIR* SVFIRBuilder::build()
55{
56 double startTime = SVFStat::getClk(true);
57
58 DBOUT(DGENERAL, outs() << pasMsg("\t Building SVFIR ...\n"));
59
60 // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
61 if (SVFIR::pagReadFromTXT())
62 {
63 PAGBuilderFromFile fileBuilder(SVFIR::pagFileName());
64 return fileBuilder.build();
65 }
66
67 // If the SVFIR has been built before, then we return the unique SVFIR of the program
68 if(pag->getNodeNumAfterPAGBuild() > 1)
69 return pag;
70
71
72 initialiseFunObjVars();
73
75 ICFGBuilder icfgbuilder;
76 pag->icfg = icfgbuilder.build();
77
80 initialiseNodes();
83 visitGlobal(svfModule);
85
86
87
89 CallGraphBuilder callGraphBuilder;
90 std::vector<const FunObjVar*> funset;
91 for (const auto& item: svfModule->getFunctionSet())
92 {
93 const Function* llvmFun = SVFUtil::cast<Function>(llvmModuleSet()->getLLVMValue(item));
94
95 funset.push_back(llvmModuleSet()->getFunObjVar(llvmFun));
96 }
97 pag->callGraph = callGraphBuilder.buildSVFIRCallGraph(funset);
98
99 CHGraph* chg = new CHGraph();
100 CHGBuilder chgbuilder(chg);
101 chgbuilder.buildCHG();
102 pag->setCHG(chg);
103
105 for (Module& M : llvmModuleSet()->getLLVMModules())
106 {
107 for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
108 {
109 const Function& fun = *F;
110 const FunObjVar* svffun = llvmModuleSet()->getFunObjVar(&fun);
112 if(!fun.isDeclaration())
113 {
119 if (fun.doesNotReturn() == false &&
120 fun.getReturnType()->isVoidTy() == false)
121 {
122 pag->addFunRet(svffun,
123 pag->getGNode(pag->getReturnNode(svffun)));
124 }
125
128 for (Function::const_arg_iterator I = fun.arg_begin(), E = fun.arg_end();
129 I != E; ++I)
130 {
131 setCurrentLocation(&*I,&fun.getEntryBlock());
132 NodeID argValNodeId = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(&*I));
133 // if this is the function does not have caller (e.g. main)
134 // or a dead function, shall we create a black hole address edge for it?
135 // it is (1) too conservative, and (2) make FormalParmVFGNode defined at blackhole address PAGEdge.
136 // if(SVFUtil::ArgInNoCallerFunction(&*I)) {
137 // if(I->getType()->isPointerTy())
138 // addBlackHoleAddrEdge(argValNodeId);
139 //}
140 pag->addFunArgs(svffun,pag->getGNode(argValNodeId));
141 }
142 }
143 for (Function::const_iterator bit = fun.begin(), ebit = fun.end();
144 bit != ebit; ++bit)
145 {
146 const BasicBlock& bb = *bit;
147 for (BasicBlock::const_iterator it = bb.begin(), eit = bb.end();
148 it != eit; ++it)
149 {
150 const Instruction& inst = *it;
151 setCurrentLocation(&inst,&bb);
152 visit(const_cast<Instruction&>(inst));
153 }
154 }
155 }
156 }
157
158 sanityCheck();
159
160 pag->initialiseCandidatePointers();
161
162 pag->setNodeNumAfterPAGBuild(pag->getTotalNodeNum());
163
164 // dump SVFIR
165 if (Options::PAGDotGraph())
166 pag->dump("svfir_initial");
167
168 // print to command line of the SVFIR graph
169 if (Options::PAGPrint())
170 pag->print();
171
172 // dump ICFG
173 if (Options::DumpICFG())
174 pag->getICFG()->dump("icfg_initial");
175
176 if (Options::LoopAnalysis())
177 {
178 LLVMLoopAnalysis loopAnalysis;
179 loopAnalysis.build(pag->getICFG());
180 }
181
182 // dump SVFIR as JSON
183 if (!Options::DumpJson().empty())
184 {
185 assert(false && "please implement SVFIRWriter::writeJsonToPath");
186 }
187
188 double endTime = SVFStat::getClk(true);
189 SVFStat::timeOfBuildingSVFIR = (endTime - startTime) / TIMEINTERVAL;
190
191 return pag;
192}

194void SVFIRBuilder::initialiseFunObjVars()
195{
196 std::vector<FunObjVar*> funset;
197 // Iterate over all object symbols in the symbol table
198 for (const auto* fun: svfModule->getFunctionSet())
199 {
200 u32_t id = llvmModuleSet()->objSyms()[fun];
201 // Debug output for adding object node
202 DBOUT(DPAGBuild, outs() << "add obj node " << id << "\n");
203 // Skip blackhole and constant symbols
204 if(id == pag->blackholeSymID() || id == pag->constantSymID())
205 continue;
206
207 // Get the LLVM value corresponding to the symbol
208 const Value* llvmValue = llvmModuleSet()->getLLVMValue(fun);
209
210 // Check if the value is a function and add a function object node
211 if (const Function* func = SVFUtil::dyn_cast<Function>(llvmValue))
212 {
213 const SVFFunction *svffunc = llvmModuleSet()->getSVFFunction(func);
214
215 NodeID id = llvmModuleSet()->getObjectNode(svffunc);
216 pag->addFunObjNode(id, pag->getObjTypeInfo(id), fun->getType(), nullptr);
217 llvmModuleSet()->LLVMFun2FunObjVar[func] = cast<FunObjVar>(pag->getGNode(id));
218
219 FunObjVar *funObjVar = SVFUtil::cast<FunObjVar>(pag->getGNode(id));
220 funset.push_back(funObjVar);
221 funObjVar->initFunObjVar(svffunc->isDecl, svffunc->intrinsic, svffunc->addrTaken,
222 svffunc->isUncalled, svffunc->isNotRet, svffunc->varArg, svffunc->funcType,
223 svffunc->loopAndDom, nullptr, svffunc->bbGraph,
224 svffunc->allArgs, svffunc->exitBlock);
226 for (auto& bb: *funObjVar->bbGraph)
227 {
228 bb.second->setFun(funObjVar);
229 }
230 llvmModuleSet()->addToSVFVar2LLVMValueMap(llvmValue, pag->getGNode(id));
231 }
232 }
233
235 for (auto& fun: funset)
236 {
237 const SVFFunction *svffunc = SVFUtil::cast<SVFFunction>(
238 llvmModuleSet()->getSVFFunction(SVFUtil::cast<Function>(llvmModuleSet()->getLLVMValue(fun))));
239 const Function *realfun = SVFUtil::cast<Function>(llvmModuleSet()->getLLVMValue(svffunc->realDefFun));
240 fun->setRelDefFun(llvmModuleSet()->getFunObjVar(realfun));
241 }
242
243 // Store annotations of functions in extapi.bc
244 for (const auto& pair : llvmModuleSet()->ExtFun2Annotations)
245 {
246 ExtAPI::getExtAPI()->setExtFuncAnnotations(llvmModuleSet()->getFunObjVar(pair.first), pair.second);
247 }

250void SVFIRBuilder::initialiseBaseObjVars()
251{
252 // Iterate over all object symbols in the symbol table
253 for (LLVMModuleSet::ValueToIDMapTy::iterator iter =
254 llvmModuleSet()->objSyms().begin(); iter != llvmModuleSet()->objSyms().end();
255 ++iter)
256 {
257 // Debug output for adding object node
258 DBOUT(DPAGBuild, outs() << "add obj node " << iter->second << "\n");
259
260 // Skip blackhole and constant symbols
261 if(iter->second == pag->blackholeSymID() || iter->second == pag->constantSymID())
262 continue;
263
264 // Get the LLVM value corresponding to the symbol
265 const Value* llvmValue = llvmModuleSet()->getLLVMValue(iter->first);
266
267 const ICFGNode* icfgNode = nullptr;
268 if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(llvmValue))
269 {
270 if(llvmModuleSet()->hasICFGNode(inst))
271 icfgNode = llvmModuleSet()->getICFGNode(inst);
272 }
273
274 // Check if the value is a function and add a function object node
275 if (SVFUtil::dyn_cast<Function>(llvmValue))
276 {
277 // already one
278 }
279 // Check if the value is a heap object and add a heap object node
280 else if (LLVMUtil::isHeapObj(llvmValue))
281 {
282 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
283 pag->addHeapObjNode(iter->second, pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
284 }
285 // Check if the value is an alloca instruction and add a stack object node
286 else if (LLVMUtil::isStackObj(llvmValue))
287 {
288 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
289 pag->addStackObjNode(iter->second, pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
290 }
291 else if (auto fpValue = SVFUtil::dyn_cast<ConstantFP>(llvmValue))
292 {
293 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
294 pag->addConstantFPObjNode(iter->second, pag->getObjTypeInfo(id), LLVMUtil::getDoubleValue(fpValue), iter->first->getType(), icfgNode);
295 }
296 else if (auto intValue = SVFUtil::dyn_cast<ConstantInt>(llvmValue))
297 {
298 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
299 pag->addConstantIntObjNode(iter->second, pag->getObjTypeInfo(id), LLVMUtil::getIntegerValue(intValue), iter->first->getType(), icfgNode);
300 }
301 else if (SVFUtil::isa<ConstantPointerNull>(llvmValue))
302 {
303 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
304 pag->addConstantNullPtrObjNode(iter->second, pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
305 }
306 else if (SVFUtil::isa<GlobalValue>(llvmValue))
307 {
308 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
309 pag->addGlobalObjNode(iter->second,
310 pag->getObjTypeInfo(id),
311 iter->first->getType(), icfgNode);
312 }
313 else if (SVFUtil::isa<ConstantData, MetadataAsValue, BlockAddress>(llvmValue))
314 {
315 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
316 pag->addConstantDataObjNode(iter->second, pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
317 }
318 else if (SVFUtil::isa<ConstantAggregate>(llvmValue))
319 {
320 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
321 pag->addConstantAggObjNode(iter->second, pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
322 }
323 // Add a generic object node for other types of values
324 else
325 {
326 NodeID id = llvmModuleSet()->getObjectNode(iter->first);
327 pag->addObjNode(iter->second,
328 pag->getObjTypeInfo(id), iter->first->getType(), icfgNode);
329 }
330 llvmModuleSet()->addToSVFVar2LLVMValueMap(llvmValue, pag->getGNode(iter->second));
331 }

335void SVFIRBuilder::initialiseValVars()
336{
337 // Iterate over all value symbols in the symbol table
338 for (LLVMModuleSet::ValueToIDMapTy::iterator iter =
339 llvmModuleSet()->valSyms().begin(); iter != llvmModuleSet()->valSyms().end();
340 ++iter)
341 {
342 // Debug output for adding value node
343 DBOUT(DPAGBuild, outs() << "add val node " << iter->second << "\n");
344
345 // Skip blackhole and null pointer symbols
346 if(iter->second == pag->blkPtrSymID() || iter->second == pag->nullPtrSymID())
347 continue;
348
349 const ICFGNode* icfgNode = nullptr;
350 auto llvmValue = llvmModuleSet()->getLLVMValue(iter->first);
351 if (const Instruction* inst =
352 SVFUtil::dyn_cast<Instruction>(llvmValue))
353 {
354 if (llvmModuleSet()->hasICFGNode(inst))
355 {
356 icfgNode = llvmModuleSet()->getICFGNode(inst);
357 }
358 }
359
360 // Check if the value is a function and get its call graph node
361 if (const Function* func = SVFUtil::dyn_cast<Function>(llvmValue))
362 {
363 // add value node representing the function
364 pag->addFunValNode(iter->second, icfgNode, llvmModuleSet()->getFunObjVar(func), iter->first->getType());
365 }
366 else if (auto argval = SVFUtil::dyn_cast<Argument>(llvmValue))
367 {
368 pag->addArgValNode(
369 iter->second, argval->getArgNo(), icfgNode,
370 llvmModuleSet()->getFunObjVar(argval->getParent()),iter->first->getType());
371 if (!argval->hasName())
372 pag->getGNode(iter->second)->setName("arg_" + std::to_string(argval->getArgNo()));
373 }
374 else if (auto fpValue = SVFUtil::dyn_cast<ConstantFP>(llvmValue))
375 {
376 pag->addConstantFPValNode(iter->second, LLVMUtil::getDoubleValue(fpValue), icfgNode, iter->first->getType());
377 }
378 else if (auto intValue = SVFUtil::dyn_cast<ConstantInt>(llvmValue))
379 {
380 pag->addConstantIntValNode(iter->second, LLVMUtil::getIntegerValue(intValue), icfgNode, iter->first->getType());
381 }
382 else if (SVFUtil::isa<ConstantPointerNull>(llvmValue))
383 {
384 pag->addConstantNullPtrValNode(iter->second, icfgNode, iter->first->getType());
385 }
386 else if (SVFUtil::isa<GlobalValue>(llvmValue))
387 {
388 pag->addGlobalValNode(iter->second, icfgNode,
389 iter->first->getType());
390 }
391 else if (SVFUtil::isa<ConstantData, MetadataAsValue, BlockAddress>(llvmValue))
392 {
393 pag->addConstantDataValNode(iter->second, icfgNode, iter->first->getType());
394 }
395 else if (SVFUtil::isa<ConstantAggregate>(llvmValue))
396 {
397 pag->addConstantAggValNode(iter->second, icfgNode, iter->first->getType());
398 }
399 else
400 {
401 // Add value node to PAG
402 pag->addValNode(iter->second, iter->first->getType(), icfgNode);
403 }
404 llvmModuleSet()->addToSVFVar2LLVMValueMap(llvmValue,
405 pag->getGNode(iter->second));
406 }
407}

413void SVFIRBuilder::initialiseNodes()
414{
415 DBOUT(DPAGBuild, outs() << "Initialise SVFIR Nodes ...\n");
416
417
418 pag->addBlackholeObjNode();
419 pag->addConstantObjNode();
420 pag->addBlackholePtrNode();
421 addNullPtrNode();
422
423 initialiseBaseObjVars();
424 initialiseValVars();
425
426 for (LLVMModuleSet::FunToIDMapTy::iterator iter =
427 llvmModuleSet()->retSyms().begin(); iter != llvmModuleSet()->retSyms().end();
428 ++iter)
429 {
430 const Value* llvmValue = llvmModuleSet()->getLLVMValue(iter->first);
431 const ICFGNode* icfgNode = nullptr;
432 if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(llvmValue))
433 {
434 if(llvmModuleSet()->hasICFGNode(inst))
435 icfgNode = llvmModuleSet()->getICFGNode(inst);
436 }
437 DBOUT(DPAGBuild, outs() << "add ret node " << iter->second << "\n");
438 pag->addRetNode(iter->second,
439 llvmModuleSet()->getFunObjVar(SVFUtil::cast<Function>(llvmValue)), iter->first->getType(), icfgNode);
440 llvmModuleSet()->addToSVFVar2LLVMValueMap(llvmValue, pag->getGNode(iter->second));
441 const FunObjVar* funObjVar = llvmModuleSet()->getFunObjVar(SVFUtil::cast<Function>(llvmValue));
442 pag->returnFunObjSymMap[funObjVar] = iter->second;
443 }
444
445 for (LLVMModuleSet::FunToIDMapTy::iterator iter =
446 llvmModuleSet()->varargSyms().begin();
447 iter != llvmModuleSet()->varargSyms().end(); ++iter)
448 {
449 const Value* llvmValue = llvmModuleSet()->getLLVMValue(iter->first);
450
451 const ICFGNode *icfgNode = nullptr;
452 if (const Instruction *inst = SVFUtil::dyn_cast<Instruction>(llvmValue))
453 {
454 if (llvmModuleSet()->hasICFGNode(inst))
455 icfgNode = llvmModuleSet()->getICFGNode(inst);
456 }
457 DBOUT(DPAGBuild, outs() << "add vararg node " << iter->second << "\n");
458 pag->addVarargNode(iter->second,
459 llvmModuleSet()->getFunObjVar(SVFUtil::cast<Function>(llvmValue)), iter->first->getType(), icfgNode);
460 llvmModuleSet()->addToSVFVar2LLVMValueMap(llvmValue, pag->getGNode(iter->second));
461 const FunObjVar* funObjVar = llvmModuleSet()->getFunObjVar(SVFUtil::cast<Function>(llvmValue));
462 pag->varargFunObjSymMap[funObjVar] = iter->second;
463 }
464
466 for (LLVMModuleSet::ValueToIDMapTy::iterator iter =
467 llvmModuleSet()->objSyms().begin(); iter != llvmModuleSet()->objSyms().end(); ++iter)
468 {
469 DBOUT(DPAGBuild, outs() << "add address edges for constant node " << iter->second << "\n");
470 const SVFLLVMValue* val = iter->first;
471 if (isConstantObjSym(val))
472 {
473 NodeID ptr = llvmModuleSet()->getValueNode(val);
474 if(ptr!= pag->getBlkPtr() && ptr!= pag->getNullPtr())
475 {
476 setCurrentLocation(val, nullptr);
477 addAddrEdge(iter->second, ptr);
478 }
479 }
480 }
481
482 assert(pag->getTotalNodeNum() >= pag->getTotalSymNum()
483 && "not all node have been initialized!!!");
484
486 for (auto& fun: svfModule->getFunctionSet())
487 {
488 const Function* llvmFun = SVFUtil::cast<Function>(llvmModuleSet()->getLLVMValue(fun));
489 for (const Argument& arg : llvmFun->args())
490 {
491 const_cast<SVFFunction *>(fun)->addArgument(
492 SVFUtil::cast<ArgValVar>(
493 pag->getGNode(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFArgument(&arg)))));
494 const_cast<FunObjVar*>(llvmModuleSet()->getFunObjVar(llvmFun))->addArgument(SVFUtil::cast<ArgValVar>(
495 pag->getGNode(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFArgument(&arg)))));
496 }
497 }

521bool SVFIRBuilder::computeGepOffset(const User *V, AccessPath& ap)
522{
523 assert(V);
524
525 const llvm::GEPOperator *gepOp = SVFUtil::dyn_cast<const llvm::GEPOperator>(V);
526 DataLayout * dataLayout = getDataLayout(llvmModuleSet()->getMainLLVMModule());
527 llvm::APInt byteOffset(dataLayout->getIndexSizeInBits(gepOp->getPointerAddressSpace()),0,true);
528 if(gepOp && dataLayout && gepOp->accumulateConstantOffset(*dataLayout,byteOffset))
529 {
530 //s32_t bo = byteOffset.getSExtValue();
531 }
532
533 bool isConst = true;
534
535 bool prevPtrOperand = false;
536 for (bridge_gep_iterator gi = bridge_gep_begin(*V), ge = bridge_gep_end(*V);
537 gi != ge; ++gi)
538 {
539 const Type* gepTy = *gi;
540 const SVFType* svfGepTy = llvmModuleSet()->getSVFType(gepTy);
541
542 assert((prevPtrOperand && svfGepTy->isPointerTy()) == false &&
543 "Expect no more than one gep operand to be of a pointer type");
544 if(!prevPtrOperand && svfGepTy->isPointerTy()) prevPtrOperand = true;
545 const Value* offsetVal = gi.getOperand();
546 const SVFLLVMValue* offsetSvfVal = llvmModuleSet()->getSVFValue(offsetVal);
547 assert(gepTy != offsetVal->getType() && "iteration and operand have the same type?");
548 ap.addOffsetVarAndGepTypePair(getPAG()->getGNode(llvmModuleSet()->getValueNode(offsetSvfVal)), svfGepTy);
549
550 //The int value of the current index operand
551 const ConstantInt* op = SVFUtil::dyn_cast<ConstantInt>(offsetVal);
552
553 // if Options::ModelConsts() is disabled. We will treat whole array as one,
554 // but we can distinguish different field of an array of struct, e.g. s[1].f1 is different from s[0].f2
555 if(const ArrayType* arrTy = SVFUtil::dyn_cast<ArrayType>(gepTy))
556 {
557 if(!op || (arrTy->getArrayNumElements() <= (u32_t)LLVMUtil::getIntegerValue(op).first))
558 continue;
559 APOffset idx = (u32_t)LLVMUtil::getIntegerValue(op).first;
560 u32_t offset = pag->getFlattenedElemIdx(llvmModuleSet()->getSVFType(arrTy), idx);
561 ap.setFldIdx(ap.getConstantStructFldIdx() + offset);
562 }
563 else if (const StructType *ST = SVFUtil::dyn_cast<StructType>(gepTy))
564 {
565 assert(op && "non-const offset accessing a struct");
566 //The actual index
567 APOffset idx = (u32_t)LLVMUtil::getIntegerValue(op).first;
568 u32_t offset = pag->getFlattenedElemIdx(llvmModuleSet()->getSVFType(ST), idx);
569 ap.setFldIdx(ap.getConstantStructFldIdx() + offset);
570 }
571 else if (gepTy->isSingleValueType())
572 {
573 // If it's a non-constant offset access
574 // If its point-to target is struct or array, it's likely an array accessing (%result = gep %struct.A* %a, i32 %non-const-index)
575 // If its point-to target is single value (pointer arithmetic), then it's a variant gep (%result = gep i8* %p, i32 %non-const-index)
576 if(!op && gepTy->isPointerTy() && gepOp->getSourceElementType()->isSingleValueType())
577 {
578 isConst = false;
579 }
580
581 // The actual index
582 //s32_t idx = op->getSExtValue();
583
584 // For pointer arithmetic we ignore the byte offset
585 // consider using inferFieldIdxFromByteOffset(geopOp,dataLayout,ap,idx)?
586 // ap.setFldIdx(ap.getConstantFieldIdx() + inferFieldIdxFromByteOffset(geopOp,idx));
587 }
588 }
589 return isConst;
590}

595void SVFIRBuilder::processCE(const Value* val)
596{
597 if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
598 {
599 if (const ConstantExpr* gepce = isGepConstantExpr(ref))
600 {
601 DBOUT(DPAGBuild, outs() << "handle gep constant expression " << llvmModuleSet()->getSVFValue(ref)->toString() << "\n");
602 const Constant* opnd = gepce->getOperand(0);
603 // handle recursive constant express case (gep (bitcast (gep X 1)) 1)
604 processCE(opnd);
605 auto &GEPOp = llvm::cast<llvm::GEPOperator>(*gepce);
606 Type *pType = GEPOp.getSourceElementType();
607 AccessPath ap(0, llvmModuleSet()->getSVFType(pType));
608 bool constGep = computeGepOffset(gepce, ap);
609 // must invoke pag methods here, otherwise it will be a dead recursion cycle
610 const SVFLLVMValue* cval = getCurrentValue();
611 const SVFBasicBlock* cbb = getCurrentBB();
612 setCurrentLocation(gepce, nullptr);
613 /*
614 * The gep edge created are like constexpr (same edge may appear at multiple callsites)
615 * so bb/inst of this edge may be rewritten several times, we treat it as global here.
616 */
617 addGepEdge(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(opnd)), llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(gepce)), ap, constGep);
618 setCurrentLocation(cval, cbb);
619 }
620 else if (const ConstantExpr* castce = isCastConstantExpr(ref))
621 {
622 DBOUT(DPAGBuild, outs() << "handle cast constant expression " << llvmModuleSet()->getSVFValue(ref)->toString() << "\n");
623 const Constant* opnd = castce->getOperand(0);
624 processCE(opnd);
625 const SVFLLVMValue* cval = getCurrentValue();
626 const SVFBasicBlock* cbb = getCurrentBB();
627 setCurrentLocation(castce, nullptr);
628 addCopyEdge(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(opnd)), llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(castce)), CopyStmt::BITCAST);
629 setCurrentLocation(cval, cbb);
630 }
631 else if (const ConstantExpr* selectce = isSelectConstantExpr(ref))
632 {
633 DBOUT(DPAGBuild, outs() << "handle select constant expression " << llvmModuleSet()->getSVFValue(ref)->toString() << "\n");
634 const Constant* src1 = selectce->getOperand(1);
635 const Constant* src2 = selectce->getOperand(2);
636 processCE(src1);
637 processCE(src2);
638 const SVFLLVMValue* cval = getCurrentValue();
639 const SVFBasicBlock* cbb = getCurrentBB();
640 setCurrentLocation(selectce, nullptr);
641 NodeID cond = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(selectce->getOperand(0)));
642 NodeID nsrc1 = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(src1));
643 NodeID nsrc2 = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(src2));
644 NodeID nres = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(selectce));
645 addSelectStmt(nres,nsrc1, nsrc2, cond);
646 setCurrentLocation(cval, cbb);
647 }
648 // if we meet a int2ptr, then it points-to black hole
649 else if (const ConstantExpr* int2Ptrce = isInt2PtrConstantExpr(ref))
650 {
651 const Constant* opnd = int2Ptrce->getOperand(0);
652 processCE(opnd);
653 const SVFBasicBlock* cbb = getCurrentBB();
654 const SVFLLVMValue* cval = getCurrentValue();
655 setCurrentLocation(int2Ptrce, nullptr);
656 addCopyEdge(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(opnd)), llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(int2Ptrce)), CopyStmt::INTTOPTR);
657 setCurrentLocation(cval, cbb);
658 }
659 else if (const ConstantExpr* ptr2Intce = isPtr2IntConstantExpr(ref))
660 {
661 const Constant* opnd = ptr2Intce->getOperand(0);
662 processCE(opnd);
663 const SVFBasicBlock* cbb = getCurrentBB();
664 const SVFLLVMValue* cval = getCurrentValue();
665 setCurrentLocation(ptr2Intce, nullptr);
666 addCopyEdge(llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(opnd)), llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(ptr2Intce)), CopyStmt::PTRTOINT);
667 setCurrentLocation(cval, cbb);
668 }
669 else if(isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
670 {
671 // we don't handle trunc and cmp instruction for now
672 const SVFLLVMValue* cval = getCurrentValue();
673 const SVFBasicBlock* cbb = getCurrentBB();
674 setCurrentLocation(ref, nullptr);
675 NodeID dst = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(ref));
676 addBlackHoleAddrEdge(dst);
677 setCurrentLocation(cval, cbb);
678 }
679 else if (isBinaryConstantExpr(ref))
680 {
681 // we don't handle binary constant expression like add(x,y) now
682 const SVFLLVMValue* cval = getCurrentValue();
683 const SVFBasicBlock* cbb = getCurrentBB();
684 setCurrentLocation(ref, nullptr);
685 NodeID dst = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(ref));
686 addBlackHoleAddrEdge(dst);
687 setCurrentLocation(cval, cbb);
688 }
689 else if (isUnaryConstantExpr(ref))
690 {
691 // we don't handle unary constant expression like fneg(x) now
692 const SVFLLVMValue* cval = getCurrentValue();
693 const SVFBasicBlock* cbb = getCurrentBB();
694 setCurrentLocation(ref, nullptr);
695 NodeID dst = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(ref));
696 addBlackHoleAddrEdge(dst);
697 setCurrentLocation(cval, cbb);
698 }
699 else if (SVFUtil::isa<ConstantAggregate>(ref))
700 {
701 // we don't handle constant aggregate like constant vectors
702 }
703 else if (SVFUtil::isa<BlockAddress>(ref))
704 {
705 // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
706 // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
707 const SVFLLVMValue* cval = getCurrentValue();
708 const SVFBasicBlock* cbb = getCurrentBB();
709 setCurrentLocation(ref, nullptr);
710 NodeID dst = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(ref));
711 addAddrEdge(pag->getConstantNode(), dst);
712 setCurrentLocation(cval, cbb);
713 }
714 else
715 {
716 if(SVFUtil::isa<ConstantExpr>(val))
717 assert(false && "we don't handle all other constant expression for now!");
718 }
719 }

742/*For global variable initialization
743 * Give a simple global variable
744 * int x = 10; // store 10 x (constant, non pointer) |
745 * int *y = &x; // store x y (pointer type)
746 * Given a struct
747 * struct Z { int s; int *t;};
748 * Global initialization:
749 * struct Z z = {10,&x}; // store x z.t (struct type)
750 * struct Z *m = &z; // store z m (pointer type)
751 * struct Z n = {10,&z.s}; // store z.s n , &z.s constant expression (constant expression)
752 */
753void SVFIRBuilder::InitialGlobal(const GlobalVariable *gvar, Constant *C,
754 u32_t offset)
755{
756 DBOUT(DPAGBuild, outs() << "global " << llvmModuleSet()->getSVFValue(gvar)->toString() << " constant initializer: " << llvmModuleSet()->getSVFValue(C)->toString() << "\n");
757 if (C->getType()->isSingleValueType())
758 {
759 NodeID src = getValueNode(C);
760 // get the field value if it is available, otherwise we create a dummy field node.
761 setCurrentLocation(gvar, nullptr);
762 NodeID field = getGlobalVarField(gvar, offset, llvmModuleSet()->getSVFType(C->getType()));
763
764 if (SVFUtil::isa<GlobalVariable, Function>(C))
765 {
766 setCurrentLocation(C, nullptr);
767 addStoreEdge(src, field);
768 }
769 else if (SVFUtil::isa<ConstantExpr>(C))
770 {
771 // add gep edge of C1 itself is a constant expression
772 processCE(C);
773 setCurrentLocation(C, nullptr);
774 addStoreEdge(src, field);
775 }
776 else if (SVFUtil::isa<BlockAddress>(C))
777 {
778 // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
779 // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
780 processCE(C);
781 setCurrentLocation(C, nullptr);
782 addAddrEdge(pag->getConstantNode(), src);
783 }
784 else
785 {
786 setCurrentLocation(C, nullptr);
787 addStoreEdge(src, field);
789 if (C->getType()->isPtrOrPtrVectorTy() && src != pag->getNullPtr())
790 addCopyEdge(pag->getNullPtr(), src, CopyStmt::COPYVAL);
791 }
792 }
793 else if (SVFUtil::isa<ConstantArray, ConstantStruct>(C))
794 {
795 if(cppUtil::isValVtbl(gvar) && !Options::VtableInSVFIR())
796 return;
797 for (u32_t i = 0, e = C->getNumOperands(); i != e; i++)
798 {
799 u32_t off = pag->getFlattenedElemIdx(llvmModuleSet()->getSVFType(C->getType()), i);
800 InitialGlobal(gvar, SVFUtil::cast<Constant>(C->getOperand(i)), offset + off);
801 }
802 }
803 else if(ConstantData* data = SVFUtil::dyn_cast<ConstantData>(C))
804 {
805 if(Options::ModelConsts())
806 {
807 if(ConstantDataSequential* seq = SVFUtil::dyn_cast<ConstantDataSequential>(data))
808 {
809 for(u32_t i = 0; i < seq->getNumElements(); i++)
810 {
811 u32_t off = pag->getFlattenedElemIdx(llvmModuleSet()->getSVFType(C->getType()), i);
812 Constant* ct = seq->getElementAsConstant(i);
813 InitialGlobal(gvar, ct, offset + off);
814 }
815 }
816 else
817 {
818 assert((SVFUtil::isa<ConstantAggregateZero, UndefValue>(data)) && "Single value type data should have been handled!");
819 }
820 }
821 }
822 else
823 {
824 //TODO:assert(SVFUtil::isa<ConstantVector>(C),"what else do we have");
825 }

831void SVFIRBuilder::visitGlobal(SVFModule* svfModule)
832{
833
835 for (Module &M : llvmModuleSet()->getLLVMModules())
836 {
837 for (Module::global_iterator I = M.global_begin(), E = M.global_end(); I != E; ++I)
838 {
839 GlobalVariable *gvar = &*I;
840 NodeID idx = getValueNode(gvar);
841 NodeID obj = getObjectNode(gvar);
842
843 setCurrentLocation(gvar, nullptr);
844 addAddrEdge(obj, idx);
845
846 if (gvar->hasInitializer())
847 {
848 Constant *C = gvar->getInitializer();

860 NodeID obj = getObjectNode(fun);

863 setCurrentLocation(fun, nullptr);
864 addAddrEdge(obj, idx);
865 }
866

869 {

871 NodeID dst = llvmModuleSet()->getValueNode(llvmModuleSet()->getSVFValue(alias));

873 processCE(alias->getAliasee());
874 setCurrentLocation(alias, nullptr);
875 addCopyEdge(src, dst, CopyStmt::COPYVAL);
876 }
877 }

1073 */
1074void SVFIRBuilder::visitCallSite(CallBase* cs)
1075{
1076
1077 // skip llvm intrinsics
1078 if(isIntrinsicInst(cs))
1079 return;
1080
1081 DBOUT(DPAGBuild,
1082 outs() << "process callsite " << svfcall->valueOnlyToString() << "\n");
1083
1084
1085 CallICFGNode* callBlockNode = llvmModuleSet()->getCallICFGNode(cs);
1086 RetICFGNode* retBlockNode = llvmModuleSet()->getRetICFGNode(cs);
1087
1088 pag->addCallSite(callBlockNode);
1089
1091 for (u32_t i = 0; i < cs->arg_size(); i++)
1092 pag->addCallSiteArgs(
1093 callBlockNode,
1094 SVFUtil::cast<ValVar>(pag->getGNode(getValueNode(cs->getArgOperand(i)))));
1095
1096 if(!cs->getType()->isVoidTy())
1097 pag->addCallSiteRets(retBlockNode,pag->getGNode(getValueNode(cs)));
1098
1099 if (callBlockNode->isVirtualCall())
1100 {
1101 const Value* value = cppUtil::getVCallVtblPtr(cs);
1102 callBlockNode->setVtablePtr(pag->getGNode(getValueNode(value)));
1103 }
1104 if (const Function *callee = LLVMUtil::getCallee(cs))
1105 {
1106 const SVFFunction* svfcallee = llvmModuleSet()->getSVFFunction(callee);
1107 if (LLVMUtil::isExtCall(svfcallee))
1108 {
1109 handleExtCall(cs, svfcallee);
1110 }
1111 else
1112 {
1113 handleDirectCall(cs, callee);
1114 }
1115 }
1116 else
1117 {
1118 //If the callee was not identified as a function (null F), this is indirect.
1119 handleIndCall(cs);
1120 }

1179void SVFIRBuilder::visitBranchInst(BranchInst &inst)
1180{
1181 NodeID brinst = getValueNode(&inst);
1182 NodeID cond;
1183 if (inst.isConditional())
1184 cond = getValueNode(inst.getCondition());
1185 else
1186 cond = pag->getNullPtr();
1187
1188 assert(inst.getNumSuccessors() <= 2 && "if/else has more than two branches?");
1189
1190 BranchStmt::SuccAndCondPairVec successors;
1191 std::vector<const Instruction*> nextInsts;
1192 LLVMUtil::getNextInsts(&inst, nextInsts);
1193 u32_t branchID = 0;
1194 for (const Instruction* succInst : nextInsts)
1195 {

1198 successors.push_back(std::make_pair(icfgNode, 1-branchID));
1199 branchID++;
1200 }
1201 addBranchStmt(brinst, cond, successors);
1203 if (inst.isConditional())
1204 {
1205 for (auto& edge : llvmModuleSet()->getICFGNode(&inst)->getOutEdges())
1206 {
1207 if (IntraCFGEdge* intraEdge = SVFUtil::dyn_cast<IntraCFGEdge>(edge))
1208 {
1209 intraEdge->setConditionVar(pag->getGNode(cond));
1210 }
1211 }
1212 }

1324void SVFIRBuilder::handleDirectCall(CallBase* cs, const Function *F)
1325{
1326
1327 assert(F);
1328 CallICFGNode* callICFGNode = llvmModuleSet()->getCallICFGNode(cs);
1329 const FunObjVar* svffun = llvmModuleSet()->getFunObjVar(F);
1330 DBOUT(DPAGBuild,
1331 outs() << "handle direct call " << LLVMUtil::dumpValue(cs) << " callee " << F->getName().str() << "\n");
1332
1333 //Only handle the ret.val. if it's used as a ptr.
1334 NodeID dstrec = getValueNode(cs);
1335 //Does it actually return a ptr?
1336 if (!cs->getType()->isVoidTy())
1337 {
1338 NodeID srcret = getReturnNode(svffun);
1339 FunExitICFGNode* exitICFGNode = pag->getICFG()->getFunExitICFGNode(svffun);
1340 addRetEdge(srcret, dstrec,callICFGNode, exitICFGNode);
1341 }
1342 //Iterators for the actual and formal parameters
1343 u32_t itA = 0, ieA = cs->arg_size();
1344 Function::const_arg_iterator itF = F->arg_begin(), ieF = F->arg_end();
1345 //Go through the fixed parameters.
1346 DBOUT(DPAGBuild, outs() << " args:");
1347 for (; itF != ieF; ++itA, ++itF)
1348 {
1349 //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
1350 if (itA == ieA)
1351 {
1352 DBOUT(DPAGBuild, outs() << " !! not enough args\n");
1353 break;
1354 }
1355 const Value* AA = cs->getArgOperand(itA), *FA = &*itF; //current actual/formal arg
1356
1357 DBOUT(DPAGBuild, outs() << "process actual parm " << llvmModuleSet()->getSVFValue(AA)->toString() << " \n");
1358
1359 NodeID dstFA = getValueNode(FA);
1360 NodeID srcAA = getValueNode(AA);
1361 FunEntryICFGNode* entry = pag->getICFG()->getFunEntryICFGNode(svffun);
1362 addCallEdge(srcAA, dstFA, callICFGNode, entry);
1363 }
1364 //Any remaining actual args must be varargs.
1365 if (F->isVarArg())
1366 {

1368 DBOUT(DPAGBuild, outs() << "\n varargs:");
