54 SVFIR* SVFIRBuilder::build()
55 {
56  double startTime = SVFStat::getClk(true);
57 
58  DBOUT(DGENERAL, outs() << pasMsg("\t Building SVFIR ...\n"));
59 
60  // Set SVFModule from SVFIRBuilder
61  pag->setModule(svfModule);
62 
63  // Build ICFG
64  ICFG* icfg = new ICFG();
65  ICFGBuilder icfgbuilder(icfg);
66  icfgbuilder.build(svfModule);
67  pag->setICFG(icfg);
68 
69  CHGraph* chg = new CHGraph(pag->getModule());
70  CHGBuilder chgbuilder(chg);
71  chgbuilder.buildCHG();
72  pag->setCHG(chg);
73 
74  // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
75  if (SVFModule::pagReadFromTXT())
76  {
77  PAGBuilderFromFile fileBuilder(SVFModule::pagFileName());
78  return fileBuilder.build();
79  }
80 
81  // If the SVFIR has been built before, then we return the unique SVFIR of the program
82  if(pag->getNodeNumAfterPAGBuild() > 1)
83  return pag;
84 
87  initialiseNodes();
90  visitGlobal(svfModule);
92 
94  for (Module& M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
95  {
96  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
97  {
98  const Function& fun = *F;
99  const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(&fun);
101  if(!fun.isDeclaration())
102  {
108  if(fun.doesNotReturn() == false && fun.getReturnType()->isVoidTy() == false)
109  pag->addFunRet(svffun,pag->getGNode(pag->getReturnNode(svffun)));
110 
113  for (Function::const_arg_iterator I = fun.arg_begin(), E = fun.arg_end();
114  I != E; ++I)
115  {
116  setCurrentLocation(&*I,&fun.getEntryBlock());
117  NodeID argValNodeId = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(&*I));
118  // if this is the function does not have caller (e.g. main)
119  // or a dead function, shall we create a black hole address edge for it?
120  // it is (1) too conservative, and (2) make FormalParmVFGNode defined at blackhole address PAGEdge.
121  // if(SVFUtil::ArgInNoCallerFunction(&*I)) {
122  // if(I->getType()->isPointerTy())
123  // addBlackHoleAddrEdge(argValNodeId);
124  //}
125  pag->addFunArgs(svffun,pag->getGNode(argValNodeId));
126  }
127  }
128  for (Function::const_iterator bit = fun.begin(), ebit = fun.end();
129  bit != ebit; ++bit)
130  {
131  const BasicBlock& bb = *bit;
132  for (BasicBlock::const_iterator it = bb.begin(), eit = bb.end();
133  it != eit; ++it)
134  {
135  const Instruction& inst = *it;
136  setCurrentLocation(&inst,&bb);
137  visit(const_cast<Instruction&>(inst));
138  }
139  }
140  }
141  }
142 
143  sanityCheck();
144 
145  pag->initialiseCandidatePointers();
146 
147  pag->setNodeNumAfterPAGBuild(pag->getTotalNodeNum());
148 
149  // dump SVFIR
150  if (Options::PAGDotGraph())
151  pag->dump("svfir_initial");
152 
153  // print to command line of the SVFIR graph
154  if (Options::PAGPrint())
155  pag->print();
156 
157  // dump ICFG
158  if (Options::DumpICFG())
159  pag->getICFG()->dump("icfg_initial");
160 
161  if (Options::LoopAnalysis())
162  {
163  LLVMLoopAnalysis loopAnalysis;
164  loopAnalysis.build(pag->getICFG());
165  }
166 
167  // dump SVFIR as JSON
168  if (!Options::DumpJson().empty())
169  {
170  SVFIRWriter::writeJsonToPath(pag, Options::DumpJson());
171  }
172 
173  double endTime = SVFStat::getClk(true);
174  SVFStat::timeOfBuildingSVFIR = (endTime - startTime) / TIMEINTERVAL;
175 
176  return pag;
177 }

187 
188  pag->addBlackholeObjNode();
189  pag->addConstantObjNode();
190  pag->addBlackholePtrNode();
191  addNullPtrNode();
192 
193  for (SymbolTableInfo::ValueToIDMapTy::iterator iter =
194  symTable->valSyms().begin(); iter != symTable->valSyms().end();
195  ++iter)
196  {
197  DBOUT(DPAGBuild, outs() << "add val node " << iter->second << "\n");
198  if(iter->second == symTable->blkPtrSymID() || iter->second == symTable->nullPtrSymID())
199  continue;
200  pag->addValNode(iter->first, iter->second);
201  }
202

204  symTable->objSyms().begin(); iter != symTable->objSyms().end();
205  ++iter)
206  {
207  DBOUT(DPAGBuild, outs() << "add obj node " << iter->second << "\n");
208  if(iter->second == symTable->blackholeSymID() || iter->second == symTable->constantSymID())
209  continue;
210  pag->addObjNode(iter->first, iter->second);
211  }
212 
213  for (SymbolTableInfo::FunToIDMapTy::iterator iter =
214  symTable->retSyms().begin(); iter != symTable->retSyms().end();
215  ++iter)
216  {
217  DBOUT(DPAGBuild, outs() << "add ret node " << iter->second << "\n");
218  pag->addRetNode(iter->first, iter->second);
219  }
220 
221  for (SymbolTableInfo::FunToIDMapTy::iterator iter =
222  symTable->varargSyms().begin();
223  iter != symTable->varargSyms().end(); ++iter)
224  {
225  DBOUT(DPAGBuild, outs() << "add vararg node " << iter->second << "\n");
226  pag->addVarargNode(iter->first, iter->second);
227  }
228 
230  for (SymbolTableInfo::ValueToIDMapTy::iterator iter =
231  symTable->objSyms().begin(); iter != symTable->objSyms().end(); ++iter)
232  {
233  DBOUT(DPAGBuild, outs() << "add address edges for constant node " << iter->second << "\n");
234  const SVFValue* val = iter->first;
235  if (isConstantObjSym(val))
236  {
237  NodeID ptr = pag->getValueNode(val);
238  if(ptr!= pag->getBlkPtr() && ptr!= pag->getNullPtr())
239  {
240  setCurrentLocation(val, nullptr);
241  addAddrEdge(iter->second, ptr);
242  }
243  }
244  }
245 
246  assert(pag->getTotalNodeNum() >= symTable->getTotalSymNum()
247  && "not all node have been initialized!!!");

271 bool SVFIRBuilder::computeGepOffset(const User *V, AccessPath& ap)
272 {
273  assert(V);
274 
275  const llvm::GEPOperator *gepOp = SVFUtil::dyn_cast<const llvm::GEPOperator>(V);
276  DataLayout * dataLayout = getDataLayout(LLVMModuleSet::getLLVMModuleSet()->getMainLLVMModule());
277  llvm::APInt byteOffset(dataLayout->getIndexSizeInBits(gepOp->getPointerAddressSpace()),0,true);
278  if(gepOp && dataLayout && gepOp->accumulateConstantOffset(*dataLayout,byteOffset))
279  {
280  //s32_t bo = byteOffset.getSExtValue();
281  }
282 
283  bool isConst = true;
284 
285  for (bridge_gep_iterator gi = bridge_gep_begin(*V), ge = bridge_gep_end(*V);
286  gi != ge; ++gi)
287  {
288  const Type* gepTy = *gi;
289  const SVFType* svfGepTy = LLVMModuleSet::getLLVMModuleSet()->getSVFType(gepTy);
290  const Value* offsetVal = gi.getOperand();
291  const SVFValue* offsetSvfVal = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(offsetVal);
292  assert(gepTy != offsetVal->getType() && "iteration and operand have the same type?");
293  ap.addOffsetVarAndGepTypePair(getPAG()->getGNode(getPAG()->getValueNode(offsetSvfVal)), svfGepTy);
294 
295  //The int value of the current index operand
296  const ConstantInt* op = SVFUtil::dyn_cast<ConstantInt>(offsetVal);
297 
298  // if Options::ModelConsts() is disabled. We will treat whole array as one,
299  // but we can distinguish different field of an array of struct, e.g. s[1].f1 is different from s[0].f2
300  if(const ArrayType* arrTy = SVFUtil::dyn_cast<ArrayType>(gepTy))
301  {
302  if(!op || (arrTy->getArrayNumElements() <= (u32_t)op->getSExtValue()))
303  continue;
304  APOffset idx = op->getSExtValue();
305  u32_t offset = pag->getSymbolInfo()->getFlattenedElemIdx(LLVMModuleSet::getLLVMModuleSet()->getSVFType(arrTy), idx);
306  ap.setFldIdx(ap.getConstantStructFldIdx() + offset);
307  }
308  else if (const StructType *ST = SVFUtil::dyn_cast<StructType>(gepTy))
309  {
310  assert(op && "non-const offset accessing a struct");
311  //The actual index
312  APOffset idx = op->getSExtValue();
313  u32_t offset = pag->getSymbolInfo()->getFlattenedElemIdx(LLVMModuleSet::getLLVMModuleSet()->getSVFType(ST), idx);
314  ap.setFldIdx(ap.getConstantStructFldIdx() + offset);
315  }
316  else if (gepTy->isSingleValueType())
317  {
318  // If it's a non-constant offset access
319  // If its point-to target is struct or array, it's likely an array accessing (%result = gep %struct.A* %a, i32 %non-const-index)
320  // If its point-to target is single value (pointer arithmetic), then it's a variant gep (%result = gep i8* %p, i32 %non-const-index)
321  if(!op && gepTy->isPointerTy() && getPtrElementType(SVFUtil::dyn_cast<PointerType>(gepTy))->isSingleValueType())
322  isConst = false;
323 
324  // The actual index
325  //s32_t idx = op->getSExtValue();
326 
327  // For pointer arithmetic we ignore the byte offset
328  // consider using inferFieldIdxFromByteOffset(geopOp,dataLayout,ap,idx)?
329  // ap.setFldIdx(ap.getConstantFieldIdx() + inferFieldIdxFromByteOffset(geopOp,idx));
330  }
331  }
332  return isConst;
333 }

338 void SVFIRBuilder::processCE(const Value* val)
339 {
340  if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
341  {
342  if (const ConstantExpr* gepce = isGepConstantExpr(ref))
343  {
344  DBOUT(DPAGBuild, outs() << "handle gep constant expression " << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref)->toString() << "\n");
345  const Constant* opnd = gepce->getOperand(0);
346  // handle recursive constant express case (gep (bitcast (gep X 1)) 1)
347  processCE(opnd);
348  auto &GEPOp = llvm::cast<llvm::GEPOperator>(*gepce);
349  Type *pType = GEPOp.getSourceElementType();
350  AccessPath ap(0, LLVMModuleSet::getLLVMModuleSet()->getSVFType(pType));
351  bool constGep = computeGepOffset(gepce, ap);
352  // must invoke pag methods here, otherwise it will be a dead recursion cycle
353  const SVFValue* cval = getCurrentValue();
354  const SVFBasicBlock* cbb = getCurrentBB();
355  setCurrentLocation(gepce, nullptr);
356  /*
357  * The gep edge created are like constexpr (same edge may appear at multiple callsites)
358  * so bb/inst of this edge may be rewritten several times, we treat it as global here.
359  */
360  addGepEdge(pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(opnd)), pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(gepce)), ap, constGep);
361  setCurrentLocation(cval, cbb);
362  }
363  else if (const ConstantExpr* castce = isCastConstantExpr(ref))
364  {
365  DBOUT(DPAGBuild, outs() << "handle cast constant expression " << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref)->toString() << "\n");
366  const Constant* opnd = castce->getOperand(0);
367  processCE(opnd);
368  const SVFValue* cval = getCurrentValue();
369  const SVFBasicBlock* cbb = getCurrentBB();
370  setCurrentLocation(castce, nullptr);
371  addCopyEdge(pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(opnd)), pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(castce)));
372  setCurrentLocation(cval, cbb);
373  }
374  else if (const ConstantExpr* selectce = isSelectConstantExpr(ref))
375  {
376  DBOUT(DPAGBuild, outs() << "handle select constant expression " << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref)->toString() << "\n");
377  const Constant* src1 = selectce->getOperand(1);
378  const Constant* src2 = selectce->getOperand(2);
379  processCE(src1);
380  processCE(src2);
381  const SVFValue* cval = getCurrentValue();
382  const SVFBasicBlock* cbb = getCurrentBB();
383  setCurrentLocation(selectce, nullptr);
384  NodeID cond = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(selectce->getOperand(0)));
385  NodeID nsrc1 = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(src1));
386  NodeID nsrc2 = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(src2));
387  NodeID nres = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(selectce));
388  addSelectStmt(nres,nsrc1, nsrc2, cond);
389  setCurrentLocation(cval, cbb);
390  }
391  // if we meet a int2ptr, then it points-to black hole
392  else if (const ConstantExpr* int2Ptrce = isInt2PtrConstantExpr(ref))
393  {
394  addGlobalBlackHoleAddrEdge(pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(int2Ptrce)), int2Ptrce);
395  }
396  else if (const ConstantExpr* ptr2Intce = isPtr2IntConstantExpr(ref))
397  {
398  const Constant* opnd = ptr2Intce->getOperand(0);
399  processCE(opnd);
400  const SVFBasicBlock* cbb = getCurrentBB();
401  const SVFValue* cval = getCurrentValue();
402  setCurrentLocation(ptr2Intce, nullptr);
403  addCopyEdge(pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(opnd)), pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ptr2Intce)));
404  setCurrentLocation(cval, cbb);
405  }
406  else if(isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
407  {
408  // we don't handle trunc and cmp instruction for now
409  const SVFValue* cval = getCurrentValue();
410  const SVFBasicBlock* cbb = getCurrentBB();
411  setCurrentLocation(ref, nullptr);
412  NodeID dst = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref));
413  addBlackHoleAddrEdge(dst);
414  setCurrentLocation(cval, cbb);
415  }
416  else if (isBinaryConstantExpr(ref))
417  {
418  // we don't handle binary constant expression like add(x,y) now
419  const SVFValue* cval = getCurrentValue();
420  const SVFBasicBlock* cbb = getCurrentBB();
421  setCurrentLocation(ref, nullptr);
422  NodeID dst = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref));
423  addBlackHoleAddrEdge(dst);
424  setCurrentLocation(cval, cbb);
425  }
426  else if (isUnaryConstantExpr(ref))
427  {
428  // we don't handle unary constant expression like fneg(x) now
429  const SVFValue* cval = getCurrentValue();
430  const SVFBasicBlock* cbb = getCurrentBB();
431  setCurrentLocation(ref, nullptr);
432  NodeID dst = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref));
433  addBlackHoleAddrEdge(dst);
434  setCurrentLocation(cval, cbb);
435  }
436  else if (SVFUtil::isa<ConstantAggregate>(ref))
437  {
438  // we don't handle constant aggregate like constant vectors
439  }
440  else if (SVFUtil::isa<BlockAddress>(ref))
441  {
442  // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
443  // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
444  const SVFValue* cval = getCurrentValue();
445  const SVFBasicBlock* cbb = getCurrentBB();
446  setCurrentLocation(ref, nullptr);
447  NodeID dst = pag->getValueNode(LLVMModuleSet::getLLVMModuleSet()->getSVFValue(ref));
448  addAddrEdge(pag->getConstantNode(), dst);
449  setCurrentLocation(cval, cbb);
450  }
451  else
452  {
453  if(SVFUtil::isa<ConstantExpr>(val))
454  assert(false && "we don't handle all other constant expression for now!");
455  }
456  }

479 /*For global variable initialization
480  * Give a simple global variable
481  * int x = 10; // store 10 x (constant, non pointer) |
482  * int *y = &x; // store x y (pointer type)
483  * Given a struct
484  * struct Z { int s; int *t;};
485  * Global initialization:
486  * struct Z z = {10,&x}; // store x z.t (struct type)
487  * struct Z *m = &z; // store z m (pointer type)
488  * struct Z n = {10,&z.s}; // store z.s n , &z.s constant expression (constant expression)
489  */
490 void SVFIRBuilder::InitialGlobal(const GlobalVariable *gvar, Constant *C,
491  u32_t offset)
492 {
493  DBOUT(DPAGBuild, outs() << "global " << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(gvar)->toString() << " constant initializer: " << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(C)->toString() << "\n");
494  if (C->getType()->isSingleValueType())
495  {
496  NodeID src = getValueNode(C);
497  // get the field value if it is available, otherwise we create a dummy field node.
498  setCurrentLocation(gvar, nullptr);
499  NodeID field = getGlobalVarField(gvar, offset, LLVMModuleSet::getLLVMModuleSet()->getSVFType(C->getType()));
500 
501  if (SVFUtil::isa<GlobalVariable, Function>(C))
502  {
503  setCurrentLocation(C, nullptr);
504  addStoreEdge(src, field);
505  }
506  else if (SVFUtil::isa<ConstantExpr>(C))
507  {
508  // add gep edge of C1 itself is a constant expression
509  processCE(C);
510  setCurrentLocation(C, nullptr);
511  addStoreEdge(src, field);
512  }
513  else if (SVFUtil::isa<BlockAddress>(C))
514  {
515  // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
516  // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
517  processCE(C);
518  setCurrentLocation(C, nullptr);
519  addAddrEdge(pag->getConstantNode(), src);
520  }
521  else
522  {
523  setCurrentLocation(C, nullptr);
524  addStoreEdge(src, field);
526  if (C->getType()->isPtrOrPtrVectorTy() && src != pag->getNullPtr())
527  addCopyEdge(pag->getNullPtr(), src);
528  }
529  }
530  else if (SVFUtil::isa<ConstantArray, ConstantStruct>(C))
531  {
532  if(LLVMUtil::isValVtbl(gvar) && !Options::VtableInSVFIR())
533  return;
534  for (u32_t i = 0, e = C->getNumOperands(); i != e; i++)
535  {
536  u32_t off = pag->getSymbolInfo()->getFlattenedElemIdx(LLVMModuleSet::getLLVMModuleSet()->getSVFType(C->getType()), i);
537  InitialGlobal(gvar, SVFUtil::cast<Constant>(C->getOperand(i)), offset + off);
538  }
539  }
540  else if(ConstantData* data = SVFUtil::dyn_cast<ConstantData>(C))
541  {
542  if(Options::ModelConsts())
543  {
544  if(ConstantDataSequential* seq = SVFUtil::dyn_cast<ConstantDataSequential>(data))
545  {
546  for(u32_t i = 0; i < seq->getNumElements(); i++)
547  {
548  u32_t off = pag->getSymbolInfo()->getFlattenedElemIdx(LLVMModuleSet::getLLVMModuleSet()->getSVFType(C->getType()), i);
549  Constant* ct = seq->getElementAsConstant(i);
550  InitialGlobal(gvar, ct, offset + off);
551  }
552  }
553  else
554  {
555  assert((SVFUtil::isa<ConstantAggregateZero, UndefValue>(data)) && "Single value type data should have been handled!");
556  }
557  }
558  }
559  else
560  {
561  //TODO:assert(SVFUtil::isa<ConstantVector>(C),"what else do we have");
562  }
563 }

597  NodeID obj = getObjectNode(fun);

1066 {

1069  if(const GetElementPtrInst* gep = SVFUtil::dyn_cast<GetElementPtrInst>(value))
1070  {
1071  APOffset totalidx = 0;
1072  for (bridge_gep_iterator gi = bridge_gep_begin(gep), ge = bridge_gep_end(gep); gi != ge; ++gi)
1073  {
1074  if(const ConstantInt* op = SVFUtil::dyn_cast<ConstantInt>(gi.getOperand()))
1075  totalidx += op->getSExtValue();

1077  if(totalidx == 0 && !SVFUtil::isa<StructType>(value->getType()))
1078  value = gep->getPointerOperand();
1079  }

1085  // %call = malloc 80
