191  static bool eqVarToAddrs(const VarToAddrs &lhs, const VarToAddrs &rhs)
192  {
193  if (lhs.size() != rhs.size()) return false;
194  for (const auto &item: lhs)
195  {
196  auto it = rhs.find(item.first);
197  if (it == rhs.end())
198  return false;
199  if (item.second.equals(it->second))
200  {
201  return false;
202  }
203  }
204  return true;
205  }
