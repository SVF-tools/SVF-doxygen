75 void MHP::analyzeInterleaving()
76 {
77  for (const std::pair<const NodeID, TCTNode*>& tpair : *tct)
78  {
79  const CxtThread& ct = tpair.second->getCxtThread();
80  NodeID rootTid = tpair.first;
81  const SVFFunction* routine = tct->getStartRoutineOfCxtThread(ct);
82  const ICFGNode* svfInst = routine->getEntryBlock()->front();
83  CxtThreadStmt rootcts(rootTid, ct.getContext(), svfInst);
84 
85  addInterleavingThread(rootcts, rootTid);
86  updateAncestorThreads(rootTid);
87  updateSiblingThreads(rootTid);
88 
89  while (!cxtStmtList.empty())
90  {
91  CxtThreadStmt cts = popFromCTSWorkList();
92  const ICFGNode* curInst = cts.getStmt();
93  DBOUT(DMTA, outs() << "-----\nMHP analysis root thread: " << rootTid << " ");
94  DBOUT(DMTA, cts.dump());
95  DBOUT(DMTA, outs() << "current thread interleaving: < ");
96  DBOUT(DMTA, dumpSet(getInterleavingThreads(cts)));
97  DBOUT(DMTA, outs() << " >\n-----\n");
98 
100  if (!tct->isCandidateFun(curInst->getFun()))
101  {
102  handleNonCandidateFun(cts);
103  }
105  else
106  {
107  if (isTDFork(curInst))
108  {
109  handleFork(cts, rootTid);
110  }
111  else if (isTDJoin(curInst))
112  {
113  handleJoin(cts, rootTid);
114  }
115  else if (tct->isCallSite(curInst) && !tct->isExtCall(curInst))
116  {
117  handleCall(cts, rootTid);
118  CallGraph::FunctionSet callees;
119  if (!tct->isCandidateFun(getCallee(SVFUtil::cast<CallICFGNode>(curInst), callees)))
120  handleIntra(cts);
121  }
122  else if (isRetInstNode(curInst))
123  {
124  handleRet(cts);
125  }
126  else
127  {
128  handleIntra(cts);
129  }
130  }
131  }
132  }
133 
135  updateNonCandidateFunInterleaving();
136 
137  if (Options::PrintInterLev())
138  printInterleaving();
139 }

144 void MHP::updateNonCandidateFunInterleaving()
145 {
146  SVFModule* module = tct->getSVFModule();
147  for (const SVFFunction* fun : module->getFunctionSet())
148  {
149  if (!tct->isCandidateFun(fun) && !isExtCall(fun))
150  {
151  const ICFGNode* entryNode = fun->getEntryBlock()->front();
152 
153  if (!hasThreadStmtSet(entryNode))
154  continue;
155 
156  const CxtThreadStmtSet& tsSet = getThreadStmtSet(entryNode);
157 
158  for (const CxtThreadStmt& cts : tsSet)
159  {
160  const CallStrCxt& curCxt = cts.getContext();
161 
162  for (const SVFBasicBlock* svfbb : fun->getBasicBlockList())
163  {
164  for (const ICFGNode* curNode : svfbb->getICFGNodeList())
165  {
166  if (curNode == entryNode)
167  continue;
168  CxtThreadStmt newCts(cts.getTid(), curCxt, curNode);
169  threadStmtToTheadInterLeav[newCts] |= threadStmtToTheadInterLeav[cts];
170  instToTSMap[curNode].insert(newCts);
171  }
172  }
173  }
174  }

181 void MHP::handleNonCandidateFun(const CxtThreadStmt& cts)
182 {
183  const ICFGNode* curInst = cts.getStmt();
184  const SVFFunction* curfun = curInst->getFun();

186  const CallStrCxt& curCxt = cts.getContext();
187  CallGraphNode* node = tcg->getCallGraphNode(curfun);
188  for (CallGraphNode::const_iterator nit = node->OutEdgeBegin(), neit = node->OutEdgeEnd(); nit != neit; nit++)
189  {
190  const SVFFunction* callee = (*nit)->getDstNode()->getFunction();
191  if (!isExtCall(callee))
192  {
193  const ICFGNode* calleeInst = callee->getEntryBlock()->front();
194  CxtThreadStmt newCts(cts.getTid(), curCxt, calleeInst);
195  addInterleavingThread(newCts, cts);
196  }

203 void MHP::handleFork(const CxtThreadStmt& cts, NodeID rootTid)
204 {
205 
206  const ICFGNode* call = cts.getStmt();
207  const CallStrCxt& curCxt = cts.getContext();
208 
209  assert(isTDFork(call));
210  const CallICFGNode* cbn = cast<CallICFGNode>(call);
211  if (tct->getThreadCallGraph()->hasCallGraphEdge(cbn))
212  {
213 
214  for (ThreadCallGraph::ForkEdgeSet::const_iterator cgIt = tcg->getForkEdgeBegin(cbn),
215  ecgIt = tcg->getForkEdgeEnd(cbn);
216  cgIt != ecgIt; ++cgIt)
217  {
218  const SVFFunction* svfroutine = (*cgIt)->getDstNode()->getFunction();
219  CallStrCxt newCxt = curCxt;
220  pushCxt(newCxt, cbn, svfroutine);
221  const ICFGNode* stmt = svfroutine->getEntryBlock()->front();
222  CxtThread ct(newCxt, call);
223  CxtThreadStmt newcts(tct->getTCTNode(ct)->getId(), ct.getContext(), stmt);
224  addInterleavingThread(newcts, cts);
225  }
226  }
227  handleIntra(cts);

233 void MHP::handleJoin(const CxtThreadStmt& cts, NodeID rootTid)
234 {
235 
236  const ICFGNode* call = cts.getStmt();
237  const CallStrCxt& curCxt = cts.getContext();
238 
239  assert(isTDJoin(call));
240 
241  NodeBS joinedTids = getDirAndIndJoinedTid(curCxt, call);
242  if (!joinedTids.empty())
243  {
244  if (fja->hasJoinLoop(call))
245  {
246  std::vector<const SVFBasicBlock*> exitbbs;
247  call->getFun()->getExitBlocksOfLoop(call->getBB(), exitbbs);
248  while (!exitbbs.empty())
249  {
250  const SVFBasicBlock* eb = exitbbs.back();
251  exitbbs.pop_back();
252  const ICFGNode* svfEntryInst = eb->front();
253  CxtThreadStmt newCts(cts.getTid(), curCxt, svfEntryInst);
254  addInterleavingThread(newCts, cts);
255  if (hasJoinInSymmetricLoop(curCxt, call))
256  rmInterleavingThread(newCts, joinedTids, call);
257  }
258  }
259  else
260  {
261  rmInterleavingThread(cts, joinedTids, call);
262  DBOUT(DMTA, outs() << "\n\t match join site " << call->toString() << " for thread " << rootTid << "\n");
263  }
264  }
267  else
268  {
269  if (fja->hasJoinLoop(call))
270  {
271  std::vector<const SVFBasicBlock*> exitbbs;
272  call->getFun()->getExitBlocksOfLoop(call->getBB(), exitbbs);
273  while (!exitbbs.empty())
274  {
275  const SVFBasicBlock* eb = exitbbs.back();
276  exitbbs.pop_back();
277  const ICFGNode* svfEntryInst = eb->front();
278  CxtThreadStmt newCts(cts.getTid(), cts.getContext(), svfEntryInst);
279  addInterleavingThread(newCts, cts);
280  }
281  }
282  }
283  handleIntra(cts);

290 {
291

318 void MHP::handleRet(const CxtThreadStmt& cts)
319 {
320  CallGraphNode* curFunNode = tcg->getCallGraphNode(cts.getStmt()->getFun());
321  for (CallGraphEdge* edge : curFunNode->getInEdges())
322  {
323  if (SVFUtil::isa<ThreadForkEdge, ThreadJoinEdge>(edge))
324  continue;
325  for (CallGraphEdge::CallInstSet::const_iterator cit = (edge)->directCallsBegin(),
326  ecit = (edge)->directCallsEnd();
327  cit != ecit; ++cit)
328  {
329  CallStrCxt newCxt = cts.getContext();
330  if (matchCxt(newCxt, *cit, curFunNode->getFunction()))
331  {
332  for(const ICFGEdge* outEdge : cts.getStmt()->getOutEdges())
333  {
334  if(outEdge->getDstNode()->getFun() == cts.getStmt()->getFun())
335  {
336  CxtThreadStmt newCts(cts.getTid(), newCxt, outEdge->getDstNode());
337  addInterleavingThread(newCts, cts);
338  }
339  }
340  }
341  }
342  for (CallGraphEdge::CallInstSet::const_iterator cit = (edge)->indirectCallsBegin(),
343  ecit = (edge)->indirectCallsEnd();
344  cit != ecit; ++cit)
345  {
346  CallStrCxt newCxt = cts.getContext();
347  if (matchCxt(newCxt, *cit, curFunNode->getFunction()))
348  {
349  for(const ICFGEdge* outEdge : cts.getStmt()->getOutEdges())
350  {
351  if(outEdge->getDstNode()->getFun() == cts.getStmt()->getFun())
352  {
353  CxtThreadStmt newCts(cts.getTid(), newCxt, outEdge->getDstNode());
354  addInterleavingThread(newCts, cts);
355  }
356  }
357  }
358  }
359  }

445 bool MHP::isRecurFullJoin(NodeID parentTid, NodeID curTid)
446 {
447  if (parentTid == curTid)
448  return true;
449 
450  const TCTNode* curNode = tct->getTCTNode(curTid);
451  FIFOWorkList<const TCTNode*> worklist;
452  worklist.push(curNode);
453  while (!worklist.empty())
454  {
455  const TCTNode* node = worklist.pop();
456  for (TCTEdge* edge : node->getInEdges())
457  {
458  NodeID srcID = edge->getSrcID();
459  if (fja->isFullJoin(srcID, node->getId()))
460  {
461  if (srcID == parentTid)
462  return true;
463  else
464  worklist.push(edge->getSrcNode());
465  }
466  else
467  {
468  return false;
469  }
470  }
471  }
472  return false;
473 }

554 bool MHP::mayHappenInParallelInst(const ICFGNode* i1, const ICFGNode* i2)
555 {
556 
558  if (!hasThreadStmtSet(i1) || !hasThreadStmtSet(i2))
559  return false;
560 
561  const CxtThreadStmtSet& tsSet1 = getThreadStmtSet(i1);
562  const CxtThreadStmtSet& tsSet2 = getThreadStmtSet(i2);
563  for (const CxtThreadStmt& ts1 : tsSet1)
564  {
565  NodeBS l1 = getInterleavingThreads(ts1);
566  for (const CxtThreadStmt& ts2 : tsSet2)
567  {
568  NodeBS l2 = getInterleavingThreads(ts2);
569  if (ts1.getTid() != ts2.getTid())
570  {
571  if (l1.test(ts2.getTid()) && l2.test(ts1.getTid()))
572  {
573  numOfMHPQueries++;
574  return true;
575  }
576  }
577  else
578  {
579  if (isMultiForkedThread(ts1.getTid()))
580  {
581  numOfMHPQueries++;
582  return true;
583  }
584  }
585  }
586  }
587  return false;
588 }

591 {

599  nonCandidateFuncMHPRelMap[funpair] = mhp;

602  else

606  return it->second;
607  }
608  }

719 void ForkJoinAnalysis::analyzeForkJoinPair()
720 {

722  {

724  const NodeID rootTid = tpair.first;

726  if (const ICFGNode* forkInst = ct.getThread())
727  {

729  const ICFGNode* exitInst = getExitInstOfParentRoutineFun(rootTid);
730 
731  for(const ICFGEdge* outEdge : forkInst->getOutEdges())
732  {
733  if(outEdge->getDstNode()->getFun() == forkInst->getFun())
734  {
735  CxtStmt newCts(forkSiteCxt, outEdge->getDstNode());
736  markCxtStmtFlag(newCts, TDAlive);
737  }
738  }
739

744  DBOUT(DMTA, outs() << "-----\nForkJoinAnalysis root thread: " << tpair.first << " ");
745  DBOUT(DMTA, cts.dump());
746  DBOUT(DMTA, outs() << "-----\n");
747  CallGraph::FunctionSet callees;
748  if (isTDFork(curInst))
749  {
750  handleFork(cts, rootTid);
751  }
752  else if (isTDJoin(curInst))
753  {
754  handleJoin(cts, rootTid);
755  }
756  else if (tct->isCallSite(curInst) && tct->isCandidateFun(getCallee(curInst, callees)))
757  {
758 
759  handleCall(cts, rootTid);
760  }
761  else if (isRetInstNode(curInst))
762  {
763  handleRet(cts);
764  }

766  {
767  handleIntra(cts);
768  }
769 
770  if (curInst == exitInst)

772  if (getMarkedFlag(cts) != TDAlive)
773  addToFullJoin(tct->getParentThread(rootTid), rootTid);
774  else
775  addToPartial(tct->getParentThread(rootTid), rootTid);

810 void ForkJoinAnalysis::handleJoin(const CxtStmt& cts, NodeID rootTid)
811 {

813  const CallStrCxt& curCxt = cts.getContext();
814 
815  assert(isTDJoin(call));
816  const CallICFGNode* cbn = cast<CallICFGNode>(call);
817  if (getTCG()->hasCallGraphEdge(cbn))
818  {
819  const ICFGNode* forkSite = tct->getTCTNode(rootTid)->getCxtThread().getThread();
820  const ICFGNode* joinSite = cts.getStmt();
821 
822  if (isAliasedForkJoin(forkSite, joinSite))
823  {
824  if (hasJoinLoop(joinSite))
825  {
826  LoopBBs& joinLoop = getJoinLoop(joinSite);
827  std::vector<const SVFBasicBlock *> exitbbs;
828  joinSite->getFun()->getExitBlocksOfLoop(joinSite->getBB(), exitbbs);
829  while (!exitbbs.empty())
830  {
831  const SVFBasicBlock* eb = exitbbs.back();
832  exitbbs.pop_back();
833  const ICFGNode* svfEntryInst = eb->front();
834  CxtStmt newCts(curCxt, svfEntryInst);
835  addDirectlyJoinTID(cts, rootTid);
836  if (isSameSCEV(forkSite, joinSite))
837  {
838  markCxtStmtFlag(newCts, TDDead);
839  addSymmetricLoopJoin(cts, joinLoop);
840  }
841  else
842  markCxtStmtFlag(cts, TDAlive);
843  }
844  }
845  else
846  {
847  markCxtStmtFlag(cts, TDDead);
848  addDirectlyJoinTID(cts, rootTid);

857  {

860  while (!exitbbs.empty())
861  {
862  const SVFBasicBlock* eb = exitbbs.back();
863  exitbbs.pop_back();
