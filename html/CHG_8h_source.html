270 inline u32_t getVirtualFunctionID(const SVFFunction* vfn) const
271 {
272 Map<const SVFFunction*, u32_t>::const_iterator it =
273 virtualFunctionToIDMap.find(vfn);
274 if (it != virtualFunctionToIDMap.end())

279 inline const SVFFunction* getVirtualFunctionBasedonID(u32_t id) const
280 {
281 Map<const SVFFunction*, u32_t>::const_iterator it, eit;
282 for (it = virtualFunctionToIDMap.begin(), eit =
283 virtualFunctionToIDMap.end(); it != eit; ++it)
284 {
285 if (it->second == id)
286 return it->first;
287 }
288 return nullptr;
