250  inline u32_t getVirtualFunctionID(const SVFFunction* vfn) const
251  {
252  Map<const SVFFunction*, u32_t>::const_iterator it =
253  virtualFunctionToIDMap.find(vfn);
254  if (it != virtualFunctionToIDMap.end())
255  return it->second;
256  else
257  return -1;

259  inline const SVFFunction* getVirtualFunctionBasedonID(u32_t id) const
260  {
261  Map<const SVFFunction*, u32_t>::const_iterator it, eit;
262  for (it = virtualFunctionToIDMap.begin(), eit =
263  virtualFunctionToIDMap.end(); it != eit; ++it)
264  {
265  if (it->second == id)
266  return it->first;
267  }
268  return nullptr;
269  }
