48 CFBasicBlockEdge* CFBasicBlockGraph::getCFBasicBlockEdge(const CFBasicBlockNode *src, const CFBasicBlockNode *dst, const ICFGEdge* icfgEdge)
49 {
50  CFBasicBlockEdge *edge = nullptr;
51  size_t counter = 0;
52  for (auto iter = src->OutEdgeBegin();
53  iter != src->OutEdgeEnd(); ++iter)
54  {
55  if ((*iter)->getDstID() == dst->getId() && (*iter)->getICFGEdge() == icfgEdge)
56  {
57  counter++;
58  edge = (*iter);
59  }
60  }
61  assert(counter <= 1 && "there's more than one edge between two nodes");
62  return edge;
63 }
