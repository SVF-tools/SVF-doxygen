74void LLVMUtil::getFunReachableBBs (const Function* fun, std::vector<const SVFBasicBlock*> &reachableBBs)
75{
76 assert(!LLVMUtil::isExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun)) && "The calling function cannot be an external function.");
77 //initial DominatorTree
78 DominatorTree& dt = LLVMModuleSet::getLLVMModuleSet()->getDomTree(fun);
79
80 Set<const BasicBlock*> visited;
81 std::vector<const BasicBlock*> bbVec;
82 bbVec.push_back(&fun->getEntryBlock());
83 while(!bbVec.empty())
84 {
85 const BasicBlock* bb = bbVec.back();
86 bbVec.pop_back();
87 const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
88 reachableBBs.push_back(svfbb);
89 if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
90 {
91 for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
92 DI != DE; ++DI)
93 {
94 const BasicBlock* succbb = (*DI)->getBlock();
95 if(visited.find(succbb)==visited.end())
96 visited.insert(succbb);
97 else
98 continue;
99 bbVec.push_back(succbb);
100 }
101 }
102 }
103}

122bool LLVMUtil::functionDoesNotRet(const Function* fun)
123{
124 const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
125 if (LLVMUtil::isExtCall(svffun))
126 {
127 return fun->getReturnType()->isVoidTy();
128 }
129 std::vector<const BasicBlock*> bbVec;
130 Set<const BasicBlock*> visited;
131 bbVec.push_back(&fun->getEntryBlock());
132 while(!bbVec.empty())
133 {
134 const BasicBlock* bb = bbVec.back();
135 bbVec.pop_back();
136 if (basicBlockHasRetInst(bb))
137 {
138 return false;
139 }
140
141 for (succ_const_iterator sit = succ_begin(bb), esit = succ_end(bb);
142 sit != esit; ++sit)
143 {
144 const BasicBlock* succbb = (*sit);
145 if(visited.find(succbb)==visited.end())
146 visited.insert(succbb);
147 else
148 continue;
149 bbVec.push_back(succbb);
150 }
151 }
152 return true;
153}

250const Value* LLVMUtil::stripAllCasts(const Value* val)
251{
252 while (true)
253 {
254 if (const CastInst *ci = SVFUtil::dyn_cast<CastInst>(val))
255 {
256 val = ci->getOperand(0);
257 }
258 else if (const ConstantExpr *ce = SVFUtil::dyn_cast<ConstantExpr>(val))
259 {
260 if(ce->isCast())
261 val = ce->getOperand(0);
262 else
263 return val;
264 }
265 else

267 return val;

269 }
270 return nullptr;

273/*
274 * Get the first dominated cast instruction for heap allocations since they typically come from void* (i8*)
275 * for example, %4 = call align 16 i8* @malloc(i64 10); %5 = bitcast i8* %4 to i32*
276 * return %5 whose type is i32* but not %4 whose type is i8*
277 */
278const Value* LLVMUtil::getFirstUseViaCastInst(const Value* val)
279{
280 assert(SVFUtil::isa<PointerType>(val->getType()) && "this value should be a pointer type!");
282 const Value *latestUse = nullptr;
283 for (const auto &it : val->uses())
284 {
285 if (SVFUtil::isa<BitCastInst>(it.getUser()))
286 latestUse = it.getUser();
287 else
288 latestUse = nullptr;
289 }
290 return latestUse;
291}

315bool LLVMUtil::isIRFile(const std::string &filename)
316{
317 llvm::LLVMContext context;
318 llvm::SMDiagnostic err;
319
320 // Parse the input LLVM IR file into a module
321 std::unique_ptr<llvm::Module> module = llvm::parseIRFile(filename, err, context);
322
323 // Check if the parsing succeeded
324 if (!module)
325 {
326 err.print("isIRFile", llvm::errs());
327 return false; // Not an LLVM IR file
328 }
329
330 return true; // It is an LLVM IR file
331}

336void LLVMUtil::processArguments(int argc, char **argv, int &arg_num, char **arg_value,
337 std::vector<std::string> &moduleNameVec)
338{
339 bool first_ir_file = true;
340 for (int i = 0; i < argc; ++i)
341 {
342 std::string argument(argv[i]);
343 if (LLVMUtil::isIRFile(argument))
344 {
345 if (find(moduleNameVec.begin(), moduleNameVec.end(), argument)
346 == moduleNameVec.end())
347 moduleNameVec.push_back(argument);
348 if (first_ir_file)
349 {
350 arg_value[arg_num] = argv[i];
351 arg_num++;
352 first_ir_file = false;
353 }
354 }
355 else
356 {
357 arg_value[arg_num] = argv[i];
358 arg_num++;
359 }
360 }

407std::string LLVMUtil::restoreFuncName(std::string funcName)
408{
409 assert(!funcName.empty() && "Empty function name");
410 // Some function names change due to mangling, such as "fopen" to "\01_fopen" on macOS.
411 // Since C function names cannot include '.', change the function name from llvm.memcpy.p0i8.p0i8.i64 to llvm_memcpy_p0i8_p0i8_i64."
412 bool hasSpecialPrefix = funcName[0] == '\01';
413 bool hasDot = funcName.find('.') != std::string::npos;
414
415 if (!hasDot && !hasSpecialPrefix)
416 return funcName;
417
418 // Remove prefix "\01_" or "\01"
419 if (hasSpecialPrefix)
420 {
421 const std::string prefix1 = "\01_";
422 const std::string prefix2 = "\01";
423 if (funcName.substr(0, prefix1.length()) == prefix1)
424 funcName = funcName.substr(prefix1.length());
425 else if (funcName.substr(0, prefix2.length()) == prefix2)
426 funcName = funcName.substr(prefix2.length());
427 }
428 // Replace '.' with '_'
429 if (hasDot)
430 std::replace(funcName.begin(), funcName.end(), '.', '_');
431
432 return funcName;
433}

456const std::string LLVMUtil::getSourceLoc(const Value* val )
457{
458 if(val==nullptr) return "{ empty val }";
459
460 std::string str;
461 std::stringstream rawstr(str);
462 rawstr << "{ ";
463
464 if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
465 {
466 if (SVFUtil::isa<AllocaInst>(inst))
467 {
468 for (llvm::DbgInfoIntrinsic *DII : FindDbgDeclareUses(const_cast<Instruction*>(inst)))
469 {
470 if (llvm::DbgDeclareInst *DDI = SVFUtil::dyn_cast<llvm::DbgDeclareInst>(DII))
471 {
472 llvm::DIVariable *DIVar = SVFUtil::cast<llvm::DIVariable>(DDI->getVariable());
473 rawstr << "\"ln\": " << DIVar->getLine() << ", \"fl\": \"" << DIVar->getFilename().str() << "\"";
474 break;
475 }
476 }
477 }
478 else if (MDNode *N = inst->getMetadata("dbg")) // Here I is an LLVM instruction
479 {
480 llvm::DILocation* Loc = SVFUtil::cast<llvm::DILocation>(N); // DILocation is in DebugInfo.h
481 unsigned Line = Loc->getLine();
482 unsigned Column = Loc->getColumn();
483 std::string File = Loc->getFilename().str();
484 //StringRef Dir = Loc.getDirectory();
485 if(File.empty() || Line == 0)
486 {
487 auto inlineLoc = Loc->getInlinedAt();
488 if(inlineLoc)
489 {
490 Line = inlineLoc->getLine();
491 Column = inlineLoc->getColumn();
492 File = inlineLoc->getFilename().str();
493 }
494 }
495 rawstr << "\"ln\": " << Line << ", \"cl\": " << Column << ", \"fl\": \"" << File << "\"";
496 }
497 }
498 else if (const Argument* argument = SVFUtil::dyn_cast<Argument>(val))
499 {
500 if (argument->getArgNo()%10 == 1)
501 rawstr << argument->getArgNo() << "st";
502 else if (argument->getArgNo()%10 == 2)
503 rawstr << argument->getArgNo() << "nd";
504 else if (argument->getArgNo()%10 == 3)
505 rawstr << argument->getArgNo() << "rd";
506 else
507 rawstr << argument->getArgNo() << "th";
508 rawstr << " arg " << argument->getParent()->getName().str() << " "
509 << getSourceLocOfFunction(argument->getParent());
510 }
511 else if (const GlobalVariable* gvar = SVFUtil::dyn_cast<GlobalVariable>(val))
512 {
513 rawstr << "Glob ";
514 NamedMDNode* CU_Nodes = gvar->getParent()->getNamedMetadata("llvm.dbg.cu");
515 if(CU_Nodes)
516 {
517 for (unsigned i = 0, e = CU_Nodes->getNumOperands(); i != e; ++i)
518 {
519 llvm::DICompileUnit *CUNode = SVFUtil::cast<llvm::DICompileUnit>(CU_Nodes->getOperand(i));
520 for (llvm::DIGlobalVariableExpression *GV : CUNode->getGlobalVariables())
521 {
522 llvm::DIGlobalVariable * DGV = GV->getVariable();
523
524 if(DGV->getName() == gvar->getName())
525 {
526 rawstr << "\"ln\": " << DGV->getLine() << ", \"fl\": \"" << DGV->getFilename().str() << "\"";
527 }
528
529 }
530 }
531 }
532 }
533 else if (const Function* func = SVFUtil::dyn_cast<Function>(val))
534 {
535 rawstr << getSourceLocOfFunction(func);
536 }
537 else if (const BasicBlock* bb = SVFUtil::dyn_cast<BasicBlock>(val))
538 {
539 rawstr << "\"basic block\": " << bb->getName().str() << ", \"location\": " << getSourceLoc(bb->getFirstNonPHI());
540 }
541 else if(LLVMUtil::isConstDataOrAggData(val))
542 {
543 rawstr << "constant data";
544 }
545 else
546 {
547 rawstr << "N/A";
548 }
549 rawstr << " }";
550
551 if(rawstr.str()=="{ }")
552 return "";
553 return rawstr.str();
554}

577void LLVMUtil::getNextInsts(const Instruction* curInst, std::vector<const Instruction*>& instList)
578{
579 if (!curInst->isTerminator())
580 {
581 const Instruction* nextInst = curInst->getNextNode();
582 if (LLVMUtil::isIntrinsicInst(nextInst))
583 getNextInsts(nextInst, instList);
584 else
585 instList.push_back(nextInst);
586 }
587 else
588 {
589 const BasicBlock *BB = curInst->getParent();
590 // Visit all successors of BB in the CFG
591 for (succ_const_iterator it = succ_begin(BB), ie = succ_end(BB); it != ie; ++it)
592 {
593 const Instruction* nextInst = &((*it)->front());
594 if (LLVMUtil::isIntrinsicInst(nextInst))
595 getNextInsts(nextInst, instList);
596 else
597 instList.push_back(nextInst);
598 }
599 }
600}

694bool LLVMUtil::isHeapObj(const Value* val)
695{
696 // Check if the value is an argument in the program entry function
697 if (ArgInProgEntryFunction(val))
698 {
699 // Return true if the value does not have a first use via cast instruction
700 return !getFirstUseViaCastInst(val);
701 }
702 // Check if the value is an instruction and if it is a heap allocation external call
703 else if (SVFUtil::isa<Instruction>(val) &&
704 LLVMUtil::isHeapAllocExtCall(SVFUtil::cast<Instruction>(val)))
705 {
706 return true;
707 }
708 // Return false if none of the above conditions are met
709 return false;
710}
