70 {
71  assert(!SVFUtil::isExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun)) && "The calling function cannot be an external function.");
72  //initial DominatorTree
73  DominatorTree dt;
74  dt.recalculate(const_cast<Function&>(*fun));
75 
76  Set<const BasicBlock*> visited;
77  std::vector<const BasicBlock*> bbVec;
78  bbVec.push_back(&fun->getEntryBlock());
79  while(!bbVec.empty())
80  {
81  const BasicBlock* bb = bbVec.back();
82  bbVec.pop_back();
83  const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
84  reachableBBs.push_back(svfbb);
85  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))

88  DI != DE; ++DI)

92  visited.insert(succbb);

461  if (glob == nullptr || !glob->hasInitializer())

465  if (ca == nullptr)

468  for (unsigned i = 0; i < ca->getNumOperands(); ++i)
469  {

471  if (structAn == nullptr)
472  continue;

475  if (expr == nullptr || expr->getOpcode() != Instruction::BitCast || expr->getOperand(0) != fun)
476  continue;
477

479  if (note->getOpcode() != Instruction::GetElementPtr)
480  continue;

483  if (annotateStr == nullptr || !annotateStr->hasInitializer())
484  continue;

487  if (data->isString())
488  {
489  std::string annotation = data->getAsString().str();
490  if (!annotation.empty())
491  annotations.push_back(annotation);
492  }
493  }
494  return annotations;

497 void LLVMUtil::removeFunAnnotations(std::vector<Function*>& removedFuncList)
498 {
499  if (removedFuncList.empty())
500  return; // No functions to remove annotations in extapi.bc module
501 
502  Module* module = removedFuncList[0]->getParent();
503  GlobalVariable* glob = module->getGlobalVariable("llvm.global.annotations");
504  if (glob == nullptr || !glob->hasInitializer())
505  return;
506 
507  ConstantArray* ca = SVFUtil::dyn_cast<ConstantArray>(glob->getInitializer());
508  if (ca == nullptr)
509  return;
510 
511  std::vector<Constant*> newAnnotations;
512  for (unsigned i = 0; i < ca->getNumOperands(); ++i)
513  {
514  ConstantStruct* structAn = SVFUtil::dyn_cast<ConstantStruct>(ca->getOperand(i));
515  if (structAn == nullptr)
516  continue;
517 
518  ConstantExpr* expr = SVFUtil::dyn_cast<ConstantExpr>(structAn->getOperand(0));
519  if (expr == nullptr || expr->getOpcode() != Instruction::BitCast)
520  {
521  // Keep the annotation if it's not created using BitCast
522  newAnnotations.push_back(structAn);
523  continue;
524  }
525 
526  Function* annotatedFunc = SVFUtil::dyn_cast<Function>(expr->getOperand(0));
527  if (annotatedFunc == nullptr || std::find(removedFuncList.begin(), removedFuncList.end(), annotatedFunc) != removedFuncList.end())
528  continue;
529 
530  // Keep the annotation for all other functions
531  newAnnotations.push_back(structAn);
532  }
533 
534  if (newAnnotations.size() == ca->getNumOperands())
535  return; // No annotations to remove
536 
537  ArrayType* annotationsType = ArrayType::get(ca->getType()->getElementType(), newAnnotations.size());
538  Constant* newCA = ConstantArray::get(annotationsType, newAnnotations);
539 
540  // Check if a global variable with the name llvm.global.annotations already exists
541  GlobalVariable* existingGlobal = module->getGlobalVariable("llvm.global.annotations");
542  if (existingGlobal)
543  // Rename the existing llvm.global.annotations to llvm.global.annotations_old
544  existingGlobal->setName("llvm.global.annotations_old");
545 
546  // Create a new global variable with the updated annotations
547  GlobalVariable* newGlobal = new GlobalVariable(*module, newCA->getType(), glob->isConstant(),
548  glob->getLinkage(), newCA, "llvm.global.annotations", glob, glob->getThreadLocalMode());
549 
550  // Copy other properties from the old global variable to the new one
551  newGlobal->setSection(glob->getSection());
552  newGlobal->setAlignment(llvm::MaybeAlign(glob->getAlignment()));
553 
554  // Remove the old global variable
555  glob->eraseFromParent();
556 }

682 
683  std::string str;
684  std::stringstream rawstr(str);
685  rawstr << "{ ";
686 
687  if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
688  {
689  if (SVFUtil::isa<AllocaInst>(inst))
690  {

700  }
701  else if (MDNode *N = inst->getMetadata("dbg")) // Here I is an LLVM instruction
702  {
703  llvm::DILocation* Loc = SVFUtil::cast<llvm::DILocation>(N); // DILocation is in DebugInfo.h
704  unsigned Line = Loc->getLine();
705  unsigned Column = Loc->getColumn();
706  std::string File = Loc->getFilename().str();
707  //StringRef Dir = Loc.getDirectory();
708  if(File.empty() || Line == 0)
709  {
710  auto inlineLoc = Loc->getInlinedAt();
711  if(inlineLoc)
712  {
713  Line = inlineLoc->getLine();
714  Column = inlineLoc->getColumn();
715  File = inlineLoc->getFilename().str();
716  }
717  }
718  rawstr << "\"ln\": " << Line << ", \"cl\": " << Column << ", \"fl\": \"" << File << "\"";
719  }
720  }
