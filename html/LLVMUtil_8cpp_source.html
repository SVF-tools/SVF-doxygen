84 void LLVMUtil::getFunReachableBBs (const Function* fun, std::vector<const SVFBasicBlock*> &reachableBBs)
85 {
86  assert(!SVFUtil::isExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun)) && "The calling function cannot be an external function.");
87  //initial DominatorTree
88  DominatorTree dt;
89  dt.recalculate(const_cast<Function&>(*fun));
90 
91  Set<const BasicBlock*> visited;
92  std::vector<const BasicBlock*> bbVec;
93  bbVec.push_back(&fun->getEntryBlock());
94  while(!bbVec.empty())
95  {
96  const BasicBlock* bb = bbVec.back();
97  bbVec.pop_back();
98  const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
99  reachableBBs.push_back(svfbb);
100  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
101  {
102  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
103  DI != DE; ++DI)
104  {
105  const BasicBlock* succbb = (*DI)->getBlock();
106  if(visited.find(succbb)==visited.end())
107  visited.insert(succbb);
108  else
109  continue;
110  bbVec.push_back(succbb);
111  }
112  }
113  }
114 }

169 bool LLVMUtil::isUncalledFunction (const Function* fun)
170 {
171  if(fun->hasAddressTaken())
172  return false;
173  if (LLVMUtil::isProgEntryFunction(fun))
174  return false;
175  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
176  {
177  if (LLVMUtil::isCallSite(*i))
178  return false;
179  }
180  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
181  {
182  const LLVMModuleSet::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
183  for (LLVMModuleSet::FunctionSetType::const_iterator it = decls.begin(),
184  eit = decls.end(); it != eit; ++it)
185  {
186  const Function* decl = *it;
187  if(decl->hasAddressTaken())
188  return false;
189  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
190  {
191  if (LLVMUtil::isCallSite(*i))
192  return false;
193  }
194  }
195  }
196  return true;

302 void LLVMUtil::getNextInsts(const Instruction* curInst, std::vector<const SVFInstruction*>& instList)
303 {
304  if (!curInst->isTerminator())
305  {
306  const Instruction* nextInst = curInst->getNextNode();
307  const SVFInstruction* svfNextInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(nextInst);
308  if (LLVMUtil::isIntrinsicInst(nextInst))
309  getNextInsts(nextInst, instList);
310  else
311  instList.push_back(svfNextInst);
312  }
313  else
314  {
315  const BasicBlock* BB = curInst->getParent();
316  // Visit all successors of BB in the CFG
317  for (succ_const_iterator it = succ_begin(BB), ie = succ_end(BB); it != ie; ++it)
318  {
319  const Instruction* nextInst = &((*it)->front());
320  const SVFInstruction* svfNextInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(nextInst);
321  if (LLVMUtil::isIntrinsicInst(nextInst))
322  getNextInsts(nextInst, instList);
323  else
324  instList.push_back(svfNextInst);
325  }
326  }

457 void LLVMUtil::removeFunAnnotations(Set<Function*>& removedFuncList)
458 {
459  if (removedFuncList.empty())
460  return; // No functions to remove annotations in extapi.bc module
461 
462  Module* module = (*removedFuncList.begin())->getParent();
463  GlobalVariable* glob = module->getGlobalVariable("llvm.global.annotations");
464  if (glob == nullptr || !glob->hasInitializer())
465  return;
466 
467  ConstantArray* ca = SVFUtil::dyn_cast<ConstantArray>(glob->getInitializer());
468  if (ca == nullptr)
469  return;
470 
471  std::vector<Constant*> newAnnotations;
472  for (unsigned i = 0; i < ca->getNumOperands(); ++i)
473  {
474  ConstantStruct* structAn = SVFUtil::dyn_cast<ConstantStruct>(ca->getOperand(i));
475  if (structAn == nullptr)
476  continue;
477 
478  Function* annotatedFunc = nullptr;
479 
480  // Non-opague pointer, try to cast to ConstantExpr and check for BitCast
481  if (ConstantExpr* expr = SVFUtil::dyn_cast<ConstantExpr>(structAn->getOperand(0)))
482  {
483  if (expr->getOpcode() == Instruction::BitCast)
484  {
485  annotatedFunc = SVFUtil::dyn_cast<Function>(expr->getOperand(0));
486  }
487  }
488 
489  // Opague pointer, If the above method didn't work, try casting directly to Function
490  if (!annotatedFunc)
491  {
492  annotatedFunc = SVFUtil::dyn_cast<Function>(structAn->getOperand(0));
493  }
494 
495  // Process the annotated function if it's not in the removed list
496  if (annotatedFunc && std::find(removedFuncList.begin(), removedFuncList.end(), annotatedFunc) == removedFuncList.end())
497  {
498  newAnnotations.push_back(structAn);
499  }
500  }
501 
502  if (newAnnotations.size() == ca->getNumOperands())
503  return; // No annotations to remove
504 
505  ArrayType* annotationsType = ArrayType::get(ca->getType()->getElementType(), newAnnotations.size());
506  Constant* newCA = ConstantArray::get(annotationsType, newAnnotations);
507 
508  // Check if a global variable with the name llvm.global.annotations already exists
509  GlobalVariable* existingGlobal = module->getGlobalVariable("llvm.global.annotations");
510  if (existingGlobal)
511  // Rename the existing llvm.global.annotations to llvm.global.annotations_old
512  existingGlobal->setName("llvm.global.annotations_old");
513 
514  // Create a new global variable with the updated annotations
515  GlobalVariable* newGlobal = new GlobalVariable(*module, newCA->getType(), glob->isConstant(),
516  glob->getLinkage(), newCA, "llvm.global.annotations", glob, glob->getThreadLocalMode());
517 
518  // Copy other properties from the old global variable to the new one
519  newGlobal->setSection(glob->getSection());
520  newGlobal->setAlignment(llvm::MaybeAlign(glob->getAlignment()));
521 
522  // Remove the old global variable
523  glob->eraseFromParent();
524 }

577 {
578  if (removedFuncList.empty())
579  return;
580 
581  Module* mod = (*removedFuncList.begin())->getParent();
582  if (mod->getName().str() != ExtAPI::getExtAPI()->getExtBcPath())
583  return;

586  LLVMUtil::removeFunAnnotations(removedFuncList);

597  {

599  for (auto& use : F->uses())

601  llvm::User* user = use.getUser();
602  if (llvm::isa<llvm::CallBase>(user))

615  func->eraseFromParent();

617  // Delete first kind functions
618  for (Function* func : funcsToKeep)
619  func->eraseFromParent();

662 const std::string LLVMUtil::getSourceLoc(const Value* val )
663 {
664  if(val==nullptr) return "{ empty val }";
665 
666  std::string str;
667  std::stringstream rawstr(str);
668  rawstr << "{ ";
669 
670  if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
671  {
672  if (SVFUtil::isa<AllocaInst>(inst))
673  {
674  for (llvm::DbgInfoIntrinsic *DII : FindDbgAddrUses(const_cast<Instruction*>(inst)))
675  {
676  if (llvm::DbgDeclareInst *DDI = SVFUtil::dyn_cast<llvm::DbgDeclareInst>(DII))
677  {
678  llvm::DIVariable *DIVar = SVFUtil::cast<llvm::DIVariable>(DDI->getVariable());
679  rawstr << "\"ln\": " << DIVar->getLine() << ", \"fl\": \"" << DIVar->getFilename().str() << "\"";
680  break;
681  }
682  }
683  }
684  else if (MDNode *N = inst->getMetadata("dbg")) // Here I is an LLVM instruction
685  {
686  llvm::DILocation* Loc = SVFUtil::cast<llvm::DILocation>(N); // DILocation is in DebugInfo.h
687  unsigned Line = Loc->getLine();
688  unsigned Column = Loc->getColumn();
689  std::string File = Loc->getFilename().str();
690  //StringRef Dir = Loc.getDirectory();
691  if(File.empty() || Line == 0)
692  {
693  auto inlineLoc = Loc->getInlinedAt();
694  if(inlineLoc)
695  {
696  Line = inlineLoc->getLine();
697  Column = inlineLoc->getColumn();
698  File = inlineLoc->getFilename().str();
699  }
700  }
701  rawstr << "\"ln\": " << Line << ", \"cl\": " << Column << ", \"fl\": \"" << File << "\"";
702  }
703  }
704  else if (const Argument* argument = SVFUtil::dyn_cast<Argument>(val))
705  {
706  if (argument->getArgNo()%10 == 1)
707  rawstr << argument->getArgNo() << "st";
708  else if (argument->getArgNo()%10 == 2)
709  rawstr << argument->getArgNo() << "nd";
710  else if (argument->getArgNo()%10 == 3)
711  rawstr << argument->getArgNo() << "rd";
712  else
713  rawstr << argument->getArgNo() << "th";
714  rawstr << " arg " << argument->getParent()->getName().str() << " "
715  << getSourceLocOfFunction(argument->getParent());
716  }
717  else if (const GlobalVariable* gvar = SVFUtil::dyn_cast<GlobalVariable>(val))
718  {
719  rawstr << "Glob ";
720  NamedMDNode* CU_Nodes = gvar->getParent()->getNamedMetadata("llvm.dbg.cu");
721  if(CU_Nodes)
722  {
723  for (unsigned i = 0, e = CU_Nodes->getNumOperands(); i != e; ++i)
724  {
