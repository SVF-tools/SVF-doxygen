64 template <class S, class T> struct Hash<std::pair<S, T>>
65 {
66  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
67  static size_t szudzik(size_t a, size_t b)
68  {
69  return a > b ? b * b + a : a * a + a + b;
70  }
71 
72  size_t operator()(const std::pair<S, T>& t) const
73  {
74  Hash<decltype(t.first)> first;
75  Hash<decltype(t.second)> second;
76  return szudzik(first(t.first), second(t.second));
77  }
