279  inline virtual bool pushContext(NodeID ctx)
280  {
281 
282  if(context.size() < maximumCxtLen)
283  {
284  context.push_back(ctx);
285 
286  if(context.size() > maximumCxt)
287  maximumCxt = context.size();
288  return true;
289  }

293  {

297  }
298  return false;
