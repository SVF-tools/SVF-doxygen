73template <typename Key, typename KeySet, typename Data, typename DataSet>
74class PTData
75{
76public:
78 enum PTDataTy
79 {

81 MutBase,

83 MutDiff,
84 DataFlow,
85 MutDataFlow,
86 MutIncDataFlow,

89 PersBase,

96 PTData(bool reversePT = true, PTDataTy ty = PTDataTy::Base) : rev(reversePT), ptdTy(ty) { }
