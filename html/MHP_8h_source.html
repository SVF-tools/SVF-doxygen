172 inline void rmInterleavingThread(const CxtThreadStmt& tgr, const NodeBS& tids, const ICFGNode* joinsite)
173 {
174 NodeBS joinedTids;
175 for(NodeBS::iterator it = tids.begin(), eit = tids.end(); it!=eit; ++it)
176 {
177 if(isMustJoin(tgr.getTid(),joinsite))
178 joinedTids.set(*it);
179 }
180 if(threadStmtToThreadInterLeav[tgr].intersectWithComplement(joinedTids))
181 {
182 pushToCTSWorkList(tgr);
183 }

408 void markCxtStmtFlag(const CxtStmt& tgr, ValDomain flag)
409 {
410 ValDomain flag_tgr = getMarkedFlag(tgr);
411 cxtStmtToAliveFlagMap[tgr] = flag;
412 if(flag_tgr!=getMarkedFlag(tgr))
413 {
414 instToCxtStmt[tgr.getStmt()].insert(tgr);
415 pushToCTSWorkList(tgr);
416 }
417
418 }
420 void markCxtStmtFlag(const CxtStmt& tgr, const CxtStmt& src)
421 {
422 ValDomain flag_tgr = getMarkedFlag(tgr);
423 ValDomain flag_src = getMarkedFlag(src);
424 if(flag_tgr == Empty)
425 {
426 cxtStmtToAliveFlagMap[tgr] = flag_src;
427 }
428 else if(flag_tgr == TDDead)
429 {
430 if(flag_src==TDAlive)
431 cxtStmtToAliveFlagMap[tgr] = TDAlive;
432 }
433 else
434 {
436 }
437 if(flag_tgr!=getMarkedFlag(tgr))
438 {
439 instToCxtStmt[tgr.getStmt()].insert(tgr);
440 pushToCTSWorkList(tgr);
441 }
