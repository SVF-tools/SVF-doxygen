269  auto sse_strlen = [&](const CallICFGNode *callNode)
270  {
271  // check the arg size
272  if (callNode->arg_size() < 1) return;
273  const SVFValue* strValue = callNode->getArgument(0);
274  AbstractState& as = getAbsStateFromTrace(callNode);
275  NodeID value_id = svfir->getValueNode(strValue);
276  u32_t lhsId = svfir->getValueNode(callNode->getCallSite());
277  u32_t dst_size = 0;
278  for (const auto& addr : as[value_id].getAddrs())
279  {
280  NodeID objId = AbstractState::getInternalID(addr);
281  if (svfir->getBaseObj(objId)->isConstantByteSize())
282  {
283  dst_size = svfir->getBaseObj(objId)->getByteSizeOfObj();
284  }
285  else
286  {
287  const ICFGNode* addrNode = SVFUtil::cast<ICFGNode>(svfir->getBaseObj(objId)->getGNode());
288  for (const SVFStmt* stmt2: addrNode->getSVFStmts())
289  {
290  if (const AddrStmt* addrStmt = SVFUtil::dyn_cast<AddrStmt>(stmt2))
291  {
292  dst_size = as.getAllocaInstByteSize(addrStmt);
293  }
294  }
295  }
296  }
297  u32_t len = 0;
298  NodeID dstid = svfir->getValueNode(strValue);
299  if (as.inVarToAddrsTable(dstid))
300  {
301  for (u32_t index = 0; index < dst_size; index++)
302  {
303  AbstractValue expr0 =
304  as.getGepObjAddrs(dstid, IntervalValue(index));
305  AbstractValue val;
306  for (const auto &addr: expr0.getAddrs())
307  {
308  val.join_with(as.load(addr));
309  }
310  if (val.getInterval().is_numeral() && (char) val.getInterval().getIntNumeral() == '\0')
311  {
312  break;
313  }
314  ++len;
315  }
316  }
317  if (len == 0)
318  {
319  as[lhsId] = IntervalValue((s64_t)0, (s64_t)Options::MaxFieldLimit());
320  }
321  else
322  {
323  as[lhsId] = IntervalValue(len);
324  }
325  };
326  func_map["strlen"] = sse_strlen;
327  func_map["wcslen"] = sse_strlen;

356 std::string AbsExtAPI::strRead(AbstractState& as, const SVFVar* rhs)
357 {
358  // sse read string nodeID->string
359  std::string str0;
360 
361  for (u32_t index = 0; index < Options::MaxFieldLimit(); index++)
362  {
363  // dead loop for string and break if there's a \0. If no \0, it will throw err.
364  if (!as.inVarToAddrsTable(rhs->getId())) continue;
365  AbstractValue expr0 =
366  as.getGepObjAddrs(rhs->getId(), IntervalValue(index));
367 
368  AbstractValue val;
369  for (const auto &addr: expr0.getAddrs())
370  {
371  val.join_with(as.load(addr));
372  }
373  if (!val.getInterval().is_numeral())
374  {
375  break;
376  }
377  if ((char) val.getInterval().getIntNumeral() == '\0')
378  {
379  break;
380  }
381  str0.push_back((char) val.getInterval().getIntNumeral());
382  }
383  return str0;
384 }

386 void AbsExtAPI::handleExtAPI(const CallICFGNode *call)
387 {
388  AbstractState& as = getAbsStateFromTrace(call);
389  const SVFFunction *fun = call->getCalledFunction();
390  assert(fun && "SVFFunction* is nullptr");
391  ExtAPIType extType = UNCLASSIFIED;
392  // get type of mem api
393  for (const std::string &annotation: ExtAPI::getExtAPI()->getExtFuncAnnotations(fun))
394  {
395  if (annotation.find("MEMCPY") != std::string::npos)
396  extType = MEMCPY;
397  if (annotation.find("MEMSET") != std::string::npos)
398  extType = MEMSET;
399  if (annotation.find("STRCPY") != std::string::npos)
400  extType = STRCPY;
401  if (annotation.find("STRCAT") != std::string::npos)
402  extType = STRCAT;
403  }
404  if (extType == UNCLASSIFIED)
405  {
406  if (func_map.find(fun->getName()) != func_map.end())
407  {
408  func_map[fun->getName()](call);
409  }
410  else
411  {
412  u32_t lhsId = svfir->getValueNode(call->getCallSite());
413  if (as.inVarToAddrsTable(lhsId))
414  {
415 
416  }
417  else
418  {
419  as[lhsId] = IntervalValue();
420  }
421  return;
422  }
423  }
424  // 1. memcpy functions like memcpy_chk, strncpy, annotate("MEMCPY"), annotate("BUF_CHECK:Arg0, Arg2"), annotate("BUF_CHECK:Arg1, Arg2")
425  else if (extType == MEMCPY)
426  {
427  IntervalValue len = as[svfir->getValueNode(call->getArgument(2))].getInterval();
428  svfir->getGNode(svfir->getValueNode(call->getArgument(0)));
429  handleMemcpy(as, getSVFVar(call->getArgument(0)), getSVFVar(call->getArgument(1)), len, 0);
430  }
431  else if (extType == MEMSET)
432  {
433  // memset dst is arg0, elem is arg1, size is arg2
434  IntervalValue len = as[svfir->getValueNode(call->getArgument(2))].getInterval();
435  IntervalValue elem = as[svfir->getValueNode(call->getArgument(1))].getInterval();
436  handleMemset(as, getSVFVar(call->getArgument(0)), elem, len);
437  }
438  else if (extType == STRCPY)
439  {
440  handleStrcpy(call);
441  }
442  else if (extType == STRCAT)
443  {
444  handleStrcat(call);
445  }
446  else
447  {
448 
449  }
450  return;
451 }

465 IntervalValue AbsExtAPI::getStrlen(AbstractState& as, const SVF::SVFVar *strValue)
466 {
467  NodeID value_id = strValue->getId();
468  u32_t dst_size = 0;
469  for (const auto& addr : as[value_id].getAddrs())
470  {
471  NodeID objId = AbstractState::getInternalID(addr);
472  if (svfir->getBaseObj(objId)->isConstantByteSize())
473  {
474  dst_size = svfir->getBaseObj(objId)->getByteSizeOfObj();
475  }
476  else
477  {
478  const ICFGNode* icfgNode = SVFUtil::cast<ICFGNode>( svfir->getBaseObj(objId)->getGNode());
479  for (const SVFStmt* stmt2: icfgNode->getSVFStmts())
480  {
481  if (const AddrStmt* addrStmt = SVFUtil::dyn_cast<AddrStmt>(stmt2))
482  {
483  dst_size = as.getAllocaInstByteSize(addrStmt);
484  }
485  }
486  }
487  }
488  u32_t len = 0;
489  u32_t elemSize = 1;
490  if (as.inVarToAddrsTable(value_id))
491  {
492  for (u32_t index = 0; index < dst_size; index++)
493  {
494  AbstractValue expr0 =
495  as.getGepObjAddrs(value_id, IntervalValue(index));
496  AbstractValue val;
497  for (const auto &addr: expr0.getAddrs())
498  {
499  val.join_with(as.load(addr));
500  }
501  if (val.getInterval().is_numeral() && (char) val.getInterval().getIntNumeral() == '\0')
502  {
503  break;
504  }
505  ++len;
506  }
507  if (strValue->getType()->isArrayTy())
508  {
509  elemSize = SVFUtil::dyn_cast<SVFArrayType>(strValue->getType())->getTypeOfElement()->getByteSize();
510  }
511  else if (strValue->getType()->isPointerTy())
512  {
513  if (const SVFType* elemType = as.getPointeeElement(value_id))
514  {
515  if (elemType->isArrayTy())
516  elemSize = SVFUtil::dyn_cast<SVFArrayType>(elemType)->getTypeOfElement()->getByteSize();
517  else
518  elemSize = elemType->getByteSize();
519  }
520  else
521  {
522  elemSize = 1;
523  }
524  }
525  else
526  {
527  assert(false && "we cannot support this type");
528  }
529  }
530  if (len == 0)
531  {
532  return IntervalValue((s64_t)0, (s64_t)Options::MaxFieldLimit());
533  }
534  else
535  {
536  return IntervalValue(len * elemSize);
537  }
538 }

543  // __strcat_chk, strcat, __wcscat_chk, wcscat, __strncat_chk, strncat, __wcsncat_chk, wcsncat
544  // to check it is strcat group or strncat group
545  AbstractState& as = getAbsStateFromTrace(call);
546  const SVFFunction *fun = call->getCalledFunction();
547  const std::vector<std::string> strcatGroup = {"__strcat_chk", "strcat", "__wcscat_chk", "wcscat"};
548  const std::vector<std::string> strncatGroup = {"__strncat_chk", "strncat", "__wcsncat_chk", "wcsncat"};
549  if (std::find(strcatGroup.begin(), strcatGroup.end(), fun->getName()) != strcatGroup.end())
550  {
551  const SVFVar* arg0Val = getSVFVar(call->getArgument(0));
552  const SVFVar* arg1Val = getSVFVar(call->getArgument(1));
553  IntervalValue strLen0 = getStrlen(as, arg0Val);
554  IntervalValue strLen1 = getStrlen(as, arg1Val);
555  IntervalValue totalLen = strLen0 + strLen1;
556  handleMemcpy(as, arg0Val, arg1Val, strLen1, strLen0.lb().getIntNumeral());
557  // do memcpy
558  }
559  else if (std::find(strncatGroup.begin(), strncatGroup.end(), fun->getName()) != strncatGroup.end())
560  {
561  const SVFVar* arg0Val = getSVFVar(call->getArgument(0));
562  const SVFVar* arg1Val = getSVFVar(call->getArgument(1));
563  const SVFVar* arg2Val = getSVFVar(call->getArgument(2));
564  IntervalValue arg2Num = as[arg2Val->getId()].getInterval();
565  IntervalValue strLen0 = getStrlen(as, arg0Val);
566  IntervalValue totalLen = strLen0 + arg2Num;
567  handleMemcpy(as, arg0Val, arg1Val, arg2Num, strLen0.lb().getIntNumeral());
568  // do memcpy
569  }
570  else
571  {
572  assert(false && "unknown strcat function, please add it to strcatGroup or strncatGroup");

576 void AbsExtAPI::handleMemcpy(AbstractState& as, const SVF::SVFVar *dst, const SVF::SVFVar *src, IntervalValue len, u32_t start_idx)
577 {
578  u32_t dstId = dst->getId(); // pts(dstId) = {objid} objbar objtypeinfo->getType().
579  u32_t srcId = src->getId();
580  u32_t elemSize = 1;
581  if (dst->getType()->isArrayTy())
582  {
583  elemSize = SVFUtil::dyn_cast<SVFArrayType>(dst->getType())->getTypeOfElement()->getByteSize();
584  }
585  // memcpy(i32*, i32*, 40)
586  else if (dst->getType()->isPointerTy())
587  {
588  if (const SVFType* elemType = as.getPointeeElement(dstId))
589  {
590  if (elemType->isArrayTy())
591  elemSize = SVFUtil::dyn_cast<SVFArrayType>(elemType)->getTypeOfElement()->getByteSize();
592  else
593  elemSize = elemType->getByteSize();
594  }
595  else
596  {
597  elemSize = 1;
598  }
599  }
600  else
601  {
602  assert(false && "we cannot support this type");
603  }
604  u32_t size = std::min((u32_t)Options::MaxFieldLimit(), (u32_t) len.lb().getIntNumeral());
605  u32_t range_val = size / elemSize;
606  if (as.inVarToAddrsTable(srcId) && as.inVarToAddrsTable(dstId))
607  {
608  for (u32_t index = 0; index < range_val; index++)
609  {
610  // dead loop for string and break if there's a \0. If no \0, it will throw err.
611  AbstractValue expr_src =
612  as.getGepObjAddrs(srcId, IntervalValue(index));
613  AbstractValue expr_dst =
614  as.getGepObjAddrs(dstId, IntervalValue(index + start_idx));
615  for (const auto &dst: expr_dst.getAddrs())
616  {
617  for (const auto &src: expr_src.getAddrs())
618  {
619  u32_t objId = AbstractState::getInternalID(src);
620  if (as.inAddrToValTable(objId))
621  {
622  as.store(dst, as.load(src));
623  }
624  else if (as.inAddrToAddrsTable(objId))
625  {
626  as.store(dst, as.load(src));
627  }
628  }
629  }
630  }
631  }

641  elemSize = SVFUtil::dyn_cast<SVFArrayType>(dst->getType())->getTypeOfElement()->getByteSize();

648  }

653  }

657  }
658

697 IntervalValue AbsExtAPI::getRangeLimitFromType(const SVFType* type)
698 {
699  if (const SVFIntegerType* intType = SVFUtil::dyn_cast<SVFIntegerType>(type))
700  {
701  u32_t bits = type->getByteSize() * 8;
702  s64_t ub = 0;
703  s64_t lb = 0;
704  if (bits >= 32)
705  {
706  if (intType->isSigned())
707  {
708  ub = static_cast<s64_t>(std::numeric_limits<s32_t>::max());
709  lb = static_cast<s64_t>(std::numeric_limits<s32_t>::min());
710  }
711  else
712  {
713  ub = static_cast<s64_t>(std::numeric_limits<u32_t>::max());
714  lb = static_cast<s64_t>(std::numeric_limits<u32_t>::min());
715  }
716  }
717  else if (bits == 16)
718  {
719  if (intType->isSigned())
720  {
721  ub = static_cast<s64_t>(std::numeric_limits<s16_t>::max());
722  lb = static_cast<s64_t>(std::numeric_limits<s16_t>::min());
723  }
724  else
725  {
726  ub = static_cast<s64_t>(std::numeric_limits<u16_t>::max());
727  lb = static_cast<s64_t>(std::numeric_limits<u16_t>::min());
728  }
729  }
730  else if (bits == 8)
731  {
732  if (intType->isSigned())
733  {
734  ub = static_cast<s64_t>(std::numeric_limits<int8_t>::max());
735  lb = static_cast<s64_t>(std::numeric_limits<int8_t>::min());
736  }
737  else
738  {
739  ub = static_cast<s64_t>(std::numeric_limits<u_int8_t>::max());
740  lb = static_cast<s64_t>(std::numeric_limits<u_int8_t>::min());
741  }
742  }
743  return IntervalValue(lb, ub);
744  }
745  else if (SVFUtil::isa<SVFOtherType>(type))
746  {
747  // handle other type like float double, set s32_t as the range
748  s64_t ub = static_cast<s64_t>(std::numeric_limits<s32_t>::max());
749  s64_t lb = static_cast<s64_t>(std::numeric_limits<s32_t>::min());
750  return IntervalValue(lb, ub);
751  }
752  else
753  {
754  return IntervalValue::top();
755  // other types, return top interval
756  }
757 }
