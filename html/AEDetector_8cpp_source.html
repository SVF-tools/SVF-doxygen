44 void BufOverflowDetector::detect(AbstractState& as, const ICFGNode* node)
45 {
46  if (!SVFUtil::isa<CallICFGNode>(node))
47  {
48  // Handle non-call nodes by analyzing GEP instructions
49  for (const SVFStmt* stmt : node->getSVFStmts())
50  {
51  if (const GepStmt* gep = SVFUtil::dyn_cast<GepStmt>(stmt))
52  {
53  SVFIR* svfir = PAG::getPAG();
54  NodeID lhs = gep->getLHSVarID();
55  NodeID rhs = gep->getRHSVarID();
56 
57  // Update the GEP object offset from its base
58  updateGepObjOffsetFromBase(as[lhs].getAddrs(), as[rhs].getAddrs(), as.getByteOffset(gep));
59 
60  IntervalValue baseObjSize = IntervalValue::bottom();
61  AddressValue objAddrs = as[gep->getRHSVarID()].getAddrs();
62  for (const auto& addr : objAddrs)
63  {
64  NodeID objId = AbstractState::getInternalID(addr);
65  u32_t size = 0;
66 
67  if (svfir->getBaseObj(objId)->isConstantByteSize())
68  {
69  size = svfir->getBaseObj(objId)->getByteSizeOfObj();
70  }
71  else
72  {
73  const ICFGNode* addrNode = SVFUtil::cast<ICFGNode>(
74  svfir->getBaseObj(objId)->getGNode());
75  for (const SVFStmt* stmt2 : addrNode->getSVFStmts())
76  {
77  if (const AddrStmt* addrStmt = SVFUtil::dyn_cast<AddrStmt>(stmt2))
78  {
79  size = as.getAllocaInstByteSize(addrStmt);
80  }
81  }
82  }
83 
84  // Calculate access offset and check for potential overflow
85  IntervalValue accessOffset = getAccessOffset(as, objId, gep);
86  if (accessOffset.ub().getIntNumeral() >= size)
87  {
88  AEException bug(stmt->toString());
89  addBugToReporter(bug, stmt->getICFGNode());
90  }
91  }
92  }
93  }
94  }
95  else
96  {
97  // Handle call nodes by checking for external API calls
98  const CallICFGNode* callNode = SVFUtil::cast<CallICFGNode>(node);
99  if (SVFUtil::isExtCall(callNode->getCalledFunction()))
100  {
101  detectExtAPI(as, callNode);
102  }
103  }
104 }

114 void BufOverflowDetector::handleStubFunctions(const SVF::CallICFGNode* callNode)
115 {
116  // get function name
117  SVFIR* svfir = PAG::getPAG();
118  std::string funcName = callNode->getCalledFunction()->getName();
119  if (funcName == "SAFE_BUFACCESS")
120  {
121  // void SAFE_BUFACCESS(void* data, int size);
122  AbstractInterpretation::getAEInstance().checkpoints.erase(callNode);
123  if (callNode->arg_size() < 2)
124  return;
125  AbstractState& as =
126  AbstractInterpretation::getAEInstance().getAbsStateFromTrace(
127  callNode);
128  u32_t size_id = svfir->getValueNode(callNode->getArgument(1));
129  IntervalValue val = as[size_id].getInterval();
130  if (val.isBottom())
131  {
132  val = IntervalValue(0);
133  assert(false && "SAFE_BUFACCESS size is bottom");
134  }
135  const SVFVar* arg0Val =
136  AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(
137  callNode->getArgument(0));
138  bool isSafe = canSafelyAccessMemory(as, arg0Val, val);
139  if (isSafe)
140  {
141  std::cout << "safe buffer access success: " << callNode->toString()
142  << std::endl;
143  return;
144  }
145  else
146  {
147  std::string err_msg = "this SAFE_BUFACCESS should be a safe access but detected buffer overflow. Pos: ";
148  err_msg += callNode->getSourceLoc();
149  std::cerr << err_msg << std::endl;
150  assert(false);
151  }
152  }
153  else if (funcName == "UNSAFE_BUFACCESS")
154  {
155  // handle other stub functions
156  //void UNSAFE_BUFACCESS(void* data, int size);
157  AbstractInterpretation::getAEInstance().checkpoints.erase(callNode);
158  if (callNode->arg_size() < 2) return;
159  AbstractState&as = AbstractInterpretation::getAEInstance().getAbsStateFromTrace(callNode);
160  u32_t size_id = svfir->getValueNode(callNode->getArgument(1));
161  IntervalValue val = as[size_id].getInterval();
162  if (val.isBottom())
163  {
164  assert(false && "UNSAFE_BUFACCESS size is bottom");
165  }
166  const SVFVar* arg0Val =
167  AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(
168  callNode->getArgument(0));
169  bool isSafe = canSafelyAccessMemory(as, arg0Val, val);
170  if (!isSafe)
171  {
172  std::cout << "detect buffer overflow success: " << callNode->toString() << std::endl;
173  return;
174  }
175  else
176  {
177  std::string err_msg = "this UNSAFE_BUFACCESS should be a buffer overflow but not detected. Pos: ";
178  err_msg += callNode->getSourceLoc();
179  std::cerr << err_msg << std::endl;
180  assert(false);
181  }
182  }

225 void BufOverflowDetector::detectExtAPI(AbstractState& as,
226  const CallICFGNode* call)
227 {
228  SVFIR* svfir = PAG::getPAG();
229  assert(call->getCalledFunction() && "SVFFunction* is nullptr");
230 
231  AbsExtAPI::ExtAPIType extType = AbsExtAPI::UNCLASSIFIED;
232 
233  // Determine the type of external memory API
234  for (const std::string &annotation : ExtAPI::getExtAPI()->getExtFuncAnnotations(call->getCalledFunction()))
235  {
236  if (annotation.find("MEMCPY") != std::string::npos)
237  extType = AbsExtAPI::MEMCPY;
238  if (annotation.find("MEMSET") != std::string::npos)
239  extType = AbsExtAPI::MEMSET;
240  if (annotation.find("STRCPY") != std::string::npos)
241  extType = AbsExtAPI::STRCPY;
242  if (annotation.find("STRCAT") != std::string::npos)
243  extType = AbsExtAPI::STRCAT;
244  }
245 
246  // Apply buffer overflow checks based on the determined API type
247  if (extType == AbsExtAPI::MEMCPY)
248  {
249  if (extAPIBufOverflowCheckRules.count(call->getCalledFunction()->getName()) == 0)
250  {
251  SVFUtil::errs() << "Warning: " << call->getCalledFunction()->getName() << " is not in the rules, please implement it\n";
252  return;
253  }
254  std::vector<std::pair<u32_t, u32_t>> args =
255  extAPIBufOverflowCheckRules.at(call->getCalledFunction()->getName());
256  for (auto arg : args)
257  {
258  IntervalValue offset = as[svfir->getValueNode(call->getArgument(arg.second))].getInterval() - IntervalValue(1);
259  const SVFVar* argVar = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(arg.first));
260  if (!canSafelyAccessMemory(as, argVar, offset))
261  {
262  AEException bug(call->toString());
263  addBugToReporter(bug, call);
264  }
265  }
266  }
267  else if (extType == AbsExtAPI::MEMSET)
268  {
269  if (extAPIBufOverflowCheckRules.count(call->getCalledFunction()->getName()) == 0)
270  {
271  SVFUtil::errs() << "Warning: " << call->getCalledFunction()->getName() << " is not in the rules, please implement it\n";
272  return;
273  }
274  std::vector<std::pair<u32_t, u32_t>> args =
275  extAPIBufOverflowCheckRules.at(call->getCalledFunction()->getName());
276  for (auto arg : args)
277  {
278  IntervalValue offset = as[svfir->getValueNode(call->getArgument(arg.second))].getInterval() - IntervalValue(1);
279  const SVFVar* argVar = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(arg.first));
280  if (!canSafelyAccessMemory(as, argVar, offset))
281  {
282  AEException bug(call->toString());
283  addBugToReporter(bug, call);
284  }
285  }
286  }
287  else if (extType == AbsExtAPI::STRCPY)
288  {
289  if (!detectStrcpy(as, call))
290  {
291  AEException bug(call->toString());
292  addBugToReporter(bug, call);
293  }
294  }
295  else if (extType == AbsExtAPI::STRCAT)
296  {
297  if (!detectStrcat(as, call))
298  {
299  AEException bug(call->toString());
300  addBugToReporter(bug, call);
301  }
302  }
303  else
304  {
305  // Handle other cases
306  }
307 }

320 IntervalValue BufOverflowDetector::getAccessOffset(SVF::AbstractState& as, SVF::NodeID objId, const SVF::GepStmt* gep)
321 {
322  SVFIR* svfir = PAG::getPAG();
323  auto obj = svfir->getGNode(objId);
324 
325  // if the object is a FIObjVar, return the byte offset directly
326  if (SVFUtil::isa<FIObjVar>(obj))
327  {
328  return as.getByteOffset(gep);
329  }
330  else if (SVFUtil::isa<GepObjVar>(obj))
331  {
332  // if the object is a GepObjVar, return the offset from the base object
333  return getGepObjOffsetFromBase(SVFUtil::cast<GepObjVar>(obj)) + as.getByteOffset(gep);
334  }
335  else
336  {
337  assert(SVFUtil::isa<DummyObjVar>(obj) && "Unknown object type");
338  return IntervalValue::top();
339  }

352 void BufOverflowDetector::updateGepObjOffsetFromBase(SVF::AddressValue gepAddrs, SVF::AddressValue objAddrs, SVF::IntervalValue offset)
353 {
354  SVFIR* svfir = PAG::getPAG();
355 
356  for (const auto& objAddr : objAddrs)
357  {
358  NodeID objId = AbstractState::getInternalID(objAddr);
359  auto obj = svfir->getGNode(objId);
360  // if the object is a FIObjVar, add the offset directly
361  if (SVFUtil::isa<FIObjVar>(obj))
362  {
363  for (const auto& gepAddr : gepAddrs)
364  {
365  NodeID gepObj = AbstractState::getInternalID(gepAddr);
366  const GepObjVar* gepObjVar = SVFUtil::cast<GepObjVar>(svfir->getGNode(gepObj));
367  addToGepObjOffsetFromBase(gepObjVar, offset);
368  }
369  }
370  else if (SVFUtil::isa<GepObjVar>(obj))
371  {
372  // if the object is a GepObjVar, add the offset from the base object
373  const GepObjVar* objVar = SVFUtil::cast<GepObjVar>(obj);
374  for (const auto& gepAddr : gepAddrs)
375  {
376  NodeID gepObj = AbstractState::getInternalID(gepAddr);
377  const GepObjVar* gepObjVar = SVFUtil::cast<GepObjVar>(svfir->getGNode(gepObj));
378  if (hasGepObjOffsetFromBase(objVar))
379  {
380  IntervalValue objOffsetFromBase = getGepObjOffsetFromBase(objVar);
381  if (!hasGepObjOffsetFromBase(gepObjVar))
382  addToGepObjOffsetFromBase(gepObjVar, objOffsetFromBase + offset);
383  }
384  else
385  {
386  assert(false && "GEP RHS object has no offset from base");
387  }
388  }
389  }
390  }

421 bool BufOverflowDetector::detectStrcat(AbstractState& as, const CallICFGNode *call)
422 {
423  const std::vector<std::string> strcatGroup = {"__strcat_chk", "strcat", "__wcscat_chk", "wcscat"};
424  const std::vector<std::string> strncatGroup = {"__strncat_chk", "strncat", "__wcsncat_chk", "wcsncat"};
425 
426  if (std::find(strcatGroup.begin(), strcatGroup.end(), call->getCalledFunction()->getName()) != strcatGroup.end())
427  {
428  const SVFVar* arg0Val = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(0));
429  const SVFVar* arg1Val = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(1));
430  IntervalValue strLen0 = AbstractInterpretation::getAEInstance().getUtils()->getStrlen(as, arg0Val);
431  IntervalValue strLen1 = AbstractInterpretation::getAEInstance().getUtils()->getStrlen(as, arg1Val);
432  IntervalValue totalLen = strLen0 + strLen1;
433  return canSafelyAccessMemory(as, arg0Val, totalLen);
434  }
435  else if (std::find(strncatGroup.begin(), strncatGroup.end(), call->getCalledFunction()->getName()) != strncatGroup.end())
436  {
437  const SVFVar* arg0Val = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(0));
438  const SVFVar* arg2Val = AbstractInterpretation::getAEInstance().getUtils()->getSVFVar(call->getArgument(2));
439  IntervalValue arg2Num = as[arg2Val->getId()].getInterval();
440  IntervalValue strLen0 = AbstractInterpretation::getAEInstance().getUtils()->getStrlen(as, arg0Val);
441  IntervalValue totalLen = strLen0 + arg2Num;
442  return canSafelyAccessMemory(as, arg0Val, totalLen);
443  }
444  else
445  {
446  assert(false && "Unknown strcat function, please add it to strcatGroup or strncatGroup");
447  abort();
448  }
449 }

462 bool BufOverflowDetector::canSafelyAccessMemory(AbstractState& as, const SVF::SVFVar* value, const SVF::IntervalValue& len)
463 {
464  SVFIR* svfir = PAG::getPAG();
465  NodeID value_id = value->getId();
466 
467  assert(as[value_id].isAddr());
468  for (const auto& addr : as[value_id].getAddrs())
469  {
470  NodeID objId = AbstractState::getInternalID(addr);
471  u32_t size = 0;
472 
473  // if the object is a constant size object, get the size directly
474  if (svfir->getBaseObj(objId)->isConstantByteSize())
475  {
476  size = svfir->getBaseObj(objId)->getByteSizeOfObj();
477  }
478  else
479  {
480  // if the object is not a constant size object, get the size from the addrStmt
481  const ICFGNode* addrNode = SVFUtil::cast<ICFGNode>(
482  svfir->getBaseObj(objId)->getGNode());
483  for (const SVFStmt* stmt2 : addrNode->getSVFStmts())
484  {
485  if (const AddrStmt* addrStmt = SVFUtil::dyn_cast<AddrStmt>(stmt2))
486  {
487  size = as.getAllocaInstByteSize(addrStmt);
488  }
489  }
490  }
491 
492  IntervalValue offset(0);
493  // if the object is a GepObjVar, get the offset from the base object
494  if (SVFUtil::isa<GepObjVar>(svfir->getGNode(objId)))
495  {
496  offset = getGepObjOffsetFromBase(SVFUtil::cast<GepObjVar>(svfir->getGNode(objId))) + len;
497  }
498  else
499  {
500  // if the object is a FIObjVar, get the offset directly
501  offset = len;
502  }
503  // if the offset is greater than the size, return false
504  if (offset.ub().getIntNumeral() >= size)
505  {
506  return false;
507  }
508  }
509  return true;
510 }
