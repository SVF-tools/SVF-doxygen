122 {
123  assert(A && B && "Pointers are not valid");
124  assert(A->getParent() == B->getParent() &&
125  "Two blocks are not in same function");
126 
127  // Use level information to go up the tree until the levels match. Then
128  // continue going up til we arrive at the same node.
129  while (A != B)
130  {
131  // no common PDominator
132  if(A == NULL) return NULL;
133  const auto lvA = getBBPDomLevel().find(A);

135  assert(lvA != getBBPDomLevel().end() && lvB != getBBPDomLevel().end());
136 
137  if (lvA->second < lvB->second) std::swap(A, B);
138 
139  const auto lvAIdom = getBB2PIdom().find(A);
140  assert(lvAIdom != getBB2PIdom().end());
