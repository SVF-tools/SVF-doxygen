117const SVFBasicBlock* SVFLoopAndDomInfo::findNearestCommonPDominator(const SVFBasicBlock* A, const SVFBasicBlock* B) const
118{
119 assert(A && B && "Pointers are not valid");
120 assert(A->getParent() == B->getParent() &&
121 "Two blocks are not in same function");
122
123 // Use level information to go up the tree until the levels match. Then
124 // continue going up til we arrive at the same node.
125 while (A != B)
126 {
127 // no common PDominator
128 if(A == NULL) return NULL;
129 const auto lvA = getBBPDomLevel().find(A);
130 const auto lvB = getBBPDomLevel().find(B);
131 assert(lvA != getBBPDomLevel().end() && lvB != getBBPDomLevel().end());
132
133 if (lvA->second < lvB->second) std::swap(A, B);
134
135 const auto lvAIdom = getBB2PIdom().find(A);
136 assert(lvAIdom != getBB2PIdom().end());
137 A = lvAIdom->second;
138 }
139
140 return A;
141}
