61 void CHGBuilder::buildCHG()
62 {
63 
64  double timeStart, timeEnd;
65  timeStart = PTAStat::getClk(true);
66  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
67  {
68  DBOUT(DGENERAL, outs() << SVFUtil::pasMsg("construct CHGraph From module "
69  + M.getName().str() + "...\n"));
70  readInheritanceMetadataFromModule(M);
71  for (Module::const_global_iterator I = M.global_begin(), E = M.global_end(); I != E; ++I)
72  buildCHGNodes(&(*I));
73  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
74  buildCHGNodes(LLVMUtil::getDefFunForMultipleModule(&(*F)));
75  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
76  buildCHGEdges(LLVMUtil::getDefFunForMultipleModule(&(*F)));
77 
78  analyzeVTables(M);
79  }
80 
81  DBOUT(DGENERAL, outs() << SVFUtil::pasMsg("build Internal Maps ...\n"));
82  buildInternalMaps();
83 
84  timeEnd = PTAStat::getClk(true);
85  chg->buildingCHGTime = (timeEnd - timeStart) / TIMEINTERVAL;
86 
87  if (Options::DumpCHA())
88  chg->dump("cha");
89 }

91 void CHGBuilder::buildCHGNodes(const GlobalValue *globalvalue)
92 {
93  if (cppUtil::isValVtbl(globalvalue) && globalvalue->getNumOperands() > 0)
94  {
95  const ConstantStruct *vtblStruct = cppUtil::getVtblStruct(globalvalue);
96  string className = getClassNameFromVtblObj(globalvalue->getName().str());
97  if (!chg->getNode(className))
98  createNode(className);
99 
100  for (unsigned int ei = 0; ei < vtblStruct->getNumOperands(); ++ei)
101  {
102  const ConstantArray *vtbl = SVFUtil::dyn_cast<ConstantArray>(vtblStruct->getOperand(ei));
103  assert(vtbl && "Element of initializer not an array?");
104  for (u32_t i = 0; i < vtbl->getNumOperands(); ++i)
105  {
106  if(const ConstantExpr *ce = isCastConstantExpr(vtbl->getOperand(i)))
107  {
108  const Value* bitcastValue = ce->getOperand(0);
109  if (const Function* func = SVFUtil::dyn_cast<Function>(bitcastValue))
110  {
111  struct DemangledName dname = demangle(func->getName().str());
112  if (!chg->getNode(dname.className))
113  createNode(dname.className);
114  }
115  }
116  }
117  }

132 void CHGBuilder::buildCHGEdges(const Function* F)
133 {
134  if (isConstructor(F) || isDestructor(F))
135  {
136  for (Function::const_iterator B = F->begin(), E = F->end(); B != E; ++B)
137  {
138  for (BasicBlock::const_iterator I = B->begin(), E = B->end(); I != E; ++I)
139  {
140  if (LLVMUtil::isCallSite(&(*I)))
141  {
142  connectInheritEdgeViaCall(F, SVFUtil::cast<CallBase>(&(*I)));

144  else if (const StoreInst *store = SVFUtil::dyn_cast<StoreInst>(&(*I)))

161 void CHGBuilder::connectInheritEdgeViaCall(const Function* caller, const CallBase* cs)
162 {
163  if (getCallee(cs) == nullptr)
164  return;
165 
166  const Function* callee = getCallee(cs);
167 
168  struct DemangledName dname = demangle(caller->getName().str());
169  if ((isConstructor(caller) && isConstructor(callee)) || (isDestructor(caller) && isDestructor(callee)))
170  {
171  if (cs->arg_size() < 1 || (cs->arg_size() < 2 && cs->paramHasAttr(0, llvm::Attribute::StructRet)))
172  return;
173  const Value* csThisPtr = cppUtil::getVCallThisPtr(cs);
174  //const Argument* consThisPtr = getConstructorThisPtr(caller);
175  //bool samePtr = isSameThisPtrInConstructor(consThisPtr, csThisPtr);
176  bool samePtrTrue = true;
177  if (csThisPtr != nullptr && samePtrTrue)
178  {
179  struct DemangledName basename = demangle(callee->getName().str());
180  if (!LLVMUtil::isCallSite(csThisPtr) &&
181  basename.className.size() > 0)
182  {
183  chg->addEdge(dname.className, basename.className, CHEdge::INHERITANCE);
184  }
185  }

189 void CHGBuilder::connectInheritEdgeViaStore(const Function* caller, const StoreInst* storeInst)

217 {
218  for (Module::const_named_metadata_iterator mdit = M.named_metadata_begin(),
219  mdeit = M.named_metadata_end(); mdit != mdeit; ++mdit)
220  {
221  const NamedMDNode *md = &*mdit;
222  string mdname = md->getName().str();

224  continue;

226  for (NamedMDNode::const_op_iterator opit = md->op_begin(),
227  opeit = md->op_end(); opit != opeit; ++opit)

229  const MDNode *N = *opit;

232  chg->addEdge(className, baseName, CHEdge::INHERITANCE);

237 CHNode *CHGBuilder::createNode(const std::string& className)
238 {
239  assert(!chg->getNode(className) && "this node should never be created before!");
240  CHNode * node = new CHNode(className, chg->classNum++);
241  chg->classNameToNodeMap[className] = node;
242  chg->addGNode(node->getId(), node);
243  if (className.size() > 0 && className[className.size() - 1] == '>')
244  {
245  string templateName = getBeforeBrackets(className);
246  CHNode* templateNode = chg->getNode(templateName);
247  if (!templateNode)
248  {
249  DBOUT(DCHA, outs() << "\t Create Template CHANode " + templateName + " for class " + className + "...\n");
250  templateNode = createNode(templateName);
251  templateNode->setTemplate();
252  }
253  chg->addEdge(className, templateName, CHEdge::INSTANTCE);
254  chg->addInstances(templateName,node);
255  }
256  return node;
257 }

264 void CHGBuilder::buildClassNameToAncestorsDescendantsMap()
265 {
266 
267  for (CHGraph::const_iterator it = chg->begin(), eit = chg->end();
268  it != eit; ++it)
269  {
270  const CHNode *node = it->second;
271  WorkList worklist;
272  CHNodeSetTy visitedNodes;
273  worklist.push(node);
274  while (!worklist.empty())
275  {
276  const CHNode *curnode = worklist.pop();
277  if (visitedNodes.find(curnode) == visitedNodes.end())
278  {
279  for (CHEdge::CHEdgeSetTy::const_iterator it =
280  curnode->getOutEdges().begin(), eit =
281  curnode->getOutEdges().end(); it != eit; ++it)
282  {
283  if ((*it)->getEdgeType() == CHEdge::INHERITANCE)
284  {
285  CHNode *succnode = (*it)->getDstNode();
286  chg->classNameToAncestorsMap[node->getName()].insert(succnode);
287  chg->classNameToDescendantsMap[succnode->getName()].insert(node);
288  worklist.push(succnode);
289  }
290  }
291  visitedNodes.insert(curnode);
292  }
293  }
294  }

332 /*
333  * do the following things:
334  * 1. initialize virtualFunctions for each class
335  * 2. mark multi-inheritance classes
336  * 3. mark pure abstract classes
337  *
338  * Layout of VTables:
339  *
340  * 1. single inheritance
341  * class A {...};
342  * class B: public A {...};
343  * B's vtable: {i8 *null, _ZTI1B, ...}
344  *
345  * 2. normal multiple inheritance
346  * class A {...};
347  * class B {...};
348  * class C: public A, public B {...};
349  * C's vtable: {i8 *null, _ZTI1C, ..., inttoptr xxx, _ZTI1C, ...}
350  * "inttoptr xxx" servers as a delimiter for dividing virtual methods inherited
351  * from "class A" and "class B"
352  *
353  * 3. virtual diamond inheritance
354  * class A {...};
355  * class B: public virtual A {...};
356  * class C: public virtual A {...};
357  * class D: public B, public C {...};
358  * D's vtable: {i8 *null, _ZTI1C, ..., inttoptr xxx, _ZTI1C, i8 *null, ...}
359  * there will several "i8 *null" following "inttoptr xxx, _ZTI1C,", and the
360  * number of "i8 *null" is the same as the number of virtual methods in
361  * "class A"
362  */
363 void CHGBuilder::analyzeVTables(const Module &M)
364 {
365  for (Module::const_global_iterator I = M.global_begin(),
366  E = M.global_end(); I != E; ++I)
367  {
368  const GlobalValue *globalvalue = SVFUtil::dyn_cast<const GlobalValue>(&(*I));
369  if (cppUtil::isValVtbl(globalvalue) && globalvalue->getNumOperands() > 0)
370  {
371  const ConstantStruct *vtblStruct = cppUtil::getVtblStruct(globalvalue);
372 
373  string vtblClassName = getClassNameFromVtblObj(globalvalue->getName().str());
374  CHNode *node = chg->getNode(vtblClassName);
375  assert(node && "node not found?");
376 
377  SVFGlobalValue* pValue =
378  LLVMModuleSet::getLLVMModuleSet()->getSVFGlobalValue(
379  globalvalue);
380  pValue->setName(vtblClassName);
381  node->setVTable(pValue);
382 
383  for (unsigned int ei = 0; ei < vtblStruct->getNumOperands(); ++ei)
384  {
385  const ConstantArray *vtbl =
386  SVFUtil::dyn_cast<ConstantArray>(vtblStruct->getOperand(ei));
387  assert(vtbl && "Element of initializer not an array?");
388 
389  /*
390  * items in vtables can be classified into 3 categories:
391  * 1. i8* null
392  * 2. i8* inttoptr xxx
393  * 3. i8* bitcast xxx
394  */
395  bool pure_abstract = true;
396  u32_t i = 0;
397  while (i < vtbl->getNumOperands())
398  {
399  CHNode::FuncVector virtualFunctions;
400  bool is_virtual = false; // virtual inheritance
401  int null_ptr_num = 0;
402  for (; i < vtbl->getNumOperands(); ++i)
403  {
404  Constant* operand = vtbl->getOperand(i);
405  if (SVFUtil::isa<ConstantPointerNull>(operand))
406  {
407  if (i > 0 && !SVFUtil::isa<ConstantPointerNull>(vtbl->getOperand(i-1)))
408  {
409  auto foo = [&is_virtual, &null_ptr_num, &vtbl, &i](const Value* val)
410  {
411  if (val->getName().str().compare(0, ztiLabel.size(), ztiLabel) == 0)
412  {
413  is_virtual = true;
414  null_ptr_num = 1;
415  while (i+null_ptr_num < vtbl->getNumOperands())
416  {
417  if (SVFUtil::isa<ConstantPointerNull>(vtbl->getOperand(i+null_ptr_num)))
418  null_ptr_num++;
419  else
420  break;
421  }
422  }
423  };
424  if (const ConstantExpr *ce =
425  SVFUtil::dyn_cast<ConstantExpr>(vtbl->getOperand(i-1)))
426  {
427  if(ce->getOpcode() == Instruction::BitCast)
428  foo(ce->getOperand(0));
429  }
430  else
431  {
432  // opaque pointer mode
433  foo(vtbl->getOperand(i - 1));
434  }
435  }
436  continue;
437  }
438 
439  auto foo = [this, &virtualFunctions, &pure_abstract, &vtblClassName](const Value* operand)
440  {
441  if (const Function* f = SVFUtil::dyn_cast<Function>(operand))
442  {
443  addFuncToFuncVector(virtualFunctions, f);
444  if (f->getName().str().compare(pureVirtualFunName) == 0)
445  {
446  pure_abstract &= true;
447  }
448  else
449  {
450  pure_abstract &= false;
451  }
452  struct DemangledName dname = demangle(f->getName().str());
453  if (dname.className.size() > 0 &&
454  vtblClassName.compare(dname.className) != 0)
455  {
456  if(!chg->getNode(dname.className)) createNode(dname.className);
457  chg->addEdge(vtblClassName, dname.className, CHEdge::INHERITANCE);
458  }
459  }
460  else
461  {
462  if (const GlobalAlias *alias =
463  SVFUtil::dyn_cast<GlobalAlias>(operand))
464  {
465  const Constant *aliasValue = alias->getAliasee();
466  if (const Function* aliasFunc =
467  SVFUtil::dyn_cast<Function>(aliasValue))
468  {
469  addFuncToFuncVector(virtualFunctions, aliasFunc);
470  }
471  else if (const ConstantExpr *aliasconst =
472  SVFUtil::dyn_cast<ConstantExpr>(aliasValue))
473  {
474  (void)aliasconst; // Suppress warning of unused variable under release build
475  assert(aliasconst->getOpcode() == Instruction::BitCast &&
476  "aliased constantexpr in vtable not a bitcast");
477  const Function* aliasbitcastfunc =
478  SVFUtil::dyn_cast<Function>(aliasconst->getOperand(0));
479  assert(aliasbitcastfunc &&
480  "aliased bitcast in vtable not a function");
481  addFuncToFuncVector(virtualFunctions, aliasbitcastfunc);
482  }
483  else
484  {
485  assert(false && "alias not function or bitcast");
486  }
487 
488  pure_abstract &= false;
489  }
490  else if (operand->getName().str().compare(0, ztiLabel.size(),
491  ztiLabel) == 0)
492  {
493  }
494  else
495  {
496  assert("what else can be in bitcast of a vtable?");
497  }
498  }
499  };
500 
516  if (const ConstantExpr *ce =
517  SVFUtil::dyn_cast<ConstantExpr>(operand))
518  {
519  u32_t opcode = ce->getOpcode();
520  assert(opcode == Instruction::IntToPtr);
521  assert(ce->getNumOperands() == 1 &&
522  "inttptr operand num not 1");
523  if (opcode == Instruction::IntToPtr)
524  {
525  node->setMultiInheritance();
526  ++i;
527  break;
528  }
529  }
530  else
531  {
532  foo(operand);
533  }
534  }
535  if (is_virtual && virtualFunctions.size() > 0)
536  {
537  for (int i = 0; i < null_ptr_num; ++i)
538  {
539  const SVFFunction* fun = virtualFunctions[i];
540  virtualFunctions.insert(virtualFunctions.begin(), fun);
541  }
542  }
543  if (virtualFunctions.size() > 0)
544  node->addVirtualFunctionVector(virtualFunctions);
545  }
546  if (pure_abstract == true)
547  {
548  node->setPureAbstract();
549  }
550  }
551  }
552  }
553 }

556 void CHGBuilder::buildVirtualFunctionToIDMap()
557 {
558  /*
559  * 1. Divide classes into groups
560  * 2. Get all virtual functions in a group
561  * 3. Assign consecutive IDs to virtual functions that have
562  * the same name (after demangling) in a group
563  */
564  CHGraph::CHNodeSetTy visitedNodes;
565  for (CHGraph::const_iterator nit = chg->begin(),
566  neit = chg->end(); nit != neit; ++nit)
567  {
568  CHNode *node = nit->second;
569  if (visitedNodes.find(node) != visitedNodes.end())
570  continue;
571 
572  string className = node->getName();
573 
574  /*
575  * get all the classes in a specific group
576  */
577  CHGraph::CHNodeSetTy group;
578  stack<const CHNode*> nodeStack;
579  nodeStack.push(node);
580  while (!nodeStack.empty())
581  {
582  const CHNode *curnode = nodeStack.top();
583  nodeStack.pop();
584  group.insert(curnode);
585  if (visitedNodes.find(curnode) != visitedNodes.end())
586  continue;
587  for (CHEdge::CHEdgeSetTy::const_iterator it = curnode->getOutEdges().begin(),
588  eit = curnode->getOutEdges().end(); it != eit; ++it)
589  {
590  CHNode *tmpnode = (*it)->getDstNode();
591  nodeStack.push(tmpnode);
592  group.insert(tmpnode);
593  }
594  for (CHEdge::CHEdgeSetTy::const_iterator it = curnode->getInEdges().begin(),
595  eit = curnode->getInEdges().end(); it != eit; ++it)
596  {
597  CHNode *tmpnode = (*it)->getSrcNode();
598  nodeStack.push(tmpnode);
599  group.insert(tmpnode);
600  }
601  visitedNodes.insert(curnode);
602  }
603 
604  /*
605  * get all virtual functions in a specific group
606  */
607  set<const SVFFunction*> virtualFunctions;
608  for (CHGraph::CHNodeSetTy::iterator it = group.begin(),
609  eit = group.end(); it != eit; ++it)
610  {
611  const vector<CHNode::FuncVector> &vecs = (*it)->getVirtualFunctionVectors();
612  for (vector<CHNode::FuncVector>::const_iterator vit = vecs.begin(),
613  veit = vecs.end(); vit != veit; ++vit)
614  {
615  for (vector<const SVFFunction*>::const_iterator fit = (*vit).begin(),
616  feit = (*vit).end(); fit != feit; ++fit)
617  {
618  virtualFunctions.insert(*fit);
619  }
620  }
621  }
622 
623  /*
624  * build a set of pairs of demangled function name and function in a
625  * specific group, items in the set will be sort by the first item of the
626  * pair, so all the virtual functions in a group will be sorted by the
627  * demangled function name
628  * <f, A::f>
629  * <f, B::f>
630  * <g, A::g>
631  * <g, B::g>
632  * <g, C::g>
633  * <~A, A::~A>
634  * <~B, B::~B>
635  * <~C, C::~C>
636  * ...
637  */
638  set<pair<string, const SVFFunction*> > fNameSet;
639  for (set<const SVFFunction*>::iterator fit = virtualFunctions.begin(),
640  feit = virtualFunctions.end(); fit != feit; ++fit)
641  {
642  const SVFFunction* f = *fit;
643  struct DemangledName dname = demangle(f->getName());
644  fNameSet.insert(pair<string, const SVFFunction*>(dname.funcName, f));
645  }
646  for (set<pair<string, const SVFFunction*>>::iterator it = fNameSet.begin(),
647  eit = fNameSet.end(); it != eit; ++it)
648  {
649  chg->virtualFunctionToIDMap[it->second] = chg->vfID++;
650  }
651  }
652 }

655 void CHGBuilder::buildCSToCHAVtblsAndVfnsMap()
656 {
657 
658  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
659  {
660  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
661  {
662  for (const_inst_iterator II = inst_begin(*F), E = inst_end(*F); II != E; ++II)
663  {
664  if(const CallBase* callInst = SVFUtil::dyn_cast<CallBase>(&*II))
665  {
666  if (cppUtil::isVirtualCallSite(callInst) == false)
