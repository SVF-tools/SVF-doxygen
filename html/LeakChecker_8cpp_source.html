40 void LeakChecker::initSrcs()
41 {
42 
43  SVFIR* pag = getPAG();
44  ICFG* icfg = pag->getICFG();
45  for(SVFIR::CSToRetMap::iterator it = pag->getCallSiteRets().begin(),
46  eit = pag->getCallSiteRets().end(); it!=eit; ++it)
47  {
48  const RetICFGNode* cs = it->first;
52  if(cs->getCallSite()->ptrInUncalledFunction() || !cs->getCallSite()->getType()->isPointerTy())
53  continue;
54 
55  PTACallGraph::FunctionSet callees;
56  getCallgraph()->getCallees(cs->getCallICFGNode(),callees);
57  for(PTACallGraph::FunctionSet::const_iterator cit = callees.begin(), ecit = callees.end(); cit!=ecit; cit++)
58  {
59  const SVFFunction* fun = *cit;
60  if (isSourceLikeFun(fun))
61  {
62  CSWorkList worklist;
63  SVFGNodeBS visited;
64  worklist.push(it->first->getCallICFGNode());
65  while (!worklist.empty())
66  {
67  const CallICFGNode* cs = worklist.pop();
68  const RetICFGNode* retBlockNode = icfg->getRetICFGNode(cs->getCallSite());
69  const PAGNode* pagNode = pag->getCallSiteRet(retBlockNode);
70  const SVFGNode* node = getSVFG()->getDefSVFGNode(pagNode);
71  if (visited.test(node->getId()) == 0)
72  visited.set(node->getId());
73  else
74  continue;
75 
76  CallSiteSet csSet;
77  // if this node is in an allocation wrapper, find all its call nodes
78  if (isInAWrapper(node, csSet))
79  {
80  for (CallSiteSet::iterator it = csSet.begin(), eit =
81  csSet.end(); it != eit; ++it)
82  {
83  worklist.push(*it);
84  }
85  }
86  // otherwise, this is the source we are interested
87  else
88  {
89  // exclude sources in dead functions or sources in functions that have summary
90  if (!cs->getCallSite()->ptrInUncalledFunction() && !isExtCall(cs->getCallSite()->getParent()->getParent()))
91  {
92  addToSources(node);
93  addSrcToCSID(node, cs);
94  }
95  }
96  }
97  }
98  }
99  }
100

106 void LeakChecker::initSnks()
107 {
108 
109  SVFIR* pag = getPAG();
110 
111  for(SVFIR::CSToArgsListMap::iterator it = pag->getCallSiteArgsMap().begin(),
112  eit = pag->getCallSiteArgsMap().end(); it!=eit; ++it)
113  {
114 
115  PTACallGraph::FunctionSet callees;
116  getCallgraph()->getCallees(it->first,callees);
117  for(PTACallGraph::FunctionSet::const_iterator cit = callees.begin(), ecit = callees.end(); cit!=ecit; cit++)
118  {
119  const SVFFunction* fun = *cit;
120  if (isSinkLikeFun(fun))
121  {
122  SVFIR::SVFVarList &arglist = it->second;
123  assert(!arglist.empty() && "no actual parameter at deallocation site?");
125  for (SVFIR::SVFVarList::const_iterator ait = arglist.begin(),
126  aeit = arglist.end(); ait != aeit; ++ait)
127  {
128  const PAGNode *pagNode = *ait;
129  if (pagNode->isPointer())
130  {
131  const SVFGNode *snk = getSVFG()->getActualParmVFGNode(pagNode, it->first);
132  addToSinks(snk);
133 
134  // For any multi-level pointer e.g., XFree(void** pagNode) that passed into a ExtAPI::EFT_FREE_MULTILEVEL function (e.g., XFree),
135  // we will add the DstNode of a load edge, i.e., dummy = *pagNode
136  SVFStmt::SVFStmtSetTy& loads = const_cast<PAGNode*>(pagNode)->getOutgoingEdges(SVFStmt::Load);
137  for(const SVFStmt* ld : loads)
138  {
139  if(SVFUtil::isa<DummyValVar>(ld->getDstNode()))
140  addToSinks(getSVFG()->getStmtVFGNode(ld));
141  }
142  }
143  }
144  }
145  }

149 void LeakChecker::reportBug(ProgSlice* slice)
150 {
151 
152  if(isAllPathReachable() == false && isSomePathReachable() == false)
153  {
154  // full leakage
155  GenericBug::EventStack eventStack =
156  {
157  SVFBugEvent(SVFBugEvent::SourceInst, getSrcCSID(slice->getSource())->getCallSite())
158  };
159  report.addSaberBug(GenericBug::NEVERFREE, eventStack);
160  }
161  else if (isAllPathReachable() == false && isSomePathReachable() == true)
162  {
163  // partial leakage
164  GenericBug::EventStack eventStack;
165  slice->evalFinalCond2Event(eventStack);
166  eventStack.push_back(
167  SVFBugEvent(SVFBugEvent::SourceInst, getSrcCSID(slice->getSource())->getCallSite()));
168  report.addSaberBug(GenericBug::PARTIALLEAK, eventStack);
169  }
170 
171  if(Options::ValidateTests())
172  testsValidation(slice);
173 }

192 void LeakChecker::validateSuccessTests(const SVFGNode* source, const SVFFunction* fun)
193 {
194 
195  const CallICFGNode* cs = getSrcCSID(source);
196 
197  bool success = false;
198 
199  if(fun->getName() == "SAFEMALLOC")

201  if(isAllPathReachable() == true && isSomePathReachable() == true)
202  success = true;
203  }
204  else if(fun->getName() == "NFRMALLOC")
205  {
206  if(isAllPathReachable() == false && isSomePathReachable() == false)
207  success = true;
208  }
209  else if(fun->getName() == "PLKMALLOC")
210  {
211  if(isAllPathReachable() == false && isSomePathReachable() == true)
212  success = true;
213  }
214  else if(fun->getName() == "CLKMALLOC")
215  {
216  if(isAllPathReachable() == false && isSomePathReachable() == false)
217  success = true;
218  }
219  else if(fun->getName() == "NFRLEAKFP" || fun->getName() == "PLKLEAKFP"
220  || fun->getName() == "LEAKFN")
221  {
222  return;
223  }
224  else
225  {
226  writeWrnMsg("\t can not validate, check function not found, please put it at the right place!!");
227  return;
228  }
229 
230  std::string funName = source->getFun()->getName();
231 
232  if (success)
233  {
234  outs() << sucMsg("\t SUCCESS :") << funName << " check <src id:" << source->getId()
235  << ", cs id:" << getSrcCSID(source)->getCallSite()->toString() << "> at ("
236  << cs->getCallSite()->getSourceLoc() << ")\n";
237  }
238  else
239  {
240  SVFUtil::errs() << errMsg("\t FAILURE :") << funName << " check <src id:" << source->getId()
241  << ", cs id:" << getSrcCSID(source)->getCallSite()->toString() << "> at ("
