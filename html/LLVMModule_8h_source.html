342 inline const SVFFunction* getSVFFunction(const std::string& name)
343 {
344 Function* fun = nullptr;
345
346 for (u32_t i = 0; i < llvmModuleSet->getModuleNum(); ++i)
347 {
348 Module* mod = llvmModuleSet->getModule(i);
349 fun = mod->getFunction(name);
350 if (fun)
351 {
352 return llvmModuleSet->getSVFFunction(fun);
353 }
354 }
355 return nullptr;
356 }
