306 inline const Function* getFunction(const std::string& name)
307 {
308 Function* fun = nullptr;
309
310 for (u32_t i = 0; i < llvmModuleSet->getModuleNum(); ++i)
311 {
312 Module* mod = llvmModuleSet->getModule(i);
313 fun = mod->getFunction(name);
314 if (fun)
315 {
316 return fun;
317 }
318 }
319 return nullptr;
