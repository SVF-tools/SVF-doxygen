305 inline const SVFFunction* getSVFFunction(const std::string& name)
306 {
307 Function* fun = nullptr;
308
309 for (u32_t i = 0; i < llvmModuleSet->getModuleNum(); ++i)
310 {
311 Module* mod = llvmModuleSet->getModule(i);
312 fun = mod->getFunction(name);
313 if (fun)
314 {
315 return llvmModuleSet->getSVFFunction(fun);
316 }
317 }
318 return nullptr;
319 }
