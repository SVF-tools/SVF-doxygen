83 const Type *infersiteToType(const Value *val)
84 {
85  assert(val && "value cannot be empty");
86  if (SVFUtil::isa<LoadInst, StoreInst>(val))
87  {
88  return llvm::getLoadStoreType(const_cast<Value *>(val));
89  }
90  else if (const GetElementPtrInst *gepInst = SVFUtil::dyn_cast<GetElementPtrInst>(val))
91  {
92  return gepInst->getSourceElementType();
93  }
94  else if (const CallBase *call = SVFUtil::dyn_cast<CallBase>(val))
95  {
96  return call->getFunctionType();
97  }
98  else if (const AllocaInst *allocaInst = SVFUtil::dyn_cast<AllocaInst>(val))
99  {
100  return allocaInst->getAllocatedType();
101  }
102  else if (const GlobalValue *globalValue = SVFUtil::dyn_cast<GlobalValue>(val))
103  {
104  return globalValue->getValueType();
105  }
106  else
107  {
108  ABORT_MSG("unknown value:" + dumpValueAndDbgInfo(val));
109  }
110 }

152 const Type *ObjTypeInference::fwInferObjType(const Value *startValue)
153 {
154  // consult cache
155  auto tIt = _valueToType.find(startValue);
156  if (tIt != _valueToType.end())
157  {
158  return tIt->second ? tIt->second : defaultType(startValue);
159  }
160 
161  // simulate the call stack, the second element indicates whether we should update valueTypes for current value
162  FILOWorkList<ValueBoolPair> workList;
163  Set<ValueBoolPair> visited;
164  workList.push({startValue, false});
165 
166  while (!workList.empty())
167  {
168  auto curPair = workList.pop();
169  if (visited.count(curPair)) continue;
170  visited.insert(curPair);
171  const Value *curValue = curPair.first;
172  bool canUpdate = curPair.second;
173  Set<const Value *> infersites;
174 
175  auto insertInferSite = [&infersites, &canUpdate](const Value *infersite)
176  {
177  if (canUpdate) infersites.insert(infersite);
178  };
179  auto insertInferSitesOrPushWorklist = [this, &infersites, &workList, &canUpdate](const auto &pUser)
180  {
181  auto vIt = _valueToInferSites.find(pUser);
182  if (canUpdate)
183  {
184  if (vIt != _valueToInferSites.end() && !vIt->second.empty())
185  {
186  infersites.insert(vIt->second.begin(), vIt->second.end());
187  }
188  }
189  else
190  {
191  if (vIt == _valueToInferSites.end()) workList.push({pUser, false});
192  }
193  };
194  if (!canUpdate && !_valueToInferSites.count(curValue))
195  {
196  workList.push({curValue, true});
197  }
198  if (const GetElementPtrInst *gepInst = SVFUtil::dyn_cast<GetElementPtrInst>(curValue))
199  insertInferSite(gepInst);
200  for (const auto &it: curValue->uses())
201  {
202  if (LoadInst *loadInst = SVFUtil::dyn_cast<LoadInst>(it.getUser()))
203  {
204  /*
205  * infer based on load, e.g.,
206  %call = call i8* malloc()
207  %1 = bitcast i8* %call to %struct.MyStruct*
208  %q = load %struct.MyStruct, %struct.MyStruct* %1
209  */
210  insertInferSite(loadInst);
211  }
212  else if (StoreInst *storeInst = SVFUtil::dyn_cast<StoreInst>(it.getUser()))
213  {
214  if (storeInst->getPointerOperand() == curValue)
215  {
216  /*
217  * infer based on store (pointer operand), e.g.,
218  %call = call i8* malloc()
219  %1 = bitcast i8* %call to %struct.MyStruct*
220  store %struct.MyStruct .., %struct.MyStruct* %1
221  */
222  insertInferSite(storeInst);
223  }
224  else
225  {
226  for (const auto &nit: storeInst->getPointerOperand()->uses())
227  {
228  /*
229  * propagate across store (value operand) and load
230  %call = call i8* malloc()
231  store i8* %call, i8** %p
232  %q = load i8*, i8** %p
233  ..infer based on %q..
234  */
235  if (SVFUtil::isa<LoadInst>(nit.getUser()))
236  insertInferSitesOrPushWorklist(nit.getUser());
237  }
238  /*
239  * infer based on store (value operand) <- gep (result element)
240  %call1 = call i8* @TYPE_MALLOC(i32 noundef 16, i32 noundef 2), !dbg !39
241  %2 = bitcast i8* %call1 to %struct.MyStruct*, !dbg !41
242  %3 = load %struct.MyStruct*, %struct.MyStruct** %p, align 8, !dbg !42
243  %next = getelementptr inbounds %struct.MyStruct, %struct.MyStruct* %3, i32 0, i32 1, !dbg !43
244  store %struct.MyStruct* %2, %struct.MyStruct** %next, align 8, !dbg !44
245  %5 = load %struct.MyStruct*, %struct.MyStruct** %p, align 8, !dbg !48
246  %next3 = getelementptr inbounds %struct.MyStruct, %struct.MyStruct* %5, i32 0, i32 1, !dbg !49
247  %6 = load %struct.MyStruct*, %struct.MyStruct** %next3, align 8, !dbg !49
248  infer site -> %f1 = getelementptr inbounds %struct.MyStruct, %struct.MyStruct* %6, i32 0, i32 0, !dbg !50
249  */
250  if (GetElementPtrInst *gepInst = SVFUtil::dyn_cast<GetElementPtrInst>(
251  storeInst->getPointerOperand()))
252  {
253  const Value *gepBase = gepInst->getPointerOperand();
254  if (!SVFUtil::isa<LoadInst>(gepBase)) continue;
255  const LoadInst *load = SVFUtil::dyn_cast<LoadInst>(gepBase);
256  for (const auto &loadUse: load->getPointerOperand()->uses())
257  {
258  if (loadUse.getUser() == load || !SVFUtil::isa<LoadInst>(loadUse.getUser()))
259  continue;
260  for (const auto &gepUse: loadUse.getUser()->uses())
261  {
262  if (!SVFUtil::isa<GetElementPtrInst>(gepUse.getUser())) continue;
263  for (const auto &loadUse2: gepUse.getUser()->uses())
264  {
265  if (SVFUtil::isa<LoadInst>(loadUse2.getUser()))
266  {
267  insertInferSitesOrPushWorklist(loadUse2.getUser());
268  }
269  }
270  }
271  }
272 
273  }
274  }
275 
276  }
277  else if (GetElementPtrInst *gepInst = SVFUtil::dyn_cast<GetElementPtrInst>(it.getUser()))
278  {
279  /*
280  * infer based on gep (pointer operand)
281  %call = call i8* malloc()
282  %1 = bitcast i8* %call to %struct.MyStruct*
283  %next = getelementptr inbounds %struct.MyStruct, %struct.MyStruct* %1, i32 0..
284  */
285  if (gepInst->getPointerOperand() == curValue)
286  insertInferSite(gepInst);
287  }
288  else if (BitCastInst *bitcast = SVFUtil::dyn_cast<BitCastInst>(it.getUser()))
289  {
290  // continue on bitcast
291  insertInferSitesOrPushWorklist(bitcast);
292  }
293  else if (PHINode *phiNode = SVFUtil::dyn_cast<PHINode>(it.getUser()))
294  {
295  // continue on bitcast
296  insertInferSitesOrPushWorklist(phiNode);
297  }
298  else if (ReturnInst *retInst = SVFUtil::dyn_cast<ReturnInst>(it.getUser()))
299  {
300  /*
301  * propagate from return to caller
302  Function Attrs: noinline nounwind optnone uwtable
303  define dso_local i8* @malloc_wrapper() #0 !dbg !22 {
304  entry:
305  %call = call i8* @malloc(i32 noundef 16), !dbg !25
306  ret i8* %call, !dbg !26
307  }
308  %call = call i8* @malloc_wrapper()
309  ..infer based on %call..
310  */
311  for (const auto &callsite: retInst->getFunction()->uses())
312  {
313  if (CallBase *callBase = SVFUtil::dyn_cast<CallBase>(callsite.getUser()))
314  {
315  // skip function as parameter
316  // e.g., call void @foo(%struct.ssl_ctx_st* %9, i32 (i8*, i32, i32, i8*)* @passwd_callback)
317  if (callBase->getCalledFunction() != retInst->getFunction()) continue;
318  insertInferSitesOrPushWorklist(callBase);
319  }
320  }
321  }
322  else if (CallBase *callBase = SVFUtil::dyn_cast<CallBase>(it.getUser()))
323  {
324  /*
325  * propagate from callsite to callee
326  %call = call i8* @malloc(i32 noundef 16)
327  %0 = bitcast i8* %call to %struct.Node*, !dbg !43
328  call void @foo(%struct.Node* noundef %0), !dbg !45
329 
330  define dso_local void @foo(%struct.Node* noundef %param) #0 !dbg !22 {...}
331  ..infer based on the formal param %param..
332  */
333  // skip global function value -> callsite
334  // e.g., def @foo() -> call @foo()
335  // we don't skip function as parameter, e.g., def @foo() -> call @bar(..., @foo)
336  if (SVFUtil::isa<Function>(curValue) && curValue == callBase->getCalledFunction()) continue;
337  // skip indirect call
338  // e.g., %0 = ... -> call %0(...)
339  if (!callBase->hasArgument(curValue)) continue;
340  if (Function *calleeFunc = callBase->getCalledFunction())
341  {
342  u32_t pos = getArgPosInCall(callBase, curValue);
343  // for variable argument, conservatively collect all params
344  if (calleeFunc->isVarArg()) pos = 0;
345  if (!calleeFunc->isDeclaration())
346  {
347  insertInferSitesOrPushWorklist(calleeFunc->getArg(pos));
348  }
349  }
350  }
351  }
352  if (canUpdate)
353  {
354  Set<const Type *> types;
355  std::transform(infersites.begin(), infersites.end(), std::inserter(types, types.begin()),
356  infersiteToType);
357  _valueToInferSites[curValue] = infersites;
358  _valueToType[curValue] = selectLargestType(types);
359  }
360  }
361  const Type *type = _valueToType[startValue];
362  if (type == nullptr)
363  {
364  type = defaultType(startValue);
365  WARN_MSG("Using default type, trace ID is " + std::to_string(traceId) + ":" + dumpValueAndDbgInfo(startValue));
366  }
367  ABORT_IFNOT(type, "type cannot be a null ptr");
368  return type;
369 }

376 Set<const Value *> ObjTypeInference::bwfindAllocations(const Value *startValue)
377 {
378 
379  // consult cache
380  auto tIt = _valueToAllocs.find(startValue);
381  if (tIt != _valueToAllocs.end())
382  {
383  WARN_IFNOT(!tIt->second.empty(), "empty type:" + dumpValueAndDbgInfo(startValue));
384  return !tIt->second.empty() ? tIt->second : Set<const Value *>({startValue});
385  }
386 
387  // simulate the call stack, the second element indicates whether we should update sources for current value
388  FILOWorkList<ValueBoolPair> workList;
389  Set<ValueBoolPair> visited;
390  workList.push({startValue, false});
391  while (!workList.empty())
392  {
393  auto curPair = workList.pop();
394  if (visited.count(curPair)) continue;
395  visited.insert(curPair);
396  const Value *curValue = curPair.first;
397  bool canUpdate = curPair.second;
398 
399  Set<const Value *> sources;
400  auto insertAllocs = [&sources, &canUpdate](const Value *source)
401  {
402  if (canUpdate) sources.insert(source);
403  };
404  auto insertAllocsOrPushWorklist = [this, &sources, &workList, &canUpdate](const auto &pUser)
405  {
406  auto vIt = _valueToAllocs.find(pUser);
407  if (canUpdate)
408  {
409  if (vIt != _valueToAllocs.end() && !vIt->second.empty())
410  {
411  sources.insert(vIt->second.begin(), vIt->second.end());
412  }
413  }
414  else
415  {
416  if (vIt == _valueToAllocs.end()) workList.push({pUser, false});
417  }
418  };
419 
420  if (!canUpdate && !_valueToAllocs.count(curValue))
421  {
422  workList.push({curValue, true});
423  }
424 
425  if (isAllocation(curValue))
426  {
427  insertAllocs(curValue);
428  }
429  else if (const BitCastInst *bitCastInst = SVFUtil::dyn_cast<BitCastInst>(curValue))
430  {
431  Value *prevVal = bitCastInst->getOperand(0);
432  insertAllocsOrPushWorklist(prevVal);
433  }
434  else if (const PHINode *phiNode = SVFUtil::dyn_cast<PHINode>(curValue))
435  {
436  for (u32_t i = 0; i < phiNode->getNumOperands(); ++i)
437  {
438  insertAllocsOrPushWorklist(phiNode->getOperand(i));
439  }
440  }
441  else if (const LoadInst *loadInst = SVFUtil::dyn_cast<LoadInst>(curValue))
442  {
443  for (const auto &use: loadInst->getPointerOperand()->uses())
444  {
445  if (const StoreInst *storeInst = SVFUtil::dyn_cast<StoreInst>(use.getUser()))
446  {
447  if (storeInst->getPointerOperand() == loadInst->getPointerOperand())
448  {
449  insertAllocsOrPushWorklist(storeInst->getValueOperand());
450  }
451  }
452  }
453  }
454  else if (const Argument *argument = SVFUtil::dyn_cast<Argument>(curValue))
455  {
456  for (const auto &use: argument->getParent()->uses())
457  {
458  if (const CallBase *callBase = SVFUtil::dyn_cast<CallBase>(use.getUser()))
459  {
460  // skip function as parameter
461  // e.g., call void @foo(%struct.ssl_ctx_st* %9, i32 (i8*, i32, i32, i8*)* @passwd_callback)
462  if (callBase->getCalledFunction() != argument->getParent()) continue;
463  u32_t pos = argument->getParent()->isVarArg() ? 0 : argument->getArgNo();
464  insertAllocsOrPushWorklist(callBase->getArgOperand(pos));
465  }
466  }
467  }
468  else if (const CallBase *callBase = SVFUtil::dyn_cast<CallBase>(curValue))
469  {
470  ABORT_IFNOT(!callBase->doesNotReturn(), "callbase does not return:" + dumpValueAndDbgInfo(callBase));
471  if (Function *callee = callBase->getCalledFunction())
472  {
473  if (!callee->isDeclaration())
474  {
475  const SVFFunction *svfFunc = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(callee);
476  const Value *pValue = LLVMModuleSet::getLLVMModuleSet()->getLLVMValue(svfFunc->getExitBB()->back());
477  const ReturnInst *retInst = SVFUtil::dyn_cast<ReturnInst>(pValue);
478  ABORT_IFNOT(retInst && retInst->getReturnValue(), "not return inst?");
479  insertAllocsOrPushWorklist(retInst->getReturnValue());
480  }
481  }
482  }
483  if (canUpdate)
484  {
485  _valueToAllocs[curValue] = sources;
486  }
487  }
488  Set<const Value *> srcs = _valueToAllocs[startValue];
489  if (srcs.empty())
490  {
491  srcs = {startValue};
492  WARN_MSG("Using default type, trace ID is " + std::to_string(traceId) + ":" + dumpValueAndDbgInfo(startValue));
493  }
494  ABORT_IFNOT(!srcs.empty(), "sources cannot be empty");
495  return srcs;
496 }

511  if (func->getName().find(TYPEMALLOC) != std::string::npos)

555 const Type *ObjTypeInference::selectLargestType(Set<const Type *> &objTys)
556 {
557  if (objTys.empty()) return nullptr;
558  // map type size to types from with key in descending order
559  OrderedMap<u32_t, Set<const Type *>, std::greater<int>> typeSzToTypes;
560  for (const Type *ty: objTys)
561  {
562  typeSzToTypes[objTyToNumFields(ty)].insert(ty);
563  }
564  assert(!typeSzToTypes.empty() && "typeSzToTypes cannot be empty");
565  Set<const Type *> largestTypes;
566  std::tie(std::ignore, largestTypes) = *typeSzToTypes.begin();
567  assert(!largestTypes.empty() && "largest element cannot be empty");
568  return *largestTypes.begin();
