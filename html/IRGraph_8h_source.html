90 inline void mapValueToEdge(const SVFValue* V, SVFStmt *edge)
91 {
92 auto inserted = valueToEdgeMap.emplace(V, SVFStmtSet{edge});
93 if (!inserted.second)
94 {
95 inserted.first->second.emplace(edge);
96 }
97 }

105 }
