67 void SaberSVFGBuilder::collectGlobals(BVDataPTAImpl* pta)
68 {
69  SVFIR* pag = svfg->getPAG();
70  NodeVector worklist;
71  for(SVFIR::iterator it = pag->begin(), eit = pag->end(); it!=eit; it++)
72  {
73  PAGNode* pagNode = it->second;
74  if (SVFUtil::isa<DummyValVar, DummyObjVar>(pagNode))
75  continue;
76 
77  if(GepObjVar* gepobj = SVFUtil::dyn_cast<GepObjVar>(pagNode))
78  {
79  if(SVFUtil::isa<DummyObjVar>(pag->getGNode(gepobj->getBaseNode())))
80  continue;
81  }
82  if(const SVFValue* val = pagNode->getValue())
83  {
84  if(SVFUtil::isa<SVFGlobalValue>(val))
85  worklist.push_back(it->first);
86  }
87  }
88 
89  NodeToPTSSMap cachedPtsMap;
90  while(!worklist.empty())
91  {
92  NodeID id = worklist.back();
93  worklist.pop_back();
94  globs.set(id);
95  const PointsTo& pts = pta->getPts(id);
96  for(PointsTo::iterator it = pts.begin(), eit = pts.end(); it!=eit; ++it)
97  {
98  globs |= CollectPtsChain(pta,*it,cachedPtsMap);
99  }
100  }
101 }

104 /*
105  * https://github.com/SVF-tools/SVF/issues/991
106  *
107  * Originally, this function will collect all base pointers with all their fields
108  * inside the points-to set of global variables. But if a global variable points
109  * to the pointer returned by malloc() at some program points, then all pointers
110  * returned by malloc() will be included in the global set because of the
111  * context-insensitive pointer analysis results. This will make saber abandon
112  * too many slicing thus miss potential bugs.
113  *
114  * We add an option "saber-collect-extret-globals" to control whether this function
115  * will collect external functions' returned pointers. This option is true by default,
116  * making it to be false will let saber analyze more slicing but cause performance downgrade.
117  *
118  */
119 PointsTo& SaberSVFGBuilder::CollectPtsChain(BVDataPTAImpl* pta, NodeID id, NodeToPTSSMap& cachedPtsMap)
120 {
121  SVFIR* pag = svfg->getPAG();
122 
123  NodeID baseId = pag->getBaseObjVar(id);
124  NodeToPTSSMap::iterator it = cachedPtsMap.find(baseId);
125  if(it!=cachedPtsMap.end())
126  {
127  return it->second;
128  }
129  else
130  {
131  PointsTo& pts = cachedPtsMap[baseId];
132  // base object
133  if (!Options::CollectExtRetGlobals())
134  {
135  if(pta->isFIObjNode(baseId) && pag->getGNode(baseId)->hasValue())
136  {
137  const SVFCallInst* inst = SVFUtil::dyn_cast<SVFCallInst>(pag->getGNode(baseId)->getValue());
138  if(inst && SVFUtil::isExtCall(inst))
139  {
140  return pts;
141  }
142  }
143  }
144 
145  pts |= pag->getFieldsAfterCollapse(baseId);
146 
147  WorkList worklist;
148  for(PointsTo::iterator it = pts.begin(), eit = pts.end(); it!=eit; ++it)
149  worklist.push(*it);
150 
151  while(!worklist.empty())
152  {
153  NodeID nodeId = worklist.pop();
154  const PointsTo& tmp = pta->getPts(nodeId);
155  for(PointsTo::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
156  {
157  pts |= CollectPtsChain(pta,*it,cachedPtsMap);
158  }
159  }
160  return pts;
161  }
162 }

177 void SaberSVFGBuilder::rmDerefDirSVFGEdges(BVDataPTAImpl* pta)
178 {
179 
180  for(SVFG::iterator it = svfg->begin(), eit = svfg->end(); it!=eit; ++it)
181  {
182  const SVFGNode* node = it->second;
183 
184  if(const StmtSVFGNode* stmtNode = SVFUtil::dyn_cast<StmtSVFGNode>(node))
185  {
187  if(SVFUtil::isa<StoreSVFGNode>(stmtNode))
188  {
189  const SVFGNode* def = svfg->getDefSVFGNode(stmtNode->getPAGDstNode());
190  if(SVFGEdge* edge = svfg->getIntraVFGEdge(def,stmtNode,SVFGEdge::IntraDirectVF))
191  svfg->removeSVFGEdge(edge);
192  else
193  assert((svfg->getKind()==VFG::FULLSVFG_OPT || svfg->getKind()==VFG::PTRONLYSVFG_OPT) && "Edge not found!");
194 
195  if(accessGlobal(pta,stmtNode->getPAGDstNode()))
196  {
197  globSVFGNodes.insert(stmtNode);
198  }
199  }
200  else if(SVFUtil::isa<LoadSVFGNode>(stmtNode))
201  {
202  const SVFGNode* def = svfg->getDefSVFGNode(stmtNode->getPAGSrcNode());
203  if(SVFGEdge* edge = svfg->getIntraVFGEdge(def,stmtNode,SVFGEdge::IntraDirectVF))
204  svfg->removeSVFGEdge(edge);
205  else
206  assert((svfg->getKind()==VFG::FULLSVFG_OPT || svfg->getKind()==VFG::PTRONLYSVFG_OPT) && "Edge not found!");

208  if(accessGlobal(pta,stmtNode->getPAGSrcNode()))

221 bool SaberSVFGBuilder::isStrongUpdate(const SVFGNode* node, NodeID& singleton, BVDataPTAImpl* pta)
222 {
223  bool isSU = false;
224  if (const StoreSVFGNode* store = SVFUtil::dyn_cast<StoreSVFGNode>(node))
225  {
226  const PointsTo& dstCPSet = pta->getPts(store->getPAGDstNodeID());
227  if (dstCPSet.count() == 1)
228  {
230  PointsTo::iterator it = dstCPSet.begin();
231  singleton = *it;
232 
233  // Strong update can be made if this points-to target is not heap, array or field-insensitive.
234  if (!pta->isHeapMemObj(singleton) && !pta->isArrayMemObj(singleton)
235  && SVFIR::getPAG()->getBaseObj(singleton)->isFieldInsensitive() == false
236  && !pta->isLocalVarInRecursiveFun(singleton))
237  {
238  isSU = true;
239  }
240  }
241  }
242  return isSU;
243 }

290 {

292  for(SVFIR::CSToArgsListMap::iterator it = pag->getCallSiteArgsMap().begin(),
293  eit = pag->getCallSiteArgsMap().end(); it!=eit; ++it)

298  ecit = callees.end(); cit != ecit; cit++)
299  {
300 
301  const SVFFunction* fun = *cit;

304  {
305  SVFIR::SVFVarList& arglist = it->second;
306  for(SVFIR::SVFVarList::const_iterator ait = arglist.begin(), aeit = arglist.end(); ait!=aeit; ++ait)
307  {

311  addActualParmVFGNode(pagNode, it->first);
312  svfg->addIntraDirectVFEdge(svfg->getDefSVFGNode(pagNode)->getId(), svfg->getActualParmVFGNode(pagNode, it->first)->getId());
