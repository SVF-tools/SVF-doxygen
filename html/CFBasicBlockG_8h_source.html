381  {
382  CFBasicBlockEdge edge(src, dst, icfgEdge);
383  CFBasicBlockEdge *outEdge = src->hasOutgoingEdge(&edge);
384  CFBasicBlockEdge *inEdge = dst->hasIncomingEdge(&edge);

410  {

424  inline void removeCFBBNode(CFBasicBlockNode *node)
425  {
426  std::set<CFBasicBlockEdge *> temp;
427  for (CFBasicBlockEdge *e: node->getInEdges())
428  temp.insert(e);
429  for (CFBasicBlockEdge *e: node->getOutEdges())
430  temp.insert(e);
431  for (CFBasicBlockEdge *e: temp)
432  {
433  removeCFBBEdge(e);
434  }
435  removeGNode(node);
436  _totalCFBasicBlockNode--;
437  }
