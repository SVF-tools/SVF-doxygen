102 void ThreadAPI::init()
103 {
104  set<TD_TYPE> t_seen;
105  TD_TYPE prev_t= TD_DUMMY;
106  t_seen.insert(TD_DUMMY);
107  for(const ei_pair *p= ei_pairs; p->n; ++p)
108  {
109  if(p->t != prev_t)
110  {
111  //This will detect if you move an entry to another block
112  // but forget to change the type.
113  if(t_seen.count(p->t))
114  {
115  fputs(p->n, stderr);
116  putc('\n', stderr);
117  assert(!"ei_pairs not grouped by type");
118  }
119  t_seen.insert(p->t);
120  prev_t= p->t;
121  }
122  if(tdAPIMap.count(p->n))
123  {
124  fputs(p->n, stderr);
125  putc('\n', stderr);
126  assert(!"duplicate name in ei_pairs");
127  }
128  tdAPIMap[p->n]= p->t;
129  }
130 }

206 const SVFVar* ThreadAPI::getJoinedThread(const CallICFGNode *cs) const
207 {
208  assert(isTDJoin(cs) && "not a thread join function!");
209  const SVFVar* join = cs->getArgument(0);
210  for(const SVFStmt* stmt : join->getInEdges())
211  {
212  if(SVFUtil::isa<LoadStmt>(stmt))
213  return stmt->getSrcNode();
214  }
215  if(SVFUtil::isa<SVFArgument>(join->getValue()))
216  return join;
217 
218  assert(false && "the value of the first argument at join is not a load instruction?");
219  return nullptr;
220 }
