347  void sanitizePts()
348  {
349  for(ConstraintGraph::iterator it = consCG->begin(), eit = consCG->end(); it!=eit; ++it)
350  {
351  const PointsTo& pts = getPts(it->first);
352  NodeBS fldInsenObjs;
353 
354  for (NodeID o : pts)
355  {
356  if(isFieldInsensitive(o))
357  fldInsenObjs.set(o);
358  }
359 
360  for (NodeID o : fldInsenObjs)
361  {
362  const NodeBS &allFields = consCG->getAllFieldsObjVars(o);
363  for (NodeID f : allFields) addPts(it->first, f);
364  }
365  }
366  }
