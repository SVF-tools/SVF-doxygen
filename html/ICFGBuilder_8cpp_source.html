41 void ICFGBuilder::build(SVFModule* svfModule)
42 {
43  DBOUT(DGENERAL, outs() << pasMsg("\t Building ICFG ...\n"));
44  // Add the unique global ICFGNode at the entry of a program (before the main method).
45  icfg->addGlobalICFGNode();
46 
47  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
48  {
49  for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
50  {
51  const Function *fun = &*F;
52  const SVFFunction* svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
53  if (svffun->isDeclaration())
54  continue;
55  WorkList worklist;
56  processFunEntry(fun,worklist);
57  processNoPrecessorBasicBlocks(fun, worklist);
58  processFunBody(worklist);
59  processFunExit(fun);
60 
61  checkICFGNodesVisited(fun);
62  }
63 
64  }
65  connectGlobalToProgEntry(svfModule);
66 }

68 void ICFGBuilder::checkICFGNodesVisited(const Function* fun)
69 {
70  for (const auto& bb: *fun)
71  {
72  for (const auto& inst: bb)
73  {
74  SVFInstruction* pInstruction =
75  LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(
76  &inst);
77  if(isIntrinsicInst(pInstruction))
78  continue;
79  assert(visited.count(&inst) && "inst never visited");
80  assert(icfg->hasICFGNode(pInstruction) && "icfgnode not created");
81  }
82  }

87 void ICFGBuilder::processFunEntry(const Function* fun, WorkList& worklist)
88 {
89  FunEntryICFGNode* FunEntryICFGNode = icfg->getFunEntryICFGNode(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun));
90  const Instruction* entryInst = &((fun->getEntryBlock()).front());
91  const SVFInstruction* svfentryInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(entryInst);
92 
93  InstVec insts;
94  if (isIntrinsicInst(svfentryInst))
95  LLVMUtil::getNextInsts(entryInst, insts);
96  else
97  insts.push_back(entryInst);
98  for (InstVec::const_iterator nit = insts.begin(), enit = insts.end();
99  nit != enit; ++nit)
100  {
101  visited.insert(*nit);
102  ICFGNode* instNode = addBlockICFGNode(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(*nit)); //add interprocedural edge
103  worklist.push(*nit);
104  icfg->addIntraEdge(FunEntryICFGNode, instNode);
105  }

114 void ICFGBuilder::processNoPrecessorBasicBlocks(const Function* fun, WorkList& worklist)
115 {
116  for (const auto& bb: *fun)
117  {
118  for (const auto& inst: bb)
119  {
120  if (LLVMUtil::isNoPrecessorBasicBlock(inst.getParent()) &&
121  !visited.count(&inst))
122  {
123  visited.insert(&inst);
124  (void)addBlockICFGNode(LLVMModuleSet::getLLVMModuleSet()
125  ->getSVFInstruction(&inst));
126  worklist.push(&inst);
127  }

135 void ICFGBuilder::processFunBody(WorkList& worklist)
136 {
138  while (!worklist.empty())
139  {
140  const Instruction* inst = worklist.pop();
141  const SVFInstruction* svfinst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(inst);
142  ICFGNode* srcNode = icfg->getICFGNode(svfinst);
143  if (svfinst->isRetInst())
144  {
145  const SVFFunction* svfFun = svfinst->getFunction();
146  FunExitICFGNode* FunExitICFGNode = icfg->getFunExitICFGNode(svfFun);
147  icfg->addIntraEdge(srcNode, FunExitICFGNode);
148  }
149  InstVec nextInsts;
150  LLVMUtil::getNextInsts(inst, nextInsts);
151  u32_t branchID = 0;
152  for (InstVec::const_iterator nit = nextInsts.begin(), enit =
153  nextInsts.end(); nit != enit; ++nit)
154  {
155  const Instruction* succ = *nit;
156  const SVFInstruction* svfsucc = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(succ);
157  ICFGNode* dstNode;
158  if (visited.find(succ) != visited.end())
159  {
160  dstNode = icfg->getICFGNode(svfsucc);
161  }
162  else
163  {
164  visited.insert(succ);
165  dstNode = addBlockICFGNode(svfsucc);
166  worklist.push(succ);
167  }
168 
169 
170  if (isNonInstricCallSite(svfinst))
171  {
172  RetICFGNode* retICFGNode = getRetICFGNode(svfinst);
173  srcNode = retICFGNode;
174  }
175 
176  if (const BranchInst* br = SVFUtil::dyn_cast<BranchInst>(inst))
177  {
178  assert(branchID <= 1 && "if/else has more than two branches?");
179  if(br->isConditional())
180  icfg->addConditionalIntraEdge(srcNode, dstNode, LLVMModuleSet::getLLVMModuleSet()->getSVFValue(br->getCondition()), 1 - branchID);
181  else
182  icfg->addIntraEdge(srcNode, dstNode);
183  }
184  else if (const SwitchInst* si = SVFUtil::dyn_cast<SwitchInst>(inst))
185  {
187  const ConstantInt* condVal = const_cast<SwitchInst*>(si)->findCaseDest(const_cast<BasicBlock*>(succ->getParent()));
189  s64_t val = -1;
190  if (condVal && condVal->getBitWidth() <= 64)
191  val = condVal->getSExtValue();
192  icfg->addConditionalIntraEdge(srcNode, dstNode, LLVMModuleSet::getLLVMModuleSet()->getSVFValue(si->getCondition()),val);
193  }
194  else
195  icfg->addIntraEdge(srcNode, dstNode);
196  branchID++;
197  }
198  }
199 }

206 void ICFGBuilder::processFunExit(const Function* f)
207 {
208  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(f);
209  FunExitICFGNode* FunExitICFGNode = icfg->getFunExitICFGNode(fun);
210 
211  for (const SVFBasicBlock* svfbb : fun->getBasicBlockList())

230 InterICFGNode* ICFGBuilder::addInterBlockICFGNode(const SVFInstruction* inst)
231 {
232  assert(SVFUtil::isCallSite(inst) && "not a call instruction?");
233  assert(SVFUtil::isNonInstricCallSite(inst) && "associating an intrinsic debug instruction with an ICFGNode!");
234  CallICFGNode* callICFGNode = icfg->addCallICFGNode(inst);
235  (void) icfg->addRetICFGNode(inst);
236  addICFGInterEdges(inst, callICFGNode->getCalledFunction()); //creating interprocedural edges
237  return callICFGNode;
238 }

243 void ICFGBuilder::addICFGInterEdges(const SVFInstruction* cs, const SVFFunction* callee)
244 {
245  CallICFGNode* callICFGNode = getCallICFGNode(cs);
246  RetICFGNode* retBlockNode = getRetICFGNode(cs);
247 
249  if(callee)
250  {
252  if (isExtCall(callee))
253  {
254  icfg->addIntraEdge(callICFGNode, retBlockNode);
255  }
257  else
258  {
259  FunEntryICFGNode* calleeEntryNode = icfg->getFunEntryICFGNode(callee);
260  FunExitICFGNode* calleeExitNode = icfg->getFunExitICFGNode(callee);
261  icfg->addCallEdge(callICFGNode, calleeEntryNode);
262  icfg->addRetEdge(calleeExitNode, retBlockNode);
263  }
264  }
266  else
267  {
268  icfg->addIntraEdge(callICFGNode, retBlockNode);
269  }
270 }
