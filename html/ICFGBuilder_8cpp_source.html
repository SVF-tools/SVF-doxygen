45ICFG* ICFGBuilder::build()
46{
47 icfg = new ICFG();
48 DBOUT(DGENERAL, outs() << pasMsg("\t Building ICFG ...\n"));
49 // Add the unique global ICFGNode at the entry of a program (before the main method).
50 addGlobalICFGNode();
51
52 // Add function entry and exit
53 for (Module &M : llvmModuleSet()->getLLVMModules())
54 {
55 for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
56 {
57 const Function *fun = &*F;
58 if (fun->isDeclaration())
59 continue;
60 addFunEntryBlock(fun);
61 addFunExitBlock(fun);
62 }
63
64 }
65
66 for (Module &M : llvmModuleSet()->getLLVMModules())
67 {
68 for (Module::const_iterator F = M.begin(), E = M.end(); F != E; ++F)
69 {
70 const Function *fun = &*F;
71 if (fun->isDeclaration())
72 continue;
73 WorkList worklist;
74 processFunEntry(fun,worklist);
75 processUnreachableFromEntry(fun, worklist);
76 processFunBody(worklist);
77 processFunExit(fun);
78
79 checkICFGNodesVisited(fun);
80 }
81
82 }
83 connectGlobalToProgEntry();
84 return icfg;
85}

87void ICFGBuilder::checkICFGNodesVisited(const Function* fun)
88{
89 for (const auto& bb: *fun)
90 {
91 for (const auto& inst: bb)
92 {
93 if(LLVMUtil::isIntrinsicInst(&inst))
94 continue;
95 assert(visited.count(&inst) && "inst never visited");
96 assert(hasICFGNode(&inst) && "icfgnode not created");
97 }
98 }

103void ICFGBuilder::processFunEntry(const Function* fun, WorkList& worklist)
104{
105 FunEntryICFGNode* FunEntryICFGNode = getFunEntryICFGNode(fun);
106 const Instruction* entryInst = &((fun->getEntryBlock()).front());
107
108 InstVec insts;
109 if (LLVMUtil::isIntrinsicInst(entryInst))
110 LLVMUtil::getNextInsts(entryInst, insts);
111 else
112 insts.push_back(entryInst);
113 for (InstVec::const_iterator nit = insts.begin(), enit = insts.end();
114 nit != enit; ++nit)
115 {
116 visited.insert(*nit);
117 ICFGNode* instNode = addBlockICFGNode(*nit); //add interprocedural edge
118 worklist.push(*nit);
119 icfg->addIntraEdge(FunEntryICFGNode, instNode);
120 }

129void ICFGBuilder::processUnreachableFromEntry(const Function* fun, WorkList& worklist)
130{
131 SVFLoopAndDomInfo* pInfo =
132 llvmModuleSet()->getFunObjVar(fun)->getLoopAndDomInfo();
133 for (const auto& bb : *fun)
134 {
135 if (pInfo->isUnreachable(llvmModuleSet()->getSVFBasicBlock(&bb)) &&
136 !visited.count(&bb.front()))
137 {
138 visited.insert(&bb.front());
139 (void)addBlockICFGNode(&bb.front());
140 worklist.push(&bb.front());
141 }
142 }

148void ICFGBuilder::processFunBody(WorkList& worklist)
149{
151 while (!worklist.empty())
152 {
153 const Instruction* inst = worklist.pop();
154 ICFGNode* srcNode = getICFGNode(inst);
155 if (SVFUtil::isa<ReturnInst>(inst))
156 {
157 FunExitICFGNode* FunExitICFGNode = getFunExitICFGNode(inst->getFunction());
158 icfg->addIntraEdge(srcNode, FunExitICFGNode);
159 }
160 InstVec nextInsts;
161 LLVMUtil::getNextInsts(inst, nextInsts);
162 s64_t branchID = 0;
163 for (InstVec::const_iterator nit = nextInsts.begin(), enit =
164 nextInsts.end(); nit != enit; ++nit)
165 {
166 const Instruction* succ = *nit;
167 ICFGNode* dstNode;
168 if (visited.find(succ) != visited.end())
169 {
170 dstNode = getICFGNode(succ);
171 }
172 else
173 {
174 visited.insert(succ);
175 dstNode = addBlockICFGNode(succ);
176 worklist.push(succ);
177 }
178
179
180 if (LLVMUtil::isNonInstricCallSite(inst))
181 {
182 RetICFGNode* retICFGNode = getRetICFGNode(inst);
183 srcNode = retICFGNode;
184 }
185
186 if (const BranchInst* br = SVFUtil::dyn_cast<BranchInst>(inst))
187 {
188 assert(branchID <= 1 && "if/else has more than two branches?");
189 if(br->isConditional())
190 icfg->addConditionalIntraEdge(srcNode, dstNode, 1 - branchID);
191 else
192 icfg->addIntraEdge(srcNode, dstNode);
193 }
194 else if (const SwitchInst* si = SVFUtil::dyn_cast<SwitchInst>(inst))
195 {
197 const ConstantInt* condVal = const_cast<SwitchInst*>(si)->findCaseDest(const_cast<BasicBlock*>(succ->getParent()));
199 s64_t val = -1;
200 if (condVal && condVal->getBitWidth() <= 64)
201 val = LLVMUtil::getIntegerValue(condVal).first;
202 icfg->addConditionalIntraEdge(srcNode, dstNode,val);
203 }
204 else
205 icfg->addIntraEdge(srcNode, dstNode);
206 branchID++;
207 }
208 }
209}

216void ICFGBuilder::processFunExit(const Function* f)
217{
218 FunExitICFGNode* FunExitICFGNode = getFunExitICFGNode(f);
219
220 for (const auto& bb : *f)
221 {
222 for (const auto& inst : bb)
223 {
224 if (SVFUtil::isa<ReturnInst>(&inst))
225 {
226 icfg->addIntraEdge(getICFGNode(&inst), FunExitICFGNode);
227 }
228 }
229 }

239InterICFGNode* ICFGBuilder::addInterBlockICFGNode(const Instruction* inst)
240{
241 assert(LLVMUtil::isCallSite(inst) && "not a call instruction?");
242 assert(LLVMUtil::isNonInstricCallSite(inst) && "associating an intrinsic debug instruction with an ICFGNode!");
243 assert(llvmModuleSet()->getCallBlock(inst)==nullptr && "duplicate CallICFGNode");
244 const CallBase* cb = SVFUtil::dyn_cast<CallBase>(inst);
245 bool isvcall = cppUtil::isVirtualCallSite(cb);
246 const FunObjVar* calledFunc = nullptr;
247 auto called_llvmval = cb->getCalledOperand()->stripPointerCasts();
248 if (const Function* called_llvmfunc = SVFUtil::dyn_cast<Function>(called_llvmval))
249 {
250 calledFunc = llvmModuleSet()->getFunObjVar(called_llvmfunc);
251 }
252 else
253 {
254 assert(SVFUtil::dyn_cast<Function>(called_llvmval) == nullptr && "must be nullptr");
255 }
256
257 SVFBasicBlock* bb = llvmModuleSet()->getSVFBasicBlock(inst->getParent());
258
259 CallICFGNode* callICFGNode = icfg->addCallICFGNode(
260 bb, llvmModuleSet()->getSVFType(inst->getType()),
261 calledFunc, cb->getFunctionType()->isVarArg(), isvcall,
262 isvcall ? cppUtil::getVCallIdx(cb) : 0,
263 isvcall ? cppUtil::getFunNameOfVCallSite(cb) : "");
264 llvmModuleSet()->addInstructionMap(inst, callICFGNode);
265
266 assert(llvmModuleSet()->getRetBlock(inst)==nullptr && "duplicate RetICFGNode");
267 RetICFGNode* retICFGNode = icfg->addRetICFGNode(callICFGNode);
268 llvmModuleSet()->addInstructionMap(inst, retICFGNode);
269
270 addICFGInterEdges(inst, LLVMUtil::getCallee(SVFUtil::cast<CallBase>(inst))); //creating interprocedural edges
271 return callICFGNode;
272}

277void ICFGBuilder::addICFGInterEdges(const Instruction* cs, const Function* callee)
278{
279
280 CallICFGNode* callICFGNode = getCallICFGNode(cs);
281 RetICFGNode* retBlockNode = getRetICFGNode(cs);
282
284 if(callee)
285 {
286 const FunObjVar* svfFun =
287 llvmModuleSet()->getFunObjVar(callee);
289 if (SVFUtil::isExtCall(svfFun))
290 {
291 icfg->addIntraEdge(callICFGNode, retBlockNode);
292 }
294 else
295 {
296 FunEntryICFGNode* calleeEntryNode = getFunEntryICFGNode(callee);
297 FunExitICFGNode* calleeExitNode = getFunExitICFGNode(callee);
298 icfg->addCallEdge(callICFGNode, calleeEntryNode);
299 icfg->addRetEdge(calleeExitNode, retBlockNode);
300 }
301 }
303 else
304 {
305 icfg->addIntraEdge(callICFGNode, retBlockNode);
306 }
307}

311void ICFGBuilder::connectGlobalToProgEntry()
312{
313 for (Module &M : llvmModuleSet()->getLLVMModules())
314 {
315 if (const Function* mainFunc = LLVMUtil::getProgEntryFunction(M))
316 {
317 // main function
318 FunEntryICFGNode* entryNode = getFunEntryICFGNode(mainFunc);
319 GlobalICFGNode* globalNode = getGlobalICFGNode();
320 IntraCFGEdge* intraEdge = new IntraCFGEdge(globalNode, entryNode);
321 icfg->addICFGEdge(intraEdge);
322 }
323 else
324 {
325 // not main function
326 }
327 }

330inline ICFGNode* ICFGBuilder::addBlockICFGNode(const Instruction* inst)
331{
332 ICFGNode* node;
333 if(LLVMUtil::isNonInstricCallSite(inst))
334 node = addInterBlockICFGNode(inst);
335 else
336 node = addIntraBlockICFGNode(inst);
337 const_cast<SVFBasicBlock*>(
338 llvmModuleSet()->getSVFBasicBlock(inst->getParent()))
339 ->addICFGNode(node);
340 return node;
341}

343IntraICFGNode* ICFGBuilder::addIntraBlockICFGNode(const Instruction* inst)
344{
345 IntraICFGNode* node = llvmModuleSet()->getIntraBlock(inst);
346 assert (node==nullptr && "no IntraICFGNode for this instruction?");
347 IntraICFGNode* sNode = icfg->addIntraICFGNode(
348 llvmModuleSet()->getSVFBasicBlock(inst->getParent()), SVFUtil::isa<ReturnInst>(inst));
349 llvmModuleSet()->addInstructionMap(inst, sNode);
350 return sNode;
351}
