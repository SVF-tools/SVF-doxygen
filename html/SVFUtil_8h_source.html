107 inline bool cmpPts (const PointsTo& lpts,const PointsTo& rpts)
108 {
109  if (lpts.count() != rpts.count())
110  return (lpts.count() < rpts.count());
111  else
112  {
113  PointsTo::iterator bit = lpts.begin(), eit = lpts.end();
114  PointsTo::iterator rbit = rpts.begin(), reit = rpts.end();
115  for (; bit != eit && rbit != reit; bit++, rbit++)
116  {
117  if (*bit != *rbit)
118  return (*bit < *rbit);
119  }
120 
121  return false;
122  }
123 }

237 inline std::vector<std::string> split(const std::string& s, char separator)
238 {
239  std::vector<std::string> output;
240  std::string::size_type prev_pos = 0, pos = 0;
241  while ((pos = s.find(separator, pos)) != std::string::npos)
242  {
243  std::string substring(s.substr(prev_pos, pos - prev_pos));
244  if (!substring.empty())
245  {
246  output.push_back(substring);
247  }
248  prev_pos = ++pos;
249  }
250  std::string lastSubstring(s.substr(prev_pos, pos - prev_pos));
251  if (!lastSubstring.empty())
252  {
253  output.push_back(lastSubstring);
254  }
255  return output;
256 }
