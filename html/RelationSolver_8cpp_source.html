33 IntervalESBase RelationSolver::bilateral(const IntervalESBase &domain, const Z3Expr& phi,
34  u32_t descend_check)
35 {
37  IntervalESBase upper = domain.top();
38  IntervalESBase lower = domain.bottom();
39  u32_t meets_in_a_row = 0;
40  z3::solver solver = Z3Expr::getSolver();
41  z3::params p(Z3Expr::getContext());
43  p.set(":timeout", static_cast<unsigned>(600)); // in milliseconds
44  solver.set(p);
45  IntervalESBase consequence;
46 
48  while (lower != upper)
49  {
50  if (meets_in_a_row == descend_check)
51  {
52  consequence = lower;
53  }
54  else
55  {
56  consequence = abstract_consequence(lower, upper, domain);
57  }
59  Z3Expr rhs = !(gamma_hat(consequence, domain));
60  solver.push();
61  solver.add(phi.getExpr() && rhs.getExpr());
62  Map<u32_t, double> solution;
63  z3::check_result checkRes = solver.check();
65  if (checkRes == z3::sat)
66  {
67  z3::model m = solver.get_model();
68  for (u32_t i = 0; i < m.size(); i++)
69  {
70  z3::func_decl v = m[i];
71  // assert(v.arity() == 0);
72  if (v.arity() != 0)
73  continue;
74  solution.emplace(std::stoi(v.name().str()),
75  m.get_const_interp(v).get_numeral_int());
76  }
77  for (const auto& item : domain.getVarToVal())
78  {
79  if (solution.find(item.first) == solution.end())
80  {
81  solution.emplace(item.first, 0);
82  }
83  }
84  solver.pop();
85  IntervalESBase newLower = domain.bottom();
86  newLower.joinWith(lower);
87  IntervalESBase rhs = beta(solution, domain);
88  newLower.joinWith(rhs);
89  lower = newLower;
90  meets_in_a_row = 0;
91  }
92  else
93  {
94  solver.pop();
95  if (checkRes == z3::unknown)
96  {
98  if (solver.reason_unknown() == "timeout")
99  return upper;
100  }
101  IntervalESBase newUpper = domain.top();
102  newUpper.meetWith(upper);
103  newUpper.meetWith(consequence);
104  upper = newUpper;
105  meets_in_a_row += 1;
106  }
107  }
108  return upper;
109 }

111 IntervalESBase RelationSolver::RSY(const IntervalESBase& domain, const Z3Expr& phi)
112 {

167 IntervalESBase RelationSolver::abstract_consequence(
168  const IntervalESBase& lower, const IntervalESBase& upper, const IntervalESBase& domain) const
169 {
170  /*Returns the "abstract consequence" of lower and upper.
171 
172  The abstract consequence must be a superset of lower and *NOT* a
173  superset of upper.
174 
175  Note that this is a fairly "simple" abstract consequence, in that it
176  sets only one variable to a non-top interval. This improves performance
177  of the SMT solver in many cases. In certain cases, other choices for
178  the abstract consequence will lead to better algorithm performance.*/
179 
180  for (auto it = domain.getVarToVal().begin();
181  it != domain.getVarToVal().end(); ++it)
183  {
184  IntervalESBase proposed = domain.top();
185  proposed._varToItvVal[it->first] = lower._varToItvVal.at(it->first);
188  if (!(proposed >= upper))
189  {
190  return proposed;
191  }
192  }
193  return lower;
194 }

196 Z3Expr RelationSolver::gamma_hat(const IntervalESBase& exeState) const
197 {
198  Z3Expr res(Z3Expr::getContext().bool_val(true));
199  for (auto& item : exeState.getVarToVal())
200  {
201  if (item.second.isBottom())
202  return Z3Expr::getContext().bool_val(false);
203  if (item.second.isTop())
204  continue;
205  Z3Expr v = toZ3Expr(item.first);
206  res = (res && v >= (int)item.second.lb().getNumeral() &&
207  v <= (int)item.second.ub().getNumeral()).simplify();
208  }
209  return res;
210 }

266 IntervalESBase RelationSolver::BS(const IntervalESBase& domain, const Z3Expr &phi)
267 {
270  u32_t bias = 0;
271  int infinity = (INT32_MAX) - 1;
272  // int infinity = 20;
273  Map<u32_t, NumericLiteral> ret;
274  Map<u32_t, NumericLiteral> low_values, high_values;
275  Z3Expr new_phi = phi;
277  for (const auto& item: domain.getVarToVal())
278  {
279  updateMap(ret, item.first, item.second.ub());
280  if (item.second.lb().is_minus_infinity())
281  updateMap(low_values, item.first, -infinity);
282  else
283  updateMap(low_values, item.first, item.second.lb());
284  if (item.second.ub().is_plus_infinity())
285  updateMap(high_values, item.first, infinity);
286  else
287  updateMap(high_values, item.first, item.second.ub());
288  if (item.first > bias)
289  bias = item.first + 1;
290  }
291  for (const auto& item: domain.getVarToVal())
292  {
294  u32_t reverse_key = item.first + bias;
295  updateMap(ret, reverse_key, -item.second.lb());
296  if (item.second.ub().is_plus_infinity())
297  updateMap(low_values, reverse_key, -infinity);
298  else
299  updateMap(low_values, reverse_key, -item.second.ub());
300  if (item.second.lb().is_minus_infinity())
301  updateMap(high_values, reverse_key, infinity);
302  else
303  updateMap(high_values, reverse_key, -item.second.lb());
305  new_phi = (new_phi && (toZ3Expr(reverse_key) == -1 * toZ3Expr(item.first)));
306  }
308  BoxedOptSolver(new_phi.simplify(), ret, low_values, high_values);
310  IntervalESBase retInv;
311  for (const auto& item: ret)
312  {
313  if (item.first >= bias)
314  {
315  if (item.second.equal(infinity))
316  retInv._varToItvVal[item.first - bias].setLb(Z3Expr::getContext().int_const("-oo"));
317  else
318  retInv._varToItvVal[item.first - bias].setLb(-item.second);
319  }
320  else
321  {
322  if (item.second.equal(infinity))
323  retInv._varToItvVal[item.first].setUb(Z3Expr::getContext().int_const("+oo"));
324  else
325  retInv._varToItvVal[item.first].setUb(item.second);
326  }
327  }
328  return retInv;
329 }

331 Map<u32_t, NumericLiteral> RelationSolver::BoxedOptSolver(const Z3Expr& phi, Map<u32_t, NumericLiteral>& ret, Map<u32_t, NumericLiteral>& low_values, Map<u32_t, NumericLiteral>& high_values)
332 {
334  Map<u32_t, Z3Expr> L_phi;
335  Map<u32_t, NumericLiteral> mid_values;
336  while (1)
337  {
338  L_phi.clear();
339  for (const auto& item : ret)
340  {
341  Z3Expr v = toZ3Expr(item.first);
342  if (low_values.at(item.first).leq(high_values.at(item.first)))
343  {
344  NumericLiteral mid = (low_values.at(item.first) + (high_values.at(item.first) - low_values.at(item.first)) / 2);
345  updateMap(mid_values, item.first, mid);
346  Z3Expr expr = ((int)mid.getNumeral() <= v && v <= (int)high_values.at(item.first).getNumeral());
347  L_phi[item.first] = expr;
348  }
349  }
350  if (L_phi.empty())
351  break;
352  else
353  decide_cpa_ext(phi, L_phi, mid_values, ret, low_values, high_values);
354  }
355  return ret;
356 }

359 void RelationSolver::decide_cpa_ext(const Z3Expr& phi,
360  Map<u32_t, Z3Expr>& L_phi,
361  Map<u32_t, NumericLiteral>& mid_values,
362  Map<u32_t, NumericLiteral>& ret,
363  Map<u32_t, NumericLiteral>& low_values,
364  Map<u32_t, NumericLiteral>& high_values)
365 {
366  while (1)
367  {
368  Z3Expr join_expr(Z3Expr::getContext().bool_val(false));
369  for (const auto& item : L_phi)
370  join_expr = (join_expr || item.second);
371  join_expr = (join_expr && phi).simplify();
372  z3::solver& solver = Z3Expr::getSolver();
373  solver.push();
374  solver.add(join_expr.getExpr());
375  Map<u32_t, double> solution;
376  z3::check_result checkRes = solver.check();
378  if (checkRes == z3::sat)
379  {
380  z3::model m = solver.get_model();
381  solver.pop();
382  for(const auto & item : L_phi)
383  {
384  u32_t id = item.first;
385  int value = m.eval(toZ3Expr(id).getExpr()).get_numeral_int();
386  // int value = m.eval(Z3Expr::getContext().int_const(std::to_string(id).c_str())).get_numeral_int();
389  Z3Expr expr = (item.second && toZ3Expr(id) == value);
390  solver.push();
391  solver.add(expr.getExpr());
392  // solution meets phi_id
393  if (solver.check() == z3::sat)
394  {
395  updateMap(ret, id, NumericLiteral(value));
396  updateMap(low_values, id, ret.at(id) + 1);
397 
398  NumericLiteral mid = (low_values.at(id) + high_values.at(id) + 1) / 2;
399  updateMap(mid_values, id, mid);
400  Z3Expr v = toZ3Expr(id);
401  // Z3Expr v = Z3Expr::getContext().int_const(std::to_string(id).c_str());
402  Z3Expr expr = ((int)mid_values.at(id).getNumeral() <= v && v <= (int)high_values.at(id).getNumeral());
403  L_phi[id] = expr;
404  }
405  solver.pop();
406  }
407  }
408  else
409  {
410  solver.pop();
411  for (const auto& item : L_phi)
412  high_values.at(item.first) = mid_values.at(item.first) - 1;
413  return;
414  }
415  }
