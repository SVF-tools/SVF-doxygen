287 const std::string SVFBugEvent::getEventDescription() const
288 {
289  switch(getEventType())
290  {
291  case SVFBugEvent::Branch:
292  {
293  if (typeAndInfoFlag & BRANCHFLAGMASK)
294  {
295  return "True";
296  }
297  else
298  {
299  return "False";
300  }
301  break;
302  }
303  case SVFBugEvent::CallSite:
304  {
305  std::string description("calls ");
306  const SVFFunction *callee = SVFUtil::getCallee(eventInst);
307  if(callee == nullptr)
308  {
309  description += "<unknown>";
310  }
311  else
312  {
313  description += callee->getName();
314  }
315  return description;
316  break;
317  }
318  case SVFBugEvent::SourceInst:
319  {
320  return "None";
321  }
322  default:
323  {
324  assert(false && "No such type of event!");
325  abort();
326  }
327  }
328 }

338 void SVFBugReport::dumpToJsonFile(const std::string& filePath) const
339 {
340  std::map<u32_t, std::string> eventType2Str =
341  {
342  {SVFBugEvent::CallSite, "call site"},
343  {SVFBugEvent::Caller, "caller"},
344  {SVFBugEvent::Loop, "loop"},
345  {SVFBugEvent::Branch, "branch"}
346  };
347 
348  ofstream jsonFile(filePath, ios::out);
349 
350  jsonFile << "{\n";
351 
353  jsonFile << "\"Defects\": [\n";
354  size_t commaCounter = bugSet.size() - 1;
355  for (auto bugPtr : bugSet)
356  {
357  cJSON *singleDefect = cJSON_CreateObject();
358 
360  cJSON *bugType = cJSON_CreateString(
361  GenericBug::BugType2Str.at(bugPtr->getBugType()).c_str());
362  cJSON_AddItemToObject(singleDefect, "DefectType", bugType);
363 
364  cJSON *bugLoc = cJSON_Parse(bugPtr->getLoc().c_str());
365  if (bugLoc == nullptr)
366  {
367  bugLoc = cJSON_CreateObject();
368  }
369  cJSON_AddItemToObject(singleDefect, "Location", bugLoc);
370 
371  cJSON *bugFunction = cJSON_CreateString(
372  bugPtr->getFuncName().c_str());
373  cJSON_AddItemToObject(singleDefect, "Function", bugFunction);
374 
375  cJSON_AddItemToObject(singleDefect, "Description",
376  bugPtr->getBugDescription());
377 
379  cJSON *eventList = cJSON_CreateArray();
380  const GenericBug::EventStack &bugEventStack = bugPtr->getEventStack();
381  if (BufferOverflowBug::classof(bugPtr))
382  {
383  // Add only when bug is context sensitive
384  for (const SVFBugEvent &event : bugEventStack)
385  {
386  if (event.getEventType() == SVFBugEvent::SourceInst)
387  {
388  continue;
389  }
390 
391  cJSON *singleEvent = cJSON_CreateObject();
392  // Event type
393  cJSON *eventType = cJSON_CreateString(
394  eventType2Str[event.getEventType()].c_str());
395  cJSON_AddItemToObject(singleEvent, "EventType", eventType);
396  // Function name
397  cJSON *eventFunc = cJSON_CreateString(
398  event.getFuncName().c_str());
399  cJSON_AddItemToObject(singleEvent, "Function", eventFunc);
400  // Event loc
401  cJSON *eventLoc = cJSON_Parse(event.getEventLoc().c_str());
402  if (eventLoc == nullptr)
403  {
404  eventLoc = cJSON_CreateObject();
405  }
406  cJSON_AddItemToObject(singleEvent, "Location", eventLoc);
407  // Event description
408  cJSON *eventDescription = cJSON_CreateString(
409  event.getEventDescription().c_str());
410  cJSON_AddItemToObject(singleEvent, "Description", eventDescription);
411 
412  cJSON_AddItemToArray(eventList, singleEvent);
413  }
414  }
415  cJSON_AddItemToObject(singleDefect, "Events", eventList);
416 
418  char *singleDefectStr = cJSON_Print(singleDefect);
419  jsonFile << singleDefectStr;
420  if (commaCounter != 0)
421  {
422  jsonFile << ",\n";
423  }
424  commaCounter--;
425 
427  cJSON_Delete(singleDefect);
428  }
429  jsonFile << "\n],\n";
430 
432  jsonFile << "\"Time\": " << time << ",\n";
433  jsonFile << "\"Memory\": \"" << mem << "\",\n";
434  jsonFile << "\"Coverage\": " << coverage << "\n";
435 
436  jsonFile << "}";
437  jsonFile.close();
438 }
