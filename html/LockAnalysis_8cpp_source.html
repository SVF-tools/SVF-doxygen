103  {

105  PTACallGraphNode* cgnode = tcg->getCallGraphNode(fun);
106  if (visited.find(cgnode) == visited.end())

108  worklist.push(cgnode);
109  visited.insert(cgnode);
110  }
111  }
112  for (InstSet::iterator it = unlocksites.begin(), eit = unlocksites.end(); it != eit; ++it)
113  {
114  const SVFFunction* fun = (*it)->getParent()->getParent();
115  PTACallGraphNode* cgnode = tcg->getCallGraphNode(fun);
116  if (visited.find(cgnode) == visited.end())
117  {
118  worklist.push(cgnode);
119  visited.insert(cgnode);
120  }
121  }
122  while (!worklist.empty())
123  {
124  const PTACallGraphNode* node = worklist.pop();
125  lockcandidateFuncSet.insert(node->getFunction());
126  for (PTACallGraphNode::const_iterator nit = node->InEdgeBegin(), neit = node->InEdgeEnd(); nit != neit; nit++)

142 void LockAnalysis::analyzeIntraProcedualLock()
143 {
144 
145  // Identify the protected Instructions.
146  for (InstSet::const_iterator it = locksites.begin(), ie = locksites.end(); it != ie; ++it)
147  {
148  const SVFInstruction* lockSite = *it;
149  assert(SVFUtil::isCallSite(lockSite) && "Lock acquire instruction must be a CallSite");
150 
151  // Perform forward traversal
152  InstSet forwardInsts;
153  InstSet backwardInsts;
154  InstSet unlockSet;
155 
156  bool forward = intraForwardTraverse(lockSite,unlockSet,forwardInsts);
157  bool backward = intraBackwardTraverse(unlockSet,backwardInsts);
158 
160  if(forward && backward)
161  addIntraLock(lockSite,forwardInsts);
162  else if(forward && !backward)
163  addCondIntraLock(lockSite,forwardInsts);
164  }

170 bool LockAnalysis::intraForwardTraverse(const SVFInstruction* lockSite, InstSet& unlockSet, InstSet& forwardInsts)
171 {
172 
173  const SVFFunction* svfFun = lockSite->getFunction();
174 
175  InstVec worklist;
176  worklist.push_back(lockSite);
177  while (!worklist.empty())
178  {
179  const SVFInstruction *I = worklist.back();
180  worklist.pop_back();
181  const SVFInstruction* exitInst = svfFun->getExitBB()->back();
182  if(exitInst == I)
183  return false;
184 
185  // Skip the visited Instructions.
186  if (forwardInsts.find(I)!=forwardInsts.end())
187  continue;
188  forwardInsts.insert(I);
189 
190  if (isTDRelease(I) && isAliasedLocks(lockSite, I))
191  {
192  unlockSet.insert(I);
193  DBOUT(DMTA, outs() << "LockAnalysis ci lock -- " << lockSite->getSourceLoc()<<"\n");
194  DBOUT(DMTA, outs() << "LockAnalysis ci unlock -- " << I->getSourceLoc()<<"\n");
195  continue;
196  }
197 
198  const InstVec& nextInsts = I->getSuccInstructions();
199  for (InstVec::const_iterator nit = nextInsts.begin(), enit = nextInsts.end(); nit != enit; ++nit)
200  {
201  worklist.push_back(*nit);
202  }
203  }
204 
205  return true;
206 }

212 bool LockAnalysis::intraBackwardTraverse(const InstSet& unlockSet, InstSet& backwardInsts)
213 {
214 
215  InstVec worklist;
216  for(InstSet::const_iterator it = unlockSet.begin(), eit = unlockSet.end(); it!=eit; ++it)
217  {
218  const SVFInstruction* unlockSite = *it;
219  const SVFInstruction* entryInst = unlockSite->getFunction()->getEntryBlock()->back();
220  worklist.push_back(*it);
221 
222  while (!worklist.empty())
223  {
224  const SVFInstruction *I = worklist.back();
225  worklist.pop_back();
226 
227  if(entryInst == I)
228  return false;
229 
230  // Skip the visited Instructions.
231  if (backwardInsts.find(I)!=backwardInsts.end())
232  continue;
233  backwardInsts.insert(I);
234 
235  if (isTDAcquire(I) && isAliasedLocks(unlockSite, I))
236  {
237  DBOUT(DMTA, outs() << "LockAnalysis ci lock -- " << I->getSourceLoc()<<"\n");
238  DBOUT(DMTA, outs() << "LockAnalysis ci unlock -- " << unlockSite->getSourceLoc()<<"\n");
239  continue;
240  }
241 
242  const InstVec& nextInsts = I->getPredInstructions();
243  for (InstVec::const_iterator nit = nextInsts.begin(), enit = nextInsts.end(); nit != enit; ++nit)
244  {
245  worklist.push_back(*nit);
246  }
247  }
248  }
249 
250  return true;

254 void LockAnalysis::collectCxtLock()
255 {
256  FunSet entryFuncSet = tct->getEntryProcs();
257  for (FunSet::const_iterator it = entryFuncSet.begin(), eit = entryFuncSet.end(); it != eit; ++it)
258  {
259  if (!isLockCandidateFun(*it))
260  continue;
261  CallStrCxt cxt;
262  CxtLockProc t(cxt, *it);
263  pushToCTPWorkList(t);
264  }
265 
266  while (!clpList.empty())
267  {
268  CxtLockProc clp = popFromCTPWorkList();
269  PTACallGraphNode* cgNode = getTCG()->getCallGraphNode(clp.getProc());
270  // lzh TODO.
271  if (!isLockCandidateFun(cgNode->getFunction()))
272  continue;
273 
274  for (PTACallGraphNode::const_iterator nit = cgNode->OutEdgeBegin(), neit = cgNode->OutEdgeEnd(); nit != neit; nit++)
275  {
276  const PTACallGraphEdge* cgEdge = (*nit);
277 
278  for (PTACallGraphEdge::CallInstSet::const_iterator cit = cgEdge->directCallsBegin(), ecit = cgEdge->directCallsEnd();
279  cit != ecit; ++cit)
280  {
281  DBOUT(DMTA,
282  outs() << "\nCollecting CxtLocks: handling direct call:" << **cit << "\t" << cgEdge->getSrcNode()->getFunction()->getName()
283  << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
284  handleCallRelation(clp, cgEdge, getSVFCallSite((*cit)->getCallSite()));

286  for (PTACallGraphEdge::CallInstSet::const_iterator ind = cgEdge->indirectCallsBegin(), eind = cgEdge->indirectCallsEnd();
287  ind != eind; ++ind)
288  {
289  DBOUT(DMTA,

294  }

325 void LockAnalysis::analyzeLockSpanCxtStmt()
326 {
327 
328  FunSet entryFuncSet = tct->getEntryProcs();
329  for (FunSet::const_iterator it = entryFuncSet.begin(), eit = entryFuncSet.end(); it != eit; ++it)
330  {
331  if (!isLockCandidateFun(*it))
332  continue;
333  CallStrCxt cxt;
334  const SVFInstruction* frontInst = (*it)->getEntryBlock()->front();
335  CxtStmt cxtstmt(cxt, frontInst);
336  pushToCTSWorkList(cxtstmt);
337  }
338 
339  while (!cxtStmtList.empty())
340  {
341  CxtStmt cts = popFromCTSWorkList();
342 
343  touchCxtStmt(cts);
344  const SVFInstruction* curInst = cts.getStmt();
345  instToCxtStmtSet[curInst].insert(cts);
346 
347  DBOUT(DMTA, outs() << "\nVisit cxtStmt: ");
348  DBOUT(DMTA, cts.dump());
349 
350  DBOUT(DMTA, outs() << "\nIts cxt lock sets: ");
351  DBOUT(DMTA, printLocks(cts));
352 
353  if (isTDFork(curInst))
354  {
355  handleFork(cts);
356  }
357  else if (isTDAcquire(curInst))
358  {
359  assert(hasCxtLock(cts) && "context-sensitive lock not found!!");
360  if(addCxtStmtToSpan(cts,cts))
361  handleIntra(cts);
362  }
363  else if (isTDRelease(curInst))
364  {
365  if(removeCxtStmtToSpan(cts,cts))
366  handleIntra(cts);
367  }
368  else if (SVFUtil::isCallSite(curInst) && !isExtCall(curInst))
369  {
370  handleCall(cts);
371  }
372  else if (curInst->isRetInst())
373  {
374  handleRet(cts);
375  }
376  else
377  {
378  handleIntra(cts);
379  }
380 
381  }

448 void LockAnalysis::handleRet(const CxtStmt& cts)
449 {
450 
451  const SVFInstruction* curInst = cts.getStmt();
452  const CallStrCxt& curCxt = cts.getContext();
453  const SVFFunction* svffun = curInst->getFunction();
454  PTACallGraphNode* curFunNode = getTCG()->getCallGraphNode(svffun);
455 
456  for (PTACallGraphNode::const_iterator it = curFunNode->getInEdges().begin(), eit = curFunNode->getInEdges().end(); it != eit; ++it)
457  {
458  PTACallGraphEdge* edge = *it;
459  if (SVFUtil::isa<ThreadForkEdge, ThreadJoinEdge>(edge))
460  continue;
461  for (PTACallGraphEdge::CallInstSet::const_iterator cit = (edge)->directCallsBegin(), ecit = (edge)->directCallsEnd(); cit != ecit;
462  ++cit)
463  {
464  CallStrCxt newCxt = curCxt;
465  const SVFInstruction* inst = (*cit)->getCallSite();
466  if (matchCxt(newCxt, inst, curFunNode->getFunction()))
467  {
468  const InstVec& nextInsts = inst->getSuccInstructions();
469  for (InstVec::const_iterator nit = nextInsts.begin(), enit = nextInsts.end(); nit != enit; ++nit)
470  {
471  CxtStmt newCts(newCxt, *nit);
472  markCxtStmtFlag(newCts, cts);
473  }
474  }
475  }
476  for (PTACallGraphEdge::CallInstSet::const_iterator cit = (edge)->indirectCallsBegin(), ecit = (edge)->indirectCallsEnd();
477  cit != ecit; ++cit)
478  {
479  CallStrCxt newCxt = curCxt;

649 bool LockAnalysis::isInSameCISpan(const SVFInstruction *i1, const SVFInstruction *i2) const
650 {
651  if(!isInsideCondIntraLock(i1) && !isInsideCondIntraLock(i2))
652  {
653  const InstSet& lockset1 = getIntraLockSet(i1);
654  const InstSet& lockset2 = getIntraLockSet(i2);
655  for (InstSet::const_iterator cil1 = lockset1.begin(), ecil1 = lockset1.end(); cil1!=ecil1; ++cil1)
656  {
657  for (InstSet::const_iterator cil2=lockset2.begin(), ecil2=lockset2.end(); cil2!=ecil2; ++cil2)
658  {
659  if (*cil1==*cil2)
