448void VFG::addVFGNodes()
449{
450
451 // initialize dummy definition null pointers in order to uniform the construction
452 // to be noted for black hole pointer it has already has address edge connected,
453 // and its definition will be set when processing addr SVFIR edge.
454 addNullPtrVFGNode(pag->getGNode(pag->getNullPtr()));
455
456 // initialize address nodes
457 SVFStmt::SVFStmtSetTy& addrs = getPAGEdgeSet(SVFStmt::Addr);
458 for (SVFStmt::SVFStmtSetTy::iterator iter = addrs.begin(), eiter =
459 addrs.end(); iter != eiter; ++iter)
460 {
461 addAddrVFGNode(SVFUtil::cast<AddrStmt>(*iter));
462 }
463
464 // initialize copy nodes
465 SVFStmt::SVFStmtSetTy& copys = getPAGEdgeSet(SVFStmt::Copy);
466 for (SVFStmt::SVFStmtSetTy::iterator iter = copys.begin(), eiter =
467 copys.end(); iter != eiter; ++iter)
468 {
469 const CopyStmt* edge = SVFUtil::cast<CopyStmt>(*iter);
470 assert(!isPhiCopyEdge(edge) && "Copy edges can not be a PhiNode (or from PhiNode)");
471 addCopyVFGNode(edge);
472 }
473
474 // initialize gep nodes
475 SVFStmt::SVFStmtSetTy& ngeps = getPAGEdgeSet(SVFStmt::Gep);
476 for (SVFStmt::SVFStmtSetTy::iterator iter = ngeps.begin(), eiter =
477 ngeps.end(); iter != eiter; ++iter)
478 {
479 addGepVFGNode(SVFUtil::cast<GepStmt>(*iter));
480 }
481
482 // initialize load nodes
483 SVFStmt::SVFStmtSetTy& loads = getPAGEdgeSet(SVFStmt::Load);
484 for (SVFStmt::SVFStmtSetTy::iterator iter = loads.begin(), eiter =
485 loads.end(); iter != eiter; ++iter)
486 {
487 addLoadVFGNode(SVFUtil::cast<LoadStmt>(*iter));
488 }
489
490 // initialize store nodes
491 SVFStmt::SVFStmtSetTy& stores = getPAGEdgeSet(SVFStmt::Store);
492 for (SVFStmt::SVFStmtSetTy::iterator iter = stores.begin(), eiter =
493 stores.end(); iter != eiter; ++iter)
494 {
495 addStoreVFGNode(SVFUtil::cast<StoreStmt>(*iter));
496 }
497
498 SVFStmt::SVFStmtSetTy& forks = getPAGEdgeSet(SVFStmt::ThreadFork);
499 for (SVFStmt::SVFStmtSetTy::iterator iter = forks.begin(), eiter =
500 forks.end(); iter != eiter; ++iter)
501 {
502 TDForkPE* forkedge = SVFUtil::cast<TDForkPE>(*iter);
503 addActualParmVFGNode(forkedge->getRHSVar(),forkedge->getCallSite());
504 }
505
506 // initialize actual parameter nodes
507 for(SVFIR::CSToArgsListMap::iterator it = pag->getCallSiteArgsMap().begin(), eit = pag->getCallSiteArgsMap().end(); it !=eit; ++it)
508 {
509
510 for(SVFIR::SVFVarList::iterator pit = it->second.begin(), epit = it->second.end(); pit!=epit; ++pit)
511 {
512 const PAGNode* pagNode = *pit;
513 if (isInterestedPAGNode(pagNode))
514 addActualParmVFGNode(pagNode,it->first);
515 }
516 }
517
518 // initialize actual return nodes (callsite return)
519 for(SVFIR::CSToRetMap::iterator it = pag->getCallSiteRets().begin(), eit = pag->getCallSiteRets().end(); it !=eit; ++it)
520 {
521
525 if(isInterestedPAGNode(it->second) == false || hasDef(it->second))
526 continue;
527
528 addActualRetVFGNode(it->second,it->first->getCallICFGNode());
529 }
530
531 // initialize formal parameter nodes
532 for(SVFIR::FunToArgsListMap::iterator it = pag->getFunArgsMap().begin(), eit = pag->getFunArgsMap().end(); it !=eit; ++it)
533 {
534 const SVFFunction* func = it->first;
535
536 for(SVFIR::SVFVarList::iterator pit = it->second.begin(), epit = it->second.end(); pit!=epit; ++pit)
537 {
538 const PAGNode* param = *pit;
539 if (isInterestedPAGNode(param) == false || hasBlackHoleConstObjAddrAsDef(param))
540 continue;
541
542 CallPESet callPEs;
543 if (param->hasIncomingEdges(SVFStmt::Call))
544 {
545 for (SVFStmt::SVFStmtSetTy::const_iterator cit = param->getIncomingEdgesBegin(SVFStmt::Call), ecit =
546 param->getIncomingEdgesEnd(SVFStmt::Call); cit != ecit; ++cit)
547 {
548 CallPE* callPE = SVFUtil::cast<CallPE>(*cit);
549 if (isInterestedPAGNode(callPE->getRHSVar()))
550 callPEs.insert(callPE);
551 }
552 }
553 addFormalParmVFGNode(param,func,callPEs);
554 }
555
556 if (func->isVarArg())
557 {
558 const PAGNode* varParam = pag->getGNode(pag->getVarargNode(func));
559 if (isInterestedPAGNode(varParam) == false || hasBlackHoleConstObjAddrAsDef(varParam))
560 continue;
561
562 CallPESet callPEs;
563 if (varParam->hasIncomingEdges(SVFStmt::Call))
564 {
565 for(SVFStmt::SVFStmtSetTy::const_iterator cit = varParam->getIncomingEdgesBegin(SVFStmt::Call),
566 ecit = varParam->getIncomingEdgesEnd(SVFStmt::Call); cit!=ecit; ++cit)
567 {
568 CallPE* callPE = SVFUtil::cast<CallPE>(*cit);
569 if(isInterestedPAGNode(callPE->getRHSVar()))
570 callPEs.insert(callPE);
571 }
572 }
573 addFormalParmVFGNode(varParam,func,callPEs);
574 }
575 }
576
577 // initialize formal return nodes (callee return)
578 for (SVFIR::FunToRetMap::iterator it = pag->getFunRets().begin(), eit = pag->getFunRets().end(); it != eit; ++it)
579 {
580 const SVFFunction* func = it->first;
581
582 const PAGNode* uniqueFunRetNode = it->second;
583
584 RetPESet retPEs;
585 if (uniqueFunRetNode->hasOutgoingEdges(SVFStmt::Ret))
586 {
587 for (SVFStmt::SVFStmtSetTy::const_iterator cit = uniqueFunRetNode->getOutgoingEdgesBegin(SVFStmt::Ret),
588 ecit = uniqueFunRetNode->getOutgoingEdgesEnd(SVFStmt::Ret);
589 cit != ecit; ++cit)
590 {
591 const RetPE* retPE = SVFUtil::cast<RetPE>(*cit);
592 if (isInterestedPAGNode(retPE->getLHSVar()))
593 retPEs.insert(retPE);
594 }
595 }
596
597 if(isInterestedPAGNode(uniqueFunRetNode))
598 addFormalRetVFGNode(uniqueFunRetNode, func, retPEs);
599 }
600
601 // initialize llvm phi nodes (phi of top level pointers)
602 SVFStmt::SVFStmtSetTy& phis = getPAGEdgeSet(SVFStmt::Phi);
603 for (SVFStmt::SVFStmtSetTy::iterator iter = phis.begin(), eiter =
604 phis.end(); iter != eiter; ++iter)
605 {
606 const PhiStmt* edge = SVFUtil::cast<PhiStmt>(*iter);
607 if(isInterestedPAGNode(edge->getRes()))
608 addIntraPHIVFGNode(edge);
609 }
610 // initialize select statement
611 SVFStmt::SVFStmtSetTy& selects = getPAGEdgeSet(SVFStmt::Select);
612 for (SVFStmt::SVFStmtSetTy::iterator iter = selects.begin(), eiter =
613 selects.end(); iter != eiter; ++iter)
614 {
615 const MultiOpndStmt* edge = SVFUtil::cast<MultiOpndStmt>(*iter);
616 if(isInterestedPAGNode(edge->getRes()))
617 addIntraPHIVFGNode(edge);
618 }
619 // initialize llvm binary nodes (binary operators)
620 SVFStmt::SVFStmtSetTy& binaryops = getPAGEdgeSet(SVFStmt::BinaryOp);
621 for (SVFStmt::SVFStmtSetTy::iterator iter = binaryops.begin(), eiter =
622 binaryops.end(); iter != eiter; ++iter)
623 {
624 const BinaryOPStmt* edge = SVFUtil::cast<BinaryOPStmt>(*iter);
625 if(isInterestedPAGNode(edge->getRes()))
626 addBinaryOPVFGNode(edge);
627 }
628 // initialize llvm unary nodes (unary operators)
629 SVFStmt::SVFStmtSetTy& unaryops = getPAGEdgeSet(SVFStmt::UnaryOp);
630 for (SVFStmt::SVFStmtSetTy::iterator iter = unaryops.begin(), eiter =
631 unaryops.end(); iter != eiter; ++iter)
632 {
633 const UnaryOPStmt* edge = SVFUtil::cast<UnaryOPStmt>(*iter);
634 if(isInterestedPAGNode(edge->getRes()))
635 addUnaryOPVFGNode(edge);
636 }
637 // initialize llvm unary nodes (unary operators)
638 SVFStmt::SVFStmtSetTy& brs = getPAGEdgeSet(SVFStmt::Branch);
639 for (SVFStmt::SVFStmtSetTy::iterator iter = brs.begin(), eiter =
640 brs.end(); iter != eiter; ++iter)
641 {
642 const BranchStmt* edge = SVFUtil::cast<BranchStmt>(*iter);
643 if(isInterestedPAGNode(edge->getBranchInst()))
644 addBranchVFGNode(edge);
645 }
646 // initialize llvm cmp nodes (comparison)
647 SVFStmt::SVFStmtSetTy& cmps = getPAGEdgeSet(SVFStmt::Cmp);
648 for (SVFStmt::SVFStmtSetTy::iterator iter = cmps.begin(), eiter =
649 cmps.end(); iter != eiter; ++iter)
650 {
651 const CmpStmt* edge = SVFUtil::cast<CmpStmt>(*iter);
652 if(isInterestedPAGNode(edge->getRes()))
653 addCmpVFGNode(edge);
654 }
655}

660VFGEdge* VFG::addIntraDirectVFEdge(NodeID srcId, NodeID dstId)
661{
662 VFGNode* srcNode = getVFGNode(srcId);
663 VFGNode* dstNode = getVFGNode(dstId);
664 checkIntraEdgeParents(srcNode, dstNode);
665 VFGEdge* edge = hasIntraVFGEdge(srcNode, dstNode, VFGEdge::IntraDirectVF);
666 if (edge != nullptr)
667 {
668 assert(edge->isDirectVFGEdge() && "this should be a direct value flow edge!");
669 return nullptr;
670 }
671 else
672 {
673 if(srcNode!=dstNode)
674 {
675 IntraDirSVFGEdge* directEdge = new IntraDirSVFGEdge(srcNode,dstNode);
676 return (addVFGEdge(directEdge) ? directEdge : nullptr);
677 }
678 else
679 return nullptr;
680 }
681}

686VFGEdge* VFG::addCallEdge(NodeID srcId, NodeID dstId, CallSiteID csId)
687{
688 VFGNode* srcNode = getVFGNode(srcId);
689 VFGNode* dstNode = getVFGNode(dstId);

691 if (edge != nullptr)
692 {

694 return nullptr;
695 }
696 else

698 CallDirSVFGEdge* callEdge = new CallDirSVFGEdge(srcNode,dstNode,csId);
699 return (addVFGEdge(callEdge) ? callEdge : nullptr);
700 }

706VFGEdge* VFG::addRetEdge(NodeID srcId, NodeID dstId, CallSiteID csId)
707{
708 VFGNode* srcNode = getVFGNode(srcId);
709 VFGNode* dstNode = getVFGNode(dstId);
710 VFGEdge* edge = hasInterVFGEdge(srcNode, dstNode, VFGEdge::RetDirVF, csId);
711 if (edge != nullptr)
712 {
713 assert(edge->isRetDirectVFGEdge() && "this should be a direct value flow edge!");
714 return nullptr;
715 }
716 else
717 {
718 RetDirSVFGEdge* retEdge = new RetDirSVFGEdge(srcNode,dstNode,csId);
719 return (addVFGEdge(retEdge) ? retEdge : nullptr);
720 }

727void VFG::connectDirectVFGEdges()
728{
729
730 for(iterator it = begin(), eit = end(); it!=eit; ++it)
731 {
732 NodeID nodeId = it->first;
733 VFGNode* node = it->second;
734
735 if(StmtVFGNode* stmtNode = SVFUtil::dyn_cast<StmtVFGNode>(node))
736 {
738 if(SVFUtil::isa<AddrVFGNode>(stmtNode))
739 continue;
741 if (stmtNode->getPAGSrcNode()->isConstDataOrAggDataButNotNullPtr() == false)
742 // for ptr vfg, we skip src node of integer type if it is at a int2ptr copystmt
743 if(isInterestedPAGNode(stmtNode->getPAGSrcNode()))
744 addIntraDirectVFEdge(getDef(stmtNode->getPAGSrcNode()), nodeId);
745 if (const GepStmt* gepStmt = SVFUtil::dyn_cast<GepStmt>(stmtNode->getPAGEdge()))
746 {
747 for (const auto &varType: gepStmt->getOffsetVarAndGepTypePairVec())
748 {
749 if(varType.first->isConstDataOrAggDataButNotNullPtr() || isInterestedPAGNode(varType.first) == false)
750 continue;
751 addIntraDirectVFEdge(getDef(varType.first), nodeId);
752 }
753 }
755 if(SVFUtil::isa<StoreVFGNode>(stmtNode) && (stmtNode->getPAGDstNode()->isConstDataOrAggDataButNotNullPtr() == false))
756 {
757 addIntraDirectVFEdge(getDef(stmtNode->getPAGDstNode()), nodeId);
758 }
759
760 }
761 else if(PHIVFGNode* phiNode = SVFUtil::dyn_cast<PHIVFGNode>(node))
762 {
763 for (PHIVFGNode::OPVers::const_iterator it = phiNode->opVerBegin(), eit = phiNode->opVerEnd(); it != eit; it++)
764 {
765 if (it->second->isConstDataOrAggDataButNotNullPtr() == false)
766 addIntraDirectVFEdge(getDef(it->second), nodeId);
767 }
768 }
769 else if(BinaryOPVFGNode* binaryNode = SVFUtil::dyn_cast<BinaryOPVFGNode>(node))
770 {
771 for (BinaryOPVFGNode::OPVers::const_iterator it = binaryNode->opVerBegin(), eit = binaryNode->opVerEnd(); it != eit; it++)
772 {
773 if (it->second->isConstDataOrAggDataButNotNullPtr() == false)
774 addIntraDirectVFEdge(getDef(it->second), nodeId);
775 }
776 }
777 else if(UnaryOPVFGNode* unaryNode = SVFUtil::dyn_cast<UnaryOPVFGNode>(node))
778 {
779 for (UnaryOPVFGNode::OPVers::const_iterator it = unaryNode->opVerBegin(), eit = unaryNode->opVerEnd(); it != eit; it++)
780 {
781 if (it->second->isConstDataOrAggDataButNotNullPtr() == false)
782 addIntraDirectVFEdge(getDef(it->second), nodeId);
783 }
784 }
785 else if(CmpVFGNode* cmpNode = SVFUtil::dyn_cast<CmpVFGNode>(node))
786 {
787 for (CmpVFGNode::OPVers::const_iterator it = cmpNode->opVerBegin(), eit = cmpNode->opVerEnd(); it != eit; it++)
788 {
789 if (it->second->isConstDataOrAggDataButNotNullPtr() == false)
790 addIntraDirectVFEdge(getDef(it->second), nodeId);
791 }
792 }
793 else if(BranchVFGNode* branchNode = SVFUtil::dyn_cast<BranchVFGNode>(node))
794 {
795 const SVFVar* cond = branchNode->getBranchStmt()->getCondition();
796 if (cond->isConstDataOrAggDataButNotNullPtr() == false)
797 addIntraDirectVFEdge(getDef(cond), nodeId);
798 }
799 else if(ActualParmVFGNode* actualParm = SVFUtil::dyn_cast<ActualParmVFGNode>(node))
800 {
801 if (actualParm->getParam()->isConstDataOrAggDataButNotNullPtr() == false)
802 addIntraDirectVFEdge(getDef(actualParm->getParam()), nodeId);
803 }
804 else if(FormalParmVFGNode* formalParm = SVFUtil::dyn_cast<FormalParmVFGNode>(node))
805 {
806 for(CallPESet::const_iterator it = formalParm->callPEBegin(), eit = formalParm->callPEEnd();
807 it!=eit; ++it)
808 {
809 const CallICFGNode* cs = (*it)->getCallSite();
810 ActualParmVFGNode* acutalParm = getActualParmVFGNode((*it)->getRHSVar(),cs);
811 addInterEdgeFromAPToFP(acutalParm,formalParm,getCallSiteID(cs, formalParm->getFun()));
812 }
813 }
814 else if(FormalRetVFGNode* calleeRet = SVFUtil::dyn_cast<FormalRetVFGNode>(node))
815 {
817 addIntraDirectVFEdge(getDef(calleeRet->getRet()), nodeId);
818
820 for(RetPESet::const_iterator it = calleeRet->retPEBegin(), eit = calleeRet->retPEEnd(); it!=eit; ++it)
821 {
822 ActualRetVFGNode* callsiteRev = getActualRetVFGNode((*it)->getLHSVar());
823 const CallICFGNode* callBlockNode = (*it)->getCallSite();
824 addInterEdgeFromFRToAR(calleeRet,callsiteRev, getCallSiteID(callBlockNode, calleeRet->getFun()));
825 }
826 }
829 }
830
832 if(Options::EnableThreadCallGraph())
833 {
835 SVFStmt::SVFStmtSetTy& forks = getPAGEdgeSet(SVFStmt::ThreadFork);
836 for (SVFStmt::SVFStmtSetTy::iterator iter = forks.begin(), eiter =
837 forks.end(); iter != eiter; ++iter)
838 {
839 TDForkPE* forkedge = SVFUtil::cast<TDForkPE>(*iter);
840 ActualParmVFGNode* acutalParm = getActualParmVFGNode(forkedge->getRHSVar(),forkedge->getCallSite());
841 FormalParmVFGNode* formalParm = getFormalParmVFGNode(forkedge->getLHSVar());
842 addInterEdgeFromAPToFP(acutalParm,formalParm,getCallSiteID(forkedge->getCallSite(), formalParm->getFun()));
843 }
845 SVFStmt::SVFStmtSetTy& joins = getPAGEdgeSet(SVFStmt::ThreadJoin);
846 for (SVFStmt::SVFStmtSetTy::iterator iter = joins.begin(), eiter =
847 joins.end(); iter != eiter; ++iter)
848 {
849 TDJoinPE* joinedge = SVFUtil::cast<TDJoinPE>(*iter);
850 NodeID callsiteRev = getDef(joinedge->getLHSVar());
851 FormalRetVFGNode* calleeRet = getFormalRetVFGNode(joinedge->getRHSVar());
852 addRetEdge(calleeRet->getId(),callsiteRev, getCallSiteID(joinedge->getCallSite(), calleeRet->getFun()));
853 }
854 }
855}

936void VFG::updateCallGraph(PointerAnalysis* pta)
937{
938 VFGEdgeSetTy vfEdgesAtIndCallSite;
939 PointerAnalysis::CallEdgeMap::const_iterator iter = pta->getIndCallMap().begin();
940 PointerAnalysis::CallEdgeMap::const_iterator eiter = pta->getIndCallMap().end();
941 for (; iter != eiter; iter++)
942 {
943 const CallICFGNode* newcs = iter->first;
944 assert(newcs->isIndirectCall() && "this is not an indirect call?");
945 const PointerAnalysis::FunctionSet & functions = iter->second;
946 for (PointerAnalysis::FunctionSet::const_iterator func_iter = functions.begin(); func_iter != functions.end(); func_iter++)
947 {
948 const SVFFunction* func = *func_iter;
949 connectCallerAndCallee(newcs, func, vfEdgesAtIndCallSite);
950 }

958void VFG::connectCallerAndCallee(const CallICFGNode* callBlockNode, const SVFFunction* callee, VFGEdgeSetTy& edges)
959{
960 SVFIR * pag = SVFIR::getPAG();
961 CallSiteID csId = getCallSiteID(callBlockNode, callee);
962 const RetICFGNode* retBlockNode = callBlockNode->getRetICFGNode();
963 // connect actual and formal param
964 if (pag->hasCallSiteArgsMap(callBlockNode) && pag->hasFunArgsList(callee) &&
965 matchArgs(callBlockNode, callee))
966 {
967 const SVFIR::SVFVarList& csArgList = pag->getCallSiteArgsList(callBlockNode);
968 const SVFIR::SVFVarList& funArgList = pag->getFunArgsList(callee);
969 SVFIR::SVFVarList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
970 SVFIR::SVFVarList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
971 for (; funArgIt != funArgEit && csArgIt != csArgEit; funArgIt++, csArgIt++)
972 {
973 const PAGNode *cs_arg = *csArgIt;
974 const PAGNode *fun_arg = *funArgIt;
975 if (isInterestedPAGNode(cs_arg) && isInterestedPAGNode(fun_arg))
976 connectAParamAndFParam(cs_arg, fun_arg, callBlockNode, csId, edges);
977 }
978 assert(funArgIt == funArgEit && "function has more arguments than call site");
979
980 if (callee->isVarArg())
981 {
982 NodeID varFunArg = pag->getVarargNode(callee);
983 const PAGNode* varFunArgNode = pag->getGNode(varFunArg);
984 if (isInterestedPAGNode(varFunArgNode))
985 {
986 for (; csArgIt != csArgEit; csArgIt++)
987 {
988 const PAGNode *cs_arg = *csArgIt;
989 if (isInterestedPAGNode(cs_arg))
990 connectAParamAndFParam(cs_arg, varFunArgNode, callBlockNode, csId, edges);
991 }
992 }
993 }
994 }
995
996 // connect actual return and formal return
997 if (pag->funHasRet(callee) && pag->callsiteHasRet(retBlockNode))
998 {
999 const PAGNode* cs_return = pag->getCallSiteRet(retBlockNode);
1000 const PAGNode* fun_return = pag->getFunRet(callee);
1001 if (isInterestedPAGNode(cs_return) && isInterestedPAGNode(fun_return))
1002 connectFRetAndARet(fun_return, cs_return, csId, edges);
1003 }
