82 PhiStmt* SVFIR::addPhiStmt(NodeID res, NodeID opnd, const ICFGNode* pred)
83 {

85  SVFVar* resNode = getGNode(res);

422 NodeID SVFIR::getGepObjVar(const MemObj* obj, const APOffset& apOffset)
423 {
424  NodeID base = obj->getId();
425 
427  if (obj->isFieldInsensitive())
428  return getFIObjVar(obj);
429 
430  APOffset newLS = pag->getSymbolInfo()->getModulusOffset(obj, apOffset);
431 
432  // Base and first field are the same memory location.
433  if (Options::FirstFieldEqBase() && newLS == 0) return base;
434 
435  NodeOffsetMap::iterator iter = GepObjVarMap.find(std::make_pair(base, newLS));
436  if (iter == GepObjVarMap.end())
437  {
438  NodeID gepId = NodeIDAllocator::get()->allocateGepObjectId(base, apOffset, Options::MaxFieldLimit());
439  return addGepObjNode(obj, newLS,gepId);
440  }
441  else
442  return iter->second;
443 
444 }

637 void SVFIR::initialiseCandidatePointers()
638 {
639  // collect candidate pointers for demand-driven analysis
640  for (iterator nIter = begin(); nIter != end(); ++nIter)
641  {
642  NodeID nodeId = nIter->first;
643  // do not compute points-to for isolated node
644  if (isValidPointer(nodeId) == false)
645  continue;
646  candidatePointers.insert(nodeId);
647  }
648 }
649 /*
650  * If this is a dummy node or node does not have incoming edges and outgoing edges we assume it is not a pointer here.
651  * However, if it is a pointer and it is an argument of a function definition, we assume it is a pointer here.
652  */
653 bool SVFIR::isValidPointer(NodeID nodeId) const
654 {
655  SVFVar* node = pag->getGNode(nodeId);
656 
657  if (node->hasValue() && node->isPointer())
658  {
659  if(const SVFArgument* arg = SVFUtil::dyn_cast<SVFArgument>(node->getValue()))
660  {
661  if (!(arg->getParent()->isDeclaration()))
662  return true;
663  }
664  }
665 
666  if ((node->getInEdges().empty() && node->getOutEdges().empty()))
667  return false;
668  return node->isPointer();
669 }
