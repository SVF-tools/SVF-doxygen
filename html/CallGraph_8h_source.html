348  inline CallSiteID addCallSite(const CallICFGNode* cs, const SVFFunction* callee)
349  {
350  std::pair<const CallICFGNode*, const SVFFunction*> newCS(std::make_pair(cs, callee));
351  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
352  //assert(it == csToIdMap.end() && "cannot add a callsite twice");
353  if(it == csToIdMap.end())
354  {
355  CallSiteID id = totalCallSiteNum++;
356  csToIdMap.insert(std::make_pair(newCS, id));
357  idToCSMap.insert(std::make_pair(id, newCS));
358  return id;
359  }
360  return it->second;
361  }
362  inline CallSiteID getCallSiteID(const CallICFGNode* cs, const SVFFunction* callee) const
363  {
364  CallSitePair newCS(std::make_pair(cs, callee));
365  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
366  assert(it != csToIdMap.end() && "callsite id not found! This maybe a partially resolved callgraph, please check the indCallEdge limit");
367  return it->second;
368  }
