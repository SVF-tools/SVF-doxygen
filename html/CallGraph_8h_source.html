277 inline CallSiteID addCallSite(const CallICFGNode* cs, const FunObjVar* callee)
278 {
279 std::pair<const CallICFGNode*, const FunObjVar*> newCS(std::make_pair(cs, callee));
280 CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
281 //assert(it == csToIdMap.end() && "cannot add a callsite twice");

283 {
284 CallSiteID id = totalCallSiteNum++;
