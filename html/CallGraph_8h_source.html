330  inline CallSiteID addCallSite(const CallICFGNode* cs, const SVFFunction* callee)
331  {
332  std::pair<const CallICFGNode*, const SVFFunction*> newCS(std::make_pair(cs, callee));
333  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
334  //assert(it == csToIdMap.end() && "cannot add a callsite twice");
335  if(it == csToIdMap.end())
336  {
337  CallSiteID id = totalCallSiteNum++;
338  csToIdMap.insert(std::make_pair(newCS, id));
339  idToCSMap.insert(std::make_pair(id, newCS));
340  return id;
341  }
342  return it->second;
343  }

348  assert(it != csToIdMap.end() && "callsite id not found! This maybe a partially resolved callgraph, please check the indCallEdge limit");
