131 inline bool removeCFLInEdge(CFLEdge* inEdge)
132 {
133 u32_t num1 = removeIncomingEdge(inEdge);
134
135 GrammarBase::Symbol s = inEdge->getEdgeKind();
136 u32_t num2 = inCFLEdges[s].erase(inEdge);
137
138 return num1 && num2;
