38 static bool hasEdge(const CHNode *src, const CHNode *dst,
39  CHEdge::CHEDGETYPE et)
40 {
41  for (CHEdge::CHEdgeSetTy::const_iterator it = src->getOutEdges().begin(),
42  eit = src->getOutEdges().end(); it != eit; ++it)
43  {
44  CHNode *node = (*it)->getDstNode();
45  CHEdge::CHEDGETYPE edgeType = (*it)->getEdgeType();
46  if (node == dst && edgeType == et)
47  return true;
48  }
49  return false;
50 }

52 static bool checkArgTypes(CallSite cs, const SVFFunction* fn)
53 {
54 
55  // here we skip the first argument (i.e., this pointer)
56  u32_t arg_size = (fn->arg_size() > cs.arg_size()) ? cs.arg_size(): fn->arg_size();
57  if(arg_size > 1)
58  {
59  for (unsigned i = 1; i < arg_size; i++)
60  {
61  auto cs_arg = cs.getArgOperand(i);
62  auto fn_arg = fn->getArg(i);
63  if (cs_arg->getType() != fn_arg->getType())
64  {
65  return false;
66  }
67  }
68  }
69 
70  return true;
71 }

73 void CHGraph::addEdge(const string className, const string baseClassName,
74  CHEdge::CHEDGETYPE edgeType)
75 {
76  CHNode *srcNode = getNode(className);
77  CHNode *dstNode = getNode(baseClassName);
78  assert(srcNode && dstNode && "node not found?");
79 
80  if (!hasEdge(srcNode, dstNode, edgeType))
81  {
82  CHEdge *edge = new CHEdge(srcNode, dstNode, edgeType);
83  srcNode->addOutgoingEdge(edge);

97  * Get virtual functions for callsite "cs" based on vtbls (calculated
98  * based on pointsto set)
99  */
100 void CHGraph::getVFnsFromVtbls(CallSite cs, const VTableSet &vtbls, VFunSet &virtualFunctions)
101 {
102 
104  size_t idx = cs.getFunIdxInVtable();
106  string funName = cs.getFunNameOfVirtualCall();
107  for (const SVFGlobalValue *vt : vtbls)
108  {
109  const CHNode *child = getNode(cppUtil::getClassNameFromVtblObj(vt->getName()));
110  if (child == nullptr)
111  continue;
112  CHNode::FuncVector vfns;
113  child->getVirtualFunctions(idx, vfns);
114  for (CHNode::FuncVector::const_iterator fit = vfns.begin(),
115  feit = vfns.end(); fit != feit; ++fit)
116  {
117  const SVFFunction* callee = *fit;
118  if (cs.arg_size() == callee->arg_size() ||
119  (cs.isVarArg() && callee->isVarArg()))
120  {
121 
122  // if argument types do not match
123  // skip this one
124  if (!checkArgTypes(cs, callee))
125  continue;
126 
127  cppUtil::DemangledName dname = cppUtil::demangle(callee->getName());
128  string calleeName = dname.funcName;
129 
130  /*
131  * The compiler will add some special suffix (e.g.,
132  * "[abi:cxx11]") to the end of some virtual function:
133  * In dealII
134  * function: FE_Q<3>::get_name
135  * will be mangled as: _ZNK4FE_QILi3EE8get_nameB5cxx11Ev
136  * after demangling: FE_Q<3>::get_name[abi:cxx11]
137  * The special suffix ("[abi:cxx11]") needs to be removed
138  */
139  const std::string suffix("[abi:cxx11]");
140  size_t suffix_pos = calleeName.rfind(suffix);
141  if (suffix_pos != string::npos)
142  calleeName.erase(suffix_pos, suffix.size());
143 
144  /*
145  * if we can't get the function name of a virtual callsite, all virtual
146  * functions calculated by idx will be valid
147  */
148  if (funName.size() == 0)
149  {
150  virtualFunctions.insert(callee);
151  }
152  else if (funName[0] == '~')
153  {
154  /*
155  * if the virtual callsite is calling a destructor, then all
156  * destructors in the ch will be valid
157  * class A { virtual ~A(){} };
158  * class B: public A { virtual ~B(){} };
159  * int main() {
160  * A *a = new B;
161  * delete a; /// the function name of this virtual callsite is ~A()
162  * }
163  */
164  if (calleeName[0] == '~')
165  {
166  virtualFunctions.insert(callee);
167  }
168  }
169  else
170  {
171  /*
172  * for other virtual function calls, the function name of the callsite
173  * and the function name of the target callee should match exactly
174  */
175  if (funName.compare(calleeName) == 0)
176  {
177  virtualFunctions.insert(callee);
178  }
179  }
180  }
181  }
182  }
183 }

196 void CHGraph::printCH()
197 {
198  for (CHGraph::const_iterator it = this->begin(), eit = this->end();
199  it != eit; ++it)
200  {
201  const CHNode *node = it->second;
202  outs() << "class: " << node->getName() << "\n";
203  for (CHEdge::CHEdgeSetTy::const_iterator it = node->OutEdgeBegin();
204  it != node->OutEdgeEnd(); ++it)
205  {
206  if ((*it)->getEdgeType() == CHEdge::INHERITANCE)
207  outs() << (*it)->getDstNode()->getName() << " --inheritance--> "
208  << (*it)->getSrcNode()->getName() << "\n";
209  else
210  outs() << (*it)->getSrcNode()->getName() << " --instance--> "
211  << (*it)->getDstNode()->getName() << "\n";
212  }
213  }
214  outs() << '\n';
215 }
