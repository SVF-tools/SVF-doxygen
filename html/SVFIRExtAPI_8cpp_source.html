44const Type* SVFIRBuilder::getBaseTypeAndFlattenedFields(const Value* V, std::vector<AccessPath> &fields, const Value* szValue)
45{
46 assert(V);
47 const Value* value = getBaseValueForExtArg(V);
48 const Type *objType = LLVMModuleSet::getLLVMModuleSet()->getTypeInference()->inferObjType(value);
49 u32_t numOfElems = pag->getNumOfFlattenElements(LLVMModuleSet::getLLVMModuleSet()->getSVFType(objType));
51 if(szValue && SVFUtil::isa<ConstantInt>(szValue))
52 {
53 auto szIntVal = LLVMUtil::getIntegerValue(SVFUtil::cast<ConstantInt>(szValue));
54 numOfElems = (numOfElems > szIntVal.first) ? szIntVal.first : numOfElems;
55 }
56
57 LLVMContext& context = LLVMModuleSet::getLLVMModuleSet()->getContext();
58 for(u32_t ei = 0; ei < numOfElems; ei++)
59 {
60 AccessPath ls(ei);
61 // make a ConstantInt and create char for the content type due to byte-wise copy
62 const ConstantInt* offset = ConstantInt::get(context, llvm::APInt(32, ei));
63 const SVFValue* svfOffset = LLVMModuleSet::getLLVMModuleSet()->getSVFValue(offset);
64 if (!llvmModuleSet()->hasValueNode(svfOffset))
65 {
66 SymbolTableBuilder builder(pag);
67 builder.collectSym(offset);
68 NodeID id = llvmModuleSet()->getValueNode(svfOffset);
69 pag->addConstantIntValNode(id, LLVMUtil::getIntegerValue(offset), nullptr, svfOffset->getType());
70 llvmModuleSet()->addToSVFVar2LLVMValueMap(offset,
71 pag->getGNode(id));
72 }
73 ls.addOffsetVarAndGepTypePair(getPAG()->getGNode(llvmModuleSet()->getValueNode(svfOffset)), nullptr);
74 fields.push_back(ls);
75 }
76 return objType;
77}

83void SVFIRBuilder::addComplexConsForExt(Value *D, Value *S, const Value* szValue)
84{
85 assert(D && S);
86 NodeID vnD= getValueNode(D), vnS= getValueNode(S);
87 if(!vnD || !vnS)
88 return;
89
90 std::vector<AccessPath> fields;
91
92 //Get the max possible size of the copy, unless it was provided.
93 std::vector<AccessPath> srcFields;
94 std::vector<AccessPath> dstFields;
95 const Type* stype = getBaseTypeAndFlattenedFields(S, srcFields, szValue);
96 const Type* dtype = getBaseTypeAndFlattenedFields(D, dstFields, szValue);
97 if(srcFields.size() > dstFields.size())
98 fields = dstFields;
99 else
100 fields = srcFields;
101
103 u32_t sz = fields.size();
104
105 if (fields.size() == 1 && (LLVMUtil::isConstDataOrAggData(D) || LLVMUtil::isConstDataOrAggData(S)))
106 {
107 NodeID dummy = pag->addDummyValNode();
108 addLoadEdge(vnD,dummy);
109 addStoreEdge(dummy,vnS);
110 return;
111 }
112
113 //For each field (i), add (Ti = *S + i) and (*D + i = Ti).
114 for (u32_t index = 0; index < sz; index++)
115 {
116 LLVMModuleSet* llvmmodule = LLVMModuleSet::getLLVMModuleSet();
117 const SVFType* dElementType = pag->getFlatternedElemType(llvmmodule->getSVFType(dtype),
118 fields[index].getConstantStructFldIdx());
119 const SVFType* sElementType = pag->getFlatternedElemType(llvmmodule->getSVFType(stype),
120 fields[index].getConstantStructFldIdx());
121 NodeID dField = getGepValVar(D,fields[index],dElementType);
122 NodeID sField = getGepValVar(S,fields[index],sElementType);
123 NodeID dummy = pag->addDummyValNode();
124 addLoadEdge(sField,dummy);
125 addStoreEdge(dummy,dField);
126 }
127}

129void SVFIRBuilder::handleExtCall(const CallBase* cs, const SVFFunction* svfCallee)
130{
131 const SVFInstruction* svfInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(cs);
132 const SVFCallInst* svfCall = SVFUtil::cast<SVFCallInst>(svfInst);
133 const CallICFGNode *callICFGNode = llvmModuleSet()->getCallICFGNode(cs);
134
135 if (isHeapAllocExtCallViaRet(callICFGNode))
136 {
137 NodeID val = llvmModuleSet()->getValueNode(svfInst);
138 NodeID obj = llvmModuleSet()->getObjectNode(svfInst);
139 addAddrWithHeapSz(obj, val, cs);
140 }
141 else if (isHeapAllocExtCallViaArg(callICFGNode))
142 {
143 u32_t arg_pos = getHeapAllocHoldingArgPosition(svfCallee);
144 const SVFValue* arg = svfCall->getArgOperand(arg_pos);
145 if (arg->getType()->isPointerTy())
146 {
147 NodeID vnArg = llvmModuleSet()->getValueNode(arg);
148 NodeID dummy = pag->addDummyValNode();
149 NodeID obj = pag->addDummyObjNode(arg->getType());
150 if (vnArg && dummy && obj)
151 {
152 addAddrWithHeapSz(obj, dummy, cs);
153 addStoreEdge(dummy, vnArg);
154 }
155 }
156 else
157 {
158 writeWrnMsg("Arg receiving new object must be pointer type");
159 }
160 }
161 else if (isMemcpyExtFun(svfCallee))
162 {
163 // Side-effects similar to void *memcpy(void *dest, const void * src, size_t n)
164 // which copies n characters from memory area 'src' to memory area 'dest'.
165 if(svfCallee->getName().find("iconv") != std::string::npos)
166 addComplexConsForExt(cs->getArgOperand(3), cs->getArgOperand(1), nullptr);
167 else if(svfCallee->getName().find("bcopy") != std::string::npos)
168 addComplexConsForExt(cs->getArgOperand(1), cs->getArgOperand(0), cs->getArgOperand(2));
169 if(svfCall->arg_size() == 3)
170 addComplexConsForExt(cs->getArgOperand(0), cs->getArgOperand(1), cs->getArgOperand(2));
171 else
172 addComplexConsForExt(cs->getArgOperand(0), cs->getArgOperand(1), nullptr);
173 if(SVFUtil::isa<PointerType>(cs->getType()))
174 addCopyEdge(getValueNode(cs->getArgOperand(0)), getValueNode(cs), CopyStmt::COPYVAL);
175 }
176 else if(isMemsetExtFun(svfCallee))
177 {
178 // Side-effects similar to memset(void *str, int c, size_t n)
179 // which copies the character c (an unsigned char) to the first n characters of the string pointed to, by the argument str
180 std::vector<AccessPath> dstFields;
181 const Type *dtype = getBaseTypeAndFlattenedFields(cs->getArgOperand(0), dstFields, cs->getArgOperand(2));
182 u32_t sz = dstFields.size();
183 //For each field (i), add store edge *(arg0 + i) = arg1
184 for (u32_t index = 0; index < sz; index++)
185 {
186 LLVMModuleSet* llvmmodule = LLVMModuleSet::getLLVMModuleSet();
187 const SVFType* dElementType = pag->getFlatternedElemType(llvmmodule->getSVFType(dtype),
188 dstFields[index].getConstantStructFldIdx());
189 NodeID dField = getGepValVar(cs->getArgOperand(0), dstFields[index], dElementType);
190 addStoreEdge(getValueNode(cs->getArgOperand(1)),dField);
191 }
192 if(SVFUtil::isa<PointerType>(cs->getType()))
193 addCopyEdge(getValueNode(cs->getArgOperand(0)), getValueNode(cs), CopyStmt::COPYVAL);
194 }
195 else if(svfCallee->getName().compare("dlsym") == 0)
196 {
197 /*
198 Side-effects of void* dlsym( void* handle, const char* funName),
199 Locate the function with the name "funName," then add a "copy" edge between the callsite and that function.
200 dlsym() example:
201 int main() {
202 // Open the shared library
203 void* handle = dlopen("./my_shared_library.so", RTLD_LAZY);
204 // Find the function address
205 void (*myFunctionPtr)() = (void (*)())dlsym(handle, "myFunction");
206 // Call the function
207 myFunctionPtr();
208 }
209 */
210 const Value* src = cs->getArgOperand(1);
211 if(const GetElementPtrInst* gep = SVFUtil::dyn_cast<GetElementPtrInst>(src))
212 src = stripConstantCasts(gep->getPointerOperand());
213
214 auto getHookFn = [](const Value* src)->const Function*
215 {
216 if (!SVFUtil::isa<GlobalVariable>(src))
217 return nullptr;
218
219 auto *glob = SVFUtil::cast<GlobalVariable>(src);
220 if (!glob->hasInitializer() || !SVFUtil::isa<ConstantDataArray>(glob->getInitializer()))
221 return nullptr;
222
223 auto *constarray = SVFUtil::cast<ConstantDataArray>(glob->getInitializer());
224 return LLVMUtil::getProgFunction(constarray->getAsCString().str());
225 };
226
227 if (const Function *fn = getHookFn(src))
228 {
229 NodeID srcNode = getValueNode(fn);
230 addCopyEdge(srcNode, getValueNode(cs), CopyStmt::COPYVAL);
231 }
232 }
233 else if(svfCallee->getName().find("_ZSt29_Rb_tree_insert_and_rebalancebPSt18_Rb_tree_node_baseS0_RS_") != std::string::npos)
234 {
235 // The purpose of this function is to insert a new node into the red-black tree and then rebalance the tree to ensure that the red-black tree properties are maintained.
236 assert(svfCall->arg_size() == 4 && "_Rb_tree_insert_and_rebalance should have 4 arguments.\n");
237
238 // We have vArg3 points to the entry of _Rb_tree_node_base { color; parent; left; right; }.
239 // Now we calculate the offset from base to vArg3
240 NodeID vnArg3 = llvmModuleSet()->getValueNode(svfCall->getArgOperand(3));
241 APOffset offset =
242 getAccessPathFromBaseNode(vnArg3).getConstantStructFldIdx();
243
244 // We get all flattened fields of base
245 vector<AccessPath> fields = pag->getTypeLocSetsMap(vnArg3).second;
246
247 // We summarize the side effects: arg3->parent = arg1, arg3->left = arg1, arg3->right = arg1
248 // Note that arg0 is aligned with "offset".
249 for (APOffset i = offset + 1; i <= offset + 3; ++i)
250 {
251 if((u32_t)i >= fields.size())
252 break;
253 const SVFType* elementType = pag->getFlatternedElemType(pag->getTypeLocSetsMap(vnArg3).first,
254 fields[i].getConstantStructFldIdx());
255 NodeID vnD = getGepValVar(cs->getArgOperand(3), fields[i], elementType);
256 NodeID vnS = llvmModuleSet()->getValueNode(svfCall->getArgOperand(1));
257 if(vnD && vnS)
258 addStoreEdge(vnS,vnD);
259 }
260 }
261
262 if (isThreadForkCall(callICFGNode))
263 {
264 const ValVar* valVar = getForkedFun(callICFGNode);
265 if (const FunValVar* funcValVar = SVFUtil::dyn_cast<FunValVar>(valVar))
266 {
267 const SVFFunction* forkedFun = funcValVar->getCallGraphNode()->getFunction()
268 ->getDefFunForMultipleModule();
269 const SVFVar* actualParm = getActualParmAtForkSite(callICFGNode);
272 assert((forkedFun->arg_size() <= 2) && "Size of formal parameter of start routine should be one");
273 if (forkedFun->arg_size() <= 2 && forkedFun->arg_size() >= 1)
274 {
275 const ArgValVar* formalParm = forkedFun->getArg(0);
277 if (actualParm->isPointer() && formalParm->getType()->isPointerTy())
278 {
279 FunEntryICFGNode *entry = pag->getICFG()->getFunEntryICFGNode(forkedFun);
280 addThreadForkEdge(actualParm->getId(), formalParm->getId(), callICFGNode, entry);
281 }
282 }
283 }
284 else
285 {
290 }
294 }
